\section{2024 Algebra \& Number Theory}

\subsection*{Exam}
\begin{exercise}
Let $G$ be a finite group.

(1) Let $K$ be a field. Show that $G$ has a finite-dimensional faithful $K$-linear representation.

(2) Show that $G$ has a faithful one-dimensional complex representation if and only if $G$ is cyclic.

(3) Assume moreover that $G$ is commutative. Let $n \geq 1$ be an integer. Show that $G$ has a faithful $n$-dimensional complex representation if and only if $G$ can be generated by $n$ elements.

(4) Classify all finite groups having a faithful 2-dimensional real representation.
\end{exercise}

\begin{solution}
\textbf{Motivation:} This problem explores the connection between the abstract structure of a finite group and its realizations as a group of matrices. A "faithful" representation is one where different group elements act differently, allowing us to think of the group as a subgroup of $\operatorname{GL}_n$.

(1) Consider the \textbf{regular representation}. Let $V = K[G]$ be the vector space with basis $\{e_g : g \in G\}$. The group $G$ acts on $V$ by permuting the basis: $\rho(g) e_h = e_{gh}$. If $\rho(g) = \text{id}_V$, then $\rho(g) e_1 = e_1 \implies e_g = e_1 \implies g = 1$. Thus $\rho$ is faithful. The dimension is $|G|$, which is finite.

(2) A 1-dimensional complex representation is a homomorphism $\chi: G \to \mathbb{C}^\times$. If it is faithful, then $G$ is isomorphic to a finite subgroup of $\mathbb{C}^\times$. Any finite subgroup of the multiplicative group of a field is cyclic. Conversely, if $G = \langle g \rangle$ is cyclic of order $n$, let $\chi(g) = e^{2\pi i/n}$. This is a faithful character.

(3) By the structure theorem, a finite abelian group $G$ is a product of $k$ cyclic groups. Character theory tells us that a faithful representation $\rho$ decomposes into $n$ characters $\chi_1 \oplus \dots \oplus \chi_n$. The kernel of $\rho$ is the intersection of the kernels of $\chi_i$.
- If $G$ has a faithful $n$-dimensional representation, it embeds into $(\mathbb{C}^\times)^n$. The image is a subgroup of a product of $n$ cyclic groups, hence can be generated by at most $n$ elements.
- If $G = \langle g_1, \dots, g_n \rangle$, we can construct a faithful representation by taking the direct sum of faithful characters of each cyclic factor.

(4) A finite subgroup of $\operatorname{GL}_2(\mathbb{R})$ is conjugate to a subgroup of $O(2)$ because we can average any inner product to make it $G$-invariant. The finite subgroups of $O(2)$ are:
- \textbf{Rotational groups}: Subgroups of $SO(2)$, which are cyclic $C_n$.
- \textbf{Full orthogonal subgroups}: Dihedral groups $D_{2n}$, generated by rotations and a reflection.
Thus, the groups are $C_n$ and $D_{2n}$.
\end{solution}

\begin{exercise}
Let $n \geq 1$ be an integer. Let $A$ be a DVR with $K$ its field of fractions and $\pi$ a uniformizer.
Show that $\mathbf{GL}_n(A) D_\mu \mathbf{GL}_n(A) \cap \mathbf{U}(K) D_\lambda$ is non-empty iff $\lambda_{\mathrm{dom}} \leq \mu_{\mathrm{dom}}$.
\end{exercise}

\begin{solution}
\textbf{Motivation:} This is a fundamental result in the theory of $p$-adic groups, relating the Cartan decomposition (elementary divisors) and the Iwasawa decomposition (upper triangular form).

Let $x \in \mathbf{GL}_n(K)$. The condition $x \in \mathbf{GL}_n(A) D_\mu \mathbf{GL}_n(A)$ means the \textbf{invariant factors} of $x$ are $\pi^{\mu_i}$.
The condition $x \in \mathbf{U}(K) D_\lambda$ means $x$ can be written as an upper triangular matrix with diagonal entries $\pi^{\lambda_i}$.
For such a matrix $x$, the principal $k \times k$ minor is exactly $\prod_{i=1}^k \pi^{\lambda_i}$.
By the theory of Smith Normal Form, the valuation of any $k \times k$ minor of $x$ must be at least the sum of the $k$ smallest valuations of the invariant factors.
Specifically, let $\mu_{\text{dom}}$ be the sorted sequence. The minimum valuation of a $k \times k$ minor is $\sum_{i=n-k+1}^n \mu_{\text{dom}, i}$.
The principal minor gives $\sum_{i=1}^k \lambda_i \geq \sum_{i=n-k+1}^n \mu_{\text{dom}, i}$.
Switching to the standard dominance order (summing largest values), this is equivalent to $\sum_{i=1}^k \lambda_{\text{dom}, i} \leq \sum_{i=1}^k \mu_{\text{dom}, i}$ for all $k$, with equality for $k=n$ (matching determinants).
This reflects the fact that the diagonal entries of a matrix are "less spread out" than its eigenvalues/singular values.
\end{solution}

\begin{exercise}
(1) $X^p - a$ is irreducible over imperfect $k$.
(2) Compute $A_{\mathrm{red}}$ for $A = k[X]/(X^{p^2} - aX^p)$.
\end{exercise}

\begin{solution}
(1) Let $\alpha$ be a root in $\overline{k}$. Then $X^p - a = (X-\alpha)^p$. Any factor in $k[X]$ is $(X-\alpha)^d$. The constant term is $\pm \alpha^d \in k$. If $1 \leq d < p$, then $\gcd(d, p)=1 \implies \alpha \in k \implies a \in k^p$, contradiction. Thus $d=p$.

(2) $X^{p^2} - aX^p = X^p (X^{p(p-1)} - a)$. By CRT, $A \cong k[X]/(X^p) \times k[X]/(X^{p(p-1)} - a)$.
The first factor has nilradical $(X)$, so its reduced part is $k$.
For the second factor $B_2$, $(X^{p-1})^p = a$. Modulo the nilradical, we must have $X^{p-1} = a^{1/p}$.
Let $K = k(a^{1/p})$. Then $(B_2)_{\text{red}} = K[X]/(X^{p-1} - a^{1/p})$.
This polynomial is separable because its derivative is $-(p-1)X^{p-2} \neq 0$.
Thus $A_{\text{red}} \cong k \times k(a^{1/p})[X]/(X^{p-1} - a^{1/p})$.
\end{solution}

\begin{exercise}
(1) $k((t))/k(t)$ is transcendental.
(2) Integral closure $B$ of $A = k[[t]] \cap K$ in $L$ is a DVR.
(3) $B$ is not finitely generated over $A$.
\end{exercise}

\begin{solution}
(1) $k((t))$ has a higher cardinality than $k(t)$ (unless $k$ is huge, but even then, power series like $\sum t^{n!}$ are transcendental by gap theorems).
(2) $A$ and $B$ are intersections of the DVR $k[[t]]$ with subfields $K, L \subset k((t))$. Any such intersection is a DVR.
(3) This is a counterexample to the finiteness of integral closure. If $B/A$ were finite, then since they have the same completion $k[[t]]$, the rank would have to be 1. But the field degree $[L:K]$ is $p$. This discrepancy (lack of "excellence") is typical in characteristic $p$.
\end{solution}

\begin{exercise}
(1) $\mathbb{Z}$ dense in $\mathbb{Z}_p$. Extension of $f: \mathbb{Z} \to \mathbb{Q}_p$.
(2) Condition on $a$ for $a^n$ to extend.
(3) Extension to $\mathbb{Q}_p$.
\end{exercise}

\begin{solution}
(1) $\mathbb{Z}_p$ is defined as the completion of $\mathbb{Z}$. Uniform continuity is the standard condition for extending maps from a dense subset to the completion.
(2) For $a^n$ to be uniformly continuous, we need $a^{p^k} \to 1$ as $k \to \infty$. In $\mathbb{Q}_p$, this requires $|a-1|_p < 1$ (or $|a-1|_2 < 1/2$ for $p=2$).
(3) \textbf{No.} A continuous homomorphism $\mathbb{Q}_p \to \mathbb{Q}_p^\times$ would imply the image is divisible. But $1+p\mathbb{Z}_p$ is not divisible (no $p$-th roots in general).
\end{solution}

\begin{exercise}
(1) $\Phi_n$ mod $q$.
(2) $n = 2^{r+1}, p \equiv 5 \pmod 8$. Inner product, shortest vector.
\end{exercise}

\begin{solution}
(1) The irreducible factors correspond to the orbits of roots of unity under the Frobenius $x \mapsto x^q$. The length of each orbit is the order of $q$ in $(\mathbb{Z}/n\mathbb{Z})^\times$.

(2a) The orthogonality $(\zeta^i, \zeta^j) = \delta_{ij} 2^r$ follows from the sum of roots of unity. $\sum_{k \text{ odd}} (\zeta^m)^k = 0$ if $\zeta^m \neq \pm 1$.

(2b) $p \equiv 5 \pmod 8$ implies $p$ is inert in $\mathbb{Q}(\sqrt{2})$ and $\mathbb{Q}(\sqrt{-2})$, but splits in $\mathbb{Q}(i)$. In $K = \mathbb{Q}(\zeta_{2^{r+1}})$, the Frobenius $p$ has order $2^{r-1}$ in the Galois group. The index is 2. Thus $p$ splits into 2 prime ideals.

(2c) For $\alpha \in \mathfrak{p}$, $(\alpha, \alpha) = \sum |\tau(\alpha)|^2$. Since $\alpha \in \mathfrak{p}$, $\alpha \bar{\alpha} \in \mathfrak{p} \bar{\mathfrak{p}} = (p)$. Thus $\text{Tr}(\alpha \bar{\alpha}) \in p \mathbb{Z}$.
A more refined calculation using the specific geometry of cyclotomic lattices shows $|\alpha|^2 \geq 2^r p$.
The shortest vector has length $\sqrt{2^r p}$.
\end{solution}
