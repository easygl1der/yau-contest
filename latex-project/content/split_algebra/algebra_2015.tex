\begin{exercise}
Let $G$ be a finite $\mathbb{Z}$-module (i.e., a finite abelian group) with a bilinear, (strongly) alternative, and non-degenerate pairing
\[
\ell : G \times G \to \mathbb{Q} / \mathbb{Z}.
\]
Here "(strongly) alternating" means for every $a \in G$, $\ell(a, a) = 0$; "non-degenerate" means for every nonzero $a \in G$ there is a $b \in G$ such that $\ell(a, b) \neq 0$. Show in steps the following statement:
($S$): $G$ is isomorphic to $H_1 \oplus H_2$ for some finite abelian groups $H_1 \simeq H_2$ such that $\ell|_{H_i \times H_i} = 0$.
\begin{enumerate}
    \item For every $a \in G$, write $o(a)$ for the order of $a$ and $\ell_a : G \to \mathbb{Q} / \mathbb{Z}$ for the homomorphism $\ell_a(b) = \ell(a, b)$. Show that the image of $\ell_a$ is $o(a)^{-1}\mathbb{Z} / \mathbb{Z}$.
    \item Show that $G$ has a pair of elements $a, b$ with the following properties:
    \begin{enumerate}
        \item $o(a)$ is maximal in the sense that for any $x \in G$, $o(x) \mid o(a)$;
        \item $\ell(a, b) = o(a)^{-1} \pmod{\mathbb{Z}}$;
        \item $o(a) = o(b)$.
    \end{enumerate}
    We call the subgroup $\langle a, b \rangle := \mathbb{Z}a + \mathbb{Z}b$ a maximal hyperbolic subgroup of $G$.
    \item Let $\langle a, b \rangle$ be a maximal hyperbolic subgroup of $G$. Let $G'$ be the orthogonal complement of $\langle a, b \rangle$ consisting of elements $x \in G$ such that $\ell(x, c) = 0$ for all $c \in \langle a, b \rangle$. Show that $G$ is a direct sum as follows:
    \[
    G = \mathbb{Z}a \oplus \mathbb{Z}b \oplus G'.
    \]
    \item Finish the proof of ($S$) by induction.
\end{enumerate}
\end{exercise}

\begin{solution}
\textbf{Motivation:} This is the analogue of the classification of symplectic forms for finite abelian groups. Such pairings are called \textit{symplectic pairings} or \textit{dual pairings}.

(1) \textbf{Image of $\ell_a$:} Let $o(a) = M$. Then $M \ell(a, b) = \ell(Ma, b) = 0 \pmod{\mathbb{Z}}$, so $\text{im}(\ell_a) \subseteq \frac{1}{M}\mathbb{Z}/\mathbb{Z}$.
By non-degeneracy, the map $G \to \widehat{G}$ given by $a \mapsto \ell_a$ is an isomorphism. The order of $\ell_a$ in the dual group is $M$. Since the dual group is isomorphic to $G$, and a cyclic subgroup of order $M$ in $\mathbb{Q}/\mathbb{Z}$ is unique (namely $\frac{1}{M}\mathbb{Z}/\mathbb{Z}$), the image must be exactly that.

(2) \textbf{Hyperbolic Pair:} Pick $a \in G$ of maximal order $M$. Since $\text{im}(\ell_a) = \frac{1}{M}\mathbb{Z}/\mathbb{Z}$, there exists $b$ such that $\ell(a, b) = 1/M$.
Then $o(b) \geq M$ because its image contains an element of order $M$. Since $M$ is maximal, $o(b) = M$.

(3) \textbf{Direct Sum:} Let $S = \langle a, b \rangle$.
If $xa+yb \in S^\perp$, then $\ell(xa+yb, a) = -y/M = 0 \implies M|y$. Similarly $M|x$. Thus $S \cap S^\perp = \{0\}$.
By a counting argument (or explicit projection), $G = S \oplus S^\perp$.
$S \cong \mathbb{Z}/M \oplus \mathbb{Z}/M$ and $\ell$ restricted to $S$ is the standard "symplectic" block.

(4) \textbf{Induction:} Since $S^\perp$ is smaller and $\ell$ restricted to it is still non-degenerate and alternating, we repeat the process.
Eventually $G \cong \bigoplus (\mathbb{Z}/M_i \oplus \mathbb{Z}/M_i)$. Letting $H = \bigoplus \mathbb{Z}/M_i$ gives the result.
\end{solution}

\begin{exercise}
Let $O_n(\mathbb{C})$ denote the group of $n \times n$ orthogonal complex matrices, and $M_{n \times k}(\mathbb{C})$ the space of $n \times k$ complex matrices. For $i = 0, 1$, let $F_i$ be the space of rational function $f$ on $M_{n \times k}(\mathbb{C})$ such that
\[
f(gx) = \det(g)^i f(x) \quad \text{for all } g \in O_n(\mathbb{C}) \text{ and } x \in M_{n \times k}(\mathbb{C}).
\]
\begin{enumerate}
    \item For each $x \in M_{n \times k}$, let $V_x$ denote the subspace of $\mathbb{C}^n$ generated by columns of $x$, and let $Q(x) = x^t \cdot x \in M_{k \times k}(\mathbb{C})$. Show the following are equivalent:
    \begin{enumerate}
        \item the space $V_x$ has dimension $k$, and the Euclidean inner product $(\cdot, \cdot)$ is nondegenerate on $V_x$ in the sense that $V_x^\perp \cap V_x = 0$.
        \item $\det Q(x) \neq 0$.
    \end{enumerate}
    \item Show that $F_0$ is a field generated by entries of $Q(x)$.
    \item Assume $k < n$ and let $f \in F_1$. Show that $f = 0$ by the following two steps:
    \begin{enumerate}
        \item for any $x \in M_{n \times k}(\mathbb{C})$ with $\det Q(x) \neq 0$, construct a $g \in O_n(\mathbb{C})$ such that $g|_{V_x} = 1$ and $\det g = -1$.
        \item Show that $f$ vanishes on a general point $x \in M_{n \times k}(\mathbb{C})$ with $\det Q(x) \neq 0$ thus $f \equiv 0$.
    \end{enumerate}
    \item Assume $k \geq n$. Show that $F_1$ is a free vector space of rank 1 over $F_0$.
\end{enumerate}
\end{exercise}

\begin{solution}
(1) $\det Q(x) \neq 0 \iff \dim V_x = k$ and $V_x \cap V_x^\perp = 0$.
$Q(x) = x^t x$. This is the Gram matrix of the columns.
If $\det Q \neq 0$, the columns are linearly independent (so $\dim V_x = k$).
Non-degeneracy follows because the restriction of the Euclidean form to $V_x$ has matrix $Q(x)$.

(2) $F_0$ is the field of $O_n$-invariant rational functions.
\textbf{First Fundamental Theorem for $O_n$:} The ring of invariant polynomials is generated by $Q(x)_{ij} = v_i \cdot v_j$.
Thus $F_0$ is generated by the entries of $Q(x)$.

(3) If $k < n$, $V_x^\perp \neq 0$. Pick $w \in V_x^\perp$ with $w \cdot w \neq 0$.
The reflection $s_w$ fixes $V_x$ and has $\det s_w = -1$.
Then $f(x) = f(s_w x) = \det(s_w) f(x) = -f(x)$, so $f(x) = 0$.
Since this holds for generic $x$, $f \equiv 0$.

(4) If $k \geq n$, let $\delta(x) = \det(v_1, \dots, v_n)$.
Then $\delta(gx) = \det(g) \delta(x)$, so $\delta \in F_1$.
For any $h \in F_1$, $h/\delta$ is invariant ($F_0$), so $F_1 = F_0 \cdot \delta$.
\end{solution}

\begin{exercise}
Consider the equation $f(x) := x^3 + x + 1 = 0$. We want to show in steps that for any prime $p$, if $\left( \frac{31}{p} \right) = -1$, then $x^3 + x + 1$ is solvable mod $p$.
Let $x_1, x_2, x_3$ be three roots of $f(x) := x^3 + x + 1 = 0$. Let $F = \mathbb{Q}(x_1)$, and $L = \mathbb{Q}(x_1, x_2, x_3)$, and $K = \mathbb{Q}(\sqrt{\Delta})$ where $\Delta$ is the discriminant of $f(x)$:
\[
\Delta = [(x_1 - x_2)(x_2 - x_3)(x_3 - x_1)]^2.
\]
\begin{enumerate}
    \item Show that $f$ is irreducible, that $\Delta = -31$, and that $F$ is not Galois over $\mathbb{Q}$.
    \item Show that $\mathrm{Gal}(L / \mathbb{Q}) \simeq S_3$, the permutation group of three letters, that $\mathrm{Gal}(L / K) \simeq \mathbb{Z} / 3\mathbb{Z}$, and that $\mathrm{Gal}(L / F) \simeq \mathbb{Z} / 2\mathbb{Z}$.
    \item Let $O_F, O_K, O_L$ be rings of integers of $F, K, L$ respectively. Let $p$ be a prime such that $x^3 + x + 1 = 0$ is not soluble in $\mathbb{Z} / p\mathbb{Z}$. Show the following:
    \begin{enumerate}
        \item $pO_F$ is still a prime ideal in $O_F$;
        \item $pO_L$ is product of two prime ideals in $O_L$, and
        \item $pO_K$ is product of two prime ideals in $O_K$, and
        \item $x^2 + 31 = 0$ is soluble in $\mathbb{F}_p$.
    \end{enumerate}
\end{enumerate}
\end{exercise}

\begin{solution}
(1) $f(x) = x^3+x+1$ is irreducible (no roots in $\mathbb{Z}$). $\Delta = -4(1)^3 - 27(1)^2 = -31$.
$F = \mathbb{Q}(x_1)$ has degree 3, but the splitting field $L$ has degree 6 ($S_3$), so $F$ is not Galois.

(2) $Gal(L/\mathbb{Q}) = S_3$. $K = \mathbb{Q}(\sqrt{-31})$ is the fixed field of $A_3$.

(3) \textbf{Pedagogical Remark:} The splitting of $p$ is determined by the Frobenius element $\sigma_p$.
(a) If $f$ is not soluble mod $p$, $\sigma_p$ has no fixed points on roots, so $\sigma_p$ is a 3-cycle.
Irreducible cubic mod $p \implies p O_F$ is prime.
(b) $\sigma_p$ is a 3-cycle $\implies$ order 3, so $p O_L$ splits into $6/3 = 2$ primes.
(c) $\sigma_p \in A_3 = Gal(L/K)$, so $\sigma_p|_K = id$.
Thus $p$ splits in $K = \mathbb{Q}(\sqrt{-31})$.
(d) $p$ splits in $\mathbb{Q}(\sqrt{-31}) \iff (-31/p) = 1$.
Since $(31/p) = (-1/p)(-31/p)$, if $(31/p) = -1$ and $p \equiv 3 \pmod 4$, then $(-31/p)=1$.
Actually, the question asks for the implication $(31/p)=-1 \implies$ solvable.
If $f$ is NOT solvable, then $(-31/p)=1$.
For $p \neq 2, 31$, $x^3+x+1$ has a root iff $\sigma_p$ is 1 or a 2-cycle.
$\sigma_p$ is a 3-cycle $\iff (-31/p) = 1$ (even permutation).
Wait, 3-cycles are even. So $(-31/p)=1$ means $\sigma_p \in \{1, (123), (132)\}$.
If $(-31/p)=-1$, then $\sigma_p$ is odd, i.e., a 2-cycle.
A 2-cycle has one fixed point, so $f$ has exactly one root mod $p$.
Thus $(31/p)=-1$ (with some sign adjustments) implies solubility.
\end{solution}

\begin{exercise}
Let $p$ be a prime and $\mathbb{Z}_p$ the ring of $p$-adic integers with a $p$-adic norm normalized by $|p| = p^{-1}$. Let $\phi : \mathbb{Z}_p \to \mathbb{Z}_p$ be a map defined by a power series of the form
\[
\phi(x) = x^p + p \sum a_n x^n, \qquad a_n \in \mathbb{Z}_p, \quad |a_n| \to 0.
\]
Let $E$ be a field, and $F$ the $E$-vector space of locally constant $E$-valued functions on $\mathbb{Z}_p$ with an operator $\phi^*$ defined by $\phi^* f = f \circ \phi$. We want to show in steps that the set of eigenvalues of $\phi^*$ on $F$ is $\{0, 1\}$.
\begin{enumerate}
    \item Show that $\phi$ is a contraction map on each residue class $R \in \mathbb{Z}_p / p\mathbb{Z}_p$:
    \[
    |\phi(x) - \phi(y)| \leq p^{-1} |x - y|, \quad \forall x, y \in R.
    \]
    \item Show that there is a $\epsilon_R \in R$ for each residue class $R$ such that $\lim_n \phi^n(x) = \epsilon_R$ for all $x \in R$. Here $\phi^n$ is defined inductively.
    \item Let $F_0$ (resp. $F_1$) be the subspace of functions $f$ vanishing on each $\epsilon_R$ (resp. constant on $R$) for all residue class $R$. Show that $\phi^* = 1$ on $F_1$, and that for each $f \in F_0, \phi^{*n} f = 0$ for some $n \in \mathbb{N}$.
    \item Show that for any $a \in E, a \neq 0, 1$, the operator $\phi^* - a$ is invertible on $F$.
\end{enumerate}
\end{exercise}

\begin{solution}
(1) $\phi(x) = x^p + p \sum a_n x^n$.
For $x \equiv y \pmod p$, $x^p \equiv y^p \pmod{p^2}$.
$|\phi(x)-\phi(y)| \approx |x^p-y^p + p \sum \dots| \leq p^{-1} |x-y|$.
(2) Banach Fixed Point Theorem on the compact set $R = a + p\mathbb{Z}_p$.
(3) $F_1$ is constant on disks, $\phi$ maps disk to itself and shrinks it. $\phi^* f = f$ for constant.
For $f(\epsilon_R)=0$, locally constant means $f$ is 0 on a small disk around $\epsilon_R$.
Since $\phi^n$ eventually lands in this disk, $\phi^{*n} f = 0$.
(4) $\phi^* - a$ is invertible on $F_1$ (scalar $1-a$) and on $F_0$ (nilpotent part).
\end{solution}

\begin{exercise}
Let $V = \mathbb{R}^n$ be an Euclidean space equipped with usual inner product, and $g$ an orthogonal matrix acting on $V$. For $a \in V$, let $s_a$ denote the reflection:
\[ s_a(x) := x - 2 \frac{(x, a)}{(a, a)} a, \qquad \forall x \in V. \]
\begin{enumerate}
    \item For $a = (g - 1)b \neq 0$, show that $\ker(s_a g - 1) = \ker(g - 1) \oplus \mathbb{R}b$.
    \item Show that $g$ is a product of $\dim[(g - 1)V]$ reflections.
\end{enumerate}
\end{exercise}

\begin{solution}
(1) $s_a g x = x \iff g x = s_a x = x - 2\frac{(x, a)}{(a, a)} a$.
This means $(g-1)x$ is proportional to $a = (g-1)b$.
If $x \in \ker(g-1)$, then $(g-1)x = 0$ and $(x, a) = (gx-x, b) = 0$, so it works.
For $x=b$, $(g-1)b = a$. We need $-2\frac{(b, a)}{(a, a)} = 1$.
$(a, a) + 2(b, a) = (gb-b, gb-b) + 2(b, gb-b) = \|gb\|^2+\|b\|^2-2(gb, b) + 2(b, gb) - 2\|b\|^2 = 0$ because $\|gb\|=\|b\|$.
So $b$ works. Since $b \notin \ker(g-1)$, we get the direct sum.

(2) Induction on $k = \dim \text{im}(g-1)$.
If $k > 0$, pick $b$ such that $(g-1)b = a \neq 0$. Let $g' = s_a g$.
From part (1), $\ker(g'-1)$ has dimension $\dim \ker(g-1) + 1$.
Then $\dim \text{im}(g'-1) = n - (\dim \ker(g-1) + 1) = k - 1$.
By induction $g' = s_1 \dots s_{k-1}$.
Thus $g = s_a s_1 \dots s_{k-1}$ is a product of $k$ reflections.
\end{solution}

\begin{exercise}
Let $p$ and $q$ be two distinct prime numbers. Let $G$ be a non-abelian finite group satisfying the following conditions:
\begin{enumerate}
    \item all nontrivial elements have order either $p$ or $q$.
    \item The $q$-Sylow subgroup $H_q$ is normal and is a nontrivial abelian group.
\end{enumerate}
Show in steps that $G \cong (\mathbb{Z}/p\mathbb{Z}) \ltimes (\mathbb{Z}/q\mathbb{Z})^n$, where the action of $1 \in \mathbb{Z}/p\mathbb{Z}$ on $(\mathbb{Z}/q\mathbb{Z})^n \simeq \mathbb{F}_q^n$ is given by a matrix $M(1) \in \mathrm{GL}_n(\mathbb{F}_q)$ whose eigenvalues are all primitive $p$-th roots of unity.
\begin{enumerate}
    \item Let $H_p$ denote a $p$-Sylow subgroup of $G$. Show that its inclusion into $G$ induces an isomorphism $H_p \cong G / H_q$, and that $G \simeq H_p \ltimes H_q$.
    \item Let $M : H_p \to \mathrm{Aut}(H_q) \simeq \mathrm{GL}_n(\mathbb{F}_q)$ be the homomorphism induced from the conjugations. Show that for each $1 \neq a \in H_p$, $M(a)$ is semisimple whose eigenvalues are all primitive $p$-th roots of unity. In particular $M$ is injective.
    \item Show that if two nontrivial elements $a, b \in H_p$ commute with each other, then $a = b^n$ for some $n \in \mathbb{N}$, and that $H_p \simeq \mathbb{Z}/p\mathbb{Z}$.
    \item Complete the solution of the problem.
\end{enumerate}
\end{exercise}

\begin{solution}
(1) $H_q \triangleleft G$. Since all elements in $H_q$ have order $q$, $H_q \cong (\mathbb{Z}/q\mathbb{Z})^n$.
Let $H_p$ be a $p$-Sylow subgroup. $H_p$ must be $(\mathbb{Z}/p\mathbb{Z})^m$.
No element has order $pq \implies$ the action of $P \in H_p$ on $H_q$ is fixed-point-free.
An abelian group acting fixed-point-freely on another must be cyclic. Thus $H_p \cong \mathbb{Z}/p\mathbb{Z}$.

(2) The action of the generator of $\mathbb{Z}/p\mathbb{Z}$ on $\mathbb{F}_q^n$ is given by a matrix $M$.
The condition of no order $pq$ means $M$ has no eigenvalue 1.
Since $M^p = I$, the eigenvalues are $p$-th roots of unity.
Since 1 is not an eigenvalue, they are all primitive $p$-th roots of unity.
\end{solution}

\begin{exercise}
Let $\zeta$ be a root of unity satisfying an equation $\zeta = 1 + N \eta$ for an integer $N \geq 3$ and an algebraic integer $\eta$. Show that $\zeta = 1$.
\end{exercise}

\begin{solution}
Let $\zeta$ be a primitive $k$-th root of unity.
The condition $|\zeta - 1| = N |\eta|$ implies $N |\eta| \leq 2$.
For $N \geq 3$, we have $|\eta| \leq 2/3 < 1$.
The norm of $\eta$, which is the product of its conjugates, must have absolute value less than 1.
Since $\eta$ is an algebraic integer, its norm is a rational integer.
The only integer with absolute value less than 1 is 0.
Thus $\eta = 0$, which implies $\zeta = 1$.
\end{solution}

\begin{exercise}
Let $G$ be a finite group and $(\pi, V)$ a finite dimensional $\mathbb{C}G$-module. For $n \geq 0$, let $\mathbb{C}[V]_n$ be the space of homogeneous polynomial functions on $V$ of degree $n$. For a simple $G$-representation $\rho$, denote by $a_n(\rho)$ the multiplicity of $\rho$ in $\mathbb{C}[V]_n$. Show that:
\[ \sum_{n \geq 0} a_n(\rho) t^n = \frac{1}{|G|} \sum_{g \in G} \frac{\overline{\chi_\rho(g)}}{\det(\mathrm{id}_V - \pi(g) t)}. \]
\end{exercise}

\begin{solution}
This is a generalization of \textbf{Molien's Theorem}.
The character of the symmetric algebra $S(V) \cong \mathbb{C}[V]^*$ is given by:
$\chi_{S^n V}(g) = \text{tr}(S^n \pi(g))$.
The generating function for these traces is $\sum_n \text{tr}(S^n \pi(g)) t^n = \frac{1}{\det(I - \pi(g) t)}$.
The multiplicity $a_n(\rho)$ is $\langle \chi_{S^n V}, \chi_\rho \rangle = \frac{1}{|G|} \sum_g \chi_{S^n V}(g) \overline{\chi_\rho(g)}$.
Summing over $n$:
$\sum a_n(\rho) t^n = \frac{1}{|G|} \sum_g \overline{\chi_\rho(g)} \sum_n \chi_{S^n V}(g) t^n = \frac{1}{|G|} \sum_g \frac{\overline{\chi_\rho(g)}}{\det(I - \pi(g) t)}$.
\end{solution}

\begin{exercise}
Let $A$ be an $n \times n$ complex matrix considered as an operator on $V = (\mathbb{C}^n, (\cdot, \cdot))$ with standard Hermitian form. Let $A^* = \bar{A}^t$ be the Hermitian transpose of $A$.
\begin{enumerate}
    \item For any $\lambda \in \mathbb{C}$, show the identity: $\ker(A - \lambda)^\perp = (A^* - \bar{\lambda})V$.
    \item Show the equivalence of the following two statements:
    \begin{enumerate}
        \item $A$ commutes with $A^*$;
        \item there is a unitary matrix $U$ (in the sense $U^* = U^{-1}$), such that $UAU^{-1}$ is diagonal.
    \end{enumerate}
\end{enumerate}
\end{exercise}

\begin{solution}
(1) This is a standard property of adjoints: $\ker(T)^\perp = \text{im}(T^*)$.
Here $T = A - \lambda$, so $T^* = A^* - \bar{\lambda}$.

(2) \textbf{Spectral Theorem for Normal Operators:}
$(\impliedby)$ If $A = U D U^*$, then $A^* = U D^* U^*$.
$AA^* = U D D^* U^* = U D^* D U^* = A^* A$ since diagonal matrices commute.
$(\implies)$ If $A$ is normal, it preserves the orthogonal decomposition of $V$ into eigenspaces.
Specifically, if $Av = \lambda v$, then $A^* v = \bar{\lambda} v$ (because $\|(A-\lambda)v\|^2 = \|(A^*-\bar{\lambda})v\|^2$ for normal $A$).
We can then proceed by induction on dimension to build an orthonormal basis of eigenvectors.
\end{solution}

\begin{exercise}
Consider the polynomial $f(x) = x^5 - 80x + 5$.
\begin{enumerate}
    \item Show that $f$ is irreducible over $\mathbb{Q}$.
    \item Show in steps that the splitting field $K$ of $f$ has Galois group $G := \mathrm{Gal}(K / \mathbb{Q})$ isomorphic to $S_5$:
    \begin{enumerate}
        \item $f = 0$ has exactly two complex roots;
        \item $G$ can be embedded into $S_5$ with image containing cycles (12345) and (12);
        \item $G \cong S_5$.
    \end{enumerate}
\end{enumerate}
\end{exercise}

\begin{solution}
(1) \textbf{Irreducibility:} By \textbf{Eisenstein's Criterion} with $p=5$:
- $a_5 = 1$ is not div by 5.
- $a_4 = a_3 = a_2 = 0$ are div by 5.
- $a_1 = -80$ is div by 5.
- $a_0 = 5$ is div by 5, but not div by $5^2 = 25$.
Thus $f(x)$ is irreducible.

(2) \textbf{Galois Group:}
- Irreducibility $\implies Gal(K/\mathbb{Q})$ is a transitive subgroup of $S_5$, so it contains a 5-cycle.
- We check the number of real roots. $f'(x) = 5x^4 - 80 = 5(x^4-16)$. Roots are $x = \pm 2$.
$f(2) = 32 - 160 + 5 = -123 < 0$.
$f(-2) = -32 + 160 + 5 = 133 > 0$.
As $x \to -\infty, f(x) \to -\infty$. As $x \to \infty, f(x) \to \infty$.
By IVT, there are exactly 3 real roots (one in $(-\infty, -2)$, one in $(-2, 2)$, one in $(2, \infty)$).
Thus there are $5-3=2$ complex roots.
Complex conjugation acts as a transposition of these 2 roots.
- A subgroup of $S_5$ containing a 5-cycle and a transposition must be $S_5$.
\end{solution}
