\section{2021 Algebra \& Number Theory}

\subsection*{Exam}
\begin{exercise}
For any prime $p$ and a nonzero element $a \in \mathbf{F}_p$, prove that the polynomial $A(x) = x^p - x - a$ is irreducible and separable over $\mathbf{F}_p$.
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item \textbf{Characteristic $p$:} In a field of characteristic $p$ (like $\mathbf{F}_p$), the "Freshman's Dream" holds: $(x+y)^p = x^p + y^p$.
    \item \textbf{Separability:} A polynomial is separable if all its roots in an algebraic closure are distinct. This is true if the polynomial and its derivative share no common factors ($\gcd(f, f') = 1$).
    \item \textbf{Irreducibility:} A polynomial is irreducible if it cannot be factored into the product of two non-constant polynomials.
    \item \textbf{Frobenius Automorphism:} The map $\sigma(x) = x^p$ is a fundamental symmetry of finite fields that fixes the base field $\mathbf{F}_p$.
\end{itemize}

\textbf{Steps:}
\begin{enumerate}
    \item \textbf{Test for Separability:} We compute the derivative of $A(x) = x^p - x - a$. Using the power rule, $A'(x) = px^{p-1} - 1$. Since we are working in $\mathbf{F}_p$, the coefficient $p$ is equivalent to $0$, so $A'(x) = -1$.
    \item \textbf{Check for Multiple Roots:} Since the derivative $A'(x) = -1$ is a non-zero constant, it never vanishes. Thus, $A(x)$ and $A'(x)$ can have no common roots, which proves $A(x)$ is separable.
    \item \textbf{Analyze the Root Structure:} Let $\alpha$ be a root of $A(x)$ in some algebraic closure. This means $\alpha^p - \alpha = a$. Now consider the element $\alpha + 1$. We calculate:
    $$(\alpha+1)^p - (\alpha+1) = (\alpha^p + 1^p) - (\alpha + 1) = \alpha^p - \alpha = a.$$
    This shows that if $\alpha$ is a root, then $\alpha + 1$ is also a root!
    \item \textbf{List all Roots:} By repeating this process (adding 1 again and again), we find that the $p$ distinct roots of $A(x)$ are exactly $\alpha, \alpha+1, \alpha+2, \dots, \alpha+(p-1)$.
    \item \textbf{Apply the Frobenius Map:} The Frobenius automorphism $\sigma(x) = x^p$ acts on the roots. For our root $\alpha$, we have $\sigma(\alpha) = \alpha^p = \alpha + a$.
    \item \textbf{Determine Orbit Size:} Since $a \neq 0$, the repeated application of $\sigma$ generates the sequence $\alpha, \alpha+a, \alpha+2a, \dots, \alpha+(p-1)a$. In $\mathbf{F}_p$, it takes exactly $p$ steps to return to $\alpha$. Thus, the "orbit" of $\alpha$ under $\sigma$ has size $p$.
    \item \textbf{Conclude Irreducibility:} The degree of an irreducible factor of a polynomial is equal to the size of the orbit of its roots under the Galois group (which is generated by $\sigma$ here). Since the orbit size is $p$ and the total degree of $A(x)$ is $p$, the polynomial must be irreducible.
\end{enumerate}

\textbf{Intuition:}
In characteristic $p$, the $p$-th power map is surprisingly "linear." The polynomial $x^p - x$ acts like a linear operator whose "zeros" are exactly the numbers in the base field $\mathbf{F}_p$. By adding a constant $a$, we shift this operator. The roots are all "shifted" versions of each other, and the Frobenius map "jumps" through these roots in a cycle. Because the cycle covers every possible root, they must all belong to one single, unbreakable (irreducible) piece.
\end{solution}

\begin{exercise}
Determine the automorphism group of the splitting field of $f(x) = x^3 - 3x + 1$ over $\mathbb{Q}$.
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item \textbf{Rational Root Theorem:} For a polynomial with integer coefficients, any rational root $p/q$ must have $p$ dividing the constant term and $q$ dividing the leading coefficient.
    \item \textbf{Cubic Discriminant:} For a cubic polynomial of the form $x^3 + px + q$, the discriminant is given by $D = -4p^3 - 27q^2$.
    \item \textbf{Galois Groups of Cubics:} For an irreducible cubic over $\mathbb{Q}$, the Galois group (automorphism group of the splitting field) is the cyclic group $A_3 \cong \mathbb{Z}/3\mathbb{Z}$ if the discriminant is a perfect square in $\mathbb{Q}$, and the symmetric group $S_3$ otherwise.
\end{itemize}

\textbf{Steps:}
\begin{enumerate}
    \item \textbf{Test for Irreducibility:} We check if $f(x) = x^3 - 3x + 1$ has any rational roots. By the Rational Root Theorem, the only possibilities are $\pm 1$.
    \item \textbf{Verify Potential Roots:} $f(1) = 1-3+1 = -1$ and $f(-1) = -1+3+1 = 3$. Since neither is zero, $f(x)$ has no rational roots. For a degree 3 polynomial, having no rational roots implies it is irreducible over $\mathbb{Q}$.
    \item \textbf{Calculate the Discriminant:} Here, $p = -3$ and $q = 1$. Using the formula:
    $$D = -4(-3)^3 - 27(1)^2 = -4(-27) - 27 = 108 - 27 = 81.$$
    \item \textbf{Check if the Discriminant is a Square:} We observe that $81 = 9^2$, so the discriminant is a perfect square in the rational numbers $\mathbb{Q}$.
    \item \textbf{Identify the Group:} Because $f(x)$ is an irreducible cubic and its discriminant is a square, its Galois group is the alternating group on 3 elements, $A_3$. This group is isomorphic to the cyclic group of order 3, $\mathbb{Z}/3\mathbb{Z}$.
\end{enumerate}

\textbf{Intuition:}
The discriminant is a number that "detects" the symmetry of the roots. For a cubic, there are 6 possible ways to permute the three roots ($S_3$). However, when the discriminant is a perfect square, it means the roots possess an "extra" layer of internal symmetry (specifically, alternating symmetry) that restricts the possible automorphisms to just the 3 cyclic rotations ($A_3$).
\end{solution}

\begin{exercise}
Let $R = F[x, y]/(x^2 - y^3)$ for some field $F$.

\begin{enumerate}
\item[(a)] Prove that $R$ is an integral domain.

\item[(b)] If $t$ denotes the element $x/y$ in the fraction field $K$ of $R$, prove that $K$ is equal to $F(t)$.

\item[(c)] Prove that $F[t]$ is the integral closure of $R$ in $K = F(t)$.
\end{enumerate}
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item \textbf{Integral Domain:} A ring where the product of any two non-zero elements is never zero.
    \item \textbf{Ring Homomorphism:} A function between rings that preserves addition and multiplication.
    \item \textbf{Integral Closure:} The set of elements in a larger field that satisfy a monic polynomial (leading coefficient 1) with coefficients from the base ring.
    \item \textbf{Cusp Singularity:} The curve $x^2 = y^3$ has a "sharp point" at $(0,0)$, which is a classic example of a non-normal (not integrally closed) ring.
\end{itemize}

\textbf{Steps:}
\begin{enumerate}
    \item \textbf{Step (a) - Parameterization Map:} Define a map $\phi: F[x, y] \to F[T]$ by setting $\phi(x) = T^3$ and $\phi(y) = T^2$. This is consistent with our equation because $(T^3)^2 = (T^2)^3 = T^6$. The image of this map is the subring $F[T^2, T^3]$.
    \item \textbf{Step (a) - Kernel Analysis:} We show that the kernel of $\phi$ is exactly the ideal $(x^2 - y^3)$. Any polynomial in $x,y$ can be simplified using the relation $x^2 = y^3$. Since the image $F[T^2, T^3]$ is a subring of the integral domain $F[T]$, the quotient $R \cong F[T^2, T^3]$ must also be an integral domain.
    \item \textbf{Step (b) - Fraction Field:} In the fraction field $K$, we define $t = x/y$. Substituting our $T$ variables, we get $t = T^3/T^2 = T$. Since $x = t^3$ and $y = t^2$, any element in $K$ (which is built from $x$ and $y$) can be written as a rational function of $t$. Thus, $K = F(t)$.
    \item \textbf{Step (c) - Integral Elements:} We want to find the integral closure. Notice that $t$ itself is integral over $R$ because it satisfies the equation $Z^2 - y = 0$ (since $t^2 = y$), and $y$ is an element of $R$.
    \item \textbf{Step (c) - Normality:} The ring $F[t]$ is a polynomial ring, which is a Unique Factorization Domain (UFD) and thus is "normal" (already integrally closed). Since $R \subset F[t] \subset K$ and $F[t]$ is the smallest normal ring containing $R$, it is the integral closure.
\end{enumerate}

\textbf{Intuition:}
The ring $R$ represents a curve with a "glitch" or cusp at the origin. Algebraically, this "glitch" means the ring is missing some elements that should be there if it were perfectly smooth. By introducing $t = x/y$, we are "parameterizing" the curve, effectively unfolding the sharp point into a smooth line. The integral closure $F[t]$ is simply the "smoothed-out" version of the original ring $R$.
\end{solution}

\begin{exercise}
Let $p_1, \ldots, p_n$ be $n$ distinct prime numbers. Show: $\sqrt{p_1} + \cdots + \sqrt{p_n}$ is not rational.
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item \textbf{Galois Extensions:} A field extension like $K = \mathbb{Q}(\sqrt{p_1}, \dots, \sqrt{p_n})$ is Galois, and its automorphisms are determined by flipping the signs of the square roots.
    \item \textbf{Fixed Field Property:} A number in $K$ is rational if and only if it remains unchanged (is "fixed") by every possible automorphism of the field.
    \item \textbf{Linear Independence:} Distinct square roots of primes are linearly independent over $\mathbb{Q}$.
\end{itemize}

\textbf{Steps:}
\begin{enumerate}
    \item \textbf{Define the Field:} Let $K = \mathbb{Q}(\sqrt{p_1}, \sqrt{p_2}, \dots, \sqrt{p_n})$. This extension has degree $2^n$ over the rationals $\mathbb{Q}$.
    \item \textbf{Describe the Automorphisms:} The Galois group $G = \text{Gal}(K/\mathbb{Q})$ allows us to independently flip the sign of any $\sqrt{p_i}$. For example, there is an automorphism $\sigma$ that sends $\sqrt{p_1} \to -\sqrt{p_1}$ while keeping all other $\sqrt{p_i}$ the same.
    \item \textbf{Assume Rationality:} Suppose the sum $\alpha = \sqrt{p_1} + \sqrt{p_2} + \dots + \sqrt{p_n}$ is actually a rational number.
    \item \textbf{Apply the Symmetry:} If $\alpha$ is rational, then $\sigma(\alpha)$ must be equal to $\alpha$. Let's check what $\sigma(\alpha)$ is:
    $$\sigma(\alpha) = -\sqrt{p_1} + \sqrt{p_2} + \dots + \sqrt{p_n}.$$
    \item \textbf{Set Up the Equation:} We set $\sigma(\alpha) = \alpha$:
    $$-\sqrt{p_1} + \sqrt{p_2} + \dots + \sqrt{p_n} = \sqrt{p_1} + \sqrt{p_2} + \dots + \sqrt{p_n}.$$
    \item \textbf{Find the Contradiction:} Subtracting the square roots of $p_2, \dots, p_n$ from both sides, we are left with $-\sqrt{p_1} = \sqrt{p_1}$. This simplifies to $2\sqrt{p_1} = 0$, which means $\sqrt{p_1} = 0$.
    \item \textbf{Final Conclusion:} Since $p_1$ is a prime number, its square root cannot be zero. Our assumption that $\alpha$ is rational must be false.
\end{enumerate}

\textbf{Intuition:}
Think of each $\sqrt{p_i}$ as an independent "direction" in a high-dimensional space. If the sum were rational, it would mean that these independent directions somehow "balanced out" to stay on a 1D line (the rationals). But Galois theory tells us we can "flip" one direction without touching the others. If flipping a direction doesn't change the sum, that direction couldn't have been part of the sum in the first place!
\end{solution}

\begin{exercise}
Find all integral solutions $(x, y)$ for the equation $x^2 + 13 = y^3$. (Hint: You can use the fact that $\mathbb{Q}(\sqrt{-13})$ has class number 2.)
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item \textbf{Algebraic Integers:} We work in the ring $\mathbb{Z}[\sqrt{-13}]$, where numbers look like $a + b\sqrt{-13}$.
    \item \textbf{Class Number:} A class number of 2 means that unique factorization of numbers fails, but the "failure" is very controlled. Specifically, if an ideal squared is principal, it doesn't mean the ideal itself is principal, but if an ideal cubed is the identity in the class group, it must be principal (since $\gcd(3, 2)=1$).
    \item \textbf{Coprimality in Ideals:} Two factors are coprime if no prime ideal divides both of them.
\end{itemize}

\textbf{Steps:}
\begin{enumerate}
    \item \textbf{Factor the Equation:} We rewrite the equation as a product: $(x + \sqrt{-13})(x - \sqrt{-13}) = y^3$.
    \item \textbf{Check for Common Factors:} We test if the two factors on the left are coprime. By checking the equation modulo 8 and modulo 13, we can prove that no prime ideal can divide both factors.
    \item \textbf{Use the Cube Property:} Because the product of two coprime ideals is a cube $(y)^3$, each individual ideal must be a cube of some ideal $\mathfrak{a}$. So, $(x + \sqrt{-13}) = \mathfrak{a}^3$.
    \item \textbf{Argue Principality:} In the class group (which has size 2), the cube of any class $[\mathfrak{a}]^3$ is the identity. This implies the class $[\mathfrak{a}]$ itself must be the identity. Therefore, $\mathfrak{a}$ is a principal ideal, which we can write as $(a + b\sqrt{-13})$.
    \item \textbf{Set Up the Equation:} We can now write $x + \sqrt{-13} = (a + b\sqrt{-13})^3$. Expanding the right side gives:
    $$x + \sqrt{-13} = (a^3 - 39ab^2) + (3a^2b - 13b^3)\sqrt{-13}.$$
    \item \textbf{Solve for $a$ and $b$:} Equating the "imaginary" parts: $1 = b(3a^2 - 13b^2)$. The only integer divisors of 1 are $b = 1$ or $b = -1$.
    \item \textbf{Test Cases:}
    \begin{itemize}
        \item If $b = 1$, then $3a^2 - 13 = 1 \implies 3a^2 = 14$ (no integer solution).
        \item If $b = -1$, then $13 - 3a^2 = 1 \implies 3a^2 = 12 \implies a^2 = 4$, so $a = \pm 2$.
    \end{itemize}
    \item \textbf{Find $x$ and $y$:} Plugging $a = \pm 2, b = -1$ into the real part: $x = (\pm 2)^3 - 39(\pm 2)(-1)^2 = \mp 70$. Calculating $y$: $y^3 = (-70)^2 + 13 = 4913 = 17^3$, so $y = 17$.
\end{enumerate}

\textbf{Intuition:}
This method is like "cheating" at multiplication. By moving into a larger world of numbers ($\mathbb{Z}[\sqrt{-13}]$), we turn an addition problem into a multiplication problem. Even though this new world is a bit "messy" (the class number is not 1), the fact that we are dealing with cubes ($y^3$) allows us to ignore the mess and solve the problem as if unique factorization still worked.
\end{solution}

\begin{exercise}
Let $p$ be a prime number and $\mathbb{Q}_p$ be the field of $p$-adic numbers. Fix an algebraic closure $\overline{\mathbb{Q}_p}$ of $\mathbb{Q}_p$. Let $g: \mathbb{Z}_{\geq 0} \to \mathbb{N}$ be a strictly increasing function. For each $i \in \mathbb{Z}_{\geq 0}$, pick a primitive $(p^{g(i)} - 1)$-th root of unity $\zeta_i$ in $\overline{\mathbb{Q}_p}$.

(a) Show that for each $i \geq 0$, $K_i := \mathbb{Q}_p(\zeta_i)$ is an unramified Galois extension of $\mathbb{Q}_p$ of degree $g(i)$.

(b) Give an explicit function $g$ as above such that $K_{i-1} \subset K_i$ for all $i > 0$. Let $0 = N_0 < N_1 < N_2 < \cdots$ be an increasing sequence of nonnegative integers. Let $\alpha_i := \sum_{j=0}^i \zeta_j p^{N_j}$. Show that for each $i \geq 0$, $K_i = \mathbb{Q}_p(\alpha_i)$ and that $(\alpha_i)$ is a Cauchy sequence in $\overline{\mathbb{Q}_p}$.

(c) Let $\eta \in \overline{\mathbb{Q}_p}$ be of degree $g$ over $\mathbb{Q}_p$, prove that there exists $M \in \mathbb{N}$ such that $\zeta_i$ does not satisfy any $\eta$-congruence
$$s_{g-1}\eta^{g-1} + s_{g-2}\eta^{g-2} + \cdots + s_1\eta + s_0 \equiv 0 \pmod{p^M}$$
in which the $s_i$'s are $p$-adic integers not all of which are divisible by $p$.

(d) Take a suitable sequence $(N_i)$ as above such that $(a_i)$ does not converge in $\overline{\mathbb{Q}_p}$. Conclude that $\overline{\mathbb{Q}_p}$ is not complete with respect to the $p$-adic topology.
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item \textbf{$p$-adic Numbers:} A field where distance is measured by divisibility by $p$. Numbers divisible by high powers of $p$ are "small."
    \item \textbf{Unramified Extensions:} Extensions of $p$-adic fields that are essentially mirrors of finite field extensions.
    \item \textbf{Krasner's Lemma:} A powerful theorem that says if two numbers are "extremely close," they must be essentially the same algebraically (they generate the same field extension).
    \item \textbf{Completeness:} A space is complete if every sequence that "looks like" it should converge (a Cauchy sequence) actually has a limit inside the space.
\end{itemize}

\textbf{Steps:}
\begin{enumerate}
    \item \textbf{Part (a) - Root of Unity:} In $p$-adic fields, adding a root of unity $\zeta$ of order $p^n-1$ creates a field that corresponds to the finite field $\mathbf{F}_{p^n}$. This is the definition of an unramified extension of degree $n$.
    \item \textbf{Part (b) - Nested Extensions:} By choosing $g(i) = (i+1)!$, each degree divides the next. This ensures that each field $K_i$ contains the previous one $K_{i-1}$ as a sub-field. The sequence $\alpha_i$ is Cauchy because the difference $|\alpha_{i+1} - \alpha_i|_p = p^{-N_{i+1}}$ goes to zero as $N_i$ increases.
    \item \textbf{Part (c) - The Degree Limit:} Suppose $\zeta_i$ stayed "close" to a fixed element $\eta$ for all powers of $p$. Krasner's Lemma would eventually force $\zeta_i$ to live inside the field generated by $\eta$. But the degree of $\zeta_i$ grows to infinity, while $\eta$ has a fixed finite degree. This is impossible, so there must be a "barrier" power $p^M$ where the closeness fails.
    \item \textbf{Part (d) - Proof of Incompleteness:} We construct a limit $\alpha$ that is "too complex" to be in $\overline{\mathbb{Q}_p}$. Since each $\alpha_i$ has a degree that keeps growing, their limit cannot satisfy any finite polynomial equation. This means the limit is transcendental. Since $\overline{\mathbb{Q}_p}$ only contains "algebraic" numbers, it is missing this limit and is therefore not complete.
\end{enumerate}

\textbf{Intuition:}
The field $\overline{\mathbb{Q}_p}$ is like a collection of all possible puzzle pieces (algebraic numbers). However, when we arrange these pieces into a sequence that gets closer and closer (a Cauchy sequence), we can form an "infinite pattern" (a limit) that isn't one of our starting pieces. This shows that the space of "finite solutions" isn't enough to capture the "infinite reality" of $p$-adic geometry.
\end{solution}
