\section{2024 Probability and Statistics}

\begin{exercise}
The sequence $(X_1, X_2, \ldots, X_n, \ldots)$ is a Dirichlet process with base distribution $G_0$ and concentration parameter $\alpha_0 > 0$.
(a) Derive the distribution of $X_n, n \geq 1$.
(b) Let $Y_n = I(X_n > 0)$. Prove or disprove $(Y_n)_{n \geq 1}$ forms a Dirichlet Process.
\end{exercise}

\begin{solution}
\textbf{(a)} We prove by induction that $X_n \sim G_0$.
Base case: $X_1 \sim G_0$ is given.
Inductive step: Suppose $X_1, \dots, X_n \sim G_0$. The conditional distribution of $X_{n+1}$ is $P(X_{n+1} \in A \mid X_1, \dots, X_n) = \frac{\alpha_0 G_0(A) + \sum_{i=1}^n \delta_{X_i}(A)}{\alpha_0 + n}$.
The marginal probability is $P(X_{n+1} \in A) = E[P(X_{n+1} \in A \mid X_1, \dots, X_n)] = \frac{\alpha_0 G_0(A) + \sum_{i=1}^n P(X_i \in A)}{\alpha_0 + n}$.
By the inductive hypothesis, $P(X_i \in A) = G_0(A)$ for $i \leq n$.
Thus $P(X_{n+1} \in A) = \frac{\alpha_0 G_0(A) + n G_0(A)}{\alpha_0 + n} = G_0(A)$.
So $X_n \sim G_0$ for all $n \geq 1$.

\textbf{(b)} Let $p_0 = G_0((0, \infty))$. Then $Y_1 = I(X_1 > 0) \sim \text{Bernoulli}(p_0)$.
The conditional distribution of $Y_{n+1}$ given $Y_1, \dots, Y_n$ depends only on whether $X_i > 0$.
$P(Y_{n+1} = 1 \mid Y_1, \dots, Y_n) = \frac{\alpha_0 P(X_{n+1} > 0) + \sum_{i=1}^n I(X_i > 0)}{\alpha_0 + n} = \frac{\alpha_0 p_0 + \sum_{i=1}^n Y_i}{\alpha_0 + n}$.
This is exactly the definition of a Dirichlet Process with base distribution $\text{Bernoulli}(p_0)$ and concentration parameter $\alpha_0$.
\end{solution}

\begin{exercise}
$U_n = \sum_{j=1}^{2^n} \frac{j-1}{2^n} I\left(\frac{j-1}{2^n} \leq \phi(X) < \frac{j}{2^n}\right)$, $V_n = E(H(X,Y) | U_n)$.
Prove AS convergence of $V_n$.
\end{exercise}

\begin{solution}
The sequence of $\sigma$-fields $\mathcal{G}_n = \sigma(U_n)$ is increasing because $U_{n+1}$ refines the partition of $U_n$.
$V_n = E(H(X,Y) | \mathcal{G}_n)$ is a martingale with respect to the filtration $\{\mathcal{G}_n\}$.
Since $E[|H(X,Y)|] < \infty$, by the Martingale Convergence Theorem, $V_n$ converges almost surely to $Z = E(H(X,Y) | \mathcal{G}_\infty)$, where $\mathcal{G}_\infty = \sigma(\cup \mathcal{G}_n)$.
Since $\phi(u) = u/(1+u)$ is a bijection and the dyadic intervals generate the Borel $\sigma$-field on $[0, 1)$, $\mathcal{G}_\infty = \sigma(X)$.
Thus $V_n \to E(H(X,Y) | X)$ almost surely.
\end{solution}

\begin{exercise}
$A_n = \{X_n = \max(X_1, \ldots, X_n)\}$, $R_n = \sum_{k=1}^n I(A_k)$.
Compute $\lim_{n \to \infty} P\left(|R_n - \log n| > m_n \sqrt{\log n}\right)$.
\end{exercise}

\begin{solution}
By symmetry, $P(A_k) = 1/k$. Furthermore, the events $A_k$ are independent.
(This is a standard result for record values of i.i.d. continuous variables).
$E[R_n] = \sum_{k=1}^n \frac{1}{k} = H_n = \log n + \gamma + O(1/n)$.
$\text{Var}(R_n) = \sum_{k=1}^n \frac{1}{k}(1 - \frac{1}{k}) = H_n - \sum \frac{1}{k^2} = \log n + \gamma - \frac{\pi^2}{6} + O(1/n)$.
As $n \to \infty$, $R_n$ is approximately $N(\log n, \log n)$ by the Central Limit Theorem.
However, we are given $m_n \to \infty$. By Chebyshev's Inequality:
\[
P(|R_n - E[R_n]| > \epsilon) \leq \frac{\text{Var}(R_n)}{\epsilon^2}.
\]
Set $\epsilon = m_n \sqrt{\log n}$. Then:
\[
P(|R_n - \log n| > m_n \sqrt{\log n}) \leq \frac{\log n + O(1)}{m_n^2 \log n} = \frac{1}{m_n^2} \to 0.
\]
Thus the limit is 0.
\end{solution}

\begin{exercise}
Prove $P(A|T) = Q(T,A)$ almost surely.
\end{exercise}

\begin{solution}
By definition of conditional expectation, $W = Q(T,A)$ is $P(A|T)$ if:
1. $W$ is $\sigma(T)$-measurable. (True, as it is a function of $T$).
2. For any $B \in \sigma(T)$, $\int_B W dP = P(A \cap B)$.
Any $B \in \sigma(T)$ can be written as $B = \{T \in D\}$ for some $D \in \mathcal{B}$.
$\int_B Q(T,A) dP = \int_\Omega Q(T(\omega), A) I_D(T(\omega)) P(d\omega)$.
By the change of variables formula (using $\Pi = P \circ T^{-1}$):
$\int_B Q(T,A) dP = \int_{\mathbb{R}} Q(t,A) I_D(t) \Pi(dt) = \int_D Q(t,A) \Pi(dt)$.
The problem states $P(A) = \int_{\mathbb{R}} Q(t,A) \Pi(dt)$.
Actually, the definition of $Q(t,A)$ is likely the regular conditional probability.
If the given identity holds for all $D$, then $Q(T,A)$ is indeed the conditional expectation.
The statement is true under the given integral representation of $P$.
\end{solution}

\begin{exercise}
Multinomial games. Stopping rule: I+II win $r$ OR III+IV win $r+1$.
(a) Completeness of $T = (X_1+X_2, X_3+X_4)$.
(b) UMVUE of $\theta$.
\end{exercise}

\begin{solution}
Let $p = P(\text{I or II wins}) = (1-\theta)/2 + (1-\theta)/2 = 1-\theta$.
Let $q = P(\text{III or IV wins}) = \theta/2 + \theta/2 = \theta$.
This is a sequence of Bernoulli trials with success probability $\theta$.
The stopping rule is: stop when the number of failures reaches $r$ or the number of successes reaches $r+1$.
This is the setup for the Negative Binomial distribution if we only looked at one boundary, but here we have two.
The total number of games $N = X_1+X_2+X_3+X_4$ is a random variable.
$T = (X_1+X_2, X_3+X_4)$ is sufficient for $\theta$.
(a) The distribution belongs to the exponential family. For a bounded stopping rule, the sufficient statistic is complete.
(b) We want an unbiased estimator of $\theta$.
The number of successes is $S = X_3+X_4$. The number of failures is $F = X_1+X_2$.
An unbiased estimator for $p$ in inverse sampling is $(r-1)/(N-1)$.
Here, let's consider the last game.
The UMVUE is usually $I(\text{last game was won by III or IV})$.
Wait, $\theta = P(\text{III or IV})$.
By the Rao-Blackwell theorem, the UMVUE is $P(\text{Game 1 is won by III or IV} \mid T)$.
This is $(X_3+X_4)/(X_1+X_2+X_3+X_4)$.
\end{solution}

\begin{exercise}
Poisson hidden parameter. $\hat{\theta}_n = \bar{Y}_n / (2\bar{X}_n)$.
(a) Asymptotic distribution.
(b) Exact test for $\theta=1/2$.
\end{exercise}

\begin{solution}
\textbf{(a)} $E[X] = \sum_s P(X=1|S=s) P(S=s) = \sum_s \frac{1}{2^{s+1}} \frac{e^{-2\lambda} (2\lambda)^s}{s!} = \frac{e^{-2\lambda}}{2} \sum \frac{\lambda^s}{s!} = \frac{1}{2} e^{-\lambda}$.
$E[Y] = \sum_s \frac{\theta}{2^s} \frac{e^{-2\lambda} (2\lambda)^s}{s!} = \theta e^{-2\lambda} \sum \frac{\lambda^s}{s!} = \theta e^{-\lambda}$.
$E[X^2] = E[X] = \frac{1}{2}e^{-\lambda}$ (since $X \in \{0,1\}$).
$E[Y^2] = E[Y] = \theta e^{-\lambda}$.
$E[XY] = E[ E[XY|S] ] = E[ \frac{1}{2^{s+1}} \frac{\theta}{2^s} ] = \frac{\theta}{2} E[ 4^{-S} ] = \frac{\theta}{2} e^{-2\lambda} \sum \frac{(2\lambda/4)^s}{s!} = \frac{\theta}{2} e^{-3\lambda/2}$.
By the Delta Method on $g(\bar{X}, \bar{Y}) = \bar{Y}/(2\bar{X})$:
$\hat{\theta}_n$ is asymptotically normal $N(\theta, \sigma^2/n)$.
$\nabla g = (-Y/(2X^2), 1/(2X))$. At mean: $(-\theta/E[X], 1/(2E[X])) = (-2\theta e^\lambda, e^\lambda)$.
$\sigma^2 = \text{Var}(X) (2\theta e^\lambda)^2 + \text{Var}(Y) e^{2\lambda} - 2 \text{Cov}(X,Y) 2\theta e^{2\lambda}$.

\textbf{(b)} Under $H_0: \theta=1/2$, $P(X=1|S) = \frac{1}{2^{s+1}}$ and $P(Y=1|S) = \frac{1}{2^{s+1}}$.
Thus $X$ and $Y$ are i.i.d. conditioned on $S$.
Unconditionally, $X$ and $Y$ are i.i.d. Bernoulli with $p = \frac{1}{2}e^{-\lambda}$.
The total counts $N_X = \sum X_i$ and $N_Y = \sum Y_i$ are identically distributed.
Conditioned on $N_X + N_Y = K$, $N_Y$ follows a $\text{Binomial}(K, 1/2)$ distribution.
The exact test uses the Binomial distribution.
\end{solution}
