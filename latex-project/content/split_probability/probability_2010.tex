\section{2010 Probability \& Statistics}

\subsection{Individual}

\begin{exercise}
Let $Z_1, \dots, Z_n$ be i.i.d. random variables with $Z_i \sim N(\mu, \sigma^2)$. Find
\[
E \left( \sum_{i=1}^n Z_i \mid Z_1 - Z_2 + Z_3 \right).
\]
\end{exercise}

\begin{solution}
Let $X = \sum_{i=1}^n Z_i$ and $Y = Z_1 - Z_2 + Z_3$. Since $Z_i$ are i.i.d. normal, $(X, Y)$ has a bivariate normal distribution.
For bivariate normal variables, the conditional expectation is given by:
\[
E[X|Y] = E[X] + \frac{\text{Cov}(X, Y)}{\text{Var}(Y)} (Y - E[Y]).
\]
We calculate the required terms:
1. $E[X] = E[\sum_{i=1}^n Z_i] = n\mu$.
2. $E[Y] = E[Z_1 - Z_2 + Z_3] = \mu - \mu + \mu = \mu$.
3. $\text{Var}(Y) = \text{Var}(Z_1 - Z_2 + Z_3) = \text{Var}(Z_1) + \text{Var}(-Z_2) + \text{Var}(Z_3) = \sigma^2 + \sigma^2 + \sigma^2 = 3\sigma^2$.
4. $\text{Cov}(X, Y) = \text{Cov}(\sum_{i=1}^n Z_i, Z_1 - Z_2 + Z_3) = \text{Cov}(Z_1+Z_2+Z_3, Z_1-Z_2+Z_3) = \text{Cov}(Z_1, Z_1) + \text{Cov}(Z_2, -Z_2) + \text{Cov}(Z_3, Z_3) = \sigma^2 - \sigma^2 + \sigma^2 = \sigma^2$.
Substituting these into the formula:
\[
E[X|Y] = n\mu + \frac{\sigma^2}{3\sigma^2} (Y - \mu) = n\mu + \frac{1}{3}(Z_1 - Z_2 + Z_3 - \mu).
\]
\end{solution}

\begin{exercise}
Let $X_1, \dots, X_n$ be pairwise independent. Further, assume that $E X_i = 1 + i^{-1}$ and that $\max_{1 \leq i \leq n} E |X_i|^{1+\epsilon} < \infty$ for some $\epsilon > 0$. Show that
\[
\frac{1}{n} \sum_{i=1}^n X_i \xrightarrow{P} 1.
\]
\end{exercise}

\begin{solution}
Let $\bar{X}_n = \frac{1}{n} \sum_{i=1}^n X_i$. We want to show $\bar{X}_n \xrightarrow{P} 1$.
First, consider the expectation:
\[
E[\bar{X}_n] = \frac{1}{n} \sum_{i=1}^n (1 + i^{-1}) = 1 + \frac{1}{n} \sum_{i=1}^n \frac{1}{i} = 1 + \frac{H_n}{n}.
\]
Since $H_n \sim \log n$, $\frac{H_n}{n} \to 0$ as $n \to \infty$. Thus $E[\bar{X}_n] \to 1$.
To show convergence in probability, we can use the Weak Law of Large Numbers for pairwise independent variables.
A sufficient condition for $\bar{X}_n - E[\bar{X}_n] \xrightarrow{P} 0$ with pairwise independence is the existence of a common bound on the $(1+\epsilon)$-th moments.
Specifically, if $X_i$ are pairwise i.i.d. or satisfy $\frac{1}{n^2} \sum \text{Var}(X_i) \to 0$, the result holds.
Given $\max E|X_i|^{1+\epsilon} < K$, the variables are uniformly integrable.
By the truncation method used in the proof of Etemadi's WLLN (which only requires pairwise independence), the condition $E|X_i| < \infty$ and uniform distribution properties are sufficient.
Here, the slightly stronger moment condition $\sup E|X_i|^{1+\epsilon} < \infty$ ensures that the variance of truncated variables is well-behaved, leading to:
\[
P(|\bar{X}_n - E[\bar{X}_n]| > \delta) \to 0.
\]
Since $E[\bar{X}_n] \to 1$, it follows that $\bar{X}_n \xrightarrow{P} 1$.
\end{solution}

\begin{exercise}
Suppose that three characteristics in a large population can be observed according to the following frequencies
\[
p_1 = \theta^3, \quad p_2 = 3\theta(1-\theta), \quad p_3 = (1-\theta)^3,
\]
where $\theta \in (0, 1)$. Let $N_j, j = 1, 2, 3$ be the observed frequencies of characteristic $j$ in a random sample of size $n$.
(a) Construct the approximate level $(1-\alpha)$ maximum likelihood confidence set for $\theta$.
(b) Derive the asymptotic distribution for the frequency substitution estimator $\hat{\theta}_2 = 1 - (N_3/n)^{1/3}$.
\end{exercise}

\begin{solution}
(a) The log-likelihood is $l(\theta) = N_1 \log \theta^3 + N_2 \log(3\theta(1-\theta)) + N_3 \log(1-\theta)^3$
$l(\theta) = (3N_1 + N_2) \log \theta + (N_2 + 3N_3) \log(1-\theta) + \text{const}$.
$\frac{dl}{d\theta} = \frac{3N_1 + N_2}{\theta} - \frac{N_2 + 3N_3}{1-\theta} = 0 \implies \hat{\theta} = \frac{3N_1 + N_2}{3N_1 + 2N_2 + 3N_3}$.
The Fisher Information $I(\theta) = -E[l''(\theta)] = \frac{3n(1+\theta-\theta^2)}{\theta(1-\theta)}$? No, let's calculate.
$E[3N_1 + N_2] = n(3\theta^3 + 3\theta - 3\theta^2) = 3n\theta(1 - \theta + \theta^2)$.
$E[N_2 + 3N_3] = n(3\theta - 3\theta^2 + 3(1-\theta)^3) = 3n(1-\theta)(1 - \theta + \theta^2)$.
$I(\theta) = \frac{3n(1-\theta+\theta^2)}{\theta} + \frac{3n(1-\theta+\theta^2)}{1-\theta} = \frac{3n(1-\theta+\theta^2)}{\theta(1-\theta)}$.
The asymptotic CI is $\hat{\theta} \pm z_{\alpha/2} \sqrt{\frac{\hat{\theta}(1-\hat{\theta})}{3n(1-\hat{\theta}+\hat{\theta}^2)}}$.

(b) Let $W = N_3/n$. $W$ is the sample mean of $I(\text{type } 3)$, so $\sqrt{n}(W - (1-\theta)^3) \xrightarrow{L} N(0, (1-\theta)^3(1-(1-\theta)^3))$.
Let $g(w) = 1 - w^{1/3}$. $\hat{\theta}_2 = g(W)$. $g'((1-\theta)^3) = -\frac{1}{3}((1-\theta)^3)^{-2/3} = -\frac{1}{3(1-\theta)^2}$.
By the Delta Method, $\sqrt{n}(\hat{\theta}_2 - \theta) \xrightarrow{L} N(0, \sigma^2)$ where
$\sigma^2 = [g'((1-\theta)^3)]^2 \text{Var}(N_3/n) = \frac{1}{9(1-\theta)^4} (1-\theta)^3 (1 - (1-\theta)^3) = \frac{1 - (1-\theta)^3}{9(1-\theta)}$.
\end{solution}

\begin{exercise}
Let $X_1, \dots, X_n$ be independent and identically distributed random variables with continuous distribution functions $F(x_1), \dots, F(x_n)$, respectively. Let $Y_1 < \dots < Y_n$ be the order statistics of $X_1, \dots, X_n$. Prove that $Z_j = F(Y_j)$ has the beta$(j, n-j+1)$ distribution ($j = 1, \dots, n$).
\end{exercise}

\begin{solution}
Since $X_i$ are i.i.d. with continuous distribution function $F$, the random variables $U_i = F(X_i)$ are i.i.d. with a Uniform$(0, 1)$ distribution. Let $U_{(1)} < U_{(2)} < \dots < U_{(n)}$ be the order statistics of $U_1, \dots, U_n$. Because $F$ is a non-decreasing function, it preserves order, so $F(Y_j) = F(X_{(j)}) = U_{(j)}$. Thus, $Z_j = U_{(j)}$.

The probability density function (PDF) of the $j$-th order statistic $U_{(j)}$ of $n$ i.i.d. $U(0, 1)$ variables is given by the general formula for order statistics:
\[
f_{U_{(j)}}(z) = \frac{n!}{(j-1)!(n-j)!} [F_U(z)]^{j-1} [1 - F_U(z)]^{n-j} f_U(z)
\]
For $U \sim U(0, 1)$, we have $F_U(z) = z$ and $f_U(z) = 1$ for $0 < z < 1$. Substituting these:
\[
f_{Z_j}(z) = \frac{n!}{(j-1)!(n-j)!} z^{j-1} (1-z)^{n-j}, \quad 0 < z < 1.
\]
The Beta distribution $\text{Beta}(\alpha, \beta)$ has the PDF:
\[
f(z; \alpha, \beta) = \frac{\Gamma(\alpha+\beta)}{\Gamma(\alpha)\Gamma(\beta)} z_{\alpha-1} (1-z)^{\beta-1}
\]
Setting $\alpha = j$ and $\beta = n-j+1$, we have $\alpha+\beta = n+1$. Using $\Gamma(k) = (k-1)!$ for integers:
\[
\frac{\Gamma(n+1)}{\Gamma(j)\Gamma(n-j+1)} = \frac{n!}{(j-1)!(n-j)!}
\]
Thus, $Z_j$ follows the Beta$(j, n-j+1)$ distribution.
\end{solution}

\begin{exercise}
Let $X_1, \dots, X_n$ be i.i.d. random variables with a continuous density $f$ at point 0. Let
\[
Y_{n,i} = \frac{3}{4b_n}\left(1 - \frac{X_i^2}{b_n^2}\right) I(|X_i| \leq b_n).
\]
Show that
\[
\frac{\sum_{i=1}^n (Y_{n,i} - E Y_{n,i})}{(b_n \sum_{i=1}^n Y_{n,i})^{1/2}} \xrightarrow{L} N(0, 3/5),
\]
provided $b_n \to 0$ and $nb_n \to \infty$.
\end{exercise}

\begin{solution}
First, we compute the moments of $Y_{n,i}$ as $n \to \infty$ (and thus $b_n \to 0$).
For the expectation:
\[
E Y_{n,i} = \int_{-b_n}^{b_n} \frac{3}{4b_n} \left(1 - \frac{x^2}{b_n^2}\right) f(x) dx \to f(0).
\]
For the second moment:
\[
b_n E Y_{n,i}^2 = b_n \int_{-b_n}^{b_n} \frac{9}{16b_n^2} \left(1 - \frac{x^2}{b_n^2}\right)^2 f(x) dx \to \frac{3}{5} f(0).
\]
The variance satisfies $\text{Var}(Y_{n,i}) \sim \frac{3f(0)}{5b_n}$.
Let $S_n = \sum (Y_{n,i} - E Y_{n,i})$. By the CLT, $S_n / \sqrt{n \text{Var}(Y_{n,i})} \xrightarrow{L} N(0, 1)$.
The denominator of the expression is $D_n = (b_n \sum Y_{n,i})^{1/2} \approx \sqrt{n b_n f(0)}$.
The ratio is:
\[
\frac{S_n}{D_n} \xrightarrow{L} N(0, 1) \cdot \sqrt{\frac{3 f(0) / (5 b_n)}{b_n f(0)}} = N(0, 3/5).
\]
\end{solution}

\begin{exercise}
Let $X_1, \dots, X_n$ be independently and identically distributed random variables with $X_i \sim N(\theta, 1)$. Suppose that it is known that $|\theta| \leq \tau$, where $\tau$ is given. Show
\[
\min_{a_1, \dots, a_{n+1}} \sup_{|\theta| \leq \tau} E\left(\sum_{i=1}^n a_i X_i + a_{n+1} - \theta\right)^2 = \frac{\tau^2 n^{-1}}{\tau^2 + n^{-1}}.
\]
\end{exercise}

\begin{solution}
Consider the estimator $\hat{\theta} = a\bar{X} + b$. The risk is $R(\theta, \hat{\theta}) = \frac{a^2}{n} + ((a-1)\theta + b)^2$.
To minimize the maximum risk over $|\theta| \leq \tau$, we set $b=0$.
Then $\sup R(\theta, \hat{\theta}) = \frac{a^2}{n} + (1-a)^2 \tau^2$.
Differentiating with respect to $a$ gives $a = \frac{\tau^2}{\tau^2 + n^{-1}}$.
Substituting back yields the minimum value $\frac{\tau^2 n^{-1}}{\tau^2 + n^{-1}}$.
\end{solution}

\begin{exercise}
Write down the likelihood for $(N_0, N_1, N_2)$ in the ``1 and 1'' shooting problem. Find the MLE $\hat{p}$, check its unbiasedness, and find its asymptotic distribution.
\end{exercise}

\begin{solution}
The probabilities are $P_0 = 1-p, P_1 = p(1-p), P_2 = p^2$.
Likelihood $L(p) \propto (1-p)^{N_0} (p(1-p))^{N_1} (p^2)^{N_2} = p^{N_1+2N_2}(1-p)^{N_0+N_1}$.
MLE is $\hat{p} = \frac{N_1+2N_2}{N_0+2N_1+2N_2}$.
For $n=1$, $E[\hat{p}] = \frac{p(1+p)}{2} \neq p$, so it is biased.
Fisher info $I(p) = \frac{n(1+p)}{p(1-p)}$.
Asymptotic distribution $\sqrt{n}(\hat{p} - p) \xrightarrow{L} N(0, \frac{p(1-p)}{1+p})$.
\end{solution}
