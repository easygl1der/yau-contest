\section{2013 Probability and Statistics}

\subsection{Individual}
Please solve 5 out of the following 6 problems.

\begin{exercise}
Let $(X_n)$ be a sequence of random variables.
\begin{enumerate}
    \item Assume that $\sum_{n=0}^\infty P(|X_n| > n) < \infty$. Prove that $\limsup_{n\to\infty} \frac{|X_n|}{n} \leq 1$ a.s.
    \item Prove that $(X_n)$ converges in probability to 0 if and only if for certain $r > 0$, $E\left[\frac{|X_n|^r}{1+|X_n|^r}\right] \to 0$.
\end{enumerate}
\end{exercise}

\begin{solution}
(1) Let $A_n$ be the event $\{|X_n| > n\}$. We are given that $\sum_{n=0}^\infty P(A_n) < \infty$. By the Borel-Cantelli Lemma, the probability that infinitely many $A_n$ occur is zero. Thus $\limsup_{n\to\infty} |X_n|/n \leq 1$ a.s.
(2) This is a standard result: $E[d(X_n, 0)] \to 0 \iff X_n \xrightarrow{P} 0$ for a bounded metric $d$.
\end{solution}

\begin{exercise}
Let $X$ and $Y$ be independent $N(0,1)$ random variables.
\begin{enumerate}
    \item Find $E[X+Y \mid X \geq 0, Y \geq 0]$.
    \item Find the distribution function of $X+Y$ given that $X \geq 0$ and $Y \geq 0$.
\end{enumerate}
Hint: For (2), using the fact that $U = (X+Y)/\sqrt{2}$ and $V = (X-Y)/\sqrt{2}$ are independent and $N(0,1)$ distributed.
\end{exercise}

\begin{solution}
(1) $E[X+Y | X\geq 0, Y\geq 0] = E[X|X\geq 0] + E[Y|Y\geq 0] = 2 \cdot \sqrt{2/\pi} = \sqrt{8/\pi}$.
(2) $P(X+Y \leq s | X\geq 0, Y\geq 0) = (2\Phi(s/\sqrt{2}) - 1)^2$ for $s \geq 0$.
\end{solution}

\begin{exercise}
Let $\{X_n\}$ be a sequence of i.i.d. continuous real valued random variables. Let
\[ A_n = \{X_n = \max\{X_1, X_2, \dots, X_n\}\}. \]
We say that a maximum record occurs at $n$ in such an event.
\begin{enumerate}
    \item Evaluate the probability $P(A_n)$.
    \item Denote by $Y_n$ the number of maximum records occurred until time $n$. Evaluate $EY_n$ and $DY_n$.
\end{enumerate}
\end{exercise}

\begin{solution}
(1) $P(A_n) = 1/n$.
(2) $EY_n = \sum_{k=1}^n 1/k = H_n$. $DY_n = \sum_{k=1}^n \frac{1}{k}(1-1/k) = H_n - \sum 1/k^2$.
\end{solution}

\begin{exercise}
Let $X = (X_1, \dots, X_n)$ be an iid sample from an exponential density with mean $\theta$. Consider testing $H_0: \theta = \theta_0$ vs. $H_1: \theta > \theta_0$. Let $P(X)$ be your p-value for an appropriate test.
\begin{enumerate}
    \item What is $E_{\theta_0}(P(X))$? Derive your answer explicitly.
    \item Derive $E_\theta(P(X))$ for $\theta \neq \theta_0$. Specifically, assuming only one sample, i.e., $n=1$, calculate $E_\theta(P(X))$ as explicitly as possible for $\theta \neq \theta_0$.
    \item When there is only one sample, is $E_\theta(P(X))$ a decreasing function of $\theta$? In general, can you prove your result for an arbitrary MLR family?
\end{enumerate}
\end{exercise}

\begin{solution}
(a) $1/2$. (b) $\theta_0/(\theta+\theta_0)$. (c) Yes.
\end{solution}

\begin{exercise}
Let $X_1, X_2$ be i.i.d. uniform on $\theta - 1/2$ to $\theta + 1/2$.
\begin{enumerate}
    \item Show that for any given $0 < \alpha < 1$, you can find $c > 0$ such that $P_\theta\{\bar{X} - c < \theta < \bar{X} + c\} = 1 - \alpha$.
    \item Show that for $\epsilon$ positive and sufficiently small, $P_\theta\{\bar{X} - c < \theta < \bar{X} + c \mid |X_2 - X_1| \geq 1 - \epsilon\} = 1$.
    \item Does the assertion in (2) contradict (1)? If your sample observations are $X_1 = 1, X_2 = 2$, would you use the confidence interval in (1)?
\end{enumerate}
\end{exercise}

\begin{solution}
Ancillary statistics like $|X_2-X_1|$ can provide information about the local confidence level.
\end{solution}

\begin{exercise}
Estimate the total number of enemy tanks $N$. Serial numbers $1, \dots, N$ are i.i.d. uniform.
\begin{enumerate}
    \item Find the complete sufficient statistic.
    \item Suggest how you may find the minimum variance unbiased estimate of $N$.
\end{enumerate}
\end{exercise}

\begin{solution}
$T = X_{(n)}$ is complete sufficient. MVUE is $\frac{T^{n+1}-(T-1)^{n+1}}{T^n-(T-1)^n}$.
\end{solution}

\subsection{Team}

\begin{exercise}
The characteristic function $f$ of a probability distribution function $F$ is defined by
\[ f(t) = \int_{-\infty}^{\infty} e^{itx} dF(x). \]
Show that $f_1(t) = (\cos t)^2$ is a characteristic function and $f_2(t) = |\cos t|$ is not a characteristic function.
\end{exercise}

\begin{exercise}
Let $I = [0, 1]$ be the unit interval and $\mathcal{B}$ the $\sigma$-algebra of Borel sets on $I$. Let $P$ be the Lebesgue measure on $I$. Show that on the probability space $(I, \mathcal{B}, P)$ the set of points of $x$ with the following property has probability 1: for all but finitely many rational numbers $p/q \in (0, 1)$,
\[ \left| x - \frac{p}{q} \right| \geq \frac{1}{(q \log q)^2}. \]
\end{exercise}

\begin{exercise}
Let $X$ be an integrable random variable, $\mathcal{G}$ a $\sigma$-algebra, and $Y = E[X \mid \mathcal{G}]$. Assume that $X$ and $Y$ have the same distribution.
\begin{enumerate}
    \item Prove that if $X$ is square-integrable, then $X = Y$, a.s.
    \item Prove that for any pair of real numbers $a < b$, we have $\min\{\max\{X, a\}, b\} = \min\{\max\{Y, a\}, b\}$, and consequently, $X = Y$, a.s.
\end{enumerate}
\end{exercise}

\begin{exercise}
Let $X_1, \dots, X_n$ be iid $N(\theta, \sigma^2)$, $\sigma^2$ known, and let $\theta$ have a double exponential distribution $\pi(\theta) = e^{-|\theta|/a} / (2a)$.
\begin{enumerate}
    \item For a constant $K$, calculate the posterior probability $P(\theta > K \mid x_1, \dots, x_n, a)$.
    \item Find $\lim_{a \to \infty} P(\theta > K \mid x_1, \dots, x_n, a)$.
    \item Compare (2) to the p-value associated with the classical hypothesis test.
\end{enumerate}
\end{exercise}

\begin{exercise}
Consider linear model $Y = X\beta + \varepsilon$, $\varepsilon \sim N(0, \sigma^2 I_n)$. Minimize $\frac{1}{2} \|Y - X\beta\|_2^2 + \lambda \|\beta\|_1$.
\begin{enumerate}
    \item Show $\|Y - X\beta\|_2^2 + 2\lambda \|\beta\|_1 = \|Y - X\hat{\beta}\|_2^2 + \|X(\beta - \hat{\beta})\|_2^2 + 2\lambda \|\beta\|_1$.
    \item If $X^T X = I_p$, show lasso estimator $\beta_{\lambda,i}^*$ is soft thresholding of $\hat{\beta}_i$.
\end{enumerate}
\end{exercise}

\begin{exercise}
Consider linear model $Y = X\beta + \varepsilon$. Let $a \neq 0$ be a vector.
\begin{enumerate}
    \item Show that, if $Xa = 0$, then $a^T \beta$ is not estimable.
    \item Prove or disprove that, if $a^T \beta$ is not estimable, then $Xa = 0$.
    \item Show that $X$ is full rank iff $a^T \beta$ are estimable for all $a$.
\end{enumerate}
\end{exercise}
