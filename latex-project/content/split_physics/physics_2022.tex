\subsection{Individual}

\begin{exercise}
Consider two decoupled harmonic oscillators: $H_0 = \sum_{i=1}^2 (\frac{p_i^2}{2m} + \frac{1}{2}m\omega^2 x_i^2)$.
Define $J_+ = a_1^\dagger a_2, J_- = a_2^\dagger a_1, J_z = \frac{1}{2}(a_1^\dagger a_1 - a_2^\dagger a_2)$.
Prove they form an $su(2)$ algebra and compute the spin for energy level $n$.
\end{exercise}

\begin{solution}
\textbf{Intuition:} This is the Schwinger boson representation of $su(2)$. By using two independent oscillators, we can map the "total number of quanta" to a total angular momentum $j$, and the "difference in quanta" to the magnetic quantum number $m$. This beautifully connects the simple harmonic oscillator to the theory of angular momentum in quantum mechanics.

\textbf{i. Commutation Relations:}
Using $[a_i, a_j^\dagger] = \delta_{ij}$:
$[J_z, J_+] = \frac{1}{2}[n_1 - n_2, a_1^\dagger a_2] = \frac{1}{2}([n_1, a_1^\dagger a_2] - [n_2, a_1^\dagger a_2])$.
$[n_1, a_1^\dagger a_2] = [a_1^\dagger a_1, a_1^\dagger] a_2 = a_1^\dagger a_2 = J_+$.
$[n_2, a_1^\dagger a_2] = a_1^\dagger [a_2^\dagger a_2, a_2] = -a_1^\dagger a_2 = -J_+$.
Thus $[J_z, J_+] = \frac{1}{2}(J_+ - (-J_+)) = J_+$. Similarly, $[J_z, J_-] = -J_-$.
$[J_+, J_-] = [a_1^\dagger a_2, a_2^\dagger a_1] = a_1^\dagger [a_2, a_2^\dagger a_1] + [a_1^\dagger, a_2^\dagger a_1] a_2 = a_1^\dagger a_1 - a_2^\dagger a_2 = 2J_z$.
This confirms the $su(2)$ algebra.

\textbf{ii. Representation:}
For a fixed energy $E_n = (n+1)\hbar\omega$, the total number of quanta is $N = n_1 + n_2 = n$.
The Casimir operator $J^2$ is $\frac{N}{2}(\frac{N}{2}+1)$.
Comparing with the standard $j(j+1)$, we find $j = n/2$.
The degeneracy is $n+1$, which matches the dimension $2j+1 = n+1$ of the spin-$j$ representation.
\end{solution}

\begin{exercise}
Prove $k^\mu P_\mu$ is conserved for a Killing vector $k^\mu$ and geodesic $P_\mu$.
\end{exercise}

\begin{solution}
\textbf{Intuition:} Killing vectors represent symmetries of the spacetime metric (isometries). According to Noether's theorem, every continuous symmetry corresponds to a conservation law. For example, a time-translation symmetry (static metric) leads to energy conservation. This problem proves that for any symmetry direction $k^\mu$, the projection of the momentum onto that direction is constant along the particle's path.

\textbf{Proof:}
Let $u^\nu$ be the 4-velocity. $P_\mu = m u_\mu$.
$\frac{d}{d\tau}(k^\mu u_\mu) = u^\nu D_\nu (k^\mu u_\mu) = (u^\nu D_\nu k^\mu) u_\mu + k^\mu (u^\nu D_\nu u_\mu)$.
The second term $u^\nu D_\nu u_\mu = 0$ because the particle follows a geodesic.
The first term is $u^\nu u^\mu D_\nu k_\mu$.
Since $k$ is a Killing vector, it satisfies $D_\nu k_\mu + D_\mu k_\nu = 0$ (antisymmetry).
Contracting an antisymmetric tensor $D_\nu k_\mu$ with a symmetric tensor $u^\nu u^\mu$ gives zero.
Thus $\frac{d}{d\tau}(k^\mu P_\mu) = 0$.
\end{solution}

\begin{exercise}
Currents $s^\mu = x_\nu \theta^{\mu\nu}$ and $K^{\lambda\mu} = x^2 \theta^{\lambda\mu} - 2x^\lambda x_\rho \theta^{\rho\mu}$.
\end{exercise}

\begin{solution}
\textbf{Intuition:} These currents correspond to scale (dilatation) and special conformal transformations. In a scale-invariant theory, the "size" of the system doesn't change the physics. The conservation of these currents is deeply tied to the tracelessness of the energy-momentum tensor, $\theta^\mu_\mu = 0$.

\textbf{Calculation:}
$\partial_\mu s^\mu = (\partial_\mu x_\nu) \theta^{\mu\nu} + x_\nu \partial_\mu \theta^{\mu\nu} = \delta_{\mu\nu} \theta^{\mu\nu} + 0 = \theta^\mu_\mu$.
For $K^{\lambda\mu}$:
$\partial_\mu K^{\lambda\mu} = 2x_\mu \theta^{\lambda\mu} + x^2 (0) - 2\delta^\lambda_\mu x_\rho \theta^{\rho\mu} - 2x^\lambda \delta_{\rho\mu} \theta^{\rho\mu} - 2x^\lambda x_\rho (0)$.
$= 2x_\rho \theta^{\lambda\rho} - 2x_\rho \theta^{\rho\lambda} - 2x^\lambda \theta^\rho_\rho = -2x^\lambda \theta^\rho_\rho$.
Both are conserved if $\theta^\mu_\mu = 0$.
In quantum theory, this is often broken by the \textit{trace anomaly}, where the renormalization scale introduces a non-zero trace related to the $\beta$-function.
\end{solution}

\begin{exercise}
Time reversal operator $\Theta$ and real eigenfunctions.
\end{exercise}

\begin{solution}
\textbf{Intuition:} Time reversal in quantum mechanics is special because it must reverse the direction of motion (momentum) while preserving position. This requires the operator to be anti-unitary (involving complex conjugation), which essentially flips the "phase" of the wave function evolved forward in time.

\textbf{(a) Anti-unitarity:} To preserve the Schrödinger equation $i\hbar \dot{\psi} = H\psi$ under $t \to -t$, we must have $\Theta i = -i \Theta$.
\textbf{(b) Conjugation:} In position space, $\langle x | \Theta \psi \rangle = \psi(x)^*$.
\textbf{(c) Real eigenfunctions:} If $H$ is time-reversal invariant and the state is non-degenerate, then $\Theta \psi = c \psi$. In position space, $\psi(x)^* = c \psi(x)$. Since $\Theta^2 = 1$ for spinless particles, $|c|=1$. By choosing a proper phase, we can make the wave function purely real. This is why we often see real wave functions for bound states in 1D.
\end{solution}

\begin{exercise}
Eddington-Finkelstein metric $ds^2 = -(1-2M/r)dv^2 + dr dv + r^2 d\Omega^2$.
\end{exercise}

\begin{solution}
\textbf{Intuition:} These coordinates are "ingoing" null coordinates. Unlike Schwarzschild coordinates, which singularize at $r=2M$, these coordinates are regular at the horizon. The vector $\partial_v$ represents time-translation symmetry, making it a Killing vector. The surface $r=2M$ is a null hypersurface, meaning the normal to the surface is itself a null vector.

\textbf{(a) Normal:} $l^\mu \propto g^{\mu\nu} \partial_\nu r$. $l$ is the normal vector to the $r=\text{const}$ surface.
\textbf{(b) Nullity:} At $r=2M$, the metric component $g^{rr} = 0$, so the norm of the normal vector vanishes ($l^2 = 0$). This proves the horizon is a null surface.
\textbf{(c) Killing:} $\partial_v g_{\mu\nu} = 0$ is immediate since no metric component depends on $v$.
\end{solution}
