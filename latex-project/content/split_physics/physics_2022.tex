\subsection{Individual}

\begin{exercise}
\label{physics:2022:1}
Consider two decoupled harmonic oscillators: $H_0 = \sum_{i=1}^2 (\frac{p_i^2}{2m} + \frac{1}{2}m\omega^2 x_i^2)$.
Define $J_+ = a_1^\dagger a_2, J_- = a_2^\dagger a_1, J_z = \frac{1}{2}(a_1^\dagger a_1 - a_2^\dagger a_2)$.
Prove they form an $su(2)$ algebra and compute the spin for energy level $n$.
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item \textbf{Boson Commutation Relations:} $[a_i, a_j^\dagger] = \delta_{ij}$.
    \item \textbf{Number Operator:} $n_i = a_i^\dagger a_i$, where $[n_i, a_i^\dagger] = a_i^\dagger$.
    \item \textbf{$su(2)$ Algebra:} Defined by $[J_z, J_\pm] = \pm J_\pm$ and $[J_+, J_-] = 2J_z$.
\end{itemize}

\textbf{Steps:}

1. \textbf{Verifying $[J_z, J_\pm]$:} Using $J_z = \frac{1}{2}(n_1 - n_2)$, we calculate:
   \[ [J_z, J_+] = \frac{1}{2}[n_1 - n_2, a_1^\dagger a_2] = \frac{1}{2}([n_1, a_1^\dagger]a_2 - a_1^\dagger[n_2, a_2]) = \frac{1}{2}(a_1^\dagger a_2 + a_1^\dagger a_2) = J_+. \]
   Similarly, $[J_z, J_-] = -J_-$.

2. \textbf{Verifying $[J_+, J_-]$:}
   \[ [J_+, J_-] = [a_1^\dagger a_2, a_2^\dagger a_1] = a_1^\dagger [a_2, a_2^\dagger a_1] + [a_1^\dagger, a_2^\dagger a_1] a_2 = a_1^\dagger a_1 - a_2^\dagger a_2 = 2J_z. \]
   Thus the operators form a valid $su(2)$ representation.

3. \textbf{Spin determination:} The total number of quanta $N = n_1 + n_2$ is fixed for a given energy level $n$. The Casimir operator $J^2 = \frac{N}{2}(\frac{N}{2}+1)$ implies $j = n/2$. The dimension $2j+1 = n+1$ matches the oscillator degeneracy.

\textbf{Intuition:}
This construction maps two independent vibration modes to a single "spinning" object. By treating one oscillator as "spin up" and the other as "spin down," any state with $n$ total vibrations corresponds to a specific spin representation. It shows that symmetry in quantum mechanics allows us to build complex angular momentum from simple components.
\end{solution}

\begin{exercise}
\label{physics:2022:2}
Prove $k^\mu P_\mu$ is conserved for a Killing vector $k^\mu$ and geodesic $P_\mu$.
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item \textbf{Killing Equation:} $\nabla_\mu k_\nu + \nabla_\nu k_\mu = 0$.
    \item \textbf{Geodesic Equation:} $u^\nu \nabla_\nu u^\mu = 0$.
    \item \textbf{Product Rule:} $\nabla_\nu (k \cdot u) = (\nabla_\nu k_\mu)u^\mu + k_\mu(\nabla_\nu u^\mu)$.
\end{itemize}

\textbf{Steps:}

1. \textbf{Set up the derivative:} We differentiate the quantity $Q = k_\mu u^\mu$ along the path of the particle (directional derivative $u^\nu \nabla_\nu$):
   \[ \frac{d Q}{d\tau} = u^\nu \nabla_\nu (k_\mu u^\mu) = u^\nu u^\mu \nabla_\nu k_\mu + u^\nu k_\mu \nabla_\nu u^\mu. \]

2. \textbf{Apply Geodesic Equation:} The second term $k_\mu (u^\nu \nabla_\nu u^\mu)$ is zero because the velocity $u$ satisfies the geodesic equation.

3. \textbf{Apply Killing Symmetry:} The first term $u^\mu u^\nu \nabla_\nu k_\mu$ involves the contraction of a symmetric tensor $u^\mu u^\nu$ with the gradient of the Killing vector. Since $\nabla_{(\mu} k_{\nu)} = 0$ by the Killing equation, any contraction with a symmetric tensor must vanish.

\textbf{Intuition:}
Killing vectors represent "directions" in which spacetime doesn't change. Just as energy is conserved if the environment is time-invariant, momentum along a specific spatial direction is conserved if the geometry is invariant under translations in that direction.
\end{solution}

\begin{exercise}
\label{physics:2022:3}
Conformal currents $s^\mu$ and $K^{\lambda\mu}$.
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item \textbf{Conservation of $\theta^{\mu\nu}$:} $\partial_\mu \theta^{\mu\nu} = 0$.
    \item \textbf{Dilatation current:} $s^\mu = x_\nu \theta^{\mu\nu}$.
    \item \textbf{Conformal current:} $K^{\lambda\mu} = x^2 \theta^{\lambda\mu} - 2x^\lambda x_\rho \theta^{\rho\mu}$.
\end{itemize}

\textbf{Steps:}

1. \textbf{Divergence of $s^\mu$:}
   \[ \partial_\mu s^\mu = (\partial_\mu x_\nu) \theta^{\mu\nu} + x_\nu (\partial_\mu \theta^{\mu\nu}) = \delta_{\mu\nu} \theta^{\mu\nu} + 0 = \theta^\mu_\mu. \]
   Thus $s^\mu$ is conserved if the energy-momentum tensor is traceless.

2. \textbf{Divergence of $K^{\lambda\mu}$:}
   \[ \partial_\mu K^{\lambda\mu} = (2x_\mu)\theta^{\lambda\mu} + x^2(0) - 2(\delta^\lambda_\mu x_\rho + x^\lambda \delta_{\rho\mu})\theta^{\rho\mu} = 2x_\rho \theta^{\lambda\rho} - 2x_\rho \theta^{\rho\lambda} - 2x^\lambda \theta^\rho_\rho. \]
   Since $\theta^{\mu\nu}$ is symmetric, the first two terms cancel, leaving $\partial_\mu K^{\lambda\mu} = -2x^\lambda \theta^\rho_\rho$.

\textbf{Intuition:}
Scale invariance means that "size" doesn't matter. For a physical theory to look identical after zooming in or out, it must not have any fundamental mass scales. Mathematically, this lack of scale is expressed by the vanishing of the trace of the energy-momentum tensor.
\end{solution}

\begin{exercise}
\label{physics:2022:4}
Time reversal operator $\Theta$.
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item \textbf{Anti-unitarity:} $\Theta i \Theta^{-1} = -i$.
    \item \textbf{Invariant Hamiltonian:} $[H, \Theta] = 0$.
\end{itemize}

\textbf{Steps:}

1. \textbf{Anti-linearity:} Since $i \to -i$ under time reversal to preserve the Schrodinger equation, $\Theta$ must involve complex conjugation.
2. \textbf{Reality of states:} If $\psi$ is a non-degenerate eigenstate, $\Theta\psi = e^{i\phi}\psi$. Taking the position representation, $\psi^*(x) = e^{i\phi}\psi(x)$. We can redefine $\psi \to e^{-i\phi/2}\psi$ to make the wavefunction purely real.

\textbf{Intuition:}
Time reversal reverses momentum but keeps position. Because the evolution operator $e^{-iHt/\hbar}$ contains $i$ and $t$, flipping $t$ is equivalent to flipping $i$. This "mirroring" of the complex phase forces simple bound states to be described by real numbers.
\end{solution}

\begin{exercise}
\label{physics:2022:5}
Eddington-Finkelstein metric.
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item \textbf{Metric:} $ds^2 = -(1-2M/r)dv^2 + 2dv dr + r^2 d\Omega^2$.
    \item \textbf{Null Surface:} Normal vector satisfies $l^2 = 0$.
\end{itemize}

\textbf{Steps:}

1. \textbf{Normal vector:} For the surface $r=2M$, the normal is $l_\mu = \partial_\mu(r-2M) = (0, 1, 0, 0)$.
2. \textbf{Norm checking:} $l^2 = g^{\mu\nu} l_\mu l_\nu = g^{rr}$. In EF coordinates, $g^{rr} = (1-2M/r)$. At $r=2M$, $g^{rr}=0$.
3. \textbf{Conclusion:} Because the norm vanishes, the horizon is a null hypersurface. The vector $\partial_v$ is Killing because $g_{\mu\nu}$ is independent of $v$.

\textbf{Intuition:}
EF coordinates are like "in-falling" clocks. While a standard coordinate system breaks down at the event horizon of a black hole, these coordinates allow us to pass through smoothly, proving that the horizon is just a boundary of light, not a physical wall.
\end{solution}
