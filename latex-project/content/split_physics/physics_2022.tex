\section{2022 Mathematical Physics}

\begin{exercise}
\label{physics:2022:1}
\begin{enumerate}[label=(\alph*)]
    \item A symmetry transformation in quantum mechanics is represented by a unitary or anti-unitary operator acting on a Hilbert space. The time reversal transformation $\Theta$ relates the wave function at time $t$ to time $-t$. Prove: $\Theta$ is an anti-unitary operator.
    \item Consider state vector $|\psi\rangle$ for a quantum system. A time reversal transformation is represented by an anti-unitary operator $\Theta$. We now consider position space wavefunction $\psi(x) = \langle x | \psi\rangle$, and $\Theta |x\rangle = |x\rangle$. Prove: the position space wave function for $\Theta|\psi\rangle$ is $\psi(x)^*$.
    \item A one-dimensional quantum system is invariant under time reversal transformation, and so its Hamiltonian satisfies $\Theta H = H \Theta$. If an energy eigenstate $|\psi\rangle$ has no degeneracy, prove: it is possible to take the position space energy eigenfunction to be real: $\psi(x)^* = \psi(x)$.
\end{enumerate}
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item \textbf{Anti-unitarity:} $\Theta i \Theta^{-1} = -i$.
    \item \textbf{Invariant Hamiltonian:} $[H, \Theta] = 0$.
\end{itemize}

\textbf{Steps:}

1. \textbf{Anti-linearity:} Since $i \to -i$ under time reversal to preserve the Schrodinger equation, $\Theta$ must involve complex conjugation.
2. \textbf{Reality of states:} If $\psi$ is a non-degenerate eigenstate, $\Theta\psi = e^{i\phi}\psi$. Taking the position representation, $\psi^*(x) = e^{i\phi}\psi(x)$. We can redefine $\psi \to e^{-i\phi/2}\psi$ to make the wavefunction purely real.

\textbf{Intuition:}
Time reversal reverses momentum but keeps position. Because the evolution operator $e^{-iHt/\hbar}$ contains $i$ and $t$, flipping $t$ is equivalent to flipping $i$. This "mirroring" of the complex phase forces simple bound states to be described by real numbers.
\end{solution}

\begin{exercise}
\label{physics:2022:2}
Consider the following quantum Hamiltonian:
\begin{equation}
    H_0 = \frac{p_1^2}{2m} + \frac{1}{2}m\omega^2 x_1^2 + \frac{p_2^2}{2m} + \frac{1}{2}m\omega^2 x_2^2
\end{equation}
This is the Hamiltonian for two decoupled harmonic oscillators.

\begin{enumerate}[label=(\alph*)]
    \item Calculate the eigenstates and eigenvalues for $H_0$ (an energy eigenstate could be labeled as $|n_1, n_2\rangle$).
    \item Assume the creation and annihilation operators for two harmonic oscillators are $a_i^\dagger, a_i, i=1,2$. Define following operators
    \begin{equation}
        J_+ = a_1^\dagger a_2, \quad J_- = a_2^\dagger a_1, \quad J_z = \frac{1}{2}(a_1^\dagger a_1 - a_2^\dagger a_2)
    \end{equation}
    \begin{enumerate}[label=\roman*.]
        \item Prove that: $[J_z, J_\pm] = \pm J_\pm$, $[J_+, J_-] = 2J_z$.
        \item Consider one eigenvalue $E_n$ of $H_0$, (here $n_1 + n_2 = n$). Prove that: all eigenstates of $E_n$ form an irreducible representation of $su(2)$ Lie algebra, and compute the spin.
    \end{enumerate}
    \item Consider following perturbed Hamiltonian ($\lambda$ is small)
    \begin{equation}
        H = H_0 + \lambda x_1^2 p_2^2
    \end{equation}
    Compute the first order correction to the energy for the energy level $n_1 + n_2 = 2$.
\end{enumerate}
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item \textbf{Boson Commutation Relations:} $[a_i, a_j^\dagger] = \delta_{ij}$.
    \item \textbf{Number Operator:} $n_i = a_i^\dagger a_i$, where $[n_i, a_i^\dagger] = a_i^\dagger$.
    \item \textbf{$su(2)$ Algebra:} Defined by $[J_z, J_\pm] = \pm J_\pm$ and $[J_+, J_-] = 2J_z$.
\end{itemize}

\textbf{Steps:}

1. \textbf{Verifying $[J_z, J_\pm]$:} Using $J_z = \frac{1}{2}(n_1 - n_2)$, we calculate:
   \[ [J_z, J_+] = \frac{1}{2}[n_1 - n_2, a_1^\dagger a_2] = \frac{1}{2}([n_1, a_1^\dagger]a_2 - a_1^\dagger[n_2, a_2]) = \frac{1}{2}(a_1^\dagger a_2 + a_1^\dagger a_2) = J_+. \]
   Similarly, $[J_z, J_-] = -J_-$.

2. \textbf{Verifying $[J_+, J_-]$:}
   \[ [J_+, J_-] = [a_1^\dagger a_2, a_2^\dagger a_1] = a_1^\dagger [a_2, a_2^\dagger a_1] + [a_1^\dagger, a_2^\dagger a_1] a_2 = a_1^\dagger a_1 - a_2^\dagger a_2 = 2J_z. \]
   Thus the operators form a valid $su(2)$ representation.

3. \textbf{Spin determination:} The total number of quanta $N = n_1 + n_2$ is fixed for a given energy level $n$. The Casimir operator $J^2 = \frac{N}{2}(\frac{N}{2}+1)$ implies $j = n/2$. The dimension $2j+1 = n+1$ matches the oscillator degeneracy.

\textbf{Intuition:}
This construction maps two independent vibration modes to a single "spinning" object. By treating one oscillator as "spin up" and the other as "spin down," any state with $n$ total vibrations corresponds to a specific spin representation. It shows that symmetry in quantum mechanics allows us to build complex angular momentum from simple components.
\end{solution}

\begin{exercise}
\label{physics:2022:3}
A Killing vector field $k^\mu \frac{\partial}{\partial x^\mu}$ satisfies the equation $k^\lambda \partial_\lambda g_{\mu\nu} + \partial_\mu k^\lambda g_{\lambda\nu} + \partial_\nu k^\lambda g_{\lambda\mu} = 0$.

\begin{enumerate}[label=(\alph*)]
    \item Prove: $\nabla_\mu k_\nu + \nabla_\nu k_\mu = 0$, here $\nabla_\mu$ is the covariant derivative.
    \item For a moving particle in gravitational background with a Killing vector field, prove: $k^\mu P_\mu$ is a conserved quantity. Here $P_\mu = m \frac{dx^\nu}{d\tau}g_{\mu\nu}$ is the momentum for the free falling particle with trajectory $x^\nu(\tau)$.
\end{enumerate}
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item \textbf{Killing Equation:} $\nabla_\mu k_\nu + \nabla_\nu k_\mu = 0$.
    \item \textbf{Geodesic Equation:} $u^\nu \nabla_\nu u^\mu = 0$.
    \item \textbf{Product Rule:} $\nabla_\nu (k \cdot u) = (\nabla_\nu k_\mu)u^\mu + k_\mu(\nabla_\nu u^\mu)$.
\end{itemize}

\textbf{Steps:}

1. \textbf{Set up the derivative:} We differentiate the quantity $Q = k_\mu u^\mu$ along the path of the particle (directional derivative $u^\nu \nabla_\nu$):
   \[ \frac{d Q}{d\tau} = u^\nu \nabla_\nu (k_\mu u^\mu) = u^\nu u^\mu \nabla_\nu k_\mu + u^\nu k_\mu \nabla_\nu u^\mu. \]

2. \textbf{Apply Geodesic Equation:} The second term $k_\mu (u^\nu \nabla_\nu u^\mu)$ is zero because the velocity $u$ satisfies the geodesic equation.

3. \textbf{Apply Killing Symmetry:} The first term $u^\mu u^\nu \nabla_\nu k_\mu$ involves the contraction of a symmetric tensor $u^\mu u^\nu$ with the gradient of the Killing vector. Since $\nabla_{(\mu} k_{\nu)} = 0$ by the Killing equation, any contraction with a symmetric tensor must vanish.

\textbf{Intuition:}
Killing vectors represent "directions" in which spacetime doesn't change. Just as energy is conserved if the environment is time-invariant, momentum along a specific spatial direction is conserved if the geometry is invariant under translations in that direction.
\end{solution}

\begin{exercise}
\label{physics:2022:4}
Consider the following metric:
\begin{equation}
    ds^2 = -(1 - \frac{2 M}{r}) dv^2 + dv dr + r^2 d\Omega^2
\end{equation}
Here $d\Omega^2$ is the standard metric on two sphere. Consider the hypersurface defined by $S = r - 2M = 0$, and a vector field $l^\mu = f(x) (g^{\mu\nu} \partial_\nu S)$, here $f(x)$ is a non-zero function. Prove:

\begin{enumerate}[label=(\alph*)]
    \item $l^\mu$ is normal to the surface $S$.
    \item $l^2 = 0$ on the surface $S$.
    \item $\partial / \partial v$ is a Killing vector field.
\end{enumerate}
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item \textbf{Metric:} $ds^2 = -(1-2M/r)dv^2 + dv dr + r^2 d\Omega^2$.
    \item \textbf{Null Surface:} Normal vector satisfies $l^2 = 0$.
\end{itemize}

\textbf{Steps:}

1. \textbf{Normal vector:} For the surface $r=2M$, the normal is $l_\mu = \partial_\mu(r-2M) = (0, 1, 0, 0)$.
2. \textbf{Norm checking:} $l^2 = g^{\mu\nu} l_\mu l_\nu = g^{rr}$. In EF coordinates, $g^{rr} = (1-2M/r)$. At $r=2M$, $g^{rr}=0$.
3. \textbf{Conclusion:} Because the norm vanishes, the horizon is a null hypersurface. The vector $\partial_v$ is Killing because $g_{\mu\nu}$ is independent of $v$.

\textbf{Intuition:}
EF coordinates are like "in-falling" clocks. While a standard coordinate system breaks down at the event horizon of a black hole, these coordinates allow us to pass through smoothly, proving that the horizon is just a boundary of light, not a physical wall.
\end{solution}

\begin{exercise}
\label{physics:2022:5}
The energy momentum tensor for a relativistic quantum field theory is denoted as $\theta^{\mu\nu}$, which is symmetric and conserved.

\begin{enumerate}[label=(\alph*)]
    \item Define new current $s^\mu = x_\nu \theta^{\mu\nu}$ and $K^{\lambda\mu} = x^2 \theta^{\lambda\mu} - 2x^\lambda x_\rho \theta^{\rho\mu}$. Compute $\partial_\mu s^\mu$ and $\partial_\mu K^{\lambda\mu}$, and explain the condition on $\theta^{\mu\nu}$ so that these new currents are conserved.
    \item Consider a scalar field $\sigma(x)$ which transforms under a scale transformation as $\delta \sigma = x^\lambda \partial_\lambda \sigma + f^{-1}$. We have following Lagrangian:
    \begin{equation}
        \mathcal{L} = \mathcal{L}_s - \frac{\mu_0^2}{2} \phi^2 e^{2f\sigma} + \frac{1}{2f^2} \partial_\mu e^{f\sigma} \partial^\mu e^{f\sigma}
    \end{equation}
    The infinitesimal scale transformation on scalar field $\phi$ is $\delta \phi = (1 + x_\lambda \partial^\lambda)\phi$. Here $\mathcal{L}_s$ is scale invariant part of the Lagrangian. Prove that: the above Lagrangian is scale invariant.
    \item Explain why a classically scale invariant Lagrangian for a quantum field theory may fail to be scale invariant quantum mechanically.
\end{enumerate}
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item \textbf{Conservation of $\theta^{\mu\nu}$:} $\partial_\mu \theta^{\mu\nu} = 0$.
    \item \textbf{Dilatation current:} $s^\mu = x_\nu \theta^{\mu\nu}$.
    \item \textbf{Conformal current:} $K^{\lambda\mu} = x^2 \theta^{\lambda\mu} - 2x^\lambda x_\rho \theta^{\rho\mu}$.
\end{itemize}

\textbf{Steps:}

1. \textbf{Divergence of $s^\mu$:}
   \[ \partial_\mu s^\mu = (\partial_\mu x_\nu) \theta^{\mu\nu} + x_\nu (\partial_\mu \theta^{\mu\nu}) = \delta_{\mu\nu} \theta^{\mu\nu} + 0 = \theta^\mu_\mu. \]
   Thus $s^\mu$ is conserved if the energy-momentum tensor is traceless.

2. \textbf{Divergence of $K^{\lambda\mu}$:}
   \[ \partial_\mu K^{\lambda\mu} = (2x_\mu)\theta^{\lambda\mu} + x^2(0) - 2(\delta^\lambda_\mu x_\rho + x^\lambda \delta_{\rho\mu})\theta^{\rho\mu} = 2x_\rho \theta^{\lambda\rho} - 2x_\rho \theta^{\rho\lambda} - 2x^\lambda \theta^\rho_\rho. \]
   Since $\theta^{\mu\nu}$ is symmetric, the first two terms cancel, leaving $\partial_\mu K^{\lambda\mu} = -2x^\lambda \theta^\rho_\rho$.

\textbf{Intuition:}
Scale invariance means that "size" doesn't matter. For a physical theory to look identical after zooming in or out, it must not have any fundamental mass scales. Mathematically, this lack of scale is expressed by the vanishing of the trace of the energy-momentum tensor.
\end{solution}

\begin{exercise}
\label{physics:2022:6}
Consider following Lagrangian for $N$ scalar fields $\phi^a, a = 1, \dots, N$:
\begin{equation}
    \mathcal{L} = \frac{1}{2} \partial_\mu \phi^a \partial^\mu \phi^a - \frac{1}{2} \mu_0^2 \phi^a \phi^a - \frac{1}{8} \lambda_0 (\phi^a \phi^a)^2
\end{equation}
Here the repeated index implies the summation over the index.

\begin{enumerate}[label=(\alph*)]
    \item Write down the propagator and interaction vertex for this model, and write down four point Feynman diagrams up to one loop level.
    \item Define $g_0 = \lambda_0 N$, and compute the order in $g_0$ and $N$ for all the diagrams listed in last question. If we fix the coupling $g_0$, and let $N$ go to infinity, list the leading order Feynman diagrams in $1/N$.
\end{enumerate}
\end{exercise}

\begin{solution}
\textbf{Large $N$ Expansion:}
In the large $N$ limit, we keep $g_0 = \lambda_0 N$ fixed. 
1. \textbf{Propagator:} $i \delta^{ab} / (p^2 - \mu_0^2)$.
2. \textbf{Vertex:} $-i \frac{\lambda_0}{2} (\delta^{ab}\delta^{cd} + \delta^{ac}\delta^{bd} + \delta^{ad}\delta^{bc})$.
3. \textbf{Leading Diagrams:} The leading diagrams are "bubble" diagrams or "cactus" diagrams where each loop provides a factor of $N$ that compensates for the $1/N$ in the coupling $\lambda_0 = g_0/N$.
\end{solution}
