\section{2023 Mathematical Physics}

\subsection{Exam}

\begin{exercise}
Consider the Lagrangian
\[
L(x, y, \dot{x}, \dot{y}) = \frac{1}{2}\frac{\dot{x}^2 + \dot{y}^2 + 2(x\dot{y} - y\dot{x})}{x^2 + y^2}.
\]

\textbf{(a)} Compute the Hamiltonian $H(x, y, p_x, p_y)$, and show the final form can be written as
\[
\frac{1}{2}f(x, y)\left[(p_x - A_x(x, y))^2 + (p_y - A_y(x, y))^2\right]
\]
for some $f, A_x, A_y$. Find the vector potential $\vec{A}$ and then compute the corresponding magnetic field away from the origin.

\textbf{(b)} Prove that the Lagrangian $L(x, y, \dot{x}, \dot{y})$ is invariant under two symmetries: rotations and scale transformations.

\textbf{(c)} Derive the conserved quantities for both symmetries.

\textbf{(d)} Rewrite the Lagrangian in polar coordinates, write down the Euler-Lagrange equations and solve them.
\end{exercise}

\begin{solution}
\textbf{(a)} The Lagrangian is 
\[ L = \frac{1}{2} \frac{\dot{x}^2 + \dot{y}^2 + 2(x\dot{y} - y\dot{x})}{x^2+y^2}. \]
The canonical momenta are
\[ p_x = \frac{\partial L}{\partial \dot{x}} = \frac{\dot{x} - y}{x^2+y^2}, \quad p_y = \frac{\partial L}{\partial \dot{y}} = \frac{\dot{y} + x}{x^2+y^2}. \]
Solving for velocities:
\[ \dot{x} = (x^2+y^2)p_x + y, \quad \dot{y} = (x^2+y^2)p_y - x. \]
The Hamiltonian is $H = p_x \dot{x} + p_y \dot{y} - L$. Substituting the expressions:
\[ H = (x^2+y^2)p_x^2 + yp_x + (x^2+y^2)p_y^2 - xp_y - \frac{1}{2(x^2+y^2)} \left[ ((x^2+y^2)p_x+y)^2 + ((x^2+y^2)p_y-x)^2 + 2(x((x^2+y^2)p_y-x) - y((x^2+y^2)p_x+y)) \right] \]
After simplification:
\[ H = \frac{1}{2}(x^2+y^2) \left[ \left( p_x - \frac{-y}{x^2+y^2} \right)^2 + \left( p_y - \frac{x}{x^2+y^2} \right)^2 \right]. \]
Thus $f(x,y) = x^2+y^2$, $A_x = -y/(x^2+y^2)$, $A_y = x/(x^2+y^2)$. The vector potential is $\vec{A} = \frac{-y\hat{i} + x\hat{j}}{x^2+y^2}$. The magnetic field is $B_z = \partial_x A_y - \partial_y A_x = \partial_x \left(\frac{x}{x^2+y^2}\right) - \partial_y \left(\frac{-y}{x^2+y^2}\right) = \frac{y^2-x^2}{(x^2+y^2)^2} - \frac{y^2-x^2}{(x^2+y^2)^2} = 0$ for $(x,y) \neq (0,0)$. However, this is the Aharonov-Bohm potential, so $\vec{B} = 2\pi \delta^2(\vec{r})\hat{k}$ via Stokes' theorem.

\textbf{(b)} Under rotations $x \to x \cos \theta - y \sin \theta$, $y \to x \sin \theta + y \cos \theta$, the combination $x^2+y^2$, $\dot{x}^2+\dot{y}^2$ and $x\dot{y}-y\dot{x}$ (the z-component of angular momentum) are invariant. Thus $L$ is invariant. Under scale transformations $x \to \lambda x$, $y \to \lambda y$, we have $\dot{x} \to \lambda \dot{x}$, $x^2+y^2 \to \lambda^2 (x^2+y^2)$, $\dot{x}^2+\dot{y}^2 \to \lambda^2 (\dot{x}^2+\dot{y}^2)$, and $x\dot{y}-y\dot{x} \to \lambda^2 (x\dot{y}-y\dot{x})$. The numerator and denominator both scale as $\lambda^2$, so $L$ is invariant.

\textbf{(c)} Rotational symmetry implies conservation of angular momentum $J = x p_y - y p_x$. Substituting $p_x, p_y$:
\[ J = x \frac{\dot{y}+x}{x^2+y^2} - y \frac{\dot{x}-y}{x^2+y^2} = \frac{x\dot{y}-y\dot{x} + x^2+y^2}{x^2+y^2} = \frac{x\dot{y}-y\dot{x}}{x^2+y^2} + 1. \]
Scale symmetry (dilatation) leads to $D = \vec{p} \cdot \vec{r}$. From Noether's theorem, the conserved quantity is $Q = p_x x + p_y y$.

\textbf{(d)} In polar coordinates $x = r \cos \theta, y = r \sin \theta$, $\dot{x}^2+\dot{y}^2 = \dot{r}^2 + r^2 \dot{\theta}^2$ and $x\dot{y}-y\dot{x} = r^2 \dot{\theta}$.
\[ L = \frac{1}{2} \frac{\dot{r}^2 + r^2 \dot{\theta}^2 + 2r^2 \dot{\theta}}{r^2} = \frac{1}{2} \left( \frac{\dot{r}^2}{r^2} + \dot{\theta}^2 + 2\dot{\theta} \right). \]
Euler-Lagrange equations: 
1) $\frac{d}{dt} \frac{\partial L}{\partial \dot{\theta}} = \frac{\partial L}{\partial \theta} \implies \ddot{\theta} = 0 \implies \dot{\theta} = \omega$ (constant).
2) $\frac{d}{dt} \frac{\partial L}{\partial \dot{r}} = \frac{\partial L}{\partial r} \implies \frac{d}{dt} \left( \frac{\dot{r}}{r^2} \right) = -\frac{\dot{r}^2}{r^3} \implies \frac{\ddot{r}}{r^2} - \frac{2\dot{r}^2}{r^3} = -\frac{\dot{r}^2}{r^3} \implies r\ddot{r} - \dot{r}^2 = 0$.
Solving $r\ddot{r} = \dot{r}^2$ gives $r(t) = r_0 e^{vt}$.
\end{solution}

\begin{exercise}
A particle of mass $m$ in 2 dimensions is confined by an isotropic harmonic oscillator potential of frequency $\omega$, while subject to a weak and anisotropic perturbation of strength $\alpha \ll 1$. The total Hamiltonian of the particle is
\[
H = H_0 + V = \frac{p_x^2}{2m} + \frac{p_y^2}{2m} + \frac{1}{2}m\omega^2(x^2 + y^2) + \alpha m\omega^2 xy.
\]

\textbf{(a)} When $\alpha = 0$, what are the energies and degeneracies of the three lowest-lying unperturbed states?

\textbf{(b)} Use perturbation theory to correct the energies of the above three states to the first order in $\alpha$.

\textbf{(c)} Find the exact spectrum of $H$. (Hint: you may want to rotate $x$ and $y$ into new coordinates.)

\textbf{(d)} Check that the perturbative results in part (b) are recovered from the exact spectrum.
\end{exercise}

\begin{solution}
\textbf{(a)} For $\alpha = 0$, $H_0 = \frac{p_x^2+p_y^2}{2m} + \frac{1}{2}m\omega^2(x^2+y^2)$. This is a 2D isotropic harmonic oscillator. The energies are $E_{n_x, n_y} = (n_x + n_y + 1)\hbar\omega$.
- Ground state: $n_x=0, n_y=0$, $E_0 = \hbar\omega$, degeneracy $g=1$.
- 1st excited: $(1,0), (0,1)$, $E_1 = 2\hbar\omega$, degeneracy $g=2$.
- 2nd excited: $(2,0), (0,2), (1,1)$, $E_2 = 3\hbar\omega$, degeneracy $g=3$.

\textbf{(b)} The perturbation is $V = \alpha m \omega^2 xy$.
- Ground state $|0,0\rangle$: $E_0^{(1)} = \langle 0,0|V|0,0\rangle = \alpha m \omega^2 \langle 0|x|0\rangle \langle 0|y|0\rangle = 0$.
- 1st excited states $\{|1,0\rangle, |0,1\rangle\}$: $V_{11} = \langle 1,0|V|1,0\rangle = 0$, $V_{22} = \langle 0,1|V|0,1\rangle = 0$, $V_{12} = \langle 1,0|V|0,1\rangle = \alpha m \omega^2 \langle 1|x|0\rangle \langle 0|y|1\rangle = \alpha m \omega^2 (\sqrt{\hbar/2m\omega})^2 = \frac{1}{2}\alpha \hbar\omega$. 
Diagonalizing $\begin{pmatrix} 0 & \frac{1}{2}\alpha \hbar\omega \\ \frac{1}{2}\alpha \hbar\omega & 0 \end{pmatrix}$ gives $\Delta E = \pm \frac{1}{2}\alpha \hbar\omega$. 
Energies: $2\hbar\omega \pm \frac{1}{2}\alpha \hbar\omega$.
- 2nd excited states: Similar analysis on the $3\times 3$ subspace. $V_{11}=V_{22}=V_{33}=0$. Non-zero elements: $\langle 1,1|V|2,0\rangle = \alpha m \omega^2 \langle 1|x|2\rangle \langle 1|y|0\rangle = \alpha m \omega^2 \sqrt{\frac{2\hbar}{2m\omega}} \sqrt{\frac{\hbar}{2m\omega}} = \frac{1}{\sqrt{2}}\alpha \hbar \omega$.
Matrix: $\begin{pmatrix} 0 & 0 & \frac{1}{\sqrt{2}}\alpha \hbar\omega \\ 0 & 0 & \frac{1}{\sqrt{2}}\alpha \hbar\omega \\ \frac{1}{\sqrt{2}}\alpha \hbar\omega & \frac{1}{\sqrt{2}}\alpha \hbar\omega & 0 \end{pmatrix}$. 
Eigenvalues: $0, \pm \alpha \hbar\omega$. Energies: $3\hbar\omega, 3\hbar\omega \pm \alpha \hbar\omega$.

\textbf{(c)} Let $x = \frac{u+v}{\sqrt{2}}$, $y = \frac{u-v}{\sqrt{2}}$. Then $x^2+y^2 = u^2+v^2$ and $xy = \frac{1}{2}(u^2-v^2)$.
$H = \frac{p_u^2+p_v^2}{2m} + \frac{1}{2}m\omega^2(u^2+v^2) + \frac{1}{2}\alpha m \omega^2 (u^2-v^2) = \frac{p_u^2}{2m} + \frac{1}{2}m\omega^2(1+\alpha)u^2 + \frac{p_v^2}{2m} + \frac{1}{2}m\omega^2(1-\alpha)v^2$.
This is two decoupled oscillators with frequencies $\omega_u = \omega \sqrt{1+\alpha}$ and $\omega_v = \omega \sqrt{1-\alpha}$.
Spectrum: $E_{n_u, n_v} = (n_u + 1/2)\hbar\omega \sqrt{1+\alpha} + (n_v + 1/2)\hbar\omega \sqrt{1-\alpha}$.

\textbf{(d)} Expand for small $\alpha$: $\sqrt{1\pm\alpha} \approx 1 \pm \alpha/2 - \alpha^2/8$.
$E \approx (n_u+n_v+1)\hbar\omega + \frac{\alpha}{2}(n_u-n_v)\hbar\omega$.
- $(0,0) \to \hbar\omega$, match.
- $(1,0), (0,1) \to 2\hbar\omega \pm \frac{\alpha}{2}\hbar\omega$, match.
- $(2,0), (0,2), (1,1) \to 3\hbar\omega + \alpha \hbar\omega, 3\hbar\omega - \alpha \hbar\omega, 3\hbar\omega$, match.
\end{solution}

\begin{exercise}
One can express the electric fields $\vec{E}$ and magnetic fields $\vec{B}$ in terms of the scalar and vector potentials $A^\mu = (\phi, \vec{A})$.

\textbf{(a)} Write down the expression of $\vec{E}$ and $\vec{B}$ in terms of $(\phi, \vec{A})$ and show that the result is unchanged under gauge transformation
\[
\phi \to \phi + \frac{\partial}{\partial t}f, \quad \vec{A} \to \vec{A} - \nabla f,
\]
where $f = f(\vec{x}, t)$ is a scalar function.

\textbf{(b)} Show that two of the 4 Maxwell equations of $\vec{E}$ and $\vec{B}$ are satisfied automatically in terms of $A^\mu = (\phi, \vec{A})$.

\textbf{(c)} Derive the equations for the scalar and vector potentials from the remaining Maxwell equations in Lorentz gauge:
\[
\frac{1}{c}\partial_t\phi + \nabla\cdot\vec{A} = 0.
\]

\textbf{(d)} Recall that the Green's function of the wave equation
\[
\left(\frac{1}{c^2}\partial_t^2 - \nabla^2\right)G(t, \vec{r}) = \delta(t)\delta^3(\vec{r})
\]
is
\[
G(t - t_0, \vec{r} - \vec{r}_0) = \frac{\theta(t - t_0)}{4\pi|\vec{r} - \vec{r}_0|}\delta\left(t - t_0 - \frac{|\vec{r} - \vec{r}_0|}{c}\right).
\]
Assume a particle of electric charge $e$ moves with trajectory $\vec{R}(t)$ with $\vec{v}(t) = d\vec{R}(t)/dt$. Use the Green's function to derive the potential $\phi(\vec{r}, t)$ and $\vec{A}(\vec{r}, t)$ of this particle at $(\vec{r}, t)$. You may assume that $|\vec{R}(t)| \ll |\vec{r}|$, $|\vec{R}(t)| \ll ct$ and $|\vec{v}(t)| \ll c$ and expand your result up to the order $\mathcal{O}(1/|\vec{r}|)$ and $\mathcal{O}(|\vec{v}(t)|/c)$. This is also called non-relativistic and far-field approximations.
\end{exercise}

\begin{solution}
\textbf{(a)} $\vec{B} = \nabla \times \vec{A}$, $\vec{E} = -\nabla \phi - \frac{1}{c}\frac{\partial \vec{A}}{\partial t}$.
Under $A^\mu \to A^\mu + \partial^\mu f$:
$\vec{B} \to \nabla \times (\vec{A} - \nabla f) = \nabla \times \vec{A} - 0 = \vec{B}$.
$\vec{E} \to -\nabla(\phi + \dot{f}/c) - \frac{1}{c}\partial_t(\vec{A} - \nabla f) = -\nabla \phi - \frac{1}{c}\dot{\vec{A}} - \frac{1}{c}\nabla \dot{f} + \frac{1}{c} \partial_t \nabla f = \vec{E}$.

\textbf{(b)} The homogeneous equations are $\nabla \cdot \vec{B} = 0$ and $\nabla \times \vec{E} + \frac{1}{c}\partial_t \vec{B} = 0$.
Substituting potentials: $\nabla \cdot (\nabla \times \vec{A}) = 0$ (identity).
$\nabla \times (-\nabla \phi - \frac{1}{c}\dot{\vec{A}}) + \frac{1}{c}\partial_t (\nabla \times \vec{A}) = 0 - \frac{1}{c}\nabla \times \dot{\vec{A}} + \frac{1}{c}\nabla \times \dot{\vec{A}} = 0$.

\textbf{(c)} Inhomogeneous equations: $\nabla \cdot \vec{E} = 4\pi \rho$ and $\nabla \times \vec{B} - \frac{1}{c}\partial_t \vec{E} = \frac{4\pi}{c}\vec{j}$.
$\nabla \cdot (-\nabla \phi - \frac{1}{c}\dot{\vec{A}}) = -\nabla^2 \phi - \frac{1}{c}\partial_t (\nabla \cdot \vec{A})$. Lorentz gauge $\nabla \cdot \vec{A} = -\frac{1}{c}\dot{\phi} \implies \frac{1}{c^2}\partial_t^2 \phi - \nabla^2 \phi = 4\pi \rho$.
Similarly, $\nabla \times (\nabla \times \vec{A}) - \frac{1}{c}\partial_t (-\nabla \phi - \frac{1}{c}\dot{\vec{A}}) = \nabla(\nabla \cdot \vec{A}) - \nabla^2 \vec{A} + \frac{1}{c}\nabla \dot{\phi} + \frac{1}{c^2}\ddot{\vec{A}} = \frac{4\pi}{c}\vec{j}$.
Using Lorentz gauge: $\frac{1}{c^2}\partial_t^2 \vec{A} - \nabla^2 \vec{A} = \frac{4\pi}{c}\vec{j}$.

\textbf{(d)} $\phi(\vec{r},t) = \int G(t-t', \vec{r}-\vec{r}') \rho(\vec{r}',t') d^3r' dt'$. For a point charge, $\rho = e \delta^3(\vec{r}' - \vec{R}(t'))$.
$\phi(\vec{r},t) = \int \frac{e \delta(t-t' - |\vec{r}-\vec{R}(t')|/c)}{4\pi |\vec{r}-\vec{R}(t')|} dt'$.
Let $t_{ret}$ be the retarded time satisfying $c(t-t_{ret}) = |\vec{r}-\vec{R}(t_{ret})|$. 
$\phi = \frac{e}{4\pi [|\vec{r}-\vec{R}| (1 - \vec{n}\cdot\vec{v}/c)]} |_{t_{ret}}$.
For $r \gg R, v \ll c$: $|\vec{r}-\vec{R}| \approx r - \vec{n}\cdot\vec{R}$.
$\phi \approx \frac{e}{4\pi r} (1 + \frac{\vec{n}\cdot\vec{R}}{r} + \frac{\vec{n}\cdot\vec{v}}{c})$.
$\vec{A} \approx \frac{e \vec{v}}{4\pi r c}$.
\end{solution}

\begin{exercise}
Consider a one-dimensional system of free massless bosons with one polarization, and the dispersion relation $E_k = \hbar v|k|$, where $v$ is the particle velocity, $k$ wavevector, $E_k$ energy. The particles are not interacting either among themselves or with external scattering potentials. If the system is in equilibrium at temperature $T$,

\textbf{(a)} Assume that the chemical potential $\mu$ is $0$, calculate the heat capacity $C$ per unit length.

\textbf{(b)} Repeat the calculations for massive fermions, for which $E_k = \hbar^2 k^2 / 2m$ and the chemical potential $\mu$ is far above the bottom of the energy spectrum: $\mu \gg k_B T$. (Consider one spin direction for the fermions, and you may make reasonable approximations.)

Hint: you might find the following formulae useful to evaluate some integrals needed:
\[
\sum_{k=1}^\infty \frac{1}{k^2} = \frac{\pi^2}{6}, \quad \sum_{k=1}^\infty \frac{(-1)^{k+1}}{k^2} = \frac{\pi^2}{12}.
\]
\end{exercise}

\begin{solution}
\textbf{(a)} For 1D massless bosons, $E = \hbar v |k|$. Density of states in $k$ space is $L/2\pi$.
Internal energy $U = \int_0^\infty \frac{\hbar v k}{e^{\beta \hbar v k}-1} \frac{L dk}{\pi}$ (counting positive and negative $k$).
Let $x = \beta \hbar v k$. $U = \frac{L \hbar v}{\pi (\beta \hbar v)^2} \int_0^\infty \frac{x}{e^x-1} dx = \frac{L (k_B T)^2}{\pi \hbar v} \frac{\pi^2}{6} = \frac{\pi L (k_B T)^2}{6 \hbar v}$.
Heat capacity $C = \partial_T U = \frac{\pi L k_B^2 T}{3 \hbar v}$. Per unit length: $C/L = \frac{\pi k_B^2 T}{3 \hbar v}$.

\textbf{(b)} For 1D massive fermions, $E = \frac{\hbar^2 k^2}{2m}$. $\mu \gg k_B T$ implies a degenerate Fermi gas.
Density of states $g(E) = \frac{L}{\pi} \frac{dk}{dE} = \frac{L}{\pi} \sqrt{\frac{m}{2\hbar^2 E}}$.
Using Sommerfeld expansion: $C \approx \frac{\pi^2}{3} k_B^2 T g(E_F)$.
$E_F \approx \mu$. $g(E_F) = \frac{L}{\pi} \sqrt{\frac{m}{2\hbar^2 \mu}}$.
$C/L = \frac{\pi k_B^2 T}{3} \sqrt{\frac{m}{2\hbar^2 \mu}}$.
\end{solution}

\begin{exercise}
Consider the vacuum Einstein's equation in four dimensional spacetime with a cosmological constant:
\[
R_{\mu\nu} - \frac{1}{2}g_{\mu\nu}R + g_{\mu\nu}\Lambda = 0.
\]

\textbf{(a)} Prove that $R_{\mu\nu} = k g_{\mu\nu}$ and find out the value of $k$.

\textbf{(b)} Now start with an ansatz of a metric in the following form:
\[
ds^2 = -f(r)dt^2 + \frac{1}{f(r)}dr^2 + r^2(d\theta^2 + \sin^2\theta\, d\phi^2),
\]
where $f(r)$ is a polynomial in $r$. Compute non-zero components of the Ricci tensor $R_{\mu\nu}$ and scalar curvature $R$ of this metric.

\textbf{(c)} Assuming that the above ansatz is a solution of the vacuum Einstein equation with cosmological constant $\Lambda$, solve $f(r)$.

\textbf{(d)} Prove that $\partial_t$ and $\partial_\phi$ are Killing vector fields.
\end{exercise}

\begin{solution}
\textbf{(a)} From Einstein's equation $R_{\mu\nu} - \frac{1}{2}g_{\mu\nu}R + \Lambda g_{\mu\nu} = 0$, take the trace (multiply by $g^{\mu\nu}$): $R - 2R + 4\Lambda = 0 \implies R = 4\Lambda$.
Substitute back: $R_{\mu\nu} - \frac{1}{2}g_{\mu\nu}(4\Lambda) + \Lambda g_{\mu\nu} = 0 \implies R_{\mu\nu} = \Lambda g_{\mu\nu}$.
Thus $k = \Lambda$.

\textbf{(b)} For the metric $ds^2 = -f dt^2 + f^{-1} dr^2 + r^2 d\Omega^2$, the Ricci tensor components are:
$R_{tt} = \frac{1}{2} f f'' + \frac{1}{r} f f'$, $R_{rr} = -\frac{1}{2} f'' - \frac{1}{r} \frac{f'}{f}$, $R_{\theta\theta} = 1 - f - rf'$, $R_{\phi\phi} = \sin^2\theta R_{\theta\theta}$.
Scalar curvature $R = -f'' - \frac{4f'}{r} - \frac{2f}{r^2} + \frac{2}{r^2}$.

\textbf{(c)} Requirement $R_{\mu\nu} = \Lambda g_{\mu\nu} \implies R_{\theta\theta} = \Lambda g_{\theta\theta} = \Lambda r^2$.
$1 - f - rf' = \Lambda r^2 \implies (rf)' = 1 - \Lambda r^2$.
Integrating: $rf = r - \frac{\Lambda r^3}{3} - 2M \implies f(r) = 1 - \frac{2M}{r} - \frac{\Lambda r^2}{3}$.
This is the Schwarzschild-de Sitter solution.

\textbf{(d)} The metric components $g_{\mu\nu}$ are independent of $t$ and $\phi$. Thus $\partial_t$ and $\partial_\phi$ are symmetries of the metric. Killing's equation is $\nabla_\mu \xi_\nu + \nabla_\nu \xi_\mu = \mathcal{L}_\xi g_{\mu\nu} = 0$. Since $\partial_\sigma g_{\mu\nu} = 0$ for $\sigma = t, \phi$, these are Killing vectors.
\end{solution}

\begin{exercise}
Consider the $\phi^3$ model with a real scalar field $\phi(x)$ in $3+1$ dimensional Minkowski spacetime with metric $(-, +, +, +)$. Its Lagrangian density is:
\[
\mathcal{L} = -\frac{1}{2}\partial_\mu\phi\partial^\mu\phi - \frac{1}{2}m^2\phi^2 - \frac{1}{6}g\phi^3,
\]
where $g$ is a coupling with dimensions of mass.

\textbf{(a)} Write down the propagator and the interaction vertex for this model in momentum space.

\textbf{(b)} Compute the one-loop self-energy graph using dimensional regularization.

\textbf{(c)} Introducing $m^2 = m_R^2 + \delta m^2$. What is the value of $\delta m^2$ if we want to write the one-loop self-energy graph as a finite function of $m_R$?

You may find the following formula useful:
\[
\int d^d k \frac{1}{(-k^2 - 2p\cdot k - M^2 + i\epsilon)^s} = (-1)^s i\pi^{d/2} \frac{\Gamma(s - d/2)}{\Gamma(s)}(-p^2 + M^2 - i\epsilon)^{d/2 - s},
\]
where $\Gamma(z)$ is the Gamma function which has a simple pole at the origin:
\[
\Gamma(z) = \frac{1}{z} - \gamma + \mathcal{O}(z),
\]
where $\gamma$ is the Euler constant.
\end{exercise}

\begin{solution}
\textbf{(a)} From $\mathcal{L}$, the propagator is $D(p) = \frac{-i}{p^2+m^2-i\epsilon}$.
The interaction term is $-\frac{g}{3!}\phi^3$, giving a vertex factor $-ig$.

\textbf{(b)} One-loop self-energy $\Pi(p^2) = \frac{1}{2} (-ig)^2 \int \frac{d^d k}{(2\pi)^d} \frac{-i}{k^2+m^2} \frac{-i}{(p-k)^2+m^2}$.
Using Feynman parametrization $\frac{1}{AB} = \int_0^1 \frac{dx}{[xA + (1-x)B]^2}$:
$\Pi(p^2) = \frac{g^2}{2} \int \frac{d^d k}{(2\pi)^d} \int_0^1 \frac{dx}{[k^2 - 2xp\cdot k + xp^2 + m^2]^2}$.
Shift $k \to k + xp$: $\Pi(p^2) = \frac{g^2}{2} \int_0^1 dx \int \frac{d^d k}{(2\pi)^d} \frac{1}{[k^2 + m^2 + x(1-x)p^2]^2}$.
Using the formula with $s=2, M^2 = m^2 + x(1-x)p^2$:
$\Pi(p^2) = \frac{ig^2}{2(4\pi)^{d/2}} \int_0^1 dx \frac{\Gamma(2-d/2)}{(m^2+x(1-x)p^2)^{2-d/2}}$.
For $d=4-\epsilon$: $\Gamma(\epsilon/2) \approx \frac{2}{\epsilon} - \gamma$.
$\Pi(p^2) \approx \frac{ig^2}{32\pi^2} \int_0^1 dx \left( \frac{2}{\epsilon} - \gamma - \ln(m^2+x(1-x)p^2) + \ln(4\pi) \right)$.

\textbf{(c)} To make the self-energy finite, $\delta m^2$ must cancel the $1/\epsilon$ divergence.
$\Pi_{finite}(p^2) = \Pi(p^2) + \delta m^2$.
Thus $\delta m^2 = - \frac{g^2}{16\pi^2 \epsilon} + \text{finite terms}$.
In the MS scheme, $\delta m^2 = - \frac{g^2}{16\pi^2 \epsilon}$.
\end{solution}
