\section{2023 Mathematical Physics}

\subsection{Classical Mechanics}

\begin{exercise}
\label{physics:2023:1}
Consider the Lagrangian
\begin{equation}
    L (x, y, \dot {x}, \dot {y}) = \frac {1}{2} \frac {\dot {x} ^ {2} + \dot {y} ^ {2} + 2 (x \dot {y} - y \dot {x})}{x ^ {2} + y ^ {2}}
\end{equation}

\begin{enumerate}[label=(\alph*)]
    \item Compute the Hamiltonian $H ( x , y , p _ { x } , p _ { y } )$ , and show the final form can be written as
    \begin{equation}
        \frac {1}{2} f (x, y) \left[ \left(p _ {x} - A _ {x} (x, y)\right) ^ {2} + \left(p _ {y} - A _ {y} (x, y)\right) ^ {2} \right]
    \end{equation}
    for some $f , A _ { x } , A _ { y }$. Find the vector potential $\vec { A }$ and then compute the corresponding magnetic field away from the origin. (Hint: recall that $\vec { B } = \mathrm { curl } \vec { A } = \nabla \times \vec { A }$ ).
    \item Prove that the Lagrangian $L ( x , y , \dot { x } , \dot { y } )$ is invariant under two symmetries: rotations and scale transformations.
    \item Derive the conserved quantities for both symmetries.
    \item Rewrite the Lagrangian in polar coordinates, write down the Euler-Lagrange equations and solve them.
\end{enumerate}
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item \textbf{Legendre Transformation:} $H = \sum p \dot{q} - L$.
    \item \textbf{Noether's Theorem:} Symmetries lead to conserved charges.
\end{itemize}

\textbf{Steps:}

1. \textbf{Momenta:} $p_x = \frac{\dot{x}-y}{x^2+y^2}$, $p_y = \frac{\dot{y}+x}{x^2+y^2}$.
2. \textbf{Hamiltonian Form:} $H = \frac{1}{2}(x^2+y^2)[(p_x - \frac{-y}{x^2+y^2})^2 + (p_y - \frac{x}{x^2+y^2})^2]$.
3. \textbf{Magnetic Field:} The vector potential $\vec{A} = \frac{-y\hat{i} + x\hat{j}}{x^2+y^2}$ gives $B_z = \nabla \times \vec{A} = 0$ everywhere except the origin. This is the Aharonov-Bohm flux.
4. \textbf{Symmetries:} The Lagrangian is invariant under rotation (manifest in the $x\dot{y}-y\dot{x}$ term) and scaling (both numerator and denominator scale as $\lambda^2$).

\textbf{Intuition:}
This system behaves like a particle in a plane with a magnetic "vortex" at the center. Even though the magnetic field is zero of the origin, the potential still affects the particle's phase and conservation laws, a classical precursor to the Aharonov-Bohm effect.
\end{solution}

\subsection{Quantum Mechanics}

\begin{exercise}
\label{physics:2023:2}
A particle of mass $m$ in 2 dimensions is confined by an isotropic harmonic oscillator potential of frequency $\omega$ , while subject to a weak and anisotropic perturbation of strength $\alpha \ll 1$ . The total Hamiltonian of the particle is
\begin{equation}
    H = H _ {0} + V = \frac {p _ {x} ^ {2}}{2 m} + \frac {p _ {y} ^ {2}}{2 m} + \frac {1}{2} m \omega^ {2} \left(x ^ {2} + y ^ {2}\right) + \alpha m \omega^ {2} x y
\end{equation}

\begin{enumerate}[label=(\alph*)]
    \item When $\alpha = 0$ , what are the energies and degeneracies of the three lowest-lying unperturbed states?
    \item Use perturbation theory to correct the energies of the above three states to the first order in $\alpha$ .
    \item Find the exact spectrum of $H$ . (Hint: you may want to rotate $x$ and $y$ into a new coordinates)
    \item Check that the perturbative results in part (b) are recovered from the exact spectrum.
\end{enumerate}
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item \textbf{Matrix Diagonalization:} For degenerate perturbation theory.
    \item \textbf{Coordinate Rotation:} To decouple quadratic forms.
\end{itemize}

\textbf{Steps:}

1. \textbf{First Order:} For $|1,0\rangle, |0,1\rangle$, the perturbation matrix has off-diagonal $V_{12} = \alpha m \omega^2 (\hbar/2m\omega) = \frac{1}{2}\alpha\hbar\omega$. Eigenvalues are $\pm \alpha\hbar\omega / 2$.
2. \textbf{Exact Solution:} Define $u, v = \frac{x \pm y}{\sqrt{2}}$. The Hamiltonian becomes $\frac{p_u^2}{2m} + \frac{1}{2}m\omega^2(1+\alpha)u^2 + \frac{p_v^2}{2m} + \frac{1}{2}m\omega^2(1-\alpha)v^2$.
3. \textbf{Spectrum:} $E = (n_u + 1/2)\hbar\omega\sqrt{1+\alpha} + (n_v + 1/2)\hbar\omega\sqrt{1-\alpha}$. Expanding for small $\alpha$ recovers the perturbative result.

\textbf{Intuition:}
Two coupled oscillators can always be viewed as two independent "normal modes." Here, the perturbation "mixes" the $x$ and $y$ motions. By rotating our perspective by 45 degrees, we find that the particle is simply vibrating in two new directions with slightly different "stiffnesses."
\end{solution}

\subsection{Electromagnetism}

\begin{exercise}
\label{physics:2023:3}
One can express the electric fields $\vec { E }$ and magnetic fields $\vec { B }$ in terms of the scalar and vector potentials, $A ^ { \mu } = \left( \phi , { \vec { A } } \right)$ .

\begin{enumerate}[label=(\alph*)]
    \item Write down the expression of $\vec { E }$ and $\vec { B }$ in terms of $( \phi , { \vec { A } } )$ and show that the result is unchanged under gauge transformation
    \begin{equation}
        \phi \rightarrow \phi + \frac {\partial}{\partial t} f, \quad \vec {A} \rightarrow \vec {A} - \nabla f,
    \end{equation}
    where $f = f ( \vec { x } , t )$ is a scalar function.
    \item Show that two of the 4 Maxwell equations of $\vec { E }$ and $\vec { B }$ are satisfied automatically in terms of $A ^ { \mu } = \left( \phi , { \vec { A } } \right)$ .
    \item Derive the equations for the scalar and vector potentials from the remaining Maxwell equations in Lorentz gauge
    \begin{equation}
        \frac {1}{c} \partial_ {t} \phi + \nabla \cdot \vec {A} = 0.
    \end{equation}
    \item Recall that the Green's function of the wave equation
    \begin{equation}
        \left(\frac {1}{c ^ {2}} \partial_ {t} ^ {2} - \nabla^ {2}\right) G (t, \vec {r}) = \delta (t) \delta^ {3} (\vec {r})
    \end{equation}
    is
    \begin{equation}
        G (t - t _ {0}, \vec {r} - \vec {r} _ {0}) = \frac {\theta (t - t _ {0})}{4 \pi | \vec {r} - \vec {r} _ {0} |} \delta \left(t - t _ {0} - \frac {| \vec {r} - \vec {r} _ {0} |}{c}\right).
    \end{equation}
    Assume a particle of electric charge $e$ moves with trajectory $\vec { R } ( t )$ with ${ \vec { v } } ( t ) = d \vec { R } ( t ) / d t$. Use the Green's function to derive the potential $\phi ( \vec { r } , t )$ and $\vec { A } ( \vec { r } , t )$ of this particle at $( \vec { r } , t )$. You may assume that $| \vec { R } ( t ) | \ll | \vec { r } |$ , $| \vec { R } ( t ) | \ll c t$ and $| \vec { v } ( t ) | \ll c$ and expand your result up to the order $\mathcal { O } ( 1 / | \vec { r } | )$ and $\mathcal { O } ( | \vec { v } ( t ) | / c )$. This is also called non-relativistic and far-field approximations.
\end{enumerate}
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item \textbf{Gauge Transformation:} $\phi \to \phi + \dot{f}, \vec{A} \to \vec{A} - \nabla f$.
    \item \textbf{Retarded Potential:} $t_{ret} = t - |\vec{r} - \vec{R}|/c$.
\end{itemize}

\textbf{Steps:}

1. \textbf{Invariance:} Since $\vec{E} = -\nabla\phi - \dot{\vec{A}}/c$, the $\nabla\dot{f}$ and $\dot{\nabla}f$ terms cancel precisely.
2. \textbf{Lorentz Gauge:} Imposing $\dot{\phi}/c + \nabla \cdot \vec{A} = 0$ decouples Maxwell's equations into two wave equations.
3. \textbf{Approximation:} For $v \ll c$ and large $r$, $\phi \approx \frac{e}{r} + \frac{e \vec{n} \cdot \vec{v}}{rc}$. This is the start of the multipole expansion for a moving charge.

\textbf{Intuition:}
Potentials are mathematical tools that make Maxwell's equations easier to solve. However, they aren't unique—we can change them (gauge transformation) without affecting the physical electric and magnetic fields. The retarded potential tells us that "news" of the charge's motion travels at the speed of light.
\end{solution}

\subsection{Statistical Mechanics}

\begin{exercise}
\label{physics:2023:4}
Consider a one-dimensional system of free massless bosons with one polarization, and the dispersion relation $E _ { k } = \hbar v | k |$ , where $v$ is the particle velocity, $k$ wavevector, $E _ { k }$ energy. The particles are not interacting either among themselves or with external scattering potentials. If the system is in equilibrium at temperature $T$ ,

\begin{enumerate}[label=(\alph*)]
    \item Assume that the chemical potential $\mu$ is $0$ , calculate the heat capacity $C$ per unit length.
    \item Repeat the calculations for massive fermions, for which $E _ { k } = \hbar ^ { 2 } k ^ { 2 } / 2 m$ and the chemical potential $\mu$ is far above the bottom of the energy spectrum: $\mu \gg k _ { B } T$ . (Consider one spin direction for the fermions, and you may make reasonable approximations.)
\end{enumerate}

Hint: you might find the following formulae useful to evaluate some integrals needed
\begin{equation}
    \sum_ {k = 1} ^ {\infty} \frac {1}{k ^ {2}} = \frac {\pi^ {2}}{6}, \quad \sum_ {k = 1} ^ {\infty} \frac {(- 1) ^ {k + 1}}{k ^ {2}} = \frac {\pi^ {2}}{1 2}.
\end{equation}
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item \textbf{Density of States:} $g(k) = L/\pi$ for 1D.
    \item \textbf{Sommerfeld Expansion:} For degenerate fermions.
\end{itemize}

\textbf{Steps:}

1. \textbf{Massless Bosons:} $U \propto \int \frac{k dk}{e^{\beta \hbar v k}-1} \propto T^2$. Thus $C = dU/dT \propto T$.
2. \textbf{Massive Fermions:} $g(E) \propto E^{-1/2}$. For $\mu \gg T$, the heat capacity is $C = \frac{\pi^2}{3} k_B^2 T g(\mu)$.
3. \textbf{Conclusion:} In 1D, both systems show a linear temperature dependence of heat capacity at low $T$, but for different physical reasons (massless nature vs. Pauli exclusion).

\textbf{Intuition:}
At low temperatures, only the states near the lowest energy (for bosons) or the Fermi surface (for fermions) contribute to the heat capacity. In 1D, the restricted space makes these excitations behave very predictably, leading to a simple linear relationship between energy and temperature.
\end{solution}

\subsection{General Relativity}

\begin{exercise}
\label{physics:2023:5}
Consider the vacuum Einstein's equation in four dimensional spacetime with a cosmological constant
\begin{equation}
    R _ {\mu \nu} - \frac {1}{2} g _ {\mu \nu} R + g _ {\mu \nu} \Lambda = 0.
\end{equation}

\begin{enumerate}[label=(\alph*)]
    \item Prove that $R _ { \mu \nu } = k g _ { \mu \nu }$ and find out the value of $k$ .
    \item Now start with an ansatz of a metric in the following form
    \begin{equation}
        d s ^ {2} = - f (r) d t ^ {2} + \frac {1}{f (r)} d r ^ {2} + r ^ {2} \left(d \theta^ {2} + \sin^ {2} \theta d \phi^ {2}\right),
    \end{equation}
    where $f ( r )$ is a polynomial in $r$ . Compute non-zero components of the Ricci tensor $R _ { \mu \nu }$ and scalar curvature $R$ of this metric.
    \item Assuming that the above ansatz is a solution of the vacuum Einstein equation with cosmological constant $\Lambda$ , solve $f ( r )$ .
    \item Prove that $\partial _ { t }$ and $\partial _ { \phi }$ are Killing vector fields.
\end{enumerate}
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item \textbf{Einstein Equation:} $G_{\mu\nu} + \Lambda g_{\mu\nu} = 0$.
    \item \textbf{Static Metric Ansatz:} $ds^2 = -f dt^2 + f^{-1} dr^2 + r^2 d\Omega^2$.
\end{itemize}

\textbf{Steps:}

1. \textbf{Einstein Tensor:} The trace of the equation in 4D gives $R = 4\Lambda$.
2. \textbf{Ricci Tensor:} For the given ansatz, $R_{\theta\theta} = 1 - f - rf' = \Lambda r^2$.
3. \textbf{Solving $f$:} Integrating $1 - rf' - f = \Lambda r^2$ leads to $rf = r - \Lambda r^3/3 - 2M$.
4. \textbf{Result:} $f(r) = 1 - \frac{2M}{r} - \frac{\Lambda r^2}{3}$, the Schwarzschild-de Sitter metric.

\textbf{Intuition:}
The cosmological constant $\Lambda$ acts like a "pressure" everywhere in space. It modifies the gravity of a point mass ($2M/r$) by adding a term that grows with distance ($\Lambda r^2$), eventually dominating the geometry of the universe at large scales.
\end{solution}

\subsection{Quantum Field Theory}

\begin{exercise}
\label{physics:2023:6}
Consider the $\phi ^ { 3 }$ model with a real scalar field $\phi ( x )$ in $3 + 1$ dimensional Minkowski spacetime with metric $( - , + , + , + )$. Its Lagrangian density is
\begin{equation}
    \mathcal {L} = - \frac {1}{2} \partial_ {\mu} \phi \partial^ {\mu} \phi - \frac {1}{2} m ^ {2} \phi^ {2} - \frac {1}{6} g \phi^ {3},
\end{equation}
where $g$ is a coupling with dimensions of mass.

\begin{enumerate}[label=(\alph*)]
    \item Write down the propagator and the interaction vertex for this model in momentum space.
    \item Compute the one-loop self-energy graph using dimensional regularization.
    \item Introducing $m ^ { 2 } = m _ { R } ^ { 2 } + \delta m ^ { 2 }$. What is the value of $\delta m ^ { 2 }$ if we want to write the one-loop self-energy graph as a finite function of $m _ { R }$?
\end{enumerate}

You may find the following formula useful:
\begin{align}
    (A B) ^ {- 1} &= \int_ {0} ^ {1} d x [ x A + (1 - x) B ] ^ {- 2}, \\
    \int d ^ {d} k \frac {1}{(- k ^ {2} - 2 p \cdot k - M ^ {2} + i \epsilon) ^ {s}} &= (- 1) ^ {s} i \pi^ {d / 2} \frac {\Gamma (s - d / 2)}{\Gamma (s)} (- p ^ {2} + M ^ {2} - i \epsilon) ^ {d / 2 - s},
\end{align}
where $\Gamma ( z )$ is the Gamma function which has a simple pole at the origin: $\Gamma (z) = \frac {1}{z} - \gamma + \mathcal {O} (z)$, where $\gamma$ is the Euler constant.
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item \textbf{Feynman Rules:} Vertex $-ig$, Propagator $-i/(k^2+m^2)$.
    \item \textbf{Dimensional Regularization:} Using $d=4-\epsilon$.
\end{itemize}

\textbf{Steps:}

1. \textbf{Diagram:} A bubble with two vertices. Symmetric factor $1/2$.
2. \textbf{Integral:} $\Pi(p^2) \propto g^2 \int \frac{d^d k}{(k^2+m^2)((p-k)^2+m^2)}$.
3. \textbf{Divergence:} The integral contains a $1/\epsilon$ pole.
4. \textbf{Counter-term:} A mass shift $\delta m^2 = - \frac{g^2}{16\pi^2\epsilon}$ is required to cancel the infinity.

\textbf{Intuition:}
Virtual particles flashing in and out of existence (loops) modify the properties of real particles. This "cloud" of virtual interactions makes the particle appear heavier. Renormalization is the process of defining our starting parameters to account for this constant background interference.
\end{solution}
