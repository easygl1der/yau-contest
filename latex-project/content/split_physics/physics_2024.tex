\section{2024 Mathematical Physics}

\subsection{Classical Mechanics}

\begin{exercise}
\label{physics:2024:1}
Consider the motion of a particle of mass $m$ in an attractive central potential of the form
\begin{equation}
    V (r) = \alpha r ^ {k},
\end{equation}
where $k$ and $\alpha$ are real constants of the same sign (both positive or both negative).

\begin{enumerate}[label=(\alph*)]
    \item Write down the Lagrangian using polar coordinates $( r , \phi )$.
    \item Using conservation of the angular momentum, reduce the problem of determining the radial motion to an effective one-dimensional problem (write down the effective Lagrangian).
    \item Determine the radius and period of the circular orbits.
    \item For which values of $k$ is the circular orbit stable?
    \item Assuming that the circular orbit is stable, consider a small perturbation around it. Find the period of the small oscillations. In the approximation of small oscillations, for which values of $k$ will the orbit close?
    \item Go back to the full 2D problem for $r$ and $\phi$ (the polar coordinates in the plane of the orbit). Eliminate the time dependence and write a differential equation for the orbit.
\end{enumerate}
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item \textbf{Lagrangian in Polar Coordinates:} $L = \frac{1}{2}m(\dot{r}^2 + r^2\dot{\phi}^2) - V(r)$.
    \item \textbf{Effective Potential:} $V_{\text{eff}} = \frac{J^2}{2mr^2} + V(r)$, where $J$ is angular momentum.
    \item \textbf{Bertrand's Theorem:} Only $1/r$ and $r^2$ potentials have all bound orbits closed.
\end{itemize}

\textbf{Steps:}

1. \textbf{Conserved J:} Scaling $\dot{\phi} = J/mr^2$ reduces the problem to 1D radial motion with $V_{\text{eff}}$.
2. \textbf{Equilibrium:} Circular orbits at $V_{\text{eff}}'(r_0) = 0 \implies k\alpha r_0^{k+2} = J^2/m$.
3. \textbf{Stability:} Stable if $V_{\text{eff}}''(r_0) > 0$. Calculation gives $k\alpha r_0^{k-2}(k+2) > 0$, requiring $k > -2$.
4. \textbf{Closing Orbits:} The ratio of oscillation to orbital frequency is $\sqrt{k+2}$. Closed orbits require this to be a rational number.

\textbf{Intuition:}
Imagine the particle "sliding" on a curve formed by the overlap of gravity (pulling in) and centrifugal force (pushing out). For a stable circular orbit, the particle must "rest" at the bottom of the resulting valley. If $k$ is too negative (steeper than $1/r^2$), the valley disappears, and the particle either crashes or flies away.
\end{solution}

\subsection{Quantum Mechanics}

\begin{exercise}
\label{physics:2024:2}
Consider a 1D quantum-mechanical harmonic oscillator with mass $m$ and resonance frequency $\omega$. The oscillator initially (at $t \to -\infty$) is in its ground state. It is then subjected to a transient perturbation $\Delta H = F(t)x$ with $F(t \to \pm \infty) = 0$.

\begin{enumerate}[label=(\alph*)]
    \item Write down the Hamiltonian $\hat{H}$ of the perturbed oscillator described above in terms of the usual ladder operators $\hat{a}$ and $\hat{a}^\dagger$, and solve their equations of motion in the Heisenberg picture. Show that the Hamiltonian at $t \to \pm \infty$ takes the form
    \begin{equation}
        \hat{H} = \hbar \omega \left(\hat{a}_{\pm \infty}^\dagger \hat{a}_{\pm \infty} + 1/2\right),
    \end{equation}
    and determine the relation between $\hat{a}_{+\infty}$ and $\hat{a}_{-\infty}$.
    \item At $t \to \pm \infty$, the ladder operators act on the state $|n_{\pm \infty}\rangle = (1/\sqrt{n!})(\hat{a}_{\pm \infty}^\dagger)^n |0_{\pm \infty}\rangle$. Here $|0_{\pm \infty}\rangle$ denote the vacuum with respect to $\hat{a}_{\pm \infty}$ and $\hat{a}_{\pm \infty}^\dagger$. Determine the probabilities $|c_n|^2$ that the oscillator has undergone a transition from the initial ground state to the $n$-th excited state at the end of the time evolution.
    \item What is the expectation value of the energy at the end of the time evolution?
    \item Now assume that $F(t) = F_0 e^{-t^2 / (2\sigma_t^2)}$, with $F_0 = \eta \hbar \omega / l$ where $\eta$ is a dimensionless parameter, and $l = \sqrt{\hbar / (m \omega)}$ is the harmonic oscillator length. For short pulses with $\sigma_t \omega \ll 1$, determine the maximum pulse strength $\eta$ for which less than 1\% of the population gets lost from the ground state. Show explicitly that in the limit $\sigma_t \omega \gg 1$, losses can be suppressed for any given value of $\eta$.
\end{enumerate}
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item \textbf{Heisenberg Equation:} $\dot{A} = \frac{1}{i\hbar}[A, H]$.
    \item \textbf{Coherent State States:} Eigenstates of the annihilation operator.
\end{itemize}

\textbf{Steps:}

1. \textbf{Operator Evolution:} $\dot{\hat{a}} = -i\omega\hat{a} - \frac{iF(t)}{\sqrt{2m\hbar\omega}}$.
2. \textbf{Integration:} Solving for $t \to \infty$ yields $\hat{a}(\infty) = e^{-i\omega t}(\hat{a}(-\infty) - i\beta)$ where $\beta \propto \text{Fourier transform of } F(t)$.
3. \textbf{Transition Probabilities:} The ground state evolves into a coherent state $|\alpha\rangle$. The probability for state $n$ is a Poisson distribution $P_n = \frac{|\beta|^{2n}}{n!}e^{-|\beta|^2}$.

\textbf{Intuition:}
A linear force simply "pulls" the center of the oscillator potential. If the force is applied and then removed, the ground state doesn't stay still but "bounces"—this quantum bouncing state is what we call a coherent state, which mimics the motion of a classical particle.
\end{solution}

\subsection{Electromagnetism}

\begin{exercise}
\label{physics:2024:3}
Consider an ohmic metal with high (but not infinite) conductivity $\sigma$ and magnetic permeability $\mu = 1$ in Heaviside-Lorentz units ($\mu = \mu_0$ in SI units).

\begin{enumerate}[label=(\alph*)]
    \item Show that for harmonic time dependence, and high conductivity $\sigma \gg \omega$ ($\frac{\sigma}{\epsilon_0} \gg \omega$ in SI units), that damped wave-like solutions propagating in $z$-direction in the metal take the approximate form
    \begin{equation}
        \mathbf{H}(t, z) = \mathbf{H}_c e^{-i\omega t + i k_c z}
    \end{equation}
    where $k_c = \frac{1+i}{\sqrt{2}} \frac{\sqrt{\sigma \omega}}{c}$ in Heaviside-Lorentz units, or $k_c = \frac{1+i}{\sqrt{2}} \frac{\sqrt{\sigma \omega / \epsilon_0}}{c}$ in SI units.
    \item The electric field obeys a similar equation, $\mathbf{E}(t, z) = \mathbf{E}_c e^{-i\omega t + i k_c z}$. Use the Maxwell equations to express the amplitude of the electric field $\mathbf{E}_c$ in terms of the magnetic field $\mathbf{H}_c$.
    \item Now consider a linearly polarized plane wave in vacuum of frequency $\omega$, which is normally incident upon a semi-infinite metal block with infinite conductivity as shown below. When the metal has infinite conductivity, the amplitude of the reflected wave equals the amplitude of the incident wave, but the polarization of the reflected wave is inverted. Explain this fact using the appropriate boundary conditions.
    \item Now consider the same reflection problem as in part (c), but this time the metal has a large (but finite) conductivity $\sigma$. Determine the electric and magnetic fields in the metal to leading order in $\omega / \sigma$. Let the amplitude of the incident wave be $E_I$.
\end{enumerate}

\begin{center}
    \includegraphics[width=0.6\textwidth]{images/physics/3d2eaed279ce1032918d6e4d495a1dcbb18a8694d0399fe20866f318e430841d.jpg}
\end{center}
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item \textbf{Ohm's Law:} $\mathbf{J} = \sigma \mathbf{E}$.
    \item \textbf{Maxwell's Curl Equations:} $\nabla \times \mathbf{E} = - \dot{\mathbf{B}}/c, \nabla \times \mathbf{H} = 4\pi\sigma\mathbf{E}/c$.
\end{itemize}

\textbf{Steps:}

1. \textbf{Wave/Diffusion Equation:} $\nabla^2 \mathbf{E} = \frac{4\pi\sigma}{c^2}\dot{\mathbf{E}}$. In metals, conductivity dominates over displacement current.
2. \textbf{Skin Depth:} For frequency $\omega$, the field penetrates with a complex wave vector $k = (1+i)/\delta$, where $\delta = c/\sqrt{2\pi\sigma\omega}$.
3. \textbf{Surface Impedance:} $Z = E/H \propto \sqrt{\omega/\sigma}$. Large $\sigma$ implies $Z \to 0$, leading to near total reflection.

\textbf{Intuition:}
When light hits a metal, the electrons try to cancel the field within. They succeed almost everywhere, except for a thin "skin" at the surface. The faster the field oscillates or the more mobile the electrons, the thinner this skin becomes because the metal responds faster and more violently.
\end{solution}

\subsection{Statistical Mechanics}

\begin{exercise}
\label{physics:2024:4}
Consider the Ising model on $N$ spins $\sigma_i = \pm 1$ in an external magnetic field $h$. Within the mean field approximation, its Hamiltonian can be written as
\begin{equation}
    H_{MF} = \frac{1}{2} N J m^2 - (J m + h) \sum_i \sigma_i,
\end{equation}
where the coordination number of the lattice has been absorbed into the coupling constant $J$ and $m$ is the magnetization. The magnetization, specific heat and magnetic susceptibility are defined, respectively, as $m = \frac{\partial f}{\partial h}$, $C = \frac{\partial U}{\partial T}$, $\chi = \frac{\partial m}{\partial h}$, where $T$ is the temperature, $U$ the internal energy, and $f$ the free energy per site.

\begin{enumerate}[label=(\alph*)]
    \item Derive the (mean field) partition function following from $H_{MF}$ and hence calculate the free energy of the system.
    \item Derive the constraint on magnetization $m$, and graphically solve it for $h=0$. Discuss the physical nature of the various solutions as a function of the temperature $T$. Identify a critical temperature $T_c$ in terms of the system parameters, and discuss its physical meaning.
    \item Assuming $\beta (Jm + h) \ll 1$, derive the expression for the dependence of the magnetization, the specific heat and the magnetic susceptibility on the quantity $t = \frac{T - T_c}{T_c}$, where $T_c$ is the critical temperature, and thereby determine the mean-field critical exponents $\alpha_c, \beta_c, \gamma_c$. For the calculation of the specific heat, note that, within the mean field approximation near the critical point, the internal energy is $U \propto Jm^2$.
\end{enumerate}
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item \textbf{Mean-Field Approximation:} Replace local interactions with an average field.
    \item \textbf{Free Energy:} $f = u - Ts$.
\end{itemize}

\textbf{Steps:}

1. \textbf{Hamiltonian:} $H \approx -M(Jm+h) + \frac{1}{2}NJm^2$, where $m$ is the average magnetization.
2. \textbf{Self-Consistency:} Minimizing free energy gives $m = \tanh(\beta J m + \beta h)$.
3. \textbf{Transition:} For $T < J/k_B$, non-zero magnetization solutions appear even if $h=0$. This is the Curie temperature $T_c$.
4. \textbf{Exponents:} Expansion near $T_c$ shows $m \propto (T_c-T)^{1/2}$ and $\chi \propto |T-T_c|^{-1}$, giving $\beta_c=1/2, \gamma_c=1$.

\textbf{Intuition:}
In this model, everyone is "connected" to everyone else. If a small group of people decides to point "up," their combined influence forces everyone else to join them. This collective pressure is what leads to a phase transition, turning a random mess into a polarized state as you cool it down.
\end{solution}

\subsection{General Relativity}

\begin{exercise}
\label{physics:2024:5}
Let $ds^2 = -(dx^0)^2 + \sum_{i=1}^n (dx^i)^2$ be the Minkowski metric of $\mathbb{R}^{1,n}$. The de Sitter space $dS_n$ is the submanifold of $\mathbb{R}^{1,n}$ defined by the following equation
\begin{equation}
    -(x^0)^2 + \sum_{i=1}^n (x^i)^2 = \alpha^2,
\end{equation}
where $\alpha$ is a nonzero real constant. Define the static coordinates $(t, r, z^2, \dots, z^n)$ as
\begin{align*}
    x^0 &= \sqrt{\alpha^2 - r^2} \sinh(t/\alpha), \\
    x^1 &= \sqrt{\alpha^2 - r^2} \cosh(t/\alpha), \\
    x^i &= r z^i \quad (2 \le i \le n),
\end{align*}
where $z^i$'s are coordinates of an $(n-2)$-sphere with radius 1 in $\mathbb{R}^{n-1}$ ($\sum_{i=2}^n (z^i)^2 = 1$).

\begin{enumerate}[label=(\alph*)]
    \item Show that $(t, r, z^2, \dots, z^n)$ is a set of local coordinates of $dS_n$.
    \item Compute the metric on $dS_n$ induced from the Minkowski metric of $\mathbb{R}^{1,n}$.
    \item Let $n=3$, compute the Ricci tensor $R_{\mu\nu}$ and scalar curvature $R$ of $dS_3$. Is $dS_3$ an Einstein metric?
    \item Are $\partial_t$ and $\partial_\phi$ Killing vector fields in static coordinates when $n=3$? Prove your answer.
\end{enumerate}
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item \textbf{Hypersurface Induced Metric:} $ds^2 = \eta_{AB} dx^A dx^B$.
    \item \textbf{Einstein Manifold:} $R_{\mu\nu} = \Lambda g_{\mu\nu}$.
\end{itemize}

\textbf{Steps:}

1. \textbf{Substitution:} Using $x^0 = \sqrt{\alpha^2-r^2}\sinh(t/\alpha)$ and $x^1 = \sqrt{\alpha^2-r^2}\cosh(t/\alpha)$ satisfies the constraint identically.
2. \textbf{Metric Derivation:} Differentials $dx^0, dx^1$ yield $ds^2 = -(1-r^2/\alpha^2)dt^2 + (1-r^2/\alpha^2)^{-1}dr^2 + r^2 d\Omega^2$.
3. \textbf{Curvature:} For this space, $R = 6/\alpha^2$, representing a universe with constant positive curvature.

\textbf{Intuition:}
Just as a sphere is a surface of constant distance from a center in 3D, de Sitter space is a "hyper-hyperboloid" in 4D spacetime. Its high symmetry makes it the simplest model of an expanding universe driven by empty-space energy (dark energy).
\end{solution}

\subsection{Quantum Field Theory}

\begin{exercise}
\label{physics:2024:6}
Consider the Lagrangian of the Yukawa theory between a real scalar field $\phi$ and a Dirac spinor field $\psi$
\begin{equation}
    \mathcal{L} = \frac{1}{2} \partial_\mu \phi \partial^\mu \phi - \frac{1}{2} m^2 \phi^2 + \bar{\psi} (i \partial\!\!\!/ - M) \psi - i g \bar{\psi} \gamma^5 \psi \phi.
\end{equation}

\begin{enumerate}[label=(\alph*)]
    \item Find all divergences in 1-loop self-energy graphs of the scalars $\phi$. What are the correct counter-terms to cancel these divergences?
    \item Find all divergences in 1-loop self-energy graphs of the fermions $\psi$. What are the correct counter-terms to cancel these divergences?
    \item Are there divergences which cannot be renormalized in this theory?
\end{enumerate}
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item \textbf{Loop Diagrams:} Self-energy $\Sigma(p)$.
    \item \textbf{Counter-terms:} Terms added to Lagrangian to cancel infinities.
\end{itemize}

\textbf{Steps:}

1. \textbf{Scalar Loop:} Fermion bubble gives quadratic divergence $\Lambda^2$ and log divergence. Requires mass counter-term $\delta m$ and kinetic $\delta Z$.
2. \textbf{Fermion Loop:} Scalar exchange gives log divergence. Requires $\delta M$ and wavefunction $\delta Z_2$.
3. \textbf{Power Counting:} The degree of divergence $D = 4 - E_B - \frac{3}{2}E_F$ shows only a finite number of types of infinity exist, so the theory is renormalizable.

\textbf{Intuition:}
Interacting particles are never truly alone—they are surrounded by a "cloud" of virtual particles. This cloud changes the particle's mass. Renormalization allows us to say "the mass we see is the bare mass plus the cloud effects," keeping the math finite and physically useful.
\end{solution}
