\section{2024 Mathematical Physics}

\subsection{Classical Mechanics}

\begin{exercise}
\label{physics:2024:1}
Particle in central potential $V(r) = \alpha r^k$. Analyze orbits and stability.
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item \textbf{Lagrangian in Polar Coordinates:} $L = \frac{1}{2}m(\dot{r}^2 + r^2\dot{\phi}^2) - V(r)$.
    \item \textbf{Effective Potential:} $V_{\text{eff}} = \frac{J^2}{2mr^2} + V(r)$, where $J$ is angular momentum.
    \item \textbf{Bertrand's Theorem:} Only $1/r$ and $r^2$ potentials have all bound orbits closed.
\end{itemize}

\textbf{Steps:}

1. \textbf{Conserved J:} Scaling $\dot{\phi} = J/mr^2$ reduces the problem to 1D radial motion with $V_{\text{eff}}$.
2. \textbf{Equilibrium:} Circular orbits at $V_{\text{eff}}'(r_0) = 0 \implies k\alpha r_0^{k+2} = J^2/m$.
3. \textbf{Stability:} Stable if $V_{\text{eff}}''(r_0) > 0$. Calculation gives $k\alpha r_0^{k-2}(k+2) > 0$, requiring $k > -2$.
4. \textbf{Closing Orbits:} The ratio of oscillation to orbital frequency is $\sqrt{k+2}$. Closed orbits require this to be a rational number.

\textbf{Intuition:}
Imagine the particle "sliding" on a curve formed by the overlap of gravity (pulling in) and centrifugal force (pushing out). For a stable circular orbit, the particle must "rest" at the bottom of the resulting valley. If $k$ is too negative (steeper than $1/r^2$), the valley disappears, and the particle either crashes or flies away.
\end{solution}

\begin{exercise}
\label{physics:2024:2}
Driven oscillator with transient force $F(t)x$.
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item \textbf{Heisenberg Equation:} $\dot{A} = \frac{1}{i\hbar}[A, H]$.
    \item \textbf{Coherent State States:} Eigenstates of the annihilation operator.
\end{itemize}

\textbf{Steps:}

1. \textbf{Operator Evolution:} $\dot{\hat{a}} = -i\omega\hat{a} - \frac{iF(t)}{\sqrt{2m\hbar\omega}}$.
2. \textbf{Integration:} Solving for $t \to \infty$ yields $\hat{a}(\infty) = e^{-i\omega t}(\hat{a}(-\infty) - i\beta)$ where $\beta \propto \text{Fourier transform of } F(t)$.
3. \textbf{Transition Probabilities:} The ground state evolves into a coherent state $|\alpha\rangle$. The probability for state $n$ is a Poisson distribution $P_n = \frac{|\beta|^{2n}}{n!}e^{-|\beta|^2}$.

\textbf{Intuition:}
A linear force simply "pulls" the center of the oscillator potential. If the force is applied and then removed, the ground state doesn't stay still but "bounces"—this quantum bouncing state is what we call a coherent state, which mimics the motion of a classical particle.
\end{solution}

\subsection{Electromagnetism}

\begin{exercise}
\label{physics:2024:3}
Skin effect in a conductor with $\sigma$.
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item \textbf{Ohm's Law:} $\mathbf{J} = \sigma \mathbf{E}$.
    \item \textbf{Maxwell's Curl Equations:} $\nabla \times \mathbf{E} = - \dot{\mathbf{B}}/c, \nabla \times \mathbf{H} = 4\pi\sigma\mathbf{E}/c$.
\end{itemize}

\textbf{Steps:}

1. \textbf{Wave/Diffusion Equation:} $\nabla^2 \mathbf{E} = \frac{4\pi\sigma}{c^2}\dot{\mathbf{E}}$. In metals, conductivity dominates over displacement current.
2. \textbf{Skin Depth:} For frequency $\omega$, the field penetrates with a complex wave vector $k = (1+i)/\delta$, where $\delta = c/\sqrt{2\pi\sigma\omega}$.
3. \textbf{Surface Impedance:} $Z = E/H \propto \sqrt{\omega/\sigma}$. Large $\sigma$ implies $Z \to 0$, leading to near total reflection.

\textbf{Intuition:}
When light hits a metal, the electrons try to cancel the field within. They succeed almost everywhere, except for a thin "skin" at the surface. The faster the field oscillates or the more mobile the electrons, the thinner this skin becomes because the metal responds faster and more violently.
\end{solution}

\begin{exercise}
\label{physics:2024:4}
Ising model with infinite-range (mean-field) interaction.
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item \textbf{Mean-Field Approximation:} Replace local interactions with an average field.
    \item \textbf{Free Energy:} $f = u - Ts$.
\end{itemize}

\textbf{Steps:}

1. \textbf{Hamiltonian:} $H \approx -M(Jm+h) + \frac{1}{2}NJm^2$, where $m$ is the average magnetization.
2. \textbf{Self-Consistency:} Minimizing free energy gives $m = \tanh(\beta J m + \beta h)$.
3. \textbf{Transition:} For $T < J/k_B$, non-zero magnetization solutions appear even if $h=0$. This is the Curie temperature $T_c$.
4. \textbf{Exponents:} Expansion near $T_c$ shows $m \propto (T_c-T)^{1/2}$ and $\chi \propto |T-T_c|^{-1}$, giving $\beta_c=1/2, \gamma_c=1$.

\textbf{Intuition:}
In this model, everyone is "connected" to everyone else. If a small group of people decides to point "up," their combined influence forces everyone else to join them. This collective pressure is what leads to a phase transition, turning a random mess into a polarized state as you cool it down.
\end{solution}

\subsection{General Relativity}

\begin{exercise}
\label{physics:2024:5}
de Sitter space embedding $-(x^0)^2 + (x^1)^2 + \sum (x^i)^2 = \alpha^2$.
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item \textbf{Hypersurface Induced Metric:} $ds^2 = \eta_{AB} dx^A dx^B$.
    \item \textbf{Einstein Manifold:} $R_{\mu\nu} = \Lambda g_{\mu\nu}$.
\end{itemize}

\textbf{Steps:}

1. \textbf{Substitution:} Using $x^0 = \sqrt{\alpha^2-r^2}\sinh(t/\alpha)$ and $x^1 = \sqrt{\alpha^2-r^2}\cosh(t/\alpha)$ satisfies the constraint identically.
2. \textbf{Metric Derivation:} Differentials $dx^0, dx^1$ yield $ds^2 = -(1-r^2/\alpha^2)dt^2 + (1-r^2/\alpha^2)^{-1}dr^2 + r^2 d\Omega^2$.
3. \textbf{Curvature:} For this space, $R = 6/\alpha^2$, representing a universe with constant positive curvature.

\textbf{Intuition:}
Just as a sphere is a surface of constant distance from a center in 3D, de Sitter space is a "hyper-hyperboloid" in 4D spacetime. Its high symmetry makes it the simplest model of an expanding universe driven by empty-space energy (dark energy).
\end{solution}

\begin{exercise}
\label{physics:2024:6}
Yukawa theory renormalization.
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item \textbf{Loop Diagrams:} Self-energy $\Sigma(p)$.
    \item \textbf{Counter-terms:} Terms added to Lagrangian to cancel infinities.
\end{itemize}

\textbf{Steps:}

1. \textbf{Scalar Loop:} Fermion bubble gives quadratic divergence $\Lambda^2$ and log divergence. Requires mass counter-term $\delta m$ and kinetic $\delta Z$.
2. \textbf{Fermion Loop:} Scalar exchange gives log divergence. Requires $\delta M$ and wavefunction $\delta Z_2$.
3. \textbf{Power Counting:} The degree of divergence $D = 4 - E_B - \frac{3}{2}E_F$ shows only a finite number of types of infinity exist, so the theory is renormalizable.

\textbf{Intuition:}
Interacting particles are never truly alone—they are surrounded by a "cloud" of virtual particles. This cloud changes the particle's mass. Renormalization allows us to say "the mass we see is the bare mass plus the cloud effects," keeping the math finite and physically useful.
\end{solution}
