\section{2017 Analysis \& Differential Equations}

\subsection{Individual}

\begin{exercise}
Suppose $f$ is integrable on $[-\pi,\pi]$. Prove that 
\[
\sum_{n=-\infty}^{\infty} a_n r^{|n|}e^{inx}
\]
tends to $f(x)$ for a.e. $x$, as $r \to 1$, $r < 1$. Here $a_n = \frac{1}{2\pi}\int_{-\pi}^{\pi}f(x)e^{-inx}\,dx$.
\end{exercise}

\begin{solution}
The sum $S_r(x) = \sum_{n=-\infty}^{\infty} a_n r^{|n|}e^{inx}$ is the Abel sum of the Fourier series of $f$.
$S_r(x) = \frac{1}{2\pi} \int_{-\pi}^{\pi} f(t) \left( \sum_{n=-\infty}^{\infty} r^{|n|} e^{in(x-t)} \right) dt = (f * P_r)(x)$, where $P_r(\theta) = \frac{1-r^2}{1-2r\cos\theta + r^2}$ is the Poisson kernel.
The Poisson kernel is an approximate identity as $r \to 1^-$.
For any $f \in L^1([-\pi, \pi])$, the convolution $f * P_r$ converges to $f(x)$ at every Lebesgue point of $f$.
Since almost every point is a Lebesgue point for an integrable function, $S_r(x) \to f(x)$ for a.e. $x$.
\end{solution}

\begin{exercise}
Let $H$ be a Hilbert space equipped with an inner product $(\cdot,\cdot)$ and a norm $\|\cdot\| = (\cdot,\cdot)^{1/2}$. A sequence $\{f_k\}$ converges to $f \in H$ if $\|f_k-f\| \to 0$. A sequence $\{f_k\} \subset H$ is said to converge weakly to $f \in H$ if $(f_k,g) \to (f,g)$ for any $g \in H$. Prove the following statements:
\begin{enumerate}
    \item $\{f_k\}$ converges to $f$ if and only if $\|f_k\| \to \|f\|$ and $\{f_k\}$ converges weakly to $f$.
    \item If $H$ is a finite dimensional Hilbert space, then the weak convergence implies convergence. Give a counterexample to show that weak convergence does not necessarily imply convergence in an infinite dimensional Hilbert space.
\end{enumerate}
\end{exercise}

\begin{solution}
a) $(\implies)$ If $\|f_k - f\| \to 0$, then $|\|f_k\| - \|f\|| \leq \|f_k - f\| \to 0$, so $\|f_k\| \to \|f\|$.
Also $|(f_k, g) - (f, g)| = |(f_k - f, g)| \leq \|f_k - f\| \|g\| \to 0$, so $f_k \rightharpoonup f$.
$(\impliedby)$ Suppose $\|f_k\| \to \|f\|$ and $f_k \rightharpoonup f$.
$\|f_k - f\|^2 = (f_k-f, f_k-f) = \|f_k\|^2 - (f_k, f) - (f, f_k) + \|f\|^2$.
As $k \to \infty$, $\|f_k\|^2 \to \|f\|^2$, $(f_k, f) \to \|f\|^2$, and $(f, f_k) \to \|f\|^2$.
Thus $\|f_k - f\|^2 \to \|f\|^2 - \|f\|^2 - \|f\|^2 + \|f\|^2 = 0$.

b) In finite dimensions, weak convergence $(f_k, e_j) \to (f, e_j)$ for an orthonormal basis implies convergence of coordinates, hence convergence in norm.
Counterexample: Let $H = \ell^2$ and $f_k = e_k$ (the standard basis).
For any $g \in \ell^2$, $(e_k, g) = g_k \to 0$, so $e_k \rightharpoonup 0$.
But $\|e_k\| = 1$, so it does not converge to 0 in norm.
\end{solution}

\begin{exercise}
Let $f:\mathbb{C}\setminus\{0\}\to\mathbb{C}$ be a holomorphic function and
\[
|f(z)| \leq |z|^2 + \frac{1}{|z|^{1/2}},
\]
for $z$ near 0. Determine all such functions.
\end{exercise}

\begin{solution}
Consider $g(z) = z f(z)$. The condition implies $|g(z)| \leq |z|^3 + |z|^{1/2}$.
As $z \to 0$, $|g(z)| \to 0$.
By Riemann's Theorem on Removable Singularities, $g$ has a removable singularity at $z=0$, and $g(0) = 0$.
Since $g$ is holomorphic at 0 and $g(0)=0$, $g(z) = z h(z)$ where $h$ is holomorphic at 0.
Thus $f(z) = h(z)$ has a removable singularity at 0.
This means $f$ is the restriction of an entire function.
The condition "near 0" doesn't further restrict $f$ because $|f(z)|$ is bounded near 0 for any entire function, while $|z|^2 + |z|^{-1/2} \to \infty$.
Thus, all such functions are entire functions restricted to $\mathbb{C} \setminus \{0\}$.
Wait, if the problem implies it is ONLY defined on $\mathbb{C} \setminus \{0\}$, then the answer is "any holomorphic function on $\mathbb{C} \setminus \{0\}$ that has a removable singularity at 0".
\end{solution}

\begin{exercise}
Find a conformal mapping which maps the region $\{z\mid |z-i|<\sqrt{2}, |z+i|<\sqrt{2}\}$ onto the unit disk.
\end{exercise}

\begin{solution}
The region is the intersection of two disks. Boundary circles are $x^2 + (y-1)^2 = 2$ and $x^2 + (y+1)^2 = 2$.
They intersect at $z = \pm 1$. The angle at the vertices is $\pi/2$.
Step 1: $w_1 = \frac{z-1}{z+1}$ maps $1 \to 0$ and $-1 \to \infty$.
The arcs map to rays. The region maps to a sector of angle $\pi/2$.
Step 2: $w_2 = w_1^2$? No, first rotate.
The image is the sector $3\pi/4 < \arg w < 5\pi/4$.
Rotate by $\pi$: $w_3 = e^{-i\pi} w_1 = \frac{1-z}{1+z}$. The sector is $|\arg w_3| < \pi/4$.
Step 3: $w_4 = w_3^2$ maps to the right half-plane.
Step 4: $w = \frac{w_4-1}{w_4+1}$ maps to the unit disk.
Substituting: $w = \frac{(\frac{1-z}{1+z})^2 - 1}{(\frac{1-z}{1+z})^2 + 1} = \frac{-4z}{2+2z^2} = \frac{-2z}{1+z^2}$.
\end{solution}

\begin{exercise}
If $E$ is a compact set in a region $\Omega$, prove that there exists a constant $M>0$, depending only on $E$ and $\Omega$, such that every positive harmonic function $u(z)$ in $\Omega$ satisfies $u(z_2)\leq Mu(z_1)$ for any two points $z_1,z_2\in E$.
\end{exercise}

\begin{solution}
This is Harnack's Inequality.
For a disk $B(a, R)$, positive harmonic functions satisfy $\frac{R-r}{R+r} u(a) \leq u(z) \leq \frac{R+r}{R-r} u(a)$.
This implies $u(z_1) \leq C u(z_2)$ for $z_1, z_2$ in a smaller disk.
Since $E$ is compact and $\Omega$ is a connected region, we can cover $E$ with finitely many such disks and connect any two points by a chain of disks.
The constant $M$ will be $C^N$, where $N$ is the number of disks in the chain.
\end{solution}

\begin{exercise}
\begin{enumerate}
    \item For any bounded domain $\Omega \subset \mathbb{R}^n$, there exists a smallest constant $C(\Omega)$, such that
    \[
    \int_{\Omega} |u|^2 dx \leq C(\Omega) \int_{\Omega} \sum_{i=1}^n \left| \frac{\partial u}{\partial x_i} \right|^2 dx
    \]
    for every function $u \in H_0^1(\Omega) = \overline{C_0^\infty(\Omega)} \subset H^1(\Omega)$, where $C_0^\infty(\Omega)$ is the space of smooth functions over $\Omega$ and vanishing on the boundary of $\Omega$. Here $H^1(\Omega)$ is the Banach space of functions $u \in L^2(\Omega), \nabla u \in L^2(\Omega)^n$ with the norm:
    \[
    \|u\|_{H^1(\Omega)}^2 = \|u\|_{L^2(\Omega)}^2 + \|\nabla u\|_{L^2(\Omega)^n}^2 = \int_{\Omega} \left( |u|^2 + \sum_{i=1}^n \left| \frac{\partial u}{\partial x_i} \right|^2 \right) dx.
    \]
    $H_0^1(\Omega)$ is the completion of $C_0^\infty(\Omega)$ in $H^1(\Omega)$ with the above norm.
    \item Let $\Pi = \{(x, y) \mid 0 < x < a, 0 < y < b\}$, show that 
    \[
    C(\Pi) \geq \frac{a^2 b^2}{\pi^2(a^2 + b^2)}.
    \]
\end{enumerate}
\end{exercise}

\begin{solution}
1) This is the Poincaré inequality. The constant $C(\Omega)$ is $1/\lambda_1(\Omega)$, where $\lambda_1$ is the first eigenvalue of the Dirichlet Laplacian.
2) For the rectangle $\Pi$, the eigenvalues are $\lambda_{m,n} = \pi^2 (m^2/a^2 + n^2/b^2)$.
The smallest eigenvalue is $\lambda_{1,1} = \pi^2 (1/a^2 + 1/b^2) = \pi^2 \frac{a^2+b^2}{a^2b^2}$.
Thus $C(\Pi) = 1/\lambda_{1,1} = \frac{a^2b^2}{\pi^2(a^2+b^2)}$.
\end{solution}
