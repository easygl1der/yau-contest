\subsection{Individual}

\begin{exercise}
Suppose $f$ is integrable on $[-\pi,\pi]$. Prove that $\sum_{n=-\infty}^{\infty} a_n r^{|n|}e^{inx}$ tends to $f(x)$ for a.e. $x$, as $r\to 1$, $r<1$. Here $a_n = \frac{1}{2\pi}\int_{-\pi}^{\pi}f(x)e^{-inx}\,dx$.
\end{exercise}

\begin{solution}
The sum $S_r(x) = \sum_{n=-\infty}^{\infty} a_n r^{|n|}e^{inx}$ is the Abel sum of the Fourier series of $f$. Substituting the definition of $a_n$, we have:
\[
S_r(x) = \sum_{n=-\infty}^{\infty} \left( \frac{1}{2\pi} \int_{-\pi}^{\pi} f(t) e^{-int} \, dt \right) r^{|n|} e^{inx} = \frac{1}{2\pi} \int_{-\pi}^{\pi} f(t) \left( \sum_{n=-\infty}^{\infty} r^{|n|} e^{in(x-t)} \right) \, dt.
\]
The inner sum is the Poisson kernel $P_r(x-t)$:
\[
P_r(\theta) = \sum_{n=-\infty}^{\infty} r^{|n|} e^{in\theta} = 1 + \sum_{n=1}^{\infty} r^n (e^{in\theta} + e^{-in\theta}) = 1 + \frac{2r\cos\theta - 2r^2}{1 - 2r\cos\theta + r^2} = \frac{1-r^2}{1-2r\cos\theta + r^2}.
\]
Thus $S_r(x) = (f * P_r)(x)$. The Poisson kernel is a family of "good kernels" (or approximate identity) as $r \to 1^-$. Specifically, it satisfies:
\begin{enumerate}
    \item $\frac{1}{2\pi} \int_{-\pi}^{\pi} P_r(\theta) \, d\theta = 1$.
    \item $P_r(\theta) > 0$.
    \item For any $\delta > 0$, $\int_{\delta \leq |\theta| \leq \pi} P_r(\theta) \, d\theta \to 0$ as $r \to 1^-$.
\end{enumerate}
A standard result in harmonic analysis states that for $f \in L^1([-\pi, \pi])$, the convolution of $f$ with an approximate identity converges to $f(x)$ at every Lebesgue point of $f$. Since almost every $x \in [-\pi, \pi]$ is a Lebesgue point for an integrable function, $S_r(x) \to f(x)$ for a.e. $x$ as $r \to 1^-$.
\end{solution}

\begin{exercise}
Let $H$ be a Hilbert space equipped with an inner product $(.,.)$ and a norm $\|.\| = (.,.)^{\frac{1}{2}}$. A sequence $\{f_k\}$ converges to $f\in H$ if $\|f_k-f\|\to 0$. A sequence $\{f_k\}\subset H$ is said to converge weakly to $f\in H$ if $(f_k,g)\to(f,g)$ for any $g\in H$. Prove the following statements:

\begin{itemize}
\item[a)] $\{f_k\}$ converges to $f$ if and only if $\|f_k\|\to\|f\|$ and $\{f_k\}$ converges weakly to $f$.
\item[b)] If $H$ is a finite dimensional Hilbert space, then the weak convergence implies convergence. Give a counterexample to show that weak convergence does not necessarily imply convergence in an infinite dimensional Hilbert space.
\end{itemize}
\end{exercise}

\begin{solution}
a) Forward direction ($\implies$): Suppose $\|f_k - f\| \to 0$. By the reverse triangle inequality, $|\|f_k\| - \|f\|| \leq \|f_k - f\| \to 0$, so $\|f_k\| \to \|f\|$. For any $g \in H$, by the Cauchy-Schwarz inequality, $|(f_k, g) - (f, g)| = |(f_k - f, g)| \leq \|f_k - f\| \|g\| \to 0$, so $f_k \rightharpoonup f$.

Backward direction ($\impliedby$): Suppose $\|f_k\| \to \|f\|$ and $f_k \rightharpoonup f$. Consider the expansion:
\[
\|f_k - f\|^2 = (f_k - f, f_k - f) = \|f_k\|^2 - (f_k, f) - (f, f_k) + \|f\|^2.
\]
Since $f_k \rightharpoonup f$, we have $(f_k, f) \to (f, f) = \|f\|^2$. Also $(f, f_k) = \overline{(f_k, f)} \to \overline{(f, f)} = \|f\|^2$. Substituting these limits:
\[
\lim_{k \to \infty} \|f_k - f\|^2 = \|f\|^2 - \|f\|^2 - \|f\|^2 + \|f\|^2 = 0.
\]
Thus $\{f_k\}$ converges to $f$.

b) Finite dimensional case: Let $\dim H = n$ and let $\{e_1, \dots, e_n\}$ be an orthonormal basis. If $f_k \rightharpoonup f$, then for each $j \in \{1, \dots, n\}$, $(f_k, e_j) \to (f, e_j)$. We can write $f_k = \sum_{j=1}^n (f_k, e_j) e_j$ and $f = \sum_{j=1}^n (f, e_j) e_j$. Then:
\[
\|f_k - f\|^2 = \sum_{j=1}^n |(f_k, e_j) - (f, e_j)|^2 \to 0,
\]
as it is a finite sum of sequences tending to 0.

Infinite dimensional counterexample: Let $H = \ell^2(\mathbb{N})$ and let $\{e_k\}$ be the standard orthonormal basis. For any $g = (g_1, g_2, \dots) \in \ell^2$, we have $(e_k, g) = g_k$. Since $\sum |g_j|^2 < \infty$, it must be that $g_k \to 0$ as $k \to \infty$. Thus $e_k \rightharpoonup 0$. However, $\|e_k - 0\| = \|e_k\| = 1$ for all $k$, so $\{e_k\}$ does not converge to 0.
\end{solution}

\begin{exercise}
Let $f:\mathbb{C}\setminus\{0\}\to\mathbb{C}$ be a holomorphic function and
\[
|f(z)| \leq |z|^2 + \frac{1}{|z|^{1/2}},
\]
for $z$ near $0$. Determine all such functions.
\end{exercise}

\begin{solution}
Consider the function $g(z) = z f(z)$. The given inequality implies:
\[
|g(z)| = |z| |f(z)| \leq |z|^3 + |z|^{1/2}.
\]
As $z \to 0$, $|g(z)| \to 0$. By Riemann's Theorem on Removable Singularities, $g$ has a removable singularity at $z=0$, and we can define $g(0) = 0$. Since $g$ is holomorphic at 0 and $g(0)=0$, we can write $g(z) = z h(z)$ where $h$ is holomorphic at 0. This implies $f(z) = h(z)$ near $z=0$, so $f$ itself has a removable singularity at $z=0$. 

Thus, $f$ can be extended to a holomorphic function on the entire complex plane (an entire function). The condition "near 0" means there exists some $\delta > 0$ such that for $0 < |z| < \delta$, $|f(z)| \leq |z|^2 + |z|^{-1/2}$. Since $f$ is continuous at 0, this inequality is easily satisfied by any entire function because $|f(z)|$ is bounded in a neighborhood of 0, while $|z|^2 + |z|^{-1/2} \to \infty$ as $z \to 0$. 

If the problem implies that $f$ is defined on $\mathbb{C} \setminus \{0\}$ and the condition holds only near 0, then the set of such functions is exactly the set of all functions that are holomorphic on $\mathbb{C} \setminus \{0\}$ and have a removable singularity at 0 (i.e., they are restrictions of entire functions to $\mathbb{C} \setminus \{0\}$).
\end{solution}

\begin{exercise}
Find a conformal mapping which maps the region $\{z\mid |z-i|<\sqrt{2}, |z+i|<\sqrt{2}\}$ onto the unit disk.
\end{exercise}

\begin{solution}
The region $\Omega$ is the intersection of two disks. The boundary circles are $C_1: x^2 + (y-1)^2 = 2$ and $C_2: x^2 + (y+1)^2 = 2$. These circles intersect at points where $(y-1)^2 = (y+1)^2$, which gives $y=0$, and thus $x^2 + 1 = 2 \implies x = \pm 1$. The vertices of the region are $z = 1$ and $z = -1$.
The angle between the circles at the vertices can be found by looking at the tangents at $(1,0)$. For $C_1$, $x + (y-1)y' = 0 \implies 1 - y' = 0 \implies y' = 1$. For $C_2$, $x + (y+1)y' = 0 \implies 1 + y' = 0 \implies y' = -1$. The slopes are $1$ and $-1$, so the angle is $\pi/2$.

Step 1: Map the vertices $1, -1$ to $0, \infty$. Let $w_1 = \frac{z-1}{z+1}$. This maps the circles through $1, -1$ to rays from the origin. As calculated, the angle between the rays is $\pi/2$. 
Specifically, $w_1(0) = -1$. The image of the arcs are lines. For $C_1$, $|\frac{1+w}{1-w} - i|^2 = 2$ leads to $v=u$. For $C_2$, $|\frac{1+w}{1-w} + i|^2 = 2$ leads to $v=-u$. The region $\Omega$ contains $z=0$, so its image contains $w_1(0)=-1$. The image is the sector $S = \{w \mid 3\pi/4 < \arg w < 5\pi/4\}$.

Step 2: Rotate and power to map the sector to a half-plane. Let $w_2 = e^{-i\pi} w_1 = -w_1 = \frac{1-z}{z+1}$. The sector becomes $|\arg w_2| < \pi/4$. Then $w_3 = w_2^2 = (\frac{1-z}{z+1})^2$ maps this to the right half-plane $Re(w_3) > 0$.

Step 3: Map the right half-plane to the unit disk. $w = \frac{w_3-1}{w_3+1}$ maps $Re(w_3) > 0$ to $|w| < 1$.
Substituting $w_3$:
\[
w = \frac{(\frac{1-z}{1+z})^2 - 1}{(\frac{1-z}{1+z})^2 + 1} = \frac{(1-z)^2 - (1+z)^2}{(1-z)^2 + (1+z)^2} = \frac{-4z}{2z^2 + 2} = \frac{-2z}{z^2+1}.
\]
The mapping $f(z) = \frac{-2z}{z^2+1}$ (or $\frac{2z}{z^2+1}$) maps $\Omega$ onto the unit disk.
\end{solution}

\begin{exercise}
If $E$ is a compact set in a region $\Omega$, prove that there exists a constant $M>0$, depending only on $E$ and $\Omega$, such that every positive harmonic function $u(z)$ in $\Omega$ satisfies $u(z_2)\leq Mu(z_1)$ for any two points $z_1,z_2\in E$.
\end{exercise}

\begin{solution}
This is a global version of Harnack's Inequality.
1. Local Harnack Inequality: Let $u$ be a positive harmonic function on a disk $B(a, R)$. For any $z \in B(a, R)$ with $|z-a| = r < R$, we have:
\[
\frac{R-r}{R+r} u(a) \leq u(z) \leq \frac{R+r}{R-r} u(a).
\]
This implies that for any $z_1, z_2 \in B(a, R/2)$, $u(z_1) \leq 3 u(a) \leq 9 u(z_2)$. Thus $u(z_1) \leq C u(z_2)$ for some universal constant $C$.

2. Covering Argument: Since $E$ is compact and $\Omega$ is open, let $d = \text{dist}(E, \partial\Omega) > 0$. Let $r = d/4$. For each $z \in E$, the disk $B(z, 2r)$ is contained in $\Omega$. By the local inequality, for any $z' \in B(z, r)$, $u(z') \leq C u(z)$.
Since $E$ is compact, we can cover $E$ with finitely many such disks $B(z_i, r)$, $i=1, \dots, N$.
If $E$ is connected, for any $z_1, z_2 \in E$, we can find a chain of overlapping disks from the covering that connects $z_1$ to $z_2$. Let the chain be $B_1, B_2, \dots, B_k$ where $k \leq N$. Then $u(z_2) \leq C^k u(z_1) \leq C^N u(z_1)$.
If $E$ is not connected, but $\Omega$ is a region (connected), we can connect any two points in $E$ by a path in $\Omega$. The path is compact, so we can cover it with disks in $\Omega$ and use the same chaining argument. The constant $M$ will depend on the number of disks needed to cover $E$ or the paths between its components, which depends only on the geometry of $E$ and $\Omega$.
\end{solution}

\begin{exercise}
\begin{itemize}
\item[1)] For any bounded domain $\Omega\subset\mathbb{R}^n$, there exists a smallest constant $C(\Omega)$, such that
\[
\int_{\Omega}|u|^2\,dx \leq C(\Omega)\int_{\Omega}\sum_{i=1}^n\left|\frac{\partial u}{\partial x_i}\right|^2\,dx
\]
for every function $u\in H_0^1(\Omega) = \overline{C_0^{\infty}(\Omega)}^{H^1(\Omega)}$, where $C_0^{\infty}(\Omega)$ is the space of smooth functions over $\Omega$ and vanishing on boundary of $\Omega$ and $H^1(\Omega)$ is the Banach space of functions $u\in L^2(\Omega)$, $\nabla u\in L^2(\Omega)^{\otimes n}$ with the norm:
\[
\|u\|_{H^1(\Omega)}^2 = \|u\|_{L^2(\Omega)}^2 + \|\nabla u\|_{L^2(\Omega)^{\otimes n}}^2 = \int_{\Omega}(|u|^2 + \sum_{i=1}^n|\frac{\partial u}{\partial x_i}|^2)\,dx.
\]

$H_0^1(\Omega)$ is the completion of $C^{\infty}(\Omega)$ in $H^1(\Omega)$ with the above norm.

\item[2)] Let $\Pi = \{(x,y)\mid 0<x<a, 0<y<b\}$. Show that
\[
C(\Pi) \geq \frac{a^2b^2}{\pi^2(a^2+b^2)}.
\]
\end{itemize}
\end{exercise}

\begin{solution}
1) This is the Poincaré inequality for $H_0^1(\Omega)$. The constant $C(\Omega)$ is related to the first eigenvalue $\lambda_1$ of the Laplacian operator $-\Delta$ with Dirichlet boundary conditions on $\Omega$. Specifically,
\[
\lambda_1(\Omega) = \inf_{u \in H_0^1(\Omega), u \neq 0} \frac{\int_{\Omega} |\nabla u|^2 \, dx}{\int_{\Omega} u^2 \, dx}.
\]
By definition, $\int_{\Omega} u^2 \leq \frac{1}{\lambda_1(\Omega)} \int_{\Omega} |\nabla u|^2$. Thus the smallest constant $C(\Omega) = 1/\lambda_1(\Omega)$. The existence of $\lambda_1 > 0$ for bounded domains is a standard result in the theory of elliptic operators.

2) For the rectangle $\Pi = (0, a) \times (0, b)$, we can explicitly find the eigenvalues of the Laplacian. The eigenfunctions for the Dirichlet problem $-\Delta u = \lambda u$ are:
\[
u_{m,n}(x,y) = \sin\left(\frac{m\pi x}{a}\right) \sin\left(\frac{n\pi y}{b}\right), \quad m,n = 1, 2, \dots
\]
The corresponding eigenvalues are:
\[
\lambda_{m,n} = \left(\frac{m\pi}{a}\right)^2 + \left(\frac{n\pi}{b}\right)^2 = \pi^2 \left( \frac{m^2}{a^2} + \frac{n^2}{b^2} \right).
\]
The smallest eigenvalue is $\lambda_{1,1}$:
\[
\lambda_1(\Pi) = \pi^2 \left( \frac{1}{a^2} + \frac{1}{b^2} \right) = \pi^2 \frac{a^2 + b^2}{a^2b^2}.
\]
The smallest constant is therefore:
\[
C(\Pi) = \frac{1}{\lambda_1(\Pi)} = \frac{a^2b^2}{\pi^2(a^2+b^2)}.
\]
Since $C(\Pi)$ is the smallest constant such that the inequality holds, it is exactly equal to this value, and thus $C(\Pi) \geq \frac{a^2b^2}{\pi^2(a^2+b^2)}$ is satisfied.
\end{solution}
