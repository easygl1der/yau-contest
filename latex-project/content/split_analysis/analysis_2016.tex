\section{2016 Analysis \& Differential Equations}

\subsection{Individual}

\begin{exercise}
Suppose that $F$ is continuous on $[a,b]$, $F'(x)$ exists for every $x\in(a,b)$, and $F'(x)$ is integrable. Prove that $F$ is absolutely continuous and
\[
F(b) - F(a) = \int_a^b F'(x)\,dx.
\]
\end{exercise}

\begin{solution}
This is a standard result in real analysis (Lebesgue's fundamental theorem of calculus for functions with derivatives existing everywhere).
Let $G(x) = \int_a^x F'(t) dt$. Since $F'$ is integrable, $G$ is absolutely continuous and $G'(x) = F'(x)$ almost everywhere.
Let $H(x) = F(x) - G(x)$. $H$ is continuous on $[a,b]$.
Since $F'(x)$ exists for every $x \in (a,b)$, $F$ has the Luzin N property (maps measure zero sets to measure zero sets). Since $G$ is absolutely continuous, it also has the Luzin N property. Thus $H$ has the Luzin N property.
$H'(x) = F'(x) - G'(x) = 0$ for almost every $x$.
A continuous function with the Luzin N property and zero derivative almost everywhere is constant.
Thus $H(x) = H(a) = F(a) - 0 = F(a)$.
So $F(x) = F(a) + \int_a^x F'(t) dt$, which implies $F$ is absolutely continuous.
\end{solution}

\begin{exercise}
Suppose that $f$ is integrable on $\mathbb{R}^n$, let $K_{\delta}(x) = \delta^{-\frac{n}{2}}e^{-\frac{\pi|x|^2}{\delta}}$ for each $\delta>0$. Prove that the convolution
\[
(f*K_{\delta})(x) = \int_{\mathbb{R}^n} f(x-y)K_{\delta}(y)\,dy
\]
is integrable and $\| (f*K_{\delta}) - f\|_{L^1(\mathbb{R}^n)} \to 0$ as $\delta\to 0$.
\end{exercise}

\begin{solution}
1) \textbf{Integrability:} By Young's Inequality for convolutions, $\|f * K_\delta\|_1 \leq \|f\|_1 \|K_\delta\|_1$.
Since $K_\delta$ is a normalized Gaussian, $\|K_\delta\|_1 = \int \delta^{-n/2} e^{-\pi |x|^2/\delta} dx = 1$ (by substitution $u = x/\sqrt{\delta}$).
Thus $\|f * K_\delta\|_1 \leq \|f\|_1$, so it is integrable.

2) \textbf{$L^1$ convergence:}
\[
\| f*K_\delta - f \|_1 = \int_{\mathbb{R}^n} \left| \int_{\mathbb{R}^n} (f(x-y)-f(x)) K_\delta(y) dy \right| dx \leq \int_{\mathbb{R}^n} K_\delta(y) \left( \int_{\mathbb{R}^n} |f(x-y) - f(x)| dx \right) dy.
\]
Let $\omega_f(y) = \|f(\cdot-y) - f(\cdot)\|_1$. We know $\omega_f(y) \to 0$ as $y \to 0$ for any $f \in L^1$.
Also $\omega_f(y) \leq 2\|f\|_1$.
For any $\epsilon > 0$, choose $\eta > 0$ such that $|y| < \eta \implies \omega_f(y) < \epsilon$.
Then the integral is $\leq \int_{|y|<\eta} K_\delta(y) \epsilon dy + \int_{|y|\geq\eta} K_\delta(y) 2\|f\|_1 dy$.
The first part is $\leq \epsilon \cdot 1 = \epsilon$.
The second part is $2\|f\|_1 \int_{|y|\geq\eta} K_\delta(y) dy$.
As $\delta \to 0$, $K_\delta$ concentrates at the origin, so $\int_{|y|\geq\eta} K_\delta(y) dy \to 0$.
Thus $\limsup_{\delta\to 0} \|f*K_\delta - f\|_1 \leq \epsilon$. Since $\epsilon$ is arbitrary, the limit is 0.
\end{solution}

\begin{exercise}
Prove that a bounded function on interval $I=[a,b]$ is Riemann integrable if and only if its set of discontinuities has measure zero.
\end{exercise}

\begin{solution}
This is Lebesgue's Criterion for Riemann Integrability.
Let $D$ be the set of discontinuities of $f$. Let $D_\epsilon = \{x \in I \mid \text{osc}(f, x) \geq \epsilon\}$.
Then $D = \bigcup_{n=1}^\infty D_{1/n}$.
1) \textbf{If $\mu(D) = 0$:}
Then $\mu(D_\epsilon) = 0$ for all $\epsilon$. $D_\epsilon$ is compact.
For any $\eta > 0$, cover $D_\epsilon$ with open intervals of total length $< \eta$.
Outside these intervals, $\text{osc}(f, x) < \epsilon$.
This allows construction of a partition where the Darboux sum difference $U(P,f) - L(P,f)$ is small.
Specifically, $U-L \leq \epsilon(b-a) + (M-m)\eta$.
Taking $\epsilon, \eta \to 0$, we get $U-L \to 0$.

2) \textbf{If $f$ is Riemann integrable:}
For any partition $P$, $U(P,f) - L(P,f) = \sum (M_i-m_i) \Delta x_i$.
If $x \in D_\epsilon$ is in the interior of a subinterval, then $M_i-m_i \geq \epsilon$.
One can show that the measure of $D_\epsilon$ must be zero, otherwise $U-L$ would be bounded away from 0.
Thus $\mu(D_\epsilon) = 0$ for all $\epsilon$, so $\mu(D) = 0$.
\end{solution}

\begin{exercise}
\begin{itemize}
    \item[1)] Let $w=\frac{1}{2}(z+\frac{1}{z})$. Show that it maps $\{|z|>1\}$ to $\mathbb{C}\setminus[-1,1]$.
    \item[2)] Compute $\int_0^\infty \frac{\log x}{x^2-1} dx$.
\end{itemize}
\end{exercise}

\begin{solution}
1) Let $z = re^{i\theta}$ with $r > 1$.
$w = \frac{1}{2}(r e^{i\theta} + \frac{1}{r} e^{-i\theta}) = \frac{1}{2}(r+1/r)\cos\theta + i\frac{1}{2}(r-1/r)\sin\theta$.
This is an ellipse with semi-axes $a = \frac{1}{2}(r+1/r)$ and $b = \frac{1}{2}(r-1/r)$.
As $r$ ranges from 1 to $\infty$, these ellipses fill the entire plane.
For $r=1$, the ellipse collapses to the segment $[-1, 1]$.
Thus the image of $|z|>1$ is $\mathbb{C} \setminus [-1, 1]$.

2) $I = \int_0^\infty \frac{\log x}{x^2-1} dx = \int_0^1 \frac{\log x}{x^2-1} dx + \int_1^\infty \frac{\log x}{x^2-1} dx$.
In the second integral, let $u = 1/x$, so $dx = -1/u^2 du$.
$\int_1^\infty \frac{\log x}{x^2-1} dx = \int_1^0 \frac{\log(1/u)}{(1/u)^2-1} (-1/u^2) du = \int_0^1 \frac{\log u}{u^2-1} du$.
Thus $I = 2 \int_0^1 \frac{\log x}{x^2-1} dx = 2 \int_0^1 \log x \sum_{n=0}^\infty (x^2)^n dx$.
$I = 2 \sum_{n=0}^\infty \int_0^1 x^{2n} \log x dx$.
Using $\int_0^1 x^m \log x dx = -1/(m+1)^2$:
$I = 2 \sum_{n=0}^\infty \frac{1}{(2n+1)^2} = 2 (\sum_{k=1}^\infty \frac{1}{k^2} - \sum_{k=1}^\infty \frac{1}{(2k)^2}) = 2 (1 - 1/4) \zeta(2) = \frac{3}{2} \cdot \frac{\pi^2}{6} = \frac{\pi^2}{4}$.
\end{solution}

\begin{exercise}
Let $f$ be a doubly periodic meromorphic function. Prove that the number of zeros and poles are equal.
\end{exercise}

\begin{solution}
By the Argument Principle, the number of zeros minus the number of poles in the fundamental parallelogram $\Omega$ is:
$Z - P = \frac{1}{2\pi i} \int_{\partial \Omega} \frac{f'(z)}{f(z)} dz$.
Since $f$ is doubly periodic, $f'/f$ is also doubly periodic.
The integral over opposite sides of the parallelogram cancel out because the path is in opposite directions and the function values are the same.
$\int_{\text{side } 1} + \int_{\text{side } 3} = \int_{z_0}^{z_0+\omega_1} \frac{f'(z)}{f(z)} dz + \int_{z_0+\omega_1+\omega_2}^{z_0+\omega_2} \frac{f'(z)}{f(z)} dz = 0$.
Similarly for the other pair of sides.
Thus $Z - P = 0 \implies Z = P$.
\end{solution}

\begin{exercise}
Prove that the spectrum of a bounded self-adjoint operator is a bounded closed subset of the real line.
\end{exercise}

\begin{solution}
1) \textbf{Bounded and Closed:} The spectrum $\sigma(A)$ is always closed and contained in $\{|\lambda| \leq \|A\|\}$ for any bounded operator.
2) \textbf{Real Line:} Let $\lambda = a + ib$ with $b \neq 0$.
$\|(A-\lambda I)x\|^2 = \|(A-aI)x - ibx\|^2 = \|(A-aI)x\|^2 + b^2\|x\|^2$ (since $A-aI$ is self-adjoint, cross terms cancel).
Thus $\|(A-\lambda I)x\| \geq |b|\|x\|$.
This implies $A-\lambda I$ is injective and has a closed range.
Since $(A-\lambda I)^* = A - \bar{\lambda} I$ also satisfies the same estimate, its kernel is $\{0\}$.
So the range of $A-\lambda I$ is dense.
A dense closed range is the whole space, so $A-\lambda I$ is invertible.
Thus $\lambda \notin \sigma(A)$, so $\sigma(A) \subset \mathbb{R}$.
\end{solution}
