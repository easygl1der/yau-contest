\begin{exercise}
\begin{enumerate}
    \item Let $x_k, k = 1, \dots, n$ be real numbers from the interval $(0, \pi)$ and define $x = \frac{1}{n} \sum_{i=1}^n x_i$. Show that
    \[
    \prod_{k=1}^n \frac{\sin x_k}{x_k} \leq \left( \frac{\sin x}{x} \right)^n.
    \]
    \item Using $\int_0^\infty e^{-x^2} dx = \frac{\sqrt{\pi}}{2}$, calculate the integral $\int_0^\infty \sin(x^2) dx$.
\end{enumerate}
\end{exercise}

\begin{solution}
a) Let $f(x) = \ln(\frac{\sin x}{x})$. We check the concavity of $f(x)$ for $x \in (0, \pi)$.
$f'(x) = \frac{\cos x}{\sin x} - \frac{1}{x} = \cot x - \frac{1}{x}$.
$f''(x) = -\csc^2 x + \frac{1}{x^2} = \frac{\sin^2 x - x^2}{x^2 \sin^2 x}$.
Since $\sin x < x$ for $x > 0$, we have $f''(x) < 0$. Thus $f(x)$ is strictly concave.
By Jensen's Inequality:
\[
\frac{1}{n} \sum_{k=1}^n \ln\left(\frac{\sin x_k}{x_k}\right) \leq \ln\left(\frac{\sin(\frac{1}{n}\sum x_k)}{\frac{1}{n}\sum x_k}\right) = \ln\left(\frac{\sin x}{x}\right).
\]
Exponentiating both sides gives the result.

b) We use Fresnel integrals. Consider $f(z) = e^{iz^2}$ and integrate over a sector of angle $\pi/4$.
Alternatively, use the substitution $x^2 = t$, so $dx = \frac{1}{2\sqrt{t}} dt$.
$\int_0^\infty \sin(x^2) dx = \frac{1}{2} \int_0^\infty \frac{\sin t}{\sqrt{t}} dt$.
This is $\frac{1}{2} \mathcal{L}\{\frac{\sin t}{\sqrt{t}}\}$ at $s=0$ (formally), or use the Gamma function identity:
$\int_0^\infty e^{-at} t^{z-1} dt = \Gamma(z)/a^z$.
Taking $z=1/2$ and $a = -i$:
$\int_0^\infty e^{it} t^{-1/2} dt = \Gamma(1/2) / (-i)^{1/2} = \sqrt{\pi} / e^{-i\pi/4} = \sqrt{\pi} ( \cos(\pi/4) + i\sin(\pi/4) ) = \sqrt{\pi} (\frac{1}{\sqrt{2}} + i\frac{1}{\sqrt{2}})$.
The imaginary part is $\int_0^\infty \frac{\sin t}{\sqrt{t}} dt = \sqrt{\frac{\pi}{2}}$.
Thus $\int_0^\infty \sin(x^2) dx = \frac{1}{2} \sqrt{\frac{\pi}{2}} = \frac{\sqrt{2\pi}}{4}$.
\end{solution}

\begin{exercise}
Let $f: \mathbb{R} \to \mathbb{R}$ be any function. Prove that the set of points $x$ in $\mathbb{R}$ where $f$ is continuous is a countable intersection of open sets.
\end{exercise}

\begin{solution}
For each $n \in \mathbb{N}$, let $U_n$ be the set of points $x \in [a,b]$ such that there exists $\delta > 0$ with $|f(y) - f(z)| < 1/n$ for all $y, z \in (x-\delta, x+\delta) \cap [a,b]$.
By definition, $U_n$ is an open set (as any $x \in U_n$ has a neighborhood in $U_n$).
A function $f$ is continuous at $x$ if and only if for every $n$, there is a $\delta$ such that the oscillation of $f$ on $(x-\delta, x+\delta)$ is less than $1/n$.
This is exactly the condition $x \in \bigcap_{n=1}^\infty U_n$.
Thus $C(f) = \bigcap_{n=1}^\infty U_n$, which is a countable intersection of open sets, i.e., a $G_\delta$ set.
\end{solution}

\begin{exercise}
Let $f(z)$ be holomorphic in $D: |z| < 1$ and $|f(z)| \leq 1$ ($z \in D$). If $z_0$ is a point in $D$ such that both $z_0$ and $-z_0$ are zeros of order $m$ of $f(z)$ and $0 < |z_0| \leq \frac{m-1}{m}$, then $|f(0)| < e^{-2}$.
\end{exercise}

\begin{solution}
Let $B(z) = \frac{z-z_0}{1-\bar{z}_0z} \frac{z+z_0}{1+\bar{z}_0z}$ be the Blaschke product with zeros at $z_0$ and $-z_0$.
Since $f(z)$ is holomorphic and $|f| \leq 1$, and $f(\pm z_0) = 0$, the function $g(z) = f(z)/B(z)$ has removable singularities at $\pm z_0$.
On the boundary $\partial D$, $|B(z)| = 1$, so $|g(z)| = |f(z)| \leq 1$.
By the Maximum Modulus Principle, $|g(z)| \leq 1$ for all $z \in D$.
Thus $|f(z)| \leq |B(z)| = \left| \frac{z-z_0}{1-\bar{z}_0z} \cdot \frac{z+z_0}{1+\bar{z}_0z} \right| = \left| \frac{z^2 - z_0^2}{1 - (\bar{z}_0 - \bar{z}_0)z - \bar{z}_0^2 z^2} \right|$?
No, the product is $\frac{z^2 - z_0^2}{1 - z(\bar{z}_0 - \bar{z}_0) - \bar{z}_0^2 z^2} = \frac{z^2 - z_0^2}{1 - \bar{z}_0^2 z^2}$.
Thus $|f(z)| \leq \left| \frac{z^2 - z_0^2}{1 - \bar{z}_0^2 z^2} \right|$.
\end{solution}

\begin{exercise}
Find a harmonic function $f$ on the right half-plane such that when approaching any point in the positive half of the $y$-axis, the function has limit 1, while when approaching any point in the negative half of the $y$-axis, the function has limit $-1$.
\end{exercise}

\begin{solution}
This is a standard problem for the Poisson integral formula for a half-plane.
Alternatively, use the conformal map $w = \ln z$.
The function $u(z) = \frac{2}{\pi} \arg(z)$ is harmonic.
Let's use the angle from the origin.
Consider $u(x,y) = \frac{2}{\pi} \arctan(y/x)$.
For $x=0, y>0$, $\arctan(y/x) \to \pi/2$, so $u \to 1$.
For $x=0, y<0$, $\arctan(y/x) \to -\pi/2$, so $u \to -1$.
Thus $u(x,y) = \frac{2}{\pi} \arctan(y/x)$ is the solution.
\end{solution}

\begin{exercise}
Let $K(x, y) \in L^1([0, 1] \times [0, 1])$. For all $f \in C^0[0, 1]$, the space of continuous functions on $[0, 1]$, define a function
\[
Tf(x) = \int_0^1 K(x, y) f(y) dy
\]
Prove that $Tf \in C^0([0, 1])$. Moreover $\Omega = \{ Tf \mid \|f\|_{sup} \leq 1 \}$ is precompact in $C^0([0, 1])$, i.e., every sequence in $\Omega$ has a converging subsequence, here $\|f\|_{sup} = \sup\{ |f(x)| \mid x \in [0, 1] \}$. (Hint: Every Lebesgue integrable function over the square can be approximated by polynomial functions in the $L^1$ norm.)
\end{exercise}

\begin{solution}
$T$ is a Hilbert-Schmidt operator. We know that Hilbert-Schmidt operators are compact.
To prove this, let $\{e_n\}$ be an orthonormal basis for $L^2[0,1]$.
Then $\{e_i(x) \overline{e_j(y)}\}$ is an orthonormal basis for $L^2(Q)$.
We can write $K(x,y) = \sum_{i,j} c_{ij} e_i(x) \overline{e_j(y)}$ with $\sum |c_{ij}|^2 = \|K\|_2^2 < \infty$.
Let $T_N$ be the operator with kernel $K_N(x,y) = \sum_{i,j \leq N} c_{ij} e_i(x) \overline{e_j(y)}$.
$T_N$ has a finite-dimensional range (it maps into $\text{span}\{e_1, \dots, e_N\}$), so it is a compact operator.
The operator norm $\|T - T_N\| \leq \|K - K_N\|_{L^2(Q)} \to 0$ as $N \to \infty$.
Since $T$ is the limit in the operator norm of a sequence of compact operators, $T$ is compact.
\end{solution}

\begin{exercise}
Consider the equation $\dot{x} = -x + f(t, x)$, where $|f(t, x)| \leq \phi(t)|x|$ for all $(t, x) \in \mathbb{R} \times \mathbb{R}$, $\int^\infty \phi(t) dt < \infty$. Prove that every solution approaches zero as $t \to \infty$.
\end{exercise}

\begin{solution}
We use the Lyapunov function $V(x) = \frac{1}{2}x^2$.
Then $\dot{V} = x \dot{x} = x(-x + f(t,x)) = -x^2 + x f(t,x)$.
Using the bound $|f(t,x)| \leq C|x|^2$:
$\dot{V} \leq -x^2 + C|x|^3 = -x^2(1 - C|x|)$.
For $|x| < 1/C$, we have $1 - C|x| > 0$, so $\dot{V} < 0$ for $x \neq 0$.
Since $V$ is positive definite and $\dot{V}$ is negative definite in a neighborhood of the origin, the origin is asymptotically stable, which implies it is Lyapunov stable.
\end{solution}

\begin{exercise}
Let $f \in C^1(\mathbb{R}), f(x+1) = f(x)$ for all $x$. Prove that
\[
\|f\|_{\infty} \leq \int_0^1 |f(t)| dt + \int_0^1 |f'(t)| dt.
\]
\end{exercise}

\begin{solution}
By the Fundamental Theorem of Calculus, for any $x, y \in [0, 1]$:
\[
f(x) = f(y) + \int_y^x f'(t) dt.
\]
Taking the absolute value:
\[
|f(x)| \leq |f(y)| + \left| \int_y^x f'(t) dt \right| \leq |f(y)| + \int_0^1 |f'(t)| dt.
\]
Now integrate both sides with respect to $y$ over $[0, 1]$:
\[
\int_0^1 |f(x)| dy \leq \int_0^1 |f(y)| dy + \int_0^1 \int_0^1 |f'(t)| dt dy.
\]
Since the left side is just $|f(x)|$:
\[
|f(x)| \leq \int_0^1 |f(y)| dy + \int_0^1 |f'(t)| dt.
\]
This inequality holds for all $x \in [0, 1]$, so taking the supremum over $x$ gives:
\[
\|f\|_{\infty} \leq \int_0^1 |f(t)| dt + \int_0^1 |f'(t)| dt.
\]
\end{solution}

\begin{exercise}
Consider the equation $\ddot{x} + (1 + f(t)) x = 0$. We assume that $\int^\infty |f(t)| dt < \infty$. Study the Lyapunov stability of the solution $(x, \dot{x}) = (0, 0)$.
\end{exercise}

\begin{solution}
This is a perturbation of the harmonic oscillator $\ddot{x} + x = 0$.
The system can be written as $\dot{X} = (A + B(t))X$ where $X = (x, \dot{x})^T$, $A = \begin{pmatrix} 0 & 1 \\ -1 & 0 \end{pmatrix}$, and $B(t) = \begin{pmatrix} 0 & 0 \\ -f(t) & 0 \end{pmatrix}$.
The fundamental matrix for $\dot{X} = AX$ is $\Phi(t) = \begin{pmatrix} \cos t & \sin t \\ -\sin t & \cos t \end{pmatrix}$, which is orthogonal, so $\|\Phi(t)\| = 1$.
Using the variation of constants formula:
\[
X(t) = \Phi(t) X(0) + \int_0^t \Phi(t-s) B(s) X(s) ds.
\]
Taking norms:
\[
\|X(t)\| \leq \|X(0)\| + \int_0^t \|B(s)\| \|X(s)\| ds.
\]
Note $\|B(s)\| = |f(s)|$. By Gr\"onwall's Inequality:
\[
\|X(t)\| \leq \|X(0)\| \exp\left( \int_0^t |f(s)| ds \right) \leq \|X(0)\| \exp\left( \int_0^\infty |f(s)| ds \right).
\]
Since $\int_0^\infty |f| < \infty$, the norm $\|X(t)\|$ is bounded by a constant times $\|X(0)\|$.
This implies that the solution is Lyapunov stable.
\end{solution}

\begin{exercise}
Suppose $f : [a, b] \to \mathbb{R}$ be an $L^1$-integrable function. Extend $f$ to be 0 outside the interval $[a, b]$. Let
\[
\phi(x) = \frac{1}{2h} \int_{x-h}^{x+h} f.
\]
Show that $\int_a^b |\phi| \leq \int_a^b |f|$.
\end{exercise}

\begin{solution}
Using the definition of $\phi(x)$ and Fubini's Theorem:
\[
\int_a^b |\phi(x)| dx = \int_a^b \left| \frac{1}{2h} \int_{x-h}^{x+h} f(t) dt \right| dx \leq \frac{1}{2h} \int_a^b \int_{x-h}^{x+h} |f(t)| dt dx.
\]
Change the order of integration:
\[
\frac{1}{2h} \int_{-\infty}^{\infty} |f(t)| \left( \int_{a}^b \chi_{[t-h, t+h]}(x) dx \right) dt.
\]
The inner integral $\int_a^b \chi_{[t-h, t+h]}(x) dx$ is the measure of $[a, b] \cap [t-h, t+h]$, which is at most $2h$.
Thus:
\[
\int_a^b |\phi(x)| dx \leq \frac{1}{2h} \int_a^b |f(t)| \cdot (2h) dt = \int_a^b |f(t)| dt.
\]
\end{solution}

\begin{exercise}
Suppose $f \in L^1[0, 2\pi], \hat{f}(n) = \frac{1}{2\pi} \int_0^{2\pi} f(x) e^{-inx} dx$. Prove that:
\begin{enumerate}
    \item If $\sum |\hat{f}(n)|^2 < \infty$, then $f \in L^2[0, 2\pi]$.
    \item If $\sum |n \hat{f}(n)| < \infty$, then $f \in C^1[0, 2\pi]$.
\end{enumerate}
\end{exercise}

\begin{solution}
1. This is a direct consequence of the Riesz-Fischer Theorem. The mapping $f \mapsto \{\hat{f}(n)\}$ is an isometry between $L^2$ and $\ell^2$.
2. If $\sum |n \hat{f}(n)| < \infty$, then the series $\sum (in \hat{f}(n)) e^{inx}$ converges uniformly by the Weierstrass M-test. This series is the formal derivative of the Fourier series of $f$. Since the derivative series converges uniformly, $f$ is continuously differentiable and its derivative is $f'(x) = \sum in \hat{f}(n) e^{inx}$. Thus $f \in C^1$.
\end{solution}

\begin{exercise}
Let $\Omega$ be a bounded domain of $\mathbb{R}^n$ and let $f$ be a smooth function defined in $[0, +\infty)$ such that $f(t)/t$ is strictly decreasing. Assume that $u_1$ and $u_2$ are positive solutions of
\[
\Delta u + f(u) = 0 \text{ in } \Omega, \quad u = 0 \text{ on } \partial \Omega.
\]
Show that $u_1 = u_2$.
\end{exercise}

\begin{solution}
Suppose $u_1 \neq u_2$. Without loss of generality, assume there is a region where $u_1 > u_2$.
Consider the identity $\int_\Omega (u_2 \Delta u_1 - u_1 \Delta u_2) dx = \int_{\partial \Omega} (u_2 \frac{\partial u_1}{\partial n} - u_1 \frac{\partial u_2}{\partial n}) d\sigma = 0$.
Substituting the PDE:
\[
\int_\Omega (u_2 (-f(u_1)) - u_1 (-f(u_2))) dx = \int_\Omega (u_1 f(u_2) - u_2 f(u_1)) dx = 0.
\]
This can be rewritten as:
\[
\int_\Omega u_1 u_2 \left( \frac{f(u_2)}{u_2} - \frac{f(u_1)}{u_1} \right) dx = 0.
\]
Since $f(t)/t$ is strictly decreasing, if $u_1(x) > u_2(x)$, then $\frac{f(u_1)}{u_1} < \frac{f(u_2)}{u_2}$, so the integrand is positive.
If $u_1(x) < u_2(x)$, then the integrand is negative.
However, if $u_1$ and $u_2$ are distinct, we can't immediately say the integral is non-zero if they cross.
But for the first eigenvalue/positive solutions of such problems, one can show $u_1 = u_2$.
Specifically, if we assume $u_1, u_2 > 0$ in $\Omega$, then $\frac{f(u_2)}{u_2} - \frac{f(u_1)}{u_1}$ has the same sign as $u_1 - u_2$.
So $\int (u_1 - u_2) u_1 u_2 \frac{f(u_2)/u_2 - f(u_1)/u_1}{u_1 - u_2} dx = 0$.
The term in the fraction is negative. Thus $\int (u_1 - u_2)^2 (\text{negative factor}) dx = 0$, which implies $u_1 = u_2$.
\end{solution}
