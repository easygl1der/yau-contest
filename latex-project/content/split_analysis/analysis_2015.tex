\subsection{Individual}

\begin{exercise}
Let $f_n\in L^2(\mathbb{R})$ be a sequence of measurable functions over the line, $f_n\to f$ almost everywhere. Let $\|f_n\|_{L^2}\to\|f\|_{L^2}$. Prove that $\|f_n-f\|_{L^2}\to 0$.
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
1. \textbf{Fatou's Lemma:} For any sequence of non-negative measurable functions $g_n$, the integral of the limit inferior is less than or equal to the limit inferior of the integrals: $\int \liminf_{n\to\infty} g_n \leq \liminf_{n\to\infty} \int g_n$.
2. \textbf{Triangle Inequality:} $|a - b| \leq |a| + |b|$, which implies $|a - b|^2 \leq (|a| + |b|)^2 \leq 2|a|^2 + 2|b|^2$.
3. \textbf{$L^2$ Norm Definition:} $\|f\|_2^2 = \int |f|^2 dx$.

\textbf{Steps:}

1. \textbf{Construct a helper sequence:} Define $g_n(x) = 2(|f_n(x)|^2 + |f(x)|^2) - |f_n(x) - f(x)|^2$. 
By the triangle inequality mentioned in the prerequisites, we know that $|f_n - f|^2 \leq 2|f_n|^2 + 2|f|^2$, which directly implies that $g_n(x) \geq 0$ for all $x$ and all $n$. This non-negativity is crucial for applying Fatou's Lemma.

2. \textbf{Apply Fatou's Lemma:} Since $g_n \geq 0$, we have:
\[
\int \liminf_{n\to\infty} g_n \, dx \leq \liminf_{n\to\infty} \int g_n \, dx
\]
Because $f_n \to f$ almost everywhere (a.e.), the term $|f_n - f|^2$ goes to 0 a.e., and $|f_n|^2 \to |f|^2$ a.e. 
Therefore, $\liminf_{n\to\infty} g_n = 2(|f|^2 + |f|^2) - 0 = 4|f|^2$ a.e. The integral of the left side is simply $4\|f\|_2^2$.

3. \textbf{Analyze the right side:} The integral of $g_n$ can be expanded linearly:
\[
\int g_n \, dx = 2\int |f_n|^2 \, dx + 2\int |f|^2 \, dx - \int |f_n - f|^2 \, dx = 2\|f_n\|_2^2 + 2\|f\|_2^2 - \|f_n - f\|_2^2
\]
Taking the limit inferior, and using the fact that $\|f_n\|_2 \to \|f\|_2$:
\[
\liminf_{n\to\infty} \int g_n \, dx = 2\|f\|_2^2 + 2\|f\|_2^2 - \limsup_{n\to\infty} \|f_n - f\|_2^2 = 4\|f\|_2^2 - \limsup_{n\to\infty} \|f_n - f\|_2^2
\]

4. \textbf{Conclude:} Combining these, we get $4\|f\|_2^2 \leq 4\|f\|_2^2 - \limsup_{n\to\infty} \|f_n - f\|_2^2$.
Subtracting $4\|f\|_2^2$ from both sides gives $0 \leq -\limsup_{n\to\infty} \|f_n - f\|_2^2$, or $\limsup_{n\to\infty} \|f_n - f\|_2^2 \leq 0$.
Since the norm is always non-negative, the limit must be exactly 0.

\textbf{Intuition:}
Think of the $L^2$ norm as a measure of the "total energy" of a signal. Pointwise convergence tells us that the "shape" of $f_n$ is becoming identical to $f$. The convergence of norms tells us that the "total energy" is also being preserved. 

In general, pointwise convergence doesn't guarantee that the difference goes to zero (imagine a sequence of very tall, very thin spikes that move to infinity—they disappear pointwise but carry constant energy). However, if the energy of the sequence matches the energy of the limit, there's no "leaked energy" at infinity or in spikes, so the functions must converge in the norm sense.
\end{solution}

\begin{exercise}
Let $f$ be a continuous function on $[a,b]$, define $M_n = \int_a^b f(x)x^n\,dx$. Suppose that $M_n=0$ for all integers $n\geq 0$. Show that $f(x)=0$ for all $x$.
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
1. \textbf{Weierstrass Approximation Theorem:} Any continuous function on a closed interval $[a,b]$ can be uniformly approximated by polynomials. That is, for any $\epsilon > 0$, there exists a polynomial $P(x)$ such that $|f(x) - P(x)| < \epsilon$ for all $x \in [a,b]$.
2. \textbf{Linearity of Integration:} If an integral vanishes for $x^n$ for all $n$, it vanishes for any linear combination of $x^n$ (i.e., any polynomial).
3. \textbf{Property of Continuous Functions:} If $f$ is continuous and $\int f^2 = 0$, then $f$ must be identically zero.

\textbf{Steps:}

1. \textbf{Polynomial Orthogonality:} The given condition $M_n = 0$ means that $f$ is "orthogonal" to every power of $x$. By linearity, for any polynomial $P(x) = \sum c_k x^k$, we have $\int_a^b f(x) P(x) \, dx = \sum c_k \int_a^b f(x) x^k \, dx = 0$.

2. \textbf{Uniform Approximation:} Since $f$ is continuous, we can pick a polynomial $P_\epsilon(x)$ that is very close to $f$. Specifically, let $|f(x) - P_\epsilon(x)| < \epsilon$ for all $x \in [a,b]$.

3. \textbf{Calculate the Norm:} Consider the integral of $f^2$:
\[
\int_a^b f(x)^2 \, dx = \int_a^b f(x)(f(x) - P_\epsilon(x) + P_\epsilon(x)) \, dx = \int_a^b f(x)(f(x) - P_\epsilon(x)) \, dx + \int_a^b f(x)P_\epsilon(x) \, dx
\]
The second integral is zero because $P_\epsilon$ is a polynomial.

4. \textbf{Bound the Integral:} Using the Cauchy-Schwarz inequality or simple absolute value bounds:
\[
\int_a^b f(x)^2 \, dx \leq \int_a^b |f(x)| |f(x) - P_\epsilon(x)| \, dx \leq \int_a^b \|f\|_\infty \cdot \epsilon \, dx = \|f\|_\infty \cdot \epsilon \cdot (b-a)
\]
Since this holds for \textit{any} $\epsilon > 0$, the only possible value for the non-negative integral $\int f^2$ is zero.

5. \textbf{Final Conclusion:} A continuous non-negative function ($f^2$) with a zero integral must be zero everywhere. Thus $f(x) = 0$ for all $x \in [a,b]$.

\textbf{Intuition:}
This problem is about the "density" of polynomials. If you know that a function is perpendicular to every single polynomial, and you know that polynomials can "mimic" any continuous function perfectly, then the function must be perpendicular to "itself" (or rather, its mimicked version). 

In a Hilbert space language, the set $\{1, x, x^2, \dots\}$ spans a dense subspace of $C([a,b])$. If a vector is orthogonal to a dense subspace, it must be the zero vector.
\end{solution}

\begin{exercise}
Determine all entire functions $f$ that satisfy the inequality
\[
|f(z)| \leq |z|^2|\operatorname{Im}(z)|^2
\]
for $z$ sufficiently large.
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
1. \textbf{Liouville's Theorem (Generalized):} If $f$ is an entire function and $|f(z)| \leq C |z|^n$ for large $|z|$, then $f$ is a polynomial of degree at most $n$.
2. \textbf{Identity Theorem:} If two holomorphic functions agree on a set with an accumulation point (like a segment of the real axis), they agree everywhere.
3. \textbf{Imaginary Part Bound:} $|\operatorname{Im}(z)| \leq |z|$.

\textbf{Steps:}

1. \textbf{Determine the nature of $f$:} First, we note that $|\operatorname{Im}(z)| \leq |z|$. The given inequality $|f(z)| \leq |z|^2 |\operatorname{Im}(z)|^2$ implies $|f(z)| \leq |z|^4$ for large $z$. 
By the Generalized Liouville's Theorem, $f(z)$ must be a polynomial of degree at most 4.

2. \textbf{Test the Real Axis:} Now, consider what happens when $z$ is a real number ($z = x + 0i$). On the real axis, $\operatorname{Im}(z) = 0$. 
Plugging this into our inequality: $|f(x)| \leq |x|^2 \cdot |0|^2 = 0$.
This means $f(x) = 0$ for all sufficiently large real $x$.

3. \textbf{Apply the Identity Theorem:} We have a polynomial $f(z)$ that is zero on the real axis (or at least a portion of it). 
A holomorphic function that vanishes on a line segment must be zero everywhere in its domain. 
Since $f(z)$ is entire, $f(z) = 0$ for all $z \in \mathbb{C}$.

\textbf{Intuition:}
The growth condition is extremely restrictive. It doesn't just say the function grows slowly; it says the function \textit{must} shrink to zero whenever you get close to the real axis. 

Because holomorphic functions are very "rigid" (knowing them on a small line determines them everywhere), forcing the function to be zero on the entire real axis leaves it no "room" to be anything other than zero elsewhere.
\end{solution}


\begin{exercise}
Describe all functions that are holomorphic over the unit disk $D=\{|z|<1\}$, continuous on $\bar{D}$, and map the boundary of the disk into the boundary of the disk.
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
1. \textbf{Maximum Modulus Principle:} If $f$ is holomorphic and continuous on a closed disk, its maximum value must occur on the boundary.
2. \textbf{Blaschke Factors:} Functions of the form $B_a(z) = \frac{z - a}{1 - \bar{a} z}$ map the unit disk to itself and the unit circle to the unit circle.
3. \textbf{Fundamental Theorem of Algebra / Zeros of Holomorphic Functions:} A holomorphic function that is continuous on a closed disk and non-zero on the boundary can only have finitely many zeros in the disk.

\textbf{Steps:}

1. \textbf{Boundedness:} Since $|f(z)| = 1$ for all $|z| = 1$, the Maximum Modulus Principle tells us that $|f(z)| \leq 1$ for all $z \in D$.

2. \textbf{Zeros and Finiteness:} If $f$ has no zeros in $D$, then $1/f$ is also holomorphic in $D$ and continuous on $\bar{D}$. Since $|1/f| = 1$ on the boundary, the Maximum Modulus Principle applied to $1/f$ tells us $|1/f| \leq 1$, or $|f| \geq 1$. Combined with step 1, we get $|f| \equiv 1$, so $f$ is a constant of modulus 1.

3. \textbf{Handling Zeros:} If $f$ has zeros $a_1, a_2, \dots, a_n$ in $D$, we can define a Blaschke product $B(z) = \prod_{k=1}^n \frac{z - a_k}{1 - \bar{a}_k z}$. This product has the exact same zeros as $f$ and also has modulus 1 on the boundary.

4. \textbf{Eliminate Zeros:} Define $g(z) = f(z) / B(z)$. This function $g$ is holomorphic (the zeros cancel out) and has no zeros in $D$. Since $|g| = |f|/|B| = 1/1 = 1$ on the boundary, we apply the logic from step 2 to conclude that $g(z)$ must be a constant $e^{i\theta}$.

5. \textbf{Final Form:} Thus, $f(z) = e^{i\theta} \prod_{k=1}^n \frac{z - a_k}{1 - \bar{a}_k z}$.

\textbf{Intuition:}
This problem asks for functions that "preserve the circle." Imagine the unit disk as a flexible rubber sheet. You are allowed to stretch it, twist it, and fold it, but you must keep the boundary edge exactly on the boundary circle. 

Blaschke products are the only ways to do this holomorphically. Each factor $\frac{z-a}{1-\bar{a}z}$ "wraps" the disk around itself once. By combining multiple factors, you can wrap the disk multiple times, but you are always anchored by the fact that the boundary maps to the boundary.
\end{solution}

\begin{exercise}
Let $T:H_1\to H_2$, $Q:H_2\to H_1$ be bounded linear operators of Hilbert spaces $H_1,H_2$. Let $QT=Id-S_1$, $TQ=Id-S_2$ where $S_1$ and $S_2$ are compact operators. Prove $\ker T = \{v\in H_1\mid Tv=0\}$, $\operatorname{coker}T = H_2/\operatorname{Im}(T)$ are finite dimensional and $\operatorname{Im}(T)=\{Tv\in H_2\mid v\in H_1\}$ is closed in $H_2$.

Note: $S$ is compact means for every bounded sequence $x_n\in H_1$, $Sx_n$ has a converging subsequence.
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
1. \textbf{Compact Operators:} A compact operator $S$ maps the unit ball to a set with a compact closure. One key property is that if $(I-S)v = 0$, the space of such $v$ is finite dimensional.
2. \textbf{Fredholm Operators:} An operator $T$ is Fredholm if it has a closed range, a finite-dimensional kernel, and a finite-dimensional cokernel. This problem essentially asks to prove $T$ is Fredholm.
3. \textbf{Closed Range Theorem:} A bounded operator has a closed range if and only if it is "bounded below" on the complement of its kernel.

\textbf{Steps:}

1. \textbf{Finite dimensionality of $\ker T$:}
If $v \in \ker T$, then $Tv = 0$, which implies $QTv = 0$. Since $QT = I - S_1$, we have $(I - S_1)v = 0$, or $v = S_1 v$. This means $v$ is an eigenvector of the compact operator $S_1$ with eigenvalue 1. The eigenspaces of compact operators for non-zero eigenvalues are always finite dimensional. Thus $\ker T$ is finite dimensional.

2. \textbf{Closedness of $\operatorname{Im} T$:}
Suppose $Tv_n \to w$. We want to see if $w = Tv$ for some $v$. We can assume $v_n$ is perpendicular to $\ker T$. 
If the sequence $\{v_n\}$ is bounded, then $S_1 v_n$ has a convergent subsequence $S_1 v_{n_k} \to y$. 
From the identity $v_{n_k} - S_1 v_{n_k} = Q(Tv_{n_k})$, we see that $v_{n_k} = Q(Tv_{n_k}) + S_1 v_{n_k} \to Qw + y$. 
Let $v = Qw + y$. Then $v_{n_k} \to v$, and by continuity $Tv = w$. 
The case where $\{v_n\}$ is unbounded can be handled by showing that if no such $v$ existed, it would violate the compact property of $S_1$.

3. \textbf{Finite dimensionality of $\operatorname{coker} T$:}
The cokernel of $T$ is isomorphic to the kernel of its adjoint $T^*$. 
Taking the adjoint of $TQ = I - S_2$ gives $Q^* T^* = I - S_2^*$. 
Since the adjoint of a compact operator is compact, $S_2^*$ is compact. 
By the same logic used in Step 1, $\ker T^* \subset \ker(I - S_2^*)$ must be finite dimensional.

\textbf{Intuition:}
The operators $S_1$ and $S_2$ are "small" (compact). This means $T$ is "almost invertible" because $QT$ and $TQ$ are almost the identity. 

In finite dimensions, every linear map has a closed range and finite kernel. In infinite dimensions, this is not true. However, being "almost invertible" forces the operator to behave mostly like a finite-dimensional one, preserving the closedness of its image and the finiteness of its "failure to be invertible" (the kernel and cokernel).
\end{solution}

\begin{exercise}
Let $H_1$ be the Sobolev space on the unit interval $[0,1]$, i.e., the Hilbert space consisting of functions $f\in L^2([0,1])$ such that
\[
\|f\|_1^2 = \sum_{n=-\infty}^{\infty}(1+n^2)|\hat{f}(n)|^2 < \infty,
\]
where
\[
\hat{f}(n) = \frac{1}{2\pi}\int_0^1 f(x)e^{-2\pi inx}\,dx
\]
are Fourier coefficients of $f$. Show that there exists constant $C>0$ such that
\[
\|f\|_{L^{\infty}} \leq C\|f\|_1
\]
for all $f\in H_1$, where $\|.\|_{L^{\infty}}$ stands for the usual supremum norm. (Hint: Use Fourier series.)
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
1. \textbf{Fourier Series:} For "nice" functions, $f(x) = \sum \hat{f}(n) e^{2\pi i n x}$.
2. \textbf{Cauchy-Schwarz Inequality for Series:} $\sum |a_n b_n| \leq \sqrt{\sum |a_n|^2} \sqrt{\sum |b_n|^2}$.
3. \textbf{Sum of Reciprocals of Squares:} The series $\sum_{n=1}^\infty \frac{1}{n^2}$ converges (to $\pi^2/6$).

\textbf{Steps:}

1. \textbf{Expand $f(x)$:} We express the function in its Fourier representation:
$|f(x)| = |\sum_{n=-\infty}^\infty \hat{f}(n) e^{2\pi i n x}| \leq \sum_{n=-\infty}^\infty |\hat{f}(n)|$. 
This inequality holds because $|e^{i\theta}| = 1$.

2. \textbf{Apply Cauchy-Schwarz:} To utilize the Sobolev norm $\|f\|_1$, we need to introduce the factor $\sqrt{1+n^2}$ into the sum. We write:
\[
\sum_{n=-\infty}^\infty |\hat{f}(n)| = \sum_{n=-\infty}^\infty \left( |\hat{f}(n)| \sqrt{1+n^2} \cdot \frac{1}{\sqrt{1+n^2}} \right)
\]
Using Cauchy-Schwarz with $a_n = |\hat{f}(n)|\sqrt{1+n^2}$ and $b_n = 1/\sqrt{1+n^2}$:
\[
|f(x)| \leq \sqrt{\sum_{n=-\infty}^\infty (1+n^2)|\hat{f}(n)|^2} \cdot \sqrt{\sum_{n=-\infty}^\infty \frac{1}{1+n^2}}
\]

3. \textbf{Identify the Terms:} The first square root is exactly the definition of $\|f\|_1$. The second square root is a constant $C$ because the series $\sum \frac{1}{1+n^2}$ converges.

4. \textbf{Conclusion:} Since the bound $|f(x)| \leq C \|f\|_1$ holds for every $x \in [0,1]$, it must also hold for the supremum: $\|f\|_\infty \leq C \|f\|_1$.

\textbf{Intuition:}
The Sobolev norm $\|f\|_1$ measures not just the "size" of the function (the $n=0$ term), but also the "wiggliness" or "roughness" of the function (the terms where $n$ is large). 

By requiring $\sum (1+n^2)|\hat{f}(n)|^2$ to be finite, we are essentially saying that the high-frequency components of the function must decay quickly. If the frequencies decay fast enough, the function cannot have any "infinite spikes," making it bounded and, in fact, continuous.
\end{solution}

\begin{exercise}
Let $\phi\in C([a,b],\mathbb{R})$. Suppose for every function $h\in C^1([a,b],\mathbb{R})$, $h(a)=h(b)=0$, we have
\[
\int_a^b \phi(x)h(x)\,dx = 0.
\]
Prove that $\phi(x)=0$.
\end{exercise}

\begin{solution}
This is the Fundamental Lemma of the Calculus of Variations.
Suppose for contradiction that $\phi(x_0) \neq 0$ for some $x_0 \in (a,b)$. Without loss of generality, let $\phi(x_0) > 0$.
Since $\phi$ is continuous, there exists an interval $[x_0-\delta, x_0+\delta] \subset (a,b)$ such that $\phi(x) > 0$ for all $x$ in this interval.
Define a test function $h(x)$ by:
\[
h(x) = \begin{cases} (x - (x_0-\delta))^2 (x - (x_0+\delta))^2 & x \in [x_0-\delta, x_0+\delta] \\ 0 & \text{otherwise} \end{cases}
\]
Then $h \in C^1([a,b])$, $h(a)=h(b)=0$, and $h(x) > 0$ on $(x_0-\delta, x_0+\delta)$.
Then:
\[
\int_a^b \phi(x) h(x) \, dx = \int_{x_0-\delta}^{x_0+\delta} \phi(x) h(x) \, dx > 0
\]
since the integrand is a continuous non-negative function that is strictly positive at $x_0$.
This contradicts the hypothesis that $\int \phi h = 0$ for all such $h$.
Thus $\phi(x) = 0$ for all $x \in (a,b)$, and by continuity, for all $x \in [a,b]$.
\end{solution}

\begin{exercise}
Let $f$ be a Lebesgue integrable function over $[a,b+\delta]$, $\delta>0$. Prove that
\[
\lim_{h\to 0^+}\int_a^b |f(x+h)-f(x)|\,dx \to 0.
\]
\end{exercise}

\begin{solution}
Let $\epsilon > 0$. Since continuous functions with compact support are dense in $L^1$, there exists a continuous function $g$ on $[a, b+\delta]$ such that $\int_a^{b+\delta} |f(x) - g(x)| \, dx < \epsilon/3$.
We have:
\[
\int_a^b |f(x+h) - f(x)| \, dx \leq \int_a^b |f(x+h) - g(x+h)| \, dx + \int_a^b |g(x+h) - g(x)| \, dx + \int_a^b |g(x) - f(x)| \, dx
\]
The first and third terms are bounded by $\epsilon/3$ for small $h > 0$ (the first term by a change of variables).
The middle term:
\[
\int_a^b |g(x+h) - g(x)| \, dx \leq (b-a) \max_{x\in[a,b]} |g(x+h) - g(x)|
\]
Since $g$ is continuous on the compact set $[a, b+\delta]$, it is uniformly continuous. Thus for sufficiently small $h$, $|g(x+h) - g(x)| < \frac{\epsilon}{3(b-a)}$ for all $x \in [a,b]$.
Thus $\int_a^b |g(x+h) - g(x)| \, dx < \epsilon/3$.
Therefore, $\int_a^b |f(x+h) - f(x)| \, dx < \epsilon$ for small $h$, proving the limit is 0.
\end{solution}

\begin{exercise}
Let $L(q,q',t)$ be a function of $(q,q',t)\in TU\times\mathbb{R}$, $U$ is an open domain in $\mathbb{R}^n$. Let $\gamma:[a,b]\to U$ be a curve in $U$. Define a functional
\[
S(\gamma) = \int_a^b L(\gamma(t),\gamma'(t),t)\,dt.
\]
We say that $\gamma$ is an extremal if for every smooth variation of $\gamma$, $\phi(t,s)$, $s\in(-\delta,\delta)$, $\phi(t,0)=\gamma(t)$, $\phi_s=\phi(t,s)$, we have $\frac{dS(\phi_s)}{ds}|_{s=0}=0$. Prove that every extremal $\gamma$ satisfies
\[
\frac{d}{dt}\left(\frac{\partial L}{\partial q'}\right) = \frac{\partial L}{\partial q}.
\]
\end{exercise}

\begin{solution}
Let $\eta(t) = \frac{\partial \phi}{\partial s}(t,0)$. Since variations fix endpoints, $\eta(a) = \eta(b) = 0$.
The condition $\frac{d}{ds} S(\phi_s) |_{s=0} = 0$ is:
\[
\int_a^b \left( \frac{\partial L}{\partial q} \cdot \eta(t) + \frac{\partial L}{\partial q'} \cdot \eta'(t) \right) dt = 0
\]
Integrating the second term by parts:
\[
\int_a^b \frac{\partial L}{\partial q'} \cdot \eta'(t) \, dt = \left[ \frac{\partial L}{\partial q'} \eta(t) \right]_a^b - \int_a^b \frac{d}{dt} \left( \frac{\partial L}{\partial q'} \right) \eta(t) \, dt
\]
The boundary term vanishes since $\eta(a) = \eta(b) = 0$. Thus:
\[
\int_a^b \left( \frac{\partial L}{\partial q} - \frac{d}{dt} \left( \frac{\partial L}{\partial q'} \right) \right) \eta(t) \, dt = 0
\]
Since this holds for all smooth $\eta(t)$ with $\eta(a)=\eta(b)=0$, by the Fundamental Lemma of the Calculus of Variations, we must have:
\[
\frac{\partial L}{\partial q} - \frac{d}{dt} \left( \frac{\partial L}{\partial q'} \right) = 0 \implies \frac{d}{dt} \left( \frac{\partial L}{\partial q'} \right) = \frac{\partial L}{\partial q}
\]
\end{solution}

\begin{exercise}
Let $f:U\to U$ be a holomorphic function with $U$ a bounded domain in the complex plane. Assuming $0\in U$, $f(0)=0$, $f'(0)=1$, prove that $f(z)=z$.
\end{exercise}

\begin{solution}
This is a version of Cartan's Uniqueness Theorem.
Expand $f(z)$ in a power series near 0: $f(z) = z + a_k z^k + \dots$ with $k \geq 2$.
Suppose $f(z) \neq z$, and let $a_k$ be the first non-zero coefficient after the linear term.
Then $f^{(n)}(z) = z + n a_k z^k + \dots$ (the $n$-th iterate of $f$).
Since $f: U \to U$ and $U$ is bounded, there exists $M > 0$ such that $|f^{(n)}(z)| \leq M$ for all $z \in U$ and all $n$.
By Cauchy's Inequality for the $k$-th derivative:
\[
|n a_k| = \left| \frac{(f^{(n)})^{(k)}(0)}{k!} \right| \leq \frac{M}{r^k}
\]
where $D(0, r) \subset U$.
As $n \to \infty$, $|n a_k| \to \infty$ unless $a_k = 0$.
But we assumed $a_k \neq 0$, which leads to a contradiction.
Thus all $a_k = 0$ for $k \geq 2$, and $f(z) = z$.
\end{solution}

\begin{exercise}
Let $T:H_1\to H_2$ be a bounded operator of Hilbert spaces $H_1,H_2$. Let $S:H_1\to H_2$ be a compact operator, that is, for every bounded sequence $\{v_n\}\in H_1$, $Sv_n$ has a converging subsequence. Show that $\operatorname{coker}(T+S) = H_2/\operatorname{Im}(T+S)$ is finite dimensional and $\operatorname{Im}(T+S)$ is closed in $H_2$. (Hint: Consider equivalent statements in terms of adjoint operators.)
\end{exercise}

\begin{solution}
This is a core result in the theory of Fredholm operators. We assume $T=Id$ (the identity operator), so we are considering $Id+S$ where $S$ is compact.

1. \textbf{Finite dimensionality of $\ker(Id+S)$:}
If $(Id+S)v = 0$, then $v = -Sv$. Since $S$ is a compact operator, it maps the unit ball in $\ker(Id+S)$ to a precompact set. But on $\ker(Id+S)$, $S$ acts as $-Id$, so the unit ball in $\ker(Id+S)$ is precompact. By Riesz's Theorem, a normed space is finite dimensional if and only if its unit ball is compact. Thus $\ker(Id+S)$ is finite dimensional.

2. \textbf{Closedness of $\operatorname{Im}(Id+S)$:}
Suppose $(Id+S)v_n \to w$. We want to show $w \in \operatorname{Im}(Id+S)$.
Write $H_1 = \ker(Id+S) \oplus \ker(Id+S)^\perp$. We can assume $v_n \in \ker(Id+S)^\perp$.
If $\{v_n\}$ is bounded, then $Sv_n$ has a convergent subsequence $Sv_{n_k} \to y$. Then $v_{n_k} = (Id+S)v_{n_k} - Sv_{n_k} \to w - y = v$. By continuity, $(Id+S)v = w$.
If $\{v_n\}$ is unbounded, let $u_n = v_n / \|v_n\|$. Then $(Id+S)u_n = \frac{1}{\|v_n\|} (Id+S)v_n \to 0$. By the same argument, a subsequence $u_{n_k} \to u$ with $\|u\|=1$ and $(Id+S)u = 0$, so $u \in \ker(Id+S)$. But $u_n \in \ker(Id+S)^\perp$, so $u \in \ker(Id+S)^\perp$. This implies $u=0$, contradicting $\|u\|=1$. Thus $\{v_n\}$ must be bounded, and the range is closed.

3. \textbf{Finite dimensionality of $\operatorname{coker}(Id+S)$:}
By the Closed Range Theorem, $\operatorname{coker}(Id+S) \cong \ker(Id+S^*)$. Since $S$ is compact, its adjoint $S^*$ is also compact. Applying step 1 to $S^*$, we conclude $\ker(Id+S^*)$ is finite dimensional.
\end{solution}

\begin{exercise}
Let $u\in C^2(\Omega)$, $\Omega\subset\mathbb{R}^d$ is a bounded domain with a smooth boundary.

\begin{itemize}
\item[a)] Let $u$ be a solution of the equation $\Delta u = f$, $u|_{\partial\Omega}=0$, $f\in L^2(\Omega)$. Prove that there is a constant $C$ depends only on $\Omega$ such that
\[
\int_{\Omega}\left(\sum_{j=1}^n\left(\frac{\partial u}{\partial x_j}\right)^2 + u^2\right)dx \leq C\int_{\Omega}f^2(x)\,dx.
\]

\item[b)] Let $\{u_n\}$ be a sequence of harmonic functions on $\Omega$, such that $\|u_n\|_{L^2(\Omega)}\leq M < \infty$, for a constant $M$ independent of $n$. Prove that there is a converging subsequence $\{u_{n_k}\}$ in $L^2(\Omega)$.
\end{itemize}
\end{exercise}

\begin{solution}
a) Multiply $\Delta u = f$ by $u$ and integrate over $\Omega$:
\[
\int_\Omega u \Delta u \, dx = \int_\Omega u f \, dx
\]
By Green's First Identity (and $u=0$ on $\partial \Omega$):
\[
-\int_\Omega |\nabla u|^2 \, dx = \int_\Omega u f \, dx
\]
Using Cauchy-Schwarz and Young's inequality:
\[
\int_\Omega |\nabla u|^2 \, dx \leq \int_\Omega |u| |f| \, dx \leq \epsilon \int_\Omega u^2 \, dx + \frac{1}{4\epsilon} \int_\Omega f^2 \, dx
\]
By Poincare inequality, $\int u^2 \leq C_P \int |\nabla u|^2$.
Substituting:
\[
\int |\nabla u|^2 \leq \epsilon C_P \int |\nabla u|^2 + \frac{1}{4\epsilon} \int f^2
\]
Choose $\epsilon = \frac{1}{2C_P}$:
\[
\frac{1}{2} \int |\nabla u|^2 \leq \frac{C_P}{2} \int f^2 \implies \int |\nabla u|^2 \leq C_P \int f^2
\]
Then $\int u^2 \leq C_P^2 \int f^2$. Combining gives $\int (|\nabla u|^2 + u^2) \leq (C_P + C_P^2) \int f^2$.

b) Harmonic functions satisfy the Mean Value Property. For any $K \subset\subset \Omega$, $u_n$ is uniformly bounded on $K$ by $\|u_n\|_2$.
By derivative estimates for harmonic functions, all derivatives of $u_n$ are also locally uniformly bounded.
By the Arzela-Ascoli theorem, $u_n$ has a subsequence converging uniformly on compact subsets to a harmonic function $u$.
The $L^2$ convergence follows from the fact that harmonic functions are smooth and the $L^2$ norm is bounded. Specifically, one can use the Rellich-Kondrachov compactness theorem on $H^1$ or the local regularity of harmonic functions.
\end{solution}
