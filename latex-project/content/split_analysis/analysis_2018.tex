\begin{exercise}
Let the sequence of functions $\{f_n\}_{n=1}^{\infty}$ in $L^2(\mathbb{R}^d)$ satisfy that $\|f_n\|_{L^2}=1$.
\begin{enumerate}
    \item Show that there exists a subsequence of function $\{f_{n_j}\}_{j=1}^{\infty}$ such that $f_{n_j}$ converges weakly to some function $f$ in $L^2(\mathbb{R}^d)$, i.e. 
    \[
    (f_{n_j}, g) \to (f,g)
    \]
    for all $g\in L^2(\mathbb{R}^d)$.
    \item If $f_n\rightharpoonup f$ weakly in $L^2(\mathbb{R}^d)$, and $\|f_n\|_{L^2}\to\|f\|_{L^2}$ as $n\to\infty$. Show that $\|f_n-f\|_{L^2}\to 0$ as $n\to\infty$.
\end{enumerate}
\end{exercise}

\begin{solution}
1) $L^2(\mathbb{R}^d)$ is a Hilbert space. By the Banach-Alaoglu Theorem, the closed unit ball of a Hilbert space is weakly compact. Since $\|f_n\|=1$, the sequence is in the unit ball, so a weakly convergent subsequence exists.
2) $\|f_n - f\|^2 = (f_n-f, f_n-f) = \|f_n\|^2 - (f_n, f) - (f, f_n) + \|f\|^2$.
As $n \to \infty$, $\|f_n\|^2 \to \|f\|^2$ and $(f_n, f) \to (f, f) = \|f\|^2$.
Thus $\lim_{n\to\infty} \|f_n - f\|^2 = \|f\|^2 - \|f\|^2 - \|f\|^2 + \|f\|^2 = 0$.
\end{solution}

\begin{exercise}
Let $f:U\to\mathbb{C}$ be a non-constant holomorphic function where $U\subset\mathbb{C}$ is the open set containing the closure $\bar{D}$ of the unit disk $D=\{z\in\mathbb{C}\mid |z|<1\}$. If $|f(z)|=1$ for all $z\in\partial D$, prove that $D\subset f(\bar{D})$.
\end{exercise}

\begin{solution}
Suppose $D \not\subset f(\bar{D})$. Then there exists $a \in D$ such that $f(z) \neq a$ for all $z \in \bar{D}$.
Consider $g(z) = \frac{f(z)-a}{1-\bar{a}f(z)}$. Since $|a|<1$ and $|f| \leq 1$ on $\bar{D}$ (by Maximum Modulus Principle), the denominator is never zero.
$g(z)$ is holomorphic on $\bar{D}$. For $z \in \partial D$, $|f(z)|=1$, so $|g(z)|=1$.
Since $f(z) \neq a$, $g(z)$ has no zeros in $D$.
Thus $1/g(z)$ is also holomorphic on $\bar{D}$.
On the boundary, $|1/g(z)| = 1/1 = 1$.
By the Maximum Modulus Principle, $|1/g(z)| \leq 1$ for all $z \in D$, so $|g(z)| \geq 1$.
But $|g(z)| \leq 1$ also by Maximum Modulus.
Thus $|g(z)| = 1$ on $D$, so $g(z)$ is constant.
This implies $f(z)$ is constant, which contradicts the assumption.
Thus $D \subset f(\bar{D})$.
\end{solution}

\begin{exercise}
Prove that if a sequence of harmonic function on the open disk converges uniformly on compact subset of the disk, then the limit is harmonic.
\end{exercise}

\begin{solution}
Let $u_n \to u$ uniformly on compact sets. Since each $u_n$ is harmonic, it satisfies the mean value property:
$u_n(x) = \frac{1}{|B_r(x)|} \int_{B_r(x)} u_n(y) dy$.
By uniform convergence, $u$ is continuous.
We can pass the limit inside the integral:
$u(x) = \lim u_n(x) = \frac{1}{|B_r(x)|} \int_{B_r(x)} \lim u_n(y) dy = \frac{1}{|B_r(x)|} \int_{B_r(x)} u(y) dy$.
Since $u$ is continuous and satisfies the mean value property, $u$ is harmonic.
\end{solution}

\begin{exercise}
Let $\mu$ be a Borel measure on $\mathbb{R}^n$. Let $\rho > 0$ be a fixed positive number, and $B_{\rho}(x) = \{y \in \mathbb{R}^n \mid d(x,y) < \rho\}$. For $x \in \mathbb{R}^n$, define a function:
\[
\theta(x) = \mu(\overline{B_{\rho}(x)})
\]
\begin{enumerate}
    \item Show that $\theta$ is upper semi-continuous, i.e., for every $x \in \mathbb{R}^n$, $\theta(x) \geq \limsup_{y \to x} \theta(y)$.
    \item Give an example of a Borel measure $\mu$, such that the function $\theta$ is not continuous.
\end{enumerate}
\end{exercise}

\begin{solution}
1) Let $x_k \to x$. We want to show $\limsup \theta(x_k) \leq \theta(x)$.
$\chi_{\overline{B_\rho(x_k)}}(y) = 1$ iff $d(y, x_k) \leq \rho$.
If $d(y, x) > \rho$, then for large $k$, $d(y, x_k) > \rho$, so $\limsup \chi_{\overline{B_\rho(x_k)}}(y) = 0$.
If $d(y, x) \leq \rho$, $\limsup \chi_{\overline{B_\rho(x_k)}}(y) \leq 1 = \chi_{\overline{B_\rho(x)}}(y)$.
By Fatou's Lemma (or dominated convergence since measure is locally finite):
$\limsup \int \chi_{\overline{B_\rho(x_k)}} d\mu \leq \int \limsup \chi_{\overline{B_\rho(x_k)}} d\mu \leq \int \chi_{\overline{B_\rho(x)}} d\mu = \theta(x)$.

2) Let $n=1, \rho=1$ and $\mu = \delta_1$ (Dirac measure at 1).
$\theta(0) = \mu([-1, 1]) = 1$.
For $x < 0$, $\theta(x) = \mu([x-1, x+1])$. Since $x+1 < 1$, $1 \notin [x-1, x+1]$, so $\theta(x) = 0$.
As $x \to 0^-$, $\theta(x) \to 0 \neq \theta(0)$.
\end{solution}

\begin{exercise}
Let $g$ denote a smooth function on $\mathbb{R}^n$ with compact support. Let $f$ denote the function given by the formula
\[
f(x) = \frac{1}{n(n-1)\alpha(n)} \int_{\mathbb{R}^n} \frac{1}{|x-y|^{n-2}} g(y) \, dy.
\]
Here $\alpha(n)$ is the volume of the unit ball in $\mathbb{R}^n$.
\begin{enumerate}
    \item Prove that the integral that defines $f$ converges for each $x \in \mathbb{R}^n$.
    \item Prove that $f$ is differentiable and that the gradient of $f$ is given by the formula
    \[
    \nabla f|_x = \frac{1}{n(n-1)\alpha(n)} \int_{\mathbb{R}^n} \frac{1}{|x-y|^{n-2}} (\nabla g)|_y \, dy.
    \]
    \item Prove that $f$ obeys $-\Delta f = g$ with $\Delta$ denoting $\frac{\partial^2}{\partial x_1^2} + \cdots + \frac{\partial^2}{\partial x_n^2}$.
\end{enumerate}
\end{exercise}

\begin{solution}
This is the Newton potential. The kernel $\Phi(x) = \frac{1}{n(n-1)\alpha(n) |x|^{n-2}}$ (with appropriate constant) is the fundamental solution of the Laplacian.
For $n > 2$, $\Delta (1/|x|^{n-2}) = 0$ for $x \neq 0$.
Using Green's second identity or distribution theory, $\Delta \Phi = -\delta$.
Thus $\Delta f = \Delta (\Phi * g) = (\Delta \Phi) * g = -\delta * g = -g$.
So $-\Delta f = g$.
\end{solution}

\begin{exercise}
If $u$ is a positive harmonic function on $\mathbb{R}^n\setminus\{0\}$ ($n\geq 2$), then there exist constants $a \geq 0$, $b \geq 0$ such that 
\[
u(x) = a + b|x|^{2-n}
\]
for all $x \in \mathbb{R}^n\setminus\{0\}$.
\end{exercise}

\begin{solution}
By Bôcher's Theorem, a positive harmonic function on a punctured ball has the form $v(x) + b \Phi(x)$ where $v$ is harmonic on the ball.
Extending this to $\mathbb{R}^n \setminus \{0\}$, $u(x) = v(x) + b |x|^{2-n}$ where $v$ is harmonic on $\mathbb{R}^n$.
Since $u > 0$ and $|x|^{2-n} \to 0$ as $|x| \to \infty$ (for $n>2$), $v$ must be bounded below.
By Liouville's Theorem, a harmonic function bounded below is constant. Thus $v(x) = a$.
Since $u > 0$, $a \geq 0$ and $b \geq 0$.
\end{solution}

\end{exercise}
