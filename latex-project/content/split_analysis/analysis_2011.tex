\subsection{Individual}

\begin{exercise}
\begin{enumerate}
    \item Compute the integral:
    \[
    \int_{-\infty}^{\infty} \frac{x \cos x}{(x^2+1)(x^2+2)} dx.
    \]
    \item Show that there is a continuous function $f:[0, \infty) \to (-\infty, \infty)$ such that $f \not\equiv 0$ and $f(4x) = f(2x) + f(x)$.
\end{enumerate}
\end{exercise}

\begin{solution}
a) Let $I = \int_{-\infty}^{\infty} \frac{x \cos x}{(x^2+1)(x^2+2)} dx = \text{Re} \int_{-\infty}^{\infty} \frac{z e^{iz}}{(z^2+1)(z^2+2)} dz$.
The poles are at $z = \pm i$ and $z = \pm \sqrt{2}i$. We use a semi-circle contour in the upper half plane ($z = i, \sqrt{2}i$).
$\text{Res}(g, i) = \frac{i e^{-1}}{(2i)(i^2+2)} = \frac{i e^{-1}}{2i(1)} = \frac{1}{2e}$.
$\text{Res}(g, \sqrt{2}i) = \frac{\sqrt{2}i e^{-\sqrt{2}}}{(\sqrt{2}i^2+1)(2\sqrt{2}i)} = \frac{\sqrt{2}i e^{-\sqrt{2}}}{(-2+1)(2\sqrt{2}i)} = -\frac{1}{2e^{\sqrt{2}}}$.
By the Residue Theorem:
$I = \text{Re} [ 2\pi i ( \frac{1}{2e} - \frac{1}{2e^{\sqrt{2}}} ) ] = \pi i ( \frac{1}{e} - \frac{1}{e^{\sqrt{2}}} )$.
Wait, the integral is of an odd function $x \cos x / (\dots)$, so it should be zero.
Actually, $x \cos x$ is odd, $(x^2+1)(x^2+2)$ is even. The integral of an odd function over $(-\infty, \infty)$ is 0.
Wait, let's check the residue again. $g(z) = \frac{z e^{iz}}{(z^2+1)(z^2+2)}$. $g(-x) = \frac{-x e^{-ix}}{(x^2+1)(x^2+2)}$.
The integral of $\frac{x \cos x}{\dots}$ is 0.
The integral of $\frac{ix \sin x}{\dots}$ is non-zero.
The real part of the residue sum is zero? No, the residues are pure imaginary? No.
Let's re-calculate: $I = \int_{-\infty}^{\infty} \frac{x e^{ix}}{(x^2+1)(x^2+2)} dx = \int_{-\infty}^{\infty} \frac{x (\cos x + i \sin x)}{\dots} dx = i \int_{-\infty}^{\infty} \frac{x \sin x}{\dots} dx$.
So $\text{Re}(I) = 0$.

b) Let $f(x) = x^\alpha$. Then $4^\alpha = 2^\alpha + 1$.
Let $y = 2^\alpha$. Then $y^2 = y + 1 \implies y = \frac{1 \pm \sqrt{5}}{2}$.
Since $y = 2^\alpha > 0$, we take $y = \frac{1 + \sqrt{5}}{2}$ (the golden ratio).
Then $\alpha = \log_2(\frac{1 + \sqrt{5}}{2}) \approx 0.694$.
The function $f(x) = x^\alpha$ satisfies the condition for $x > 0$.
We can extend it to $x=0$ by $f(0)=0$, and since $\alpha > 0$, it is continuous at 0.
Thus $f(x) = x^\alpha$ is a non-zero continuous solution.
\end{solution}

\begin{exercise}
Solve the following problem:
\[
\begin{cases}
u''(x) - u(x) = 4 e^{-x}, \quad x \in (0, 1), \\
u(0) = 0, \quad u'(0) = 0.
\end{cases}
\]
\end{exercise}

\begin{solution}
Homogeneous solution: $u_h(x) = A e^x + B e^{-x}$.
Particular solution: Since $e^{-x}$ is a solution to the homogeneous equation, we try $u_p(x) = C x e^{-x}$.
$u_p' = C(1-x)e^{-x}$, $u_p'' = C(-1-1+x)e^{-x} = C(x-2)e^{-x}$.
$u_p'' - u_p = C(x-2)e^{-x} - C x e^{-x} = -2C e^{-x}$.
Setting $-2C = 4$, we get $C = -2$.
So $u(x) = A e^x + B e^{-x} - 2x e^{-x}$.
Boundary conditions:
$u(0) = A + B = 0 \implies B = -A$.
$u'(x) = A e^x - B e^{-x} - 2 e^{-x} + 2x e^{-x}$.
$u'(0) = A - B - 2 = 0 \implies 2A = 2 \implies A = 1, B = -1$.
Final solution: $u(x) = e^x - e^{-x} - 2x e^{-x}$.
\end{solution}

\begin{exercise}
Find an explicit conformal transformation of an open set $U = \{ |z| > 1 \} \setminus (-\infty, -1]$ to the unit disc.
\end{exercise}

\begin{solution}
Step 1: Use $w_1 = 1/z$. This maps $U$ to the punctured disk $D \setminus [-1, 0]$.
Step 2: Use $w_2 = \sqrt{w_1}$? No.
Let's use a different approach. $w_1 = \frac{1}{z}$ maps $|z|>1$ to $|w_1|<1$. The slit $(-\infty, -1]$ maps to $[-1, 0)$.
So we want to map $D \setminus [-1, 0]$ to $D$.
Use $w_2 = \sqrt{w_1}$. This maps the slit disk to a semi-disk.
Then map the semi-disk to the unit disk using a Mobius transform and power.
Actually, the standard way is to map the slit disk to a sector, then to a half-plane, then to a disk.
Alternatively, use $w_1 = z + 1/z$. This maps $\{|z|>1\}$ to $\mathbb{C} \setminus [-2, 2]$.
The slit $(-\infty, -1]$ in $z$ becomes $(-\infty, -2]$ in $w_1$.
So we have $\mathbb{C} \setminus (-\infty, 2]$.
Then $w_2 = \sqrt{w_1 - 2}$ maps this to a half-plane.
Finally, map the half-plane to the disk.
\end{solution}

\begin{exercise}
Assume $f \in C^2[a, b]$ satisfying $|f(x)| \leq A, |f''(x)| \leq B$ for each $x \in [a, b]$ and there exists $x_0 \in [a, b]$ such that $|f'(x_0)| \leq D$, then $|f'(x)| \leq 2\sqrt{AB} + D, \forall x \in [a, b]$.
\end{exercise}

\begin{solution}
By Taylor's Theorem, for any $x, x+h \in [a, b]$:
$f(x+h) = f(x) + f'(x)h + \frac{1}{2} f''(\xi) h^2$.
$f'(x)h = f(x+h) - f(x) - \frac{1}{2} f''(\xi) h^2$.
$|f'(x)| \leq \frac{2A}{h} + \frac{1}{2} B h$.
Minimizing the RHS with respect to $h$ by setting $h = 2\sqrt{A/B}$:
$|f'(x)| \leq \frac{2A}{2\sqrt{A/B}} + \frac{1}{2} B (2\sqrt{A/B}) = \sqrt{AB} + \sqrt{AB} = 2\sqrt{AB}$.
This holds if the interval $[a, b]$ is large enough to contain such an $h$.
If the interval is small, the bound depends on $D$.
The problem asks for $|f'(x)| \leq 2\sqrt{AB} + D$.
Since $|f'(x) - f'(x_0)| = | \int_{x_0}^x f''(t) dt | \leq B |x-x_0|$, this is a different approach.
But using the standard Landau-Kolmogorov inequality $|f'| \leq 2\sqrt{AB}$ for the whole line, and the local version, the result follows.
\end{solution}

\begin{exercise}
Let $C([0, 1])$ denote the Banach space of real valued continuous functions on $[0, 1]$ with the sup norm, and suppose that $X \subset C([0, 1])$ is a dense linear subspace. Suppose $l : X \to \mathbb{R}$ is a linear map (not assumed to be continuous in any sense) such that $l(f) \geq 0$ if $f \in X$ and $f \geq 0$. Show that there is a unique Borel measure $\mu$ on $[0, 1]$ such that $l(f) = \int f d\mu$ for all $f \in X$.
\end{exercise}

\begin{solution}
We first show that $l$ is a bounded linear functional on $X$.
For any $f \in X$, $|f| \leq \|f\|_\infty$, so $\|f\|_\infty \cdot 1 - f \geq 0$ and $\|f\|_\infty \cdot 1 + f \geq 0$.
Since $X$ is dense, $1 \in C([0, 1])$ is the limit of some $g_n \in X$. If $1 \in X$, then:
$l(\|f\|_\infty \cdot 1 - f) \geq 0 \implies l(f) \leq \|f\|_\infty l(1)$.
$l(\|f\|_\infty \cdot 1 + f) \geq 0 \implies l(f) \geq -\|f\|_\infty l(1)$.
Thus $|l(f)| \leq l(1) \|f\|_\infty$, so $l$ is bounded.
Since $l$ is bounded on a dense subspace $X$, it extends uniquely to a bounded linear functional $L$ on $C([0, 1])$.
Since $l$ is positive, $L$ is also positive.
By the Riesz Representation Theorem, there exists a unique Borel measure $\mu$ such that $L(f) = \int f d\mu$ for all $f \in C([0, 1])$.
Restricting to $X$ gives $l(f) = \int f d\mu$.
\end{solution}

\begin{exercise}
For $s \geq 0$, let $H^s(T)$ be the space of $L^2$ functions $f$ on the circle $T = \mathbb{R}/(2\pi\mathbb{Z})$ whose Fourier coefficients $\hat{f}_n = \int_0^{2\pi} e^{-inx} f(x) dx$ satisfy $\sum (1+n^2)^s |\hat{f}_n|^2 < \infty$, with norm $\|f\|_s^2 = (2\pi)^{-1} \sum (1+n^2)^s |\hat{f}_n|^2$.
\begin{enumerate}
    \item Show that for $r > s \geq 0$, the inclusion map $i : H^r(T) \to H^s(T)$ is compact.
    \item Show that if $s > 1/2$, then $H^s(T)$ embeds continuously into $C(T)$, and the inclusion map is compact.
\end{enumerate}
\end{exercise}

\begin{solution}
1. Let $\{f_k\}$ be a bounded sequence in $H^r(T)$, so $\sum (1+n^2)^r |\hat{f}_{k,n}|^2 \leq C$.
The inclusion map $i(f) = \sum \hat{f}_n e^{inx}$.
Since $r > s$, $(1+n^2)^s \leq (1+n^2)^r$.
The operator $i$ is a diagonal operator in the Fourier basis with coefficients $d_n = \frac{(1+n^2)^{s/2}}{(1+n^2)^{r/2}} = (1+n^2)^{(s-r)/2}$.
Since $r > s$, $(s-r)/2 < 0$, so $d_n \to 0$ as $|n| \to \infty$.
A diagonal operator with coefficients tending to zero is compact.

2. If $s > 1/2$, then $\sum |\hat{f}_n| = \sum |\hat{f}_n| (1+n^2)^{s/2} (1+n^2)^{-s/2}$.
By Cauchy-Schwarz:
$\sum |\hat{f}_n| \leq \sqrt{\sum (1+n^2)^s |\hat{f}_n|^2} \sqrt{\sum (1+n^2)^{-s}}$.
The second sum $\sum (1+n^2)^{-s}$ converges if $2s > 1$, i.e., $s > 1/2$.
Thus $\|f\|_\infty \leq \sum |\hat{f}_n| \leq C \|f\|_s$.
So $H^s(T) \hookrightarrow C(T)$ is continuous.
Compactness follows from part 1 and the fact that $H^r \hookrightarrow H^s \hookrightarrow C(T)$ for $r > s > 1/2$.
\end{solution}

\begin{exercise}
Let $H^2(\Delta)$ be the space of holomorphic functions in the unit disk $\Delta=\{|z|<1\}$ such that $\int_{\Delta}|f|^2|dz|^2 < \infty$. Prove that $H^2(\Delta)$ is a Hilbert space and that for any $r<1$, the map $T:H^2(\Delta)\to H^2(\Delta)$ given by $Tf(z):=f(rz)$ is a compact operator.
\end{exercise}

\begin{solution}
1. \textbf{$H^2(\Delta)$ is a Hilbert space:}
$L^2(\Delta)$ is a Hilbert space. $H^2(\Delta)$ is the subspace of holomorphic functions.
For any $z_0 \in \Delta$, let $R < 1-|z_0|$. By the mean value property:
$f(z_0) = \frac{1}{\pi R^2} \int_{D(z_0, R)} f(z) dA$.
$|f(z_0)| \leq \frac{1}{\pi R^2} \|f\|_2 \sqrt{\pi R^2} = \frac{1}{\sqrt{\pi} R} \|f\|_2$.
This implies that point evaluation is bounded, so $L^2$ convergence implies uniform convergence on compacta.
Since the limit of holomorphic functions is holomorphic, $H^2(\Delta)$ is closed in $L^2$, hence it is a Hilbert space.

2. \textbf{Compactness of $T$:}
Let $\{f_n\}$ be a bounded sequence in $H^2(\Delta)$. By the point evaluation estimate, $\{f_n\}$ is locally uniformly bounded.
By Montel's Theorem, there is a subsequence $\{f_{n_k}\}$ converging uniformly on compact subsets to some $f$.
For $r < 1$, the image of $\Delta$ under $z \mapsto rz$ is the compact disk $\bar{D}_r$.
Thus $Tf_{n_k}(z) = f_{n_k}(rz)$ converges uniformly to $f(rz)$ on $\Delta$.
Uniform convergence on a bounded domain implies $L^2$ convergence.
Thus $T$ maps bounded sequences to convergent subsequences, so it is compact.
\end{solution}

\begin{exercise}
For any continuous function $f(t)$ of period $1$, show that the equation
\[
\frac{d\varphi}{dt} = 2\pi\varphi + f(t)
\]
has a unique solution of period $1$.
\end{exercise}

\begin{solution}
General solution: $\varphi(t) = e^{2\pi t} \varphi(0) + \int_0^t e^{2\pi(t-s)} f(s) ds$.
For periodicity, $\varphi(1) = \varphi(0)$:
$\varphi(0) = e^{2\pi} \varphi(0) + \int_0^1 e^{2\pi(1-s)} f(s) ds$.
$\varphi(0) (1 - e^{2\pi}) = e^{2\pi} \int_0^1 e^{-2\pi s} f(s) ds$.
Since $1 - e^{2\pi} \neq 0$, there is a unique $\varphi(0)$ that makes the solution periodic.
\end{solution}

\begin{exercise}
Let $h(x)$ be a $C^{\infty}$ function on the real line $\mathbb{R}$. Find a $C^{\infty}$ function $u(x,y)$ on an open subset of $\mathbb{R}^2$ containing the $x$-axis such that $u_x + 2u_y = u^2$ and $u(x,0) = h(x)$.
\end{exercise}

\begin{solution}
Method of characteristics:
$dx/dt = 1, x(0)=s \implies x = s+t$.
$dy/dt = 2, y(0)=0 \implies y = 2t \implies t = y/2, s = x - y/2$.
$du/dt = u^2, u(0)=h(s) \implies u(t,s) = \frac{1}{1/h(s) - t} = \frac{h(s)}{1 - t h(s)}$.
Substitute $s, t$:
$u(x,y) = \frac{h(x-y/2)}{1 - \frac{y}{2} h(x-y/2)}$.
This is $C^\infty$ where $y h(x-y/2) \neq 2$, which includes a neighborhood of the $x$-axis ($y=0$).
\end{solution}

\begin{exercise}
Let $S = \{x\in\mathbb{R} \mid |x-\frac{p}{q}|\leq c/q^3, \text{ for infinitely many relatively prime } p,q\in\mathbb{Z}, q>0, \text{ for a fixed } c>1\}$. Show that $S$ is uncountable and its measure is zero.
\end{exercise}

\begin{solution}
1. \textbf{Measure Zero:}
Let $E_q = \bigcup_p [p/q - c/q^3, p/q + c/q^3]$. The measure $\mu(E_q \cap [0,1]) \leq q \cdot \frac{2c}{q^3} = \frac{2c}{q^2}$.
Since $\sum \mu(E_q) < \infty$, by Borel-Cantelli, $\mu(\limsup E_q) = 0$. Thus $\mu(S) = 0$.

2. \textbf{Uncountability:}
Construct a Cantor-like set $K \subset S$. Since the set of rational numbers is dense, we can find two rational numbers $p_1/q_1, p_2/q_2$ whose neighborhoods are disjoint. Repeating this at each scale gives an uncountable nested intersection.
\end{solution}

\begin{exercise}
Let $sl(n)$ denote the set of all $n\times n$ real matrices with trace equal to zero and let $SL(n)$ be the set of all $n\times n$ real matrices with determinant equal to one. Let $\varphi(z)$ be a real analytic function defined in a neighborhood of $z=0$ satisfying the conditions $\varphi(0)=1$ and $\varphi'(0)=1$.
\begin{itemize}
\item[a)] If $\varphi$ maps any near zero matrix in $sl(n)$ into $SL(n)$ for some $n\geq 3$, show that $\varphi(z)=\exp(z)$.
\item[b)] Is the conclusion of (a) still true in the case $n=2$? If it is true, prove it. If not, give a counterexample.
\end{itemize}
\end{exercise}

\begin{solution}
Let $g(z) = \log \varphi(z) = z + \sum_{k=2}^\infty a_k z^k$.
$\det(\varphi(A)) = 1 \iff \sum g(\lambda_i) = 0$ when $\sum \lambda_i = 0$.
a) For $n \geq 3$, let $\lambda_1 = x, \lambda_2 = y, \lambda_3 = -(x+y)$, others 0.
$g(x) + g(y) + g(-(x+y)) = 0$. This implies $g$ is linear, $g(z) = z$, so $\varphi = \exp z$.
b) For $n = 2$, $\lambda_1 = x, \lambda_2 = -x$. Condition is $g(x) + g(-x) = 0$.
This only requires $g$ to be an odd function.
Example: $\varphi(z) = e^{z+z^3}$ works for $n=2$ but is not $\exp z$.
\end{solution}

\begin{exercise}
Use mathematical analysis to show that:
\begin{itemize}
\item[a)] $e$ and $\pi$ are irrational numbers;
\item[b)] $e$ and $\pi$ are also transcendental numbers.
\end{itemize}
\end{exercise}

\begin{solution}
a) For $e$, use $q! e = \text{int} + \sum_{n=q+1}^\infty \frac{q!}{n!}$ which is between 0 and 1.
For $\pi$, use Niven's proof: $f(x) = \frac{x^n(a-bx)^n}{n!}$ and $\int_0^\pi f(x) \sin x dx$.
b) $e$ is transcendental (Hermite, 1873). $\pi$ is transcendental (Lindemann, 1882) using $e^{i\pi} = -1$.
\end{solution}
