\section{2022 Geometry \& Topology}

\subsection{Exam}

\begin{exercise}
Let $n > 1$.
(i) Is the tangent bundle of $S^{2n} \times S^{2n}$ trivial? 
(ii) Is the tangent bundle of $S^{2n} \times S^{2n-1}$ trivial?
\end{exercise}

\begin{solution}
\textbf{Geometric Intuition:} A manifold with a trivial tangent bundle is called \textit{parallelizable}. This means you can "comb" the manifold in $d$ independent directions everywhere. The "Hairy Ball Theorem" says $S^2$ is not parallelizable. 

\textbf{Formal Proof:}
(i) No. For a compact manifold $M$, $TM$ being trivial implies $\chi(M) = 0$.
$\chi(S^{2n} \times S^{2n}) = \chi(S^{2n})^2 = 2^2 = 4 \neq 0$.
(ii) Yes. A product of spheres $S^n \times S^m$ is parallelizable if at least one factor is an odd-dimensional sphere. 
\textbf{Pedagogical Note:} $S^{2n-1}$ is "almost" parallelizable (it admits $2n-1$ vector fields in $\mathbb{R}^{2n}$ but not all stay tangent and independent in a way that trivializes the bundle alone). However, once you multiply by another sphere, the extra "room" allows for a trivialization.
\end{solution}

\begin{exercise}
The Berger metrics $g_{\epsilon}$ on $S^{3}$ are obtained by making $\epsilon^{-1}X_{1}, X_{2}, X_{3}$ orthonormal.
(i) Prove $\nabla_{X_i} X_i = 0$. (ii) Compute scalar curvature.
\end{exercise}

\begin{solution}
\textbf{Context:} Berger metrics represent a family of metrics where the $S^1$ fibers of the Hopf fibration $S^1 \to S^3 \to S^2$ are scaled by $\epsilon$. When $\epsilon=1$, we have the round sphere. When $\epsilon \to 0$, the $S^3$ "collapses" to $S^2$.

\textbf{Formal Proof:}
(i) Using the Koszul formula for left-invariant fields: $2\langle \nabla_X Y, Z \rangle = \langle [X,Y], Z \rangle - \langle [Y,Z], X \rangle + \langle [Z,X], Y \rangle$.
If $X=Y$, the first term is 0 and the last two cancel: $2\langle \nabla_X X, Z \rangle = 0$. Thus $\nabla_{X_i} X_i = 0$.
(ii) Using the frame $E_1 = \epsilon^{-1} X_1, E_2 = X_2, E_3 = X_3$, and brackets $[X_1, X_2]=2X_3$ etc., we find:
$[E_1, E_2] = 2\epsilon^{-1} E_3$, $[E_2, E_3] = 2\epsilon E_1$, $[E_3, E_1] = 2\epsilon^{-1} E_2$.
The sectional curvatures are $K_{12} = \epsilon^2, K_{23} = 4-3\epsilon^2, K_{31} = \epsilon^2$.
The scalar curvature is $S = 2(K_{12} + K_{23} + K_{31}) = 2(4 - \epsilon^2) = 8 - 2\epsilon^2$.
\end{solution}

\begin{exercise}
(i) Compute $\pi_2(U(n))$ and $\pi_3(U(n))$. (ii) Show $\det : U(n) \to S^1$ induces isomorphism on $\pi_1$.
\end{exercise}

\begin{solution}
(i) By Bott periodicity, $\pi_k(U(n))$ for large $n$ is $0$ if $k$ even and $\mathbb{Z}$ if $k$ odd.
- $\pi_2(U(n)) = 0$ for all $n$.
- $\pi_3(U(n)) = \mathbb{Z}$ for $n \geq 2$.
(ii) $SU(n)$ is the kernel of $\det$. For $n \geq 2$, $SU(n)$ is simply connected ($\pi_1(SU(n))=0$). From the long exact sequence of the fibration $SU(n) \to U(n) \to S^1$, we get $0 \to \pi_1(U(n)) \to \pi_1(S^1) \to 0$, so $\det_*$ is an isomorphism.
\end{solution}

\begin{exercise}
Prove $\mathbb{CP}^2$ does not immerse into $\mathbb{R}^6$.
\end{exercise}

\begin{solution}
\textbf{Context:} The Whitney Immersion Theorem says any $n$-manifold immerses into $\mathbb{R}^{2n-1}$. For $n=4$, this gives $\mathbb{R}^7$. This problem shows that $2n-2$ (which is 6) is not always enough.

\textbf{Formal Proof:}
Suppose $f: \mathbb{CP}^2 \to \mathbb{R}^6$ is an immersion. The normal bundle $\nu$ has rank 2.
We have $p(T\mathbb{CP}^2) p(\nu) = 1$.
$p(T\mathbb{CP}^2) = (1+a^2)^3 = 1 + 3a^2$ where $a^2$ is the generator of $H^4$.
Thus $p_1(\nu) = -3a^2$.
For a rank 2 real bundle $\nu$, $p_1(\nu) = e(\nu)^2$, where $e$ is the Euler class.
Let $e(\nu) = m a$ for $m \in \mathbb{Z}$. Then $m^2 a^2 = -3a^2$, so $m^2 = -3$.
No such integer $m$ exists. Thus the immersion is impossible.
\end{solution}

\begin{exercise}
 conformal change $\widetilde{g} = e^{2u}g$. (i) Prove $\Delta u - K + \widetilde{K}e^{2u} = 0$. (ii) If $\chi(M)=0$, show $\int \widetilde{K} e^{2f} < 0$.
\end{exercise}

\begin{solution}
(i) Standard identity for curvature under conformal scaling.
(ii) Since $\chi(M)=0$, $\int K = 0$. Let $\Delta f = K$.
The equation becomes $\Delta(u-f) = -\widetilde{K} e^{2u}$.
Integrating $\frac{\Delta(u-f)}{e^{2(u-f)}}$ and using integration by parts shows $\int \widetilde{K} e^{2f} \leq 0$.
Equality holds only if $u-f$ is constant, implying $\widetilde{K}=0$.
\end{solution}

\begin{exercise}
Fixed points of $\mathbb{CP}^n$.
\end{exercise}

\begin{solution}
\textbf{Pedagogical Note:} The Lefschetz number $L(f)$ tells us about fixed points. $L(f) \neq 0$ implies at least one fixed point.
(i) $\mathbb{CP}^n$: cohomology is $1, x, x^2, \dots, x^n$. For $n$ even, $L(f)$ is always non-zero ($1+1+\dots+1 = n+1$ or $1-1+1-\dots+1=1$). For $n$ odd, $L(f)$ can be 0.
Example for $n=1$: $\mathbb{CP}^1 \cong S^2$. The antipodal map has no fixed points.
(ii) $\mathbb{CP}^2 \times \mathbb{CP}^2$: $\chi = 9$. However, $L(f)$ can be zero by choosing a map that acts as a "rotation-like" automorphism on the product cohomology.
\end{solution}
