\section{2022 Geometry \& Topology}

\subsection{Exam}

\begin{exercise}
Let $n > 1$.
(i) Is the tangent bundle of $S^{2n} \times S^{2n}$ trivial? 
(ii) Is the tangent bundle of $S^{2n} \times S^{2n-1}$ trivial?
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}

1. \textbf{Tangent Bundle and Parallelizability:} The tangent bundle $TM$ of a manifold $M$ is considered trivial if the manifold is parallelizable. This means there is a smooth, globally defined basis of tangent vectors at every point.

2. \textbf{Euler Characteristic Constraint:} A well-known theorem (Poincaré-Hopf) states that a compact manifold can only have a nowhere vanishing vector field if its Euler characteristic $\chi(M)$ is exactly zero. A trivial tangent bundle requires such vector fields, making $\chi(M) = 0$ a strict prerequisite.

3. \textbf{Euler Characteristics of Spheres and Products:} For an $m$-dimensional sphere $S^m$, the Euler characteristic is $\chi(S^m) = 1 + (-1)^m$. Thus, even-dimensional spheres have $\chi=2$, and odd-dimensional spheres have $\chi=0$. For a product manifold, $\chi(M \times N) = \chi(M) \chi(N)$.

\textbf{Intuition:}

Imagine trying to comb the hair flat on different shapes. A trivial tangent bundle means you can smoothly comb the hair everywhere without creating any cowlicks or singularities. 

For a 2D sphere (like Earth), the "Hairy Ball Theorem" proves this is impossible; there will always be a swirl. This obstruction exists for all even-dimensional spheres. When you multiply two even-dimensional spheres together, the resulting shape inherits this inability to be combed smoothly.

However, odd-dimensional spheres can be combed smoothly. When you combine an even-dimensional sphere with an odd-dimensional one, the odd sphere provides enough extra "room" or directional freedom to allow the entire product shape to be perfectly combed, resulting in a trivial tangent bundle.

\textbf{Steps:}

\textbf{Part (i): Is the tangent bundle of $S^{2n} \times S^{2n}$ trivial?}

\textbf{Step 1: Calculate the Euler characteristic of the product.}
Using the product formula, we compute $\chi(S^{2n} \times S^{2n}) = \chi(S^{2n}) \times \chi(S^{2n})$.

\textbf{Step 2: Evaluate the sphere components.}
Since $2n$ is always an even number, $\chi(S^{2n}) = 2$.
Therefore, the Euler characteristic of the product is $2 \times 2 = 4$.

\textbf{Step 3: Check against the necessary condition.}
For the tangent bundle to be trivial on a compact manifold, the Euler characteristic must be $0$. Since $4 \neq 0$, the tangent bundle cannot be trivial. The answer is no.

\textbf{Part (ii): Is the tangent bundle of $S^{2n} \times S^{2n-1}$ trivial?}

\textbf{Step 1: Verify the Euler characteristic condition.}
First, we check $\chi(S^{2n} \times S^{2n-1}) = \chi(S^{2n}) \times \chi(S^{2n-1})$. Since $2n-1$ is odd, $\chi(S^{2n-1}) = 0$. Thus, $2 \times 0 = 0$. The necessary condition is met, so we proceed to a deeper theorem.

\textbf{Step 2: Apply the Kervaire-Milnor Theorem.}
A profound result in differential topology states that the product of any two spheres $S^k \times S^l$ is parallelizable (meaning it has a trivial tangent bundle) if and only if at least one of the spheres is odd-dimensional. 

\textbf{Step 3: Conclude based on the dimensions.}
In our case, the factor $S^{2n-1}$ provides the required odd-dimensional sphere. Therefore, the tangent bundle of the entire product space is trivial. The answer is yes.
\end{solution}

\begin{exercise}
The Berger metrics $g_{\epsilon}$ on $S^{3}$ are obtained by making $\epsilon^{-1}X_{1}, X_{2}, X_{3}$ orthonormal.
(i) Prove $\nabla_{X_i} X_i = 0$. (ii) Compute scalar curvature.
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}

1. \textbf{Lie Groups and Left-Invariant Vector Fields:} For a Lie group like $SU(2)$ (which is diffeomorphic to the 3-sphere $S^3$), left-invariant vector fields form a Lie algebra. Their behavior is determined by the Lie bracket $[X, Y]$.

2. \textbf{Koszul Formula:} This fundamental formula connects the Levi-Civita connection $\nabla$ to the Lie bracket and the metric $\langle \cdot, \cdot \rangle$. For left-invariant vector fields, it is:
$2\langle \nabla_X Y, Z \rangle = \langle [X,Y], Z \rangle - \langle [Y,Z], X \rangle + \langle [Z,X], Y \rangle$.

3. \textbf{Sectional and Scalar Curvature:} For an orthonormal frame $\{E_1, E_2, E_3\}$, the scalar curvature $S$ is given by twice the sum of the sectional curvatures of the three coordinate planes: $S = 2(K_{12} + K_{23} + K_{31})$.

\textbf{Intuition:}

The Berger metrics give us a way to "squash" or deform the round 3-sphere. Think of the 3-sphere as being made up of many circles (fibers) wrapping around a 2-sphere base (this is called the Hopf fibration).

The parameter $\epsilon$ controls the size of these circular fibers. When $\epsilon = 1$, everything is perfectly round. As $\epsilon$ approaches 0, these circles shrink, and the 3-sphere flattens out toward the 2-sphere. This problem asks us to calculate exactly how this geometric squashing changes the intrinsic curvature of the space.

\textbf{Steps:}

\textbf{Part (i): Prove that $\nabla_{X_i} X_i = 0$.}

\textbf{Step 1: Apply the Koszul formula.}
We want to evaluate $\nabla_{X_i} X_i$. We can find it by taking its inner product with an arbitrary vector field $Z$ using the Koszul formula. Let $X = Y = X_i$.

\textbf{Step 2: Simplify the terms.}
The Koszul formula becomes:
$2\langle \nabla_{X_i} X_i, Z \rangle = \langle [X_i, X_i], Z \rangle - \langle [X_i, Z], X_i \rangle + \langle [Z, X_i], X_i \rangle$.
The Lie bracket of any vector field with itself is zero, so $[X_i, X_i] = 0$, eliminating the first term.

\textbf{Step 3: Evaluate the remaining terms.}
Using the anti-symmetry of the Lie bracket ($[Z, X_i] = -[X_i, Z]$), the remaining terms become $-2\langle [X_i, Z], X_i \rangle$. For the standard basis of $SU(2)$, the bracket $[X_i, Z]$ is always orthogonal to $X_i$ for any basis vector $Z$. Thus, this inner product is zero. Since this holds for all $Z$, we conclude $\nabla_{X_i} X_i = 0$.

\textbf{Part (ii): Compute the scalar curvature.}

\textbf{Step 1: Set up the orthonormal frame.}
We define a new orthonormal frame $E_1 = \epsilon^{-1} X_1$, $E_2 = X_2$, and $E_3 = X_3$.
The standard Lie brackets for $SU(2)$ are $[X_1, X_2] = 2X_3$, $[X_2, X_3] = 2X_1$, and $[X_3, X_1] = 2X_2$.

\textbf{Step 2: Calculate brackets for the new frame.}
Translating these to our new orthonormal frame yields:
$[E_1, E_2] = [\epsilon^{-1} X_1, X_2] = 2\epsilon^{-1} E_3$.
$[E_2, E_3] = [X_2, X_3] = 2X_1 = 2\epsilon E_1$.
$[E_3, E_1] = [X_3, \epsilon^{-1} X_1] = 2\epsilon^{-1} E_2$.

\textbf{Step 3: Find the sectional curvatures.}
Using standard curvature formulas for 3-dimensional Lie groups with these brackets, the sectional curvatures of the coordinate planes are known to be:
$K_{12} = \epsilon^2$
$K_{23} = 4 - 3\epsilon^2$
$K_{31} = \epsilon^2$

\textbf{Step 4: Compute the final scalar curvature.}
The scalar curvature $S$ is twice the sum of these sectional curvatures:
$S = 2(K_{12} + K_{23} + K_{31})$.
Substituting our values:
$S = 2(\epsilon^2 + 4 - 3\epsilon^2 + \epsilon^2) = 2(4 - \epsilon^2)$.
Finally, we arrive at $S = 8 - 2\epsilon^2$.
\end{solution}

\begin{exercise}
(i) Compute $\pi_2(U(n))$ and $\pi_3(U(n))$. (ii) Show $\det : U(n) \to S^1$ induces isomorphism on $\pi_1$.
\end{exercise}

\begin{solution}
(i) By Bott periodicity, $\pi_k(U(n))$ for large $n$ is $0$ if $k$ even and $\mathbb{Z}$ if $k$ odd.
- $\pi_2(U(n)) = 0$ for all $n$.
- $\pi_3(U(n)) = \mathbb{Z}$ for $n \geq 2$.
(ii) $SU(n)$ is the kernel of $\det$. For $n \geq 2$, $SU(n)$ is simply connected ($\pi_1(SU(n))=0$). From the long exact sequence of the fibration $SU(n) \to U(n) \to S^1$, we get $0 \to \pi_1(U(n)) \to \pi_1(S^1) \to 0$, so $\det_*$ is an isomorphism.
\end{solution}

\begin{exercise}
Prove $\mathbb{CP}^2$ does not immerse into $\mathbb{R}^6$.
\end{exercise}

\begin{solution}
\textbf{Context:} The Whitney Immersion Theorem says any $n$-manifold immerses into $\mathbb{R}^{2n-1}$. For $n=4$, this gives $\mathbb{R}^7$. This problem shows that $2n-2$ (which is 6) is not always enough.

\textbf{Formal Proof:}
Suppose $f: \mathbb{CP}^2 \to \mathbb{R}^6$ is an immersion. The normal bundle $\nu$ has rank 2.
We have $p(T\mathbb{CP}^2) p(\nu) = 1$.
$p(T\mathbb{CP}^2) = (1+a^2)^3 = 1 + 3a^2$ where $a^2$ is the generator of $H^4$.
Thus $p_1(\nu) = -3a^2$.
For a rank 2 real bundle $\nu$, $p_1(\nu) = e(\nu)^2$, where $e$ is the Euler class.
Let $e(\nu) = m a$ for $m \in \mathbb{Z}$. Then $m^2 a^2 = -3a^2$, so $m^2 = -3$.
No such integer $m$ exists. Thus the immersion is impossible.
\end{solution}

\begin{exercise}
 conformal change $\widetilde{g} = e^{2u}g$. (i) Prove $\Delta u - K + \widetilde{K}e^{2u} = 0$. (ii) If $\chi(M)=0$, show $\int \widetilde{K} e^{2f} < 0$.
\end{exercise}

\begin{solution}
(i) Standard identity for curvature under conformal scaling.
(ii) Since $\chi(M)=0$, $\int K = 0$. Let $\Delta f = K$.
The equation becomes $\Delta(u-f) = -\widetilde{K} e^{2u}$.
Integrating $\frac{\Delta(u-f)}{e^{2(u-f)}}$ and using integration by parts shows $\int \widetilde{K} e^{2f} \leq 0$.
Equality holds only if $u-f$ is constant, implying $\widetilde{K}=0$.
\end{solution}

\begin{exercise}
Fixed points of $\mathbb{CP}^n$.
\end{exercise}

\begin{solution}
\textbf{Pedagogical Note:} The Lefschetz number $L(f)$ tells us about fixed points. $L(f) \neq 0$ implies at least one fixed point.
(i) $\mathbb{CP}^n$: cohomology is $1, x, x^2, \dots, x^n$. For $n$ even, $L(f)$ is always non-zero ($1+1+\dots+1 = n+1$ or $1-1+1-\dots+1=1$). For $n$ odd, $L(f)$ can be 0.
Example for $n=1$: $\mathbb{CP}^1 \cong S^2$. The antipodal map has no fixed points.
(ii) $\mathbb{CP}^2 \times \mathbb{CP}^2$: $\chi = 9$. However, $L(f)$ can be zero by choosing a map that acts as a "rotation-like" automorphism on the product cohomology.
\end{solution}
