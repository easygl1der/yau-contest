\subsection{Individual Exam}

\begin{exercise}
Let $X$ be the quotient space of $S^{2}$ under the identifications $x \sim -x$ for $x$ in the equator $S^{1}$. Compute the homology groups $H_{n}(X)$. Do the same for $S^{3}$ with antipodal points of the equator $S^{2} \subset S^{3}$ identified.
\end{exercise}

\begin{solution}
\textbf{Geometric Intuition:} Identifying antipodal points on the equator of a sphere $S^2$ is like turning the equator into a projective line $\mathbb{R}P^1$. This "pinches" the sphere along its middle. The resulting space consists of two "caps" (disks) glued to a single circle in a non-standard way. Each cap becomes a projective plane $\mathbb{R}P^2$, and the two projective planes share a common boundary.

\textbf{Homology of $X$:}
Using Mayer-Vietoris with $A \simeq \mathbb{R}P^2$ and $B \simeq \mathbb{R}P^2$ joined along $S^1$:
\begin{itemize}
    \item $H_0(X) \cong \mathbb{Z}$.
    \item $H_1(X)$: The boundary map $\Phi: H_1(S^1) \to H_1(\mathbb{R}P^2) \oplus H_1(\mathbb{R}P^2)$ sends $1 \mapsto (1, 1) \in \mathbb{Z}_2 \oplus \mathbb{Z}_2$. The quotient is $\mathbb{Z}_2$.
    \item $H_2(X)$: The kernel of $\Phi$ is the subgroup of $\mathbb{Z}$ generated by $2$. Thus $H_2(X) \cong \mathbb{Z}$.
\end{itemize}
Summary for $X$: $H_0 \cong \mathbb{Z}, H_1 \cong \mathbb{Z}_2, H_2 \cong \mathbb{Z}$.

\textbf{Homology of $X'$:}
Similarly, for $S^3$, the "equator" $S^2$ becomes $\mathbb{R}P^2$. The space $X'$ is two 3-disks glued along $\mathbb{R}P^2$. Each disk becomes $\mathbb{R}P^3$ with a ball removed.
Summary for $X'$: $H_0 \cong \mathbb{Z}, H_1 \cong \mathbb{Z}_2, H_2 \cong 0, H_3 \cong \mathbb{Z}^2$.
\end{solution}

\begin{exercise}
Let $M \to \mathbb{R}^{3}$ be a graph defined by $z = f(u, v)$ where $\{u, v, z\}$ is a Descartes coordinate system in $\mathbb{R}^{3}$. Suppose that $M$ is a minimal surface. Prove that:
(a) The Gauss curvature $K$ of $M$ can be expressed as
\[ K = \Delta \log\!\left(1 + \frac{1}{W}\right), \qquad W := \sqrt{1 + \left(\frac{\partial f}{\partial u}\right)^{2} + \left(\frac{\partial f}{\partial v}\right)^{2}}, \]
where $\Delta$ denotes the Laplacian with respect to the induced metric on $M$.
(b) If $f$ is defined on the whole $uv$-plane, then $f$ is a linear function (Bernstein theorem).
\end{exercise}

\begin{solution}
\textbf{Geometric Intuition:} Bernstein's Theorem is a remarkable result showing that "flatness" at infinity for a minimal surface forces flatness everywhere. Minimal surfaces are "stretched" as much as possible to minimize area; if you stretch them across the whole infinite plane, they have no room to bend.

\textbf{Proof Sketch:}
(a) For a minimal surface, the coordinates are harmonic and the Gauss map is conformal. The support function $p = \langle N, (0,0,1) \rangle = 1/W$ satisfies a specific differential equation involving the Laplacian and the curvature. The formula $K = \Delta \log(1+1/W)$ reflects this deep connection between the curvature and the orientation of the normal vector.
(b) If $f$ is defined globally, the normal vector $N$ is always "pointing up" ($N_z > 0$). This means the image of the Gauss map avoids a whole hemisphere. A theorem by Osserman states that such a complete minimal surface must be a plane.
\end{solution}

\begin{exercise}
Let $M = \mathbb{R}^{2} / \mathbb{Z}^{2}$ be the two-dimensional torus, $L$ the line $3x = 7y$ in $\mathbb{R}^{2}$, and $S = \pi(L) \subset M$ where $\pi : \mathbb{R}^{2} \to M$ is the projection map. Find a differential form on $M$ which represents the Poincaré dual of $S$.
\end{exercise}

\begin{solution}
\textbf{Geometric Intuition:} The Poincaré dual represents the cycle $S$ as a "flux" density. Integrating a 1-form $\omega$ over the curve $S$ is equivalent to measuring the total "overlap" of $\omega$ with the dual form $\eta$ over the whole torus.

\textbf{Proof:} A parameterization of $S$ is $\gamma(t) = (7t, 3t)$ for $t \in [0, 1]$.
For any basis form $\omega = dx$ or $dy$, we have $\int_S dx = 7$ and $\int_S dy = 3$.
We seek $\eta = a dx + b dy$ such that $\int_M \omega \wedge \eta = \int_S \omega$.
If $\omega = dx$, $\int_M dx \wedge (a dx + b dy) = b \cdot \text{Area}(M) = b$. Thus $b = 7$.
If $\omega = dy$, $\int_M dy \wedge (a dx + b dy) = -a$. Thus $-a = 3$, so $a = -3$.
The dual form is $\eta = -3 dx + 7 dy$.
\end{solution}

\begin{exercise}
Let $p : (\tilde{M}, \tilde{g}) \to (M, g)$ be a Riemannian submersion. This is a submersion $p : \tilde{M} \to M$ such that for each $x \in \tilde{M}$, $dp : \ker^{\perp}(dp) \to T_{p(x)}(M)$ is a linear isometry.
(a) Show that $p$ shortens distances.
(b) If $(\tilde{M}, \tilde{g})$ is complete, so is $(M, g)$.
(c) Show by example that if $(M, g)$ is complete, $(\tilde{M}, \tilde{g})$ may not be complete.
\end{exercise}

\begin{solution}
\textbf{Geometric Intuition:} A Riemannian submersion is like a projection that perfectly preserves lengths in the "horizontal" directions. It's the geometric equivalent of a quotient map. Because you're discarding the "vertical" movement, distances in the quotient space are always shorter than or equal to distances in the original space.

\textbf{Proof:}
(a) For any curve $\tilde{\gamma}$ in $\tilde{M}$, $\|dp(\tilde{\gamma}')\| \le \|\tilde{\gamma}'\|$ because $dp$ is an isometry on the horizontal part and zero on the vertical part. Thus $L(p \circ \tilde{\gamma}) \le L(\tilde{\gamma})$, implying $d(p(x), p(y)) \le d(x, y)$.
(b) Geodesics in $M$ can be horizontally lifted to geodesics in $\tilde{M}$. If $\tilde{M}$ is complete, the lift exists for all time, so the original geodesic in $M$ also exists for all time.
(c) Consider the projection $p: (0, 1) \times S^1 \to S^1$. $S^1$ is complete, but the product is an open cylinder, which is not.
\end{solution}

\begin{exercise}
Let $\Psi : M \to \mathbb{R}^{3}$ be an isometric immersion of a compact surface $M$ into $\mathbb{R}^{3}$. Prove that
\[ \int_{M} H^{2}\,d\sigma \geq 4\pi, \]
where $H$ is the mean curvature of $M$ and $d\sigma$ is the area element of $M$.
\end{exercise}

\begin{solution}
\textbf{Geometric Intuition:} This is the Willmore inequality for the sphere (and a general bound for surfaces). It states that the "average" bending of a closed surface must be at least that of a sphere. The $4\pi$ comes from the fact that the Gauss map of a closed surface must cover the unit sphere at least once, and for a sphere, $H^2 = K$.

\textbf{Proof:} For any point, $H^2 \ge K$. On the part of the surface where $K > 0$ (the "outer" shell), the integral of $H^2$ is bounded by the integral of $K$. For a compact surface in $\mathbb{R}^3$, the Gauss map $N: M \to S^2$ must be surjective, and the integral of $K$ over the region where $K>0$ must be at least the area of $S^2$, which is $4\pi$. Thus $\int H^2 \ge \int_{K>0} K \ge 4\pi$.
\end{solution}

\begin{exercise}
The unit tangent bundle of $S^{2}$ is the subset
\[ T^{1}(S^{2}) = \left\{(p, v) \in \mathbb{R}^{3} \times \mathbb{R}^{3} \mid \|p\| = 1,\; \langle p, v \rangle = 0,\; \|v\| = 1\right\}. \]
Show that it is a smooth submanifold of the tangent bundle $T(S^{2})$ of $S^{2}$ and $T^{1}(S^{2})$ is diffeomorphic to $\mathbb{R}P^{3}$.
\end{exercise}

\begin{solution}
\textbf{Geometric Intuition:} A point in the unit tangent bundle of $S^2$ is an orthonormal pair of vectors $(p, v)$. Adding a third vector $p \times v$ gives a full rotation matrix in $SO(3)$. Thus $T^1(S^2)$ is naturally identified with the rotation group $SO(3)$. Since rotations can be represented by quaternions up to a sign, $SO(3) \cong \mathbb{R}P^3$.

\textbf{Proof Sketch:} The three constraint equations have linearly independent gradients, confirming the submanifold status. The map $(p, v) \mapsto [p, v, p \times v] \in SO(3)$ is a diffeomorphism. Since $SO(3) \cong S^3 / \{\pm I\} \cong \mathbb{R}P^3$, the result follows.
\end{solution}

\begin{exercise}
Compute the fundamental and homology groups of the wedge sum of a circle $S^{1}$ and a torus $T = S^{1} \times S^{1}$.
\end{exercise}

\begin{solution}
\textbf{Geometric Intuition:} A wedge sum $X \vee Y$ is two spaces joined at a single point. Its fundamental group is the free product (you can loop around one or the other in any order), and its homology is the direct sum (the cycles in each stay independent).

\textbf{Result:} $\pi_1 \cong \mathbb{Z} * (\mathbb{Z} \oplus \mathbb{Z})$. $H_0 \cong \mathbb{Z}, H_1 \cong \mathbb{Z}^3, H_2 \cong \mathbb{Z}$.
\end{solution}

\begin{exercise}
Given a properly discontinuous action $F : G \times M \to M$ on a smooth manifold $M$, show that $M/G$ is orientable if and only if $M$ is orientable and $F(g, \cdot)$ preserves the orientation of $M$. Use this statement to show that the Möbius band is not orientable and that $\mathbb{R}P^{n}$ is orientable if and only if $n$ is odd.
\end{exercise}

\begin{solution}
\textbf{Geometric Intuition:} Orientation is about a consistent choice of "handedness." If you have a global choice on $M$, it descends to the quotient only if every symmetry in $G$ respects that choice. If a symmetry "flips" the handedness, the quotient becomes non-orientable.

\textbf{Application:}
1. Möbius band: The defining symmetry $(x, y) \mapsto (x+1, -y)$ has a determinant of $-1$ (a flip), so the result is non-orientable.
2. $\mathbb{R}P^n$: The antipodal map $x \mapsto -x$ flips $n+1$ coordinates. It is orientation-preserving iff $n+1$ is even, so $n$ must be odd.
\end{solution}

\begin{exercise}
(a) Consider the space $Y$ obtained from $S^{2} \times [0, 1]$ by identifying $(x, 0)$ with $(-x, 0)$ and also identifying $(x, 1) \in S^{2} \times \{1\}$ with $(-x, 1)$, for all $x \in S^{2}$. Show that $Y$ is homeomorphic to the connected sum $\mathbb{R}P^{3} \# \mathbb{R}P^{3}$.
(b) Show that $S^{2} \times S^{1}$ is a double cover of the connected sum $\mathbb{R}P^{3} \# \mathbb{R}P^{3}$.
\end{exercise}

\begin{solution}
\textbf{Geometric Intuition:} $\mathbb{R}P^3$ is a 3-ball with antipodal boundary points identified. Taking the connected sum $\mathbb{R}P^3 \# \mathbb{R}P^3$ is like taking a cylinder $S^2 \times [0, 1]$ and "capping" both ends with projective planes.

\textbf{Proof Sketch:}
(a) $S^2 \times [0,1]$ with the specified boundary identifications is exactly the result of removing a ball from two copies of $\mathbb{R}P^3$ and gluing them along the boundary spheres.
(b) The action $(x, \theta) \mapsto (-x, -\theta)$ on $S^2 \times S^1$ is free and reverses orientation in a way that yields the connected sum of two projective spaces.
\end{solution}

\begin{exercise}
Prove that a bi-invariant metric on a Lie group $G$ has nonnegative sectional curvature.
\end{exercise}

\begin{solution}
\textbf{Geometric Intuition:} Bi-invariant metrics on Lie groups are highly symmetric. The curvature is essentially determined by the Lie bracket $[X, Y]$, which measures the non-commutativity. The positive sectional curvature reflects the fact that these groups are "curved towards themselves" by the group operation.

\textbf{Proof:} Using the Levi-Civita connection $\nabla_X Y = \frac{1}{2}[X, Y]$, the sectional curvature is $K(X, Y) = \frac{1}{4}\|[X, Y]\|^2$. Since the norm squared is always non-negative, $K \ge 0$.
\end{solution}

\begin{exercise}
Let $M$ be the upper half-plane $\mathbb{R}_{+}^{2}$ with the metric
\[ ds^{2} = \frac{dx^{2} + dy^{2}}{y^{k}}. \]
For which values of $k$ is $M$ complete?
\end{exercise}

\begin{solution}
\textbf{Geometric Intuition:} Completeness means you can never reach the "edge" in finite time. In the upper half plane, the "edges" are $y=0$ and $y=\infty$. The metric $y^{-k}$ acts as a "speed limit." If $k$ is too large, you reach infinity too fast; if $k$ is too small, you reach zero too fast. $k=2$ is the perfect balance (Hyperbolic space).

\textbf{Result:} $k=2$.
\end{solution}
