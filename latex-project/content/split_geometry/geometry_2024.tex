\begin{exercise}
\label{geometry:2024:1}
The topological space $X$ is obtained by gluing two tetrahedra as illustrated. The resulting complex has 2 tetrahedra, 4 triangles, 2 edges and 1 vertex. Show that $X$ cannot have the homotopy type of a compact manifold without boundary.
\end{exercise}

\end{center}

\end{center}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item \textbf{Euler Characteristic for CW Complexes:} For a 3-dimensional complex, $\chi(X) = V - E + F - T$, where $V$ is the number of vertices, $E$ is edges, $F$ is faces, and $T$ is tetrahedra (3-cells).
    \item \textbf{Link of a Vertex:} In a piecewise linear (PL) complex $X$, the \textbf{Link} of a vertex $v$, denoted $\operatorname{Lk}(v)$, is the set of all simplices that do not contain $v$ but are faces of simplices that do.
    \item \textbf{Manifold Condition:} For a 3-dimensional complex $X$ to be a manifold without boundary, the Link of every vertex must be homeomorphic to a 2-sphere $S^2$.
    \item \textbf{Poincaré Duality:} For a closed, connected, oriented $n$-manifold, the Euler characteristic must satisfy $\chi(M) = 0$ if $n$ is odd.
\end{itemize}

\textbf{Steps:}
\begin{enumerate}
    \item \textbf{Calculate the Euler Characteristic of $X$:} 
    The space $X$ is given to have $V = 1$ vertex, $E = 2$ edges, $F = 4$ triangles (faces), and $T = 2$ tetrahedra. 
    \[ \chi(X) = V - E + F - T = 1 - 2 + 4 - 2 = 1. \]

    \item \textbf{Topological Obstruction from Poincaré Duality:} 
    If $X$ were a compact 3-manifold without boundary (a "closed 3-manifold"), its Euler characteristic $\chi(X)$ would necessarily be zero. This is a direct consequence of Poincaré Duality ($\chi(M) = \sum (-1)^i b_i$ and $b_i = b_{n-i}$ for oriented manifolds; for non-oriented ones, we use $\mathbb{Z}_2$ coefficients or the oriented double cover). Since $1 \neq 0$, $X$ cannot be a manifold.

    \item \textbf{Local Analysis via the Link (The "Pinching" Proof):}
    Let's examine the local structure at the single vertex $v$. For $X$ to be a manifold, the Link of $v$ must be $S^2$. 
    \begin{itemize}
        \item In our complex, the 3-cells are two tetrahedra. Each tetrahedron has 4 faces, 6 edges, and 4 vertices.
        \item When we glue these, we end up with 4 triangles, 2 edges, and 1 vertex.
        \item We can calculate the Euler characteristic of the link $\operatorname{Lk}(v)$ using the fact that for a 3-dimensional CW complex:
        \[ \chi(\operatorname{Lk}(v)) = 2 - \chi(X) \cdot (\text{multiplicity factor}) \]
        More directly, the link of a vertex in a 3-complex is a 2D complex (a union of triangles). The number of vertices, edges, and faces in $\operatorname{Lk}(v)$ corresponds to the number of edges, faces, and 3-cells incident to $v$ in $X$.
        \item In $X$, there are $E_{inc} = 2$ edges incident to $v$, $F_{inc} = 4$ faces incident to $v$, and $T_{inc} = 2$ tetrahedra incident to $v$.
        \item Thus, in $\operatorname{Lk}(v)$, we have $V_{link} = 2$, $E_{link} = 4$, and $F_{link} = 2$.
        \item $\chi(\operatorname{Lk}(v)) = 2 - 4 + 2 = 0$\sidenote{Calculation for the link: $\chi = V - E + F = 2 - 4 + 2 = 0$. Since $\chi(S^2) = 2$, the link cannot be a sphere.}.
        \item A 2D complex with $\chi = 0$ is typically a torus $T^2$ or a Klein bottle, not a sphere $S^2$.
    \end{itemize}
    \item \textbf{Conclusion:} Since the link of the vertex is not a sphere, the space is not locally Euclidean at the vertex. It is a "pseudo-manifold" with a singularity at the point $v$.
\end{enumerate}
\textbf{Intuition:}
Imagine the Euler characteristic as a "scaffolding balance." For a smooth, well-behaved 3D shape like a manifold, the number of "points, lines, and surfaces" must perfectly cancel out the "volume" pieces to reach zero. 

In this problem, the way the tetrahedra are glued together at a single vertex and two edges creates a "pinched" structure. This pinching is so severe that the balance is ruined ($\chi = 1$), meaning the space doesn't have the local uniformity required to be a manifold. It is "too crowded" at that single vertex.
\end{solution}

\begin{exercise}
\label{geometry:2024:2}
Suppose $(M, h)$ is a closed Riemannian manifold with $\sec(h) \leq -1$. Suppose $\Sigma$ is a closed minimal surface with genus $g$ in $(M, h)$. Show that $\operatorname{Area}(\Sigma) \leq 4\pi(g - 1)$.
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item \textbf{Sectional Curvature ($\sec$):} For a 2-plane $P \subset T_pM$ spanned by orthonormal vectors $\{u, v\}$, $\sec(P) = \langle R(u, v)v, u \rangle$.
    \item \textbf{Minimal Surface:} A surface $\Sigma$ such that the trace of its second fundamental form $h$ is zero ($H = \operatorname{tr}(h) = 0$).
    \item \textbf{Gauss Equation:} $K_\Sigma(P) = K_M(P) + \det(h|_P)$. For a surface, $\det(h) = \kappa_1 \kappa_2$, where $\kappa_i$ are principal curvatures.
    \item \textbf{Gauss-Bonnet Theorem:} $\int_\Sigma K_\Sigma \, dA = 2\pi \chi(\Sigma) = 4\pi(1 - g)$ for a closed surface of genus $g$.
\end{itemize}

\textbf{Steps:}
\begin{enumerate}
    \item \textbf{Analyze the Surface Curvature via the Gauss Equation:} 
    Let $\{e_1, e_2\}$ be an orthonormal basis for $T\Sigma$. The Gaussian curvature $K_\Sigma$ is related to the ambient sectional curvature $\sec_M(T\Sigma)$ by:
    \[ K_\Sigma = \sec_M(T\Sigma) + \kappa_1 \kappa_2. \]
    Since $\Sigma$ is a minimal surface, its mean curvature $H = \kappa_1 + \kappa_2$ vanishes. This implies $\kappa_2 = -\kappa_1$.
    \item \textbf{Determine the Curvature Bound:} 
    Substituting $\kappa_2 = -\kappa_1$ into the product:
    \[ \kappa_1 \kappa_2 = \kappa_1 (-\kappa_1) = -\kappa_1^2 \leq 0. \]
    We are given that $\sec_M \leq -1$. Combining these inequalities:
    \[ K_\Sigma = \sec_M(T\Sigma) - \kappa_1^2 \leq -1 - \kappa_1^2 \leq -1. \]
    This shows that the internal curvature of a minimal surface in a negatively curved space is always \textit{more negative} than the ambient curvature\sidenote{$K_\Sigma \leq -1 - \kappa_1^2 \leq -1$. The term $-\kappa_1^2$ represents the "intrinsic penalty" for the surface to bend within the ambient space.}.

    \item \textbf{Integrate over the Surface:} 
    Integrating the inequality $K_\Sigma \leq -1$ over the manifold $\Sigma$:
    \[ \int_\Sigma K_\Sigma \, dA \leq \int_\Sigma (-1) \, dA = -\operatorname{Area}(\Sigma). \]
    By the Gauss-Bonnet Theorem, the left-hand side is $2\pi \chi(\Sigma)$.
    \[ 2\pi(2 - 2g) \leq -\operatorname{Area}(\Sigma). \]
    
    \item \textbf{Final Algebraic Manipulation:} 
    Multiplying by $-1$ and rearranging:
    \[ \operatorname{Area}(\Sigma) \leq -2\pi(2 - 2g) = 4\pi(g - 1). \]
    Since the area is positive, this also implies $g \geq 1$ (a minimal surface in $\sec \leq -1$ cannot be a sphere).
\end{enumerate}
\textbf{Intuition:}
Imagine trying to stretch a rubber sheet (the surface) inside a saddle-shaped room (negative curvature). Because the room is "flaring out" in all directions, the rubber sheet has to cover a lot of "ground" just to close back up on itself and form a loop (a surface of genus $g$). 

However, since the sheet is "minimal" (pulled tight), it tries to take the most efficient path. The math shows that the "tightness" of the sheet combined with the "flare" of the room puts a strict upper limit on how much area the sheet can occupy for a given number of holes ($g$).
\end{solution}

\begin{exercise}
\label{geometry:2024:3}
Show that $\mathsf{SP}^{n}(\mathbb{CP}^{1})$ is homeomorphic to $\mathbb{CP}^{n}$.
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item \textbf{Symmetric Product ($\mathsf{SP}^n$):} The space of unordered sets of $n$ points. If $X$ is a space, $\mathsf{SP}^n(X)$ is the quotient $(X \times \dots \times X) / S_n$, where the symmetric group $S_n$ permutes the coordinates.
    \item \textbf{Projective Line ($\mathbb{CP}^1$):} Points are $[z : w]$, where $z, w \in \mathbb{C}$ are not both zero. This is the Riemann sphere.
    \item \textbf{Fundamental Theorem of Algebra:} Every non-zero homogeneous polynomial $P(z, w)$ of degree $n$ over $\mathbb{C}$ can be factored uniquely (up to scaling) as a product of $n$ linear factors: $P(z, w) = \prod_{i=1}^n (\alpha_i z - \beta_i w)$.
    \item \textbf{Vieta's Formulas:} Relate the coefficients of a polynomial to symmetric functions of its roots.
\end{itemize}

\textbf{Steps:}
\begin{enumerate}
    \item \textbf{Map from $\mathsf{SP}^n(\mathbb{CP}^1)$ to $\mathbb{CP}^n$ (The Root-to-Coefficient Map):} 
    An element in $\mathsf{SP}^n(\mathbb{CP}^1)$ is an unordered set of $n$ points $\{[u_1 : v_1], [u_2 : v_2], \dots, [u_n : v_n]\}$. 
    We define a map $\Phi: \mathsf{SP}^n(\mathbb{CP}^1) \to \mathbb{CP}^n$ by constructing a homogeneous polynomial $P(z, w)$ whose roots are exactly these points:
    \[ P(z, w) = \prod_{i=1}^n (v_i z - u_i w) = a_0 z^n + a_1 z^{n-1} w + \dots + a_n w^n. \]
    The output of the map is the point in $\mathbb{CP}^n$ defined by the coefficients: $[a_0 : a_1 : \dots : a_n]$.
    \item \textbf{Well-definedness and Homogeneity:}
    \begin{itemize}
        \item \textbf{Independence of Order:} Since the product $\prod (v_i z - u_i w)$ is commutative, the result is independent of the ordering of the $n$ points. Thus, the map is well-defined on the quotient by $S_n$.
        \item \textbf{Independence of Scaling:} If we scale a single root $[u_j : v_j]$ to $[\lambda u_j : \lambda v_j]$, the coefficient vector $(a_0, \dots, a_n)$ is scaled by $\lambda$. In $\mathbb{CP}^n$, this represents the same point.
    \end{itemize}

    \item \textbf{Bijectivity via the Fundamental Theorem of Algebra:}
    \begin{itemize}
        \item \textbf{Injectivity:} If two sets of roots produce the same polynomial (up to scaling), the roots themselves must be identical as unordered sets (counting multiplicity).
        \item \textbf{Surjectivity:} Every point $[a_0 : \dots : a_n] \in \mathbb{CP}^n$ corresponds to a non-zero homogeneous polynomial $P(z, w) = \sum a_i z^{n-i} w^i$. By the Fundamental Theorem of Algebra, this polynomial factors into $n$ linear factors $L_i(z, w) = (v_i z - u_i w)$. These $n$ factors define $n$ points $[u_i : v_i] \in \mathbb{CP}^1$.
    \end{itemize}

    \item \textbf{Continuity and Homeomorphism:}
    The coefficients $a_i$ are elementary symmetric polynomials in the roots $(u_i, v_i)$. These are continuous functions. 
    Since $(\mathbb{CP}^1)^n$ is compact, its quotient $\mathsf{SP}^n(\mathbb{CP}^1)$ is compact. 
    $\mathbb{CP}^n$ is a Hausdorff space. A continuous bijection from a compact space to a Hausdorff space is a homeomorphism.
\end{enumerate}

\textbf{Intuition:}
This theorem is essentially a geometric way of saying that "a polynomial is defined by its roots." 

If you have $n$ loose points on a sphere, you can "melt" them down into a single object: a polynomial whose zeros are those points. Conversely, if you have a polynomial, you can "crack it open" to find exactly $n$ points inside. The math tells us that the space of "collections of points" and the space of "polynomial shapes" are actually the exact same shape, just described in different languages.
\end{solution}

\begin{exercise}
\label{geometry:2024:4}
Show that if a complete noncompact Riemannian manifold $M$ does not have the geodesic loops to infinity property, then there is a line in the universal cover $\widetilde{M}$.
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item \textbf{Complete Noncompact Manifold:} A manifold that has no boundary but extends infinitely (like a paraboloid or a plane).
    \item \textbf{Geodesic:} A curve $\gamma(t)$ such that $\nabla_{\dot{\gamma}} \dot{\gamma} = 0$.
    \item \textbf{Line:} A geodesic $\gamma: \mathbb{R} \to M$ that is distance-minimizing between any two of its points: $d(\gamma(t_1), \gamma(t_2)) = |t_1 - t_2|$ for all $t_1, t_2 \in \mathbb{R}$.
    \item \textbf{Geodesic Loops to Infinity Property:} A manifold $M$ has this property if for any compact set $K \subset M$ and any non-trivial class $g \in \pi_1(M)$, there exists a geodesic loop in the class $g$ that does not intersect $K$.
\end{itemize}

\textbf{Steps:}
\begin{enumerate}
    \item \textbf{Formalize the Assumption:} 
    Suppose $M$ lacks the "geodesic loops to infinity" property. There exists a non-trivial element $g \in \pi_1(M)$ and a compact set $K$ such that \textbf{every} geodesic loop in the conjugacy class of $g$ must intersect $K$.
    \item \textbf{Construct a Sequence of Minimal Loops:} 
    Let $x_i$ be a sequence of points in $M$ such that $d(x_i, K) \to \infty$. For each $x_i$, there exists a shortest loop $\gamma_i$ based at $x_i$ representing the class $g$. 
    By our assumption, each $\gamma_i$ must pass through $K$. Let $y_i \in \gamma_i \cap K$.
    \item \textbf{Apply Compactness to Find a Limit Point:}
    Since $K$ is compact, the sequence of points $\{y_i\}$ has a subsequence (which we still call $y_i$) that converges to a point $y_\infty \in K$. 
    For each $i$, we can parameterize $\gamma_i$ by arc length such that $\gamma_i(0) = y_i$. 
    The total length $L_i$ of the loop $\gamma_i$ must go to infinity, because $L_i = \operatorname{length}(\gamma_i) \geq 2 d(x_i, y_i) \to \infty$\sidenote{Since $y_i \in K$ and $x_i \to \infty$, the distance $d(x_i, y_i)$ grows without bound. The loop must go from $y_i$ to $x_i$ and back, so its length is at least twice the distance.}.
    \item \textbf{Lift to the Universal Cover:}
    Lift each loop $\gamma_i$ to a geodesic segment $\tilde{\gamma}_i: [-a_i, b_i] \to \widetilde{M}$ in the universal cover, where $\tilde{\gamma}_i(0) = \tilde{y}_i$ (a lift of $y_i$) and the endpoints satisfy $\tilde{\gamma}_i(b_i) = g \cdot \tilde{\gamma}_i(-a_i)$. 
    Since $y_i \to y_\infty$, we can choose lifts such that $\tilde{y}_i \to \tilde{y}_\infty$.
    \item \textbf{Extract the Limit (Arzelà-Ascoli):}
    The sequence of unit-speed geodesics $\tilde{\gamma}_i$ is equicontinuous and their starting points are bounded. By the Arzelà-Ascoli theorem, a subsequence converges uniformly on compact intervals to an infinite geodesic $\sigma: \mathbb{R} \to \widetilde{M}$ passing through $\tilde{y}_\infty$.
    \item \textbf{Prove $\sigma$ is a Line:}
    Each $\tilde{\gamma}_i$ was a lift of a \textit{shortest} loop in its class, meaning it is a distance-minimizing segment in $\widetilde{M}$ between its endpoints. 
    In the limit, the geodesic $\sigma$ inherits this property: for any $T > 0$, $\sigma|_{[-T, T]}$ is the limit of minimizing segments, and thus is itself a minimizing segment. 
    An infinite geodesic that is minimizing on all sub-segments is a \textbf{line}.
\end{enumerate}

\textbf{Intuition:}
Think of the manifold as having a "narrow waist" or a "hook" represented by $K$. The loop class $g$ is like a rubber band that is caught on this hook. 

Even if you try to pull the basepoint of the rubber band out to the edge of the universe (infinity), the rubber band is still "anchored" to the hook $K$. As you pull harder and farther, the two strands of the rubber band stretch out more and more. In the limit, these two strands become a single straight line that runs from one end of the universe to the other, passing right through our "hook" $K$. This straight line is exactly the "line in the universal cover" we were looking for.
\end{solution}

\begin{exercise}
\label{geometry:2024:5}
(a) Show that $\pi_1$ of an H-space is Abelian.
(b) Show $S^{2022}$ is not an H-space.
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item \textbf{H-Space:} A topological space $X$ with a continuous "multiplication" map $\mu: X \times X \to X$ and an identity element $e$, such that multiplying by $e$ is homotopic to doing nothing (the identity map).
    \item \textbf{Fundamental Group ($\pi_1$):} The group of loops in a space. A group is Abelian if the order of multiplication doesn't matter ($a \cdot b = b \cdot a$).
    \item \textbf{Cohomology Ring ($H^*$):} A way to assign algebraic rings to spaces. The "cup product" ($\cup$) multiplies two cohomology classes.
    \item \textbf{Tensor Product ($\otimes$):} A way to combine algebraic structures. The Künneth formula says $H^*(X \times X) \cong H^*(X) \otimes H^*(X)$.
\end{itemize}

\textbf{Steps:}

\textbf{Part (a): $\pi_1$ is Abelian}
1. \textbf{Setup the Two Operations:} 
In an H-space, we have two ways to "multiply" loops: 
First, the standard concatenation of loops (do loop $\alpha$, then loop $\beta$), denoted $[\alpha] \cdot [\beta]$. 
Second, using the H-space multiplication $\mu$ pointwise: $\mu_*([\alpha], [\beta])$.

2. \textbf{Eckmann-Hilton Argument:} 
Because the identity element $e$ acts as a constant loop, we have $\mu_*([\alpha], [e]) = [\alpha]$ and $\mu_*([e], [\beta]) = [\beta]$. 
Now, we write out the standard product using this property:
\[ [\alpha] \cdot [\beta] = \mu_*([\alpha], [e]) \cdot \mu_*([e], [\beta]). \]
Because $\mu$ is a continuous map, we can "slide" the standard loop concatenation inside the $\mu$ operation:
\[ \dots = \mu_*([\alpha] \cdot [e], [e] \cdot [\beta]) = \mu_*([\alpha], [\beta]). \]
We can do the same trick in reverse order:
\[ [\beta] \cdot [\alpha] = \mu_*([e], [\beta]) \cdot \mu_*([\alpha], [e]) = \mu_*([e] \cdot [\alpha], [\beta] \cdot [e]) = \mu_*([\alpha], [\beta]). \]
Since both orderings equal $\mu_*([\alpha], [\beta])$, they must equal each other: $[\alpha] \cdot [\beta] = [\beta] \cdot [\alpha]$.

\textbf{Part (b): $S^{2022}$ is not an H-space}
1. \textbf{Examine Cohomology:} 
Suppose a sphere of dimension $k$, $S^k$, is an H-space. Let $x$ be the generator of its top cohomology group $H^k(S^k)$. 
The multiplication map $\mu: S^k \times S^k \to S^k$ induces a map backward on cohomology: $\mu^*: H^k(S^k) \to H^k(S^k \times S^k)$.

2. \textbf{Use the Identity Property:} 
Because multiplying by $e$ is the identity, $\mu^*(x)$ must have a specific form in $H^*(S^k) \otimes H^*(S^k)$:
\[ \mu^*(x) = 1 \otimes x + x \otimes 1. \]

3. \textbf{Cup Product Contradiction:} 
We know that in a sphere, the product of the top generator with itself is zero because it would live in dimension $2k$: $x \cup x = 0$. 
So, $\mu^*(x \cup x) = \mu^*(0) = 0$. 
But if we compute it using the formula from Step 2:
\[ \mu^*(x \cup x) = (1 \otimes x + x \otimes 1) \cup (1 \otimes x + x \otimes 1). \]
When we expand this, we must remember the graded commutativity rule: $(a \otimes b) \cup (c \otimes d) = (-1)^{|b||c|} (a \cup c) \otimes (b \cup d)$. 
This gives us: $1 \otimes (x \cup x) + (x \cup 1) \otimes (1 \cup x) + (-1)^{k^2} (x \cup x) \otimes 1 + (x \cup x) \otimes 1$. 
Since $x \cup x = 0$, the only surviving middle term is:
\[ x \otimes x + (-1)^{k^2} x \otimes x = (1 + (-1)^{k^2}) x \otimes x. \]
For this to equal $0$, we must have $1 + (-1)^{k^2} = 0$, which means $k^2$ must be odd, so $k$ must be odd. Since $2022$ is even, $S^{2022}$ cannot be an H-space.

\textbf{Intuition:}
For part (a), having a "multiplication" on the space itself gives you so much freedom to slide loops around each other that they no longer care which one came first.

For part (b), an H-space structure puts a severe restriction on the "algebraic skeleton" (cohomology) of the space. For spheres, this skeleton forces the dimension to be odd because of how minus signs work when you swap coordinates. (A deeper theorem by Adams shows it's actually only possible for dimensions 1, 3, and 7!)
\end{solution}

\begin{exercise}
\label{geometry:2024:6}
(a) Show $S^n(\sqrt{2n})$ is a shrinker. (b) Compact shrinker intersects $S^n(\sqrt{2n})$.
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item \textbf{Shrinker:} A surface that satisfies the equation $H = \frac{1}{2} \langle x, \mathbf{n} \rangle$, where $H$ is the mean curvature, $x$ is the position vector, and $\mathbf{n}$ is the outward normal vector. These surfaces model singularities in mean curvature flow.
    \item \textbf{Mean Curvature ($H$):} The sum of the principal curvatures. For a sphere of radius $R$ in $\mathbb{R}^{n+1}$, $H = n/R$.
    \item \textbf{Maximum Principle:} A fundamental tool in elliptic PDEs. It says that at a local maximum of a function on a closed surface, its Laplacian must be $\leq 0$, and at a minimum, it must be $\geq 0$.
\end{itemize}

\textbf{Steps:}

\textbf{Part (a): $S^n(\sqrt{2n})$ is a shrinker}
1. \textbf{Calculate Mean Curvature:} 
For a sphere of radius $R$ in $\mathbb{R}^{n+1}$, the principal curvatures are all $1/R$. 
The mean curvature is the sum of these $n$ principal curvatures: $H = n/R$.

2. \textbf{Calculate the Normal Vector Term:} 
For a sphere centered at the origin, the outward normal vector $\mathbf{n}$ at a point $x$ points straight out from the origin. Thus, $\mathbf{n} = x/R$. 
Taking the dot product with $x$: $\langle x, \mathbf{n} \rangle = \langle x, x/R \rangle = |x|^2/R = R^2/R = R$.

3. \textbf{Solve the Shrinker Equation:} 
Plug these into the shrinker equation: $H = \frac{1}{2} \langle x, \mathbf{n} \rangle$.
\[ \frac{n}{R} = \frac{1}{2} R \implies R^2 = 2n \implies R = \sqrt{2n}. \]
Thus, the sphere of radius $\sqrt{2n}$ exactly satisfies the equation.

\textbf{Part (b): Compact shrinker intersects $S^n(\sqrt{2n})$}
1. \textbf{Use the Laplacian of the Squared Distance:} 
For any hypersurface $\Sigma$, there's a geometric identity for how the squared distance from the origin changes along the surface:
\[ \Delta_\Sigma |x|^2 = 2n - 2H \langle x, \mathbf{n} \rangle. \]
Since $\Sigma$ is a shrinker, we substitute $H = \frac{1}{2} \langle x, \mathbf{n} \rangle$:
\[ \Delta_\Sigma |x|^2 = 2n - 2\left(\frac{1}{2} \langle x, \mathbf{n} \rangle\right) \langle x, \mathbf{n} \rangle = 2n - \langle x, \mathbf{n} \rangle^2. \]

2. \textbf{Apply the Maximum Principle at the Furthest Point:} 
Let $x_{max}$ be the point on $\Sigma$ furthest from the origin. Here, $|x|^2$ is maximized. 
By the Maximum Principle, $\Delta_\Sigma |x|^2 \leq 0$ at this point. 
Also, at the furthest point, the surface must be tangent to a sphere centered at the origin, meaning the position vector $x_{max}$ is perfectly aligned with the normal vector $\mathbf{n}$. So, $\langle x_{max}, \mathbf{n} \rangle^2 = |x_{max}|^2 = R_{max}^2$.
Plugging this into our identity:
\[ 0 \geq \Delta_\Sigma |x|^2 = 2n - R_{max}^2 \implies R_{max}^2 \geq 2n \implies R_{max} \geq \sqrt{2n}. \]

3. \textbf{Apply the Maximum Principle at the Closest Point:} 
Similarly, let $x_{min}$ be the point closest to the origin. Here, $\Delta_\Sigma |x|^2 \geq 0$ and $\langle x_{min}, \mathbf{n} \rangle^2 = R_{min}^2$.
\[ 0 \leq \Delta_\Sigma |x|^2 = 2n - R_{min}^2 \implies R_{min}^2 \leq 2n \implies R_{min} \leq \sqrt{2n}. \]

4. \textbf{Conclusion:} 
The closest point to the origin is inside or on the sphere $S^n(\sqrt{2n})$, and the furthest point is outside or on it. Therefore, the continuous surface $\Sigma$ must intersect this sphere somewhere.

\textbf{Intuition:}
A "shrinker" is a surface trying to collapse into the origin, but its geometry precisely balances this inward pull. 

The sphere $S^n(\sqrt{2n})$ is the perfect "equilibrium" bubble. If you have any other closed, lumpy shrinker, it can't be entirely tucked inside this equilibrium bubble (it would collapse too fast), nor can it sit entirely outside it (it wouldn't collapse fast enough). It must straddle the boundary of the equilibrium sphere to maintain the delicate balance required by the shrinker equation.
\end{solution}
