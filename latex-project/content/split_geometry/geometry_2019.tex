\begin{exercise}
Let $\mathrm{Conf}$ be the following submanifold of $\mathbb{C}^{n}$:
\[
\mathrm{Conf}_{n} = \left\{\left(z_{1}, z_{2}, \dots, z_{n}\right) \in \mathbb{C}^{n} \mid z_{i} \neq z_{j} \text{ for any } i \neq j \right\}.
\]
For every pair $(i, j)$ with $i \neq j$, we define the complex valued 1-form
\[
\omega_{ij} := \frac{dz_{i} - dz_{j}}{z_{i} - z_{j}}.
\]
(a) Show that for any $i \neq j$, $\omega_{ij}$ represents a non-zero de Rham cohomology class in $H^{1}(\mathrm{Conf}_{n}, \mathbb{C})$.
(b) Show that for any pairwise distinct indices $i, j, k$,
\[
\omega_{ij} \wedge \omega_{jk} + \omega_{jk} \wedge \omega_{ki} + \omega_{ki} \wedge \omega_{ij} = 0.
\]
\end{exercise}

\begin{solution}
(a) Let $i, j$ be fixed. Consider the map $f: \mathrm{Conf}_n \to \mathbb{C}^*$ given by $f(z) = z_i - z_j$. This map is well-defined because $z_i \neq z_j$. Then $\omega_{ij} = f^*(dz/z)$. 
The form $dz/z$ is the generator of $H^1(\mathbb{C}^*, \mathbb{C}) \cong \mathbb{C}$. To show $\omega_{ij}$ is non-zero in $H^1(\mathrm{Conf}_n)$, we integrate it over a loop $\gamma$. 
Let $\gamma(t)$ be a path where $z_i$ and $z_j$ rotate around each other once, keeping other coordinates constant (e.g., $z_i(t) = e^{it}, z_j(t) = 0$). 
Then $\int_\gamma \omega_{ij} = \int_0^{2\pi} \frac{ie^{it}}{e^{it}} dt = 2\pi i \neq 0$. Thus $[\omega_{ij}] \neq 0$.

(b) This is the \textit{Arnold relation} in the cohomology of the braid group. 
Let $a = z_i - z_j$, $b = z_j - z_k$, then $z_i - z_k = a+b$.
Note $dz_i - dz_j = da$, $dz_j - dz_k = db$, and $dz_k - dz_i = -(da+db)$.
Then:
$\omega_{ij} = \frac{da}{a}$, $\omega_{jk} = \frac{db}{b}$, $\omega_{ki} = \frac{da+db}{a+b}$.
The expression is:
\[
\frac{da}{a} \wedge \frac{db}{b} + \frac{db}{b} \wedge \frac{da+db}{a+b} + \frac{da+db}{a+b} \wedge \frac{da}{a}
\]
\[
= \frac{1}{ab} da \wedge db + \frac{1}{b(a+b)} db \wedge da + \frac{1}{a(a+b)} db \wedge da \quad (\text{since } da \wedge da = 0, db \wedge db = 0)
\]
\[
= \left( \frac{1}{ab} - \frac{1}{b(a+b)} - \frac{1}{a(a+b)} \right) da \wedge db
\]
The coefficient is:
\[
\frac{(a+b) - a - b}{ab(a+b)} = 0.
\]
Thus the sum is zero.
\end{solution}

\begin{exercise}
Let $M$ be a compact oriented manifold of (real) dimension 4. Consider the following symmetric bilinear form on $H^{2}(M)$
\[
H^{2}(M) \times H^{2}(M) \to \mathbb{R}, \quad ([\alpha], [\beta]) \mapsto \int_{M} \alpha \wedge \beta.
\]
Let $\tau(M)$ be the signature of this bilinear form. Compute $\tau(M)$ for $M = S^{4}, \mathbb{CP}^{2}$ and $S^{2} \times S^{2}$.
\end{exercise}

\begin{solution}
\textbf{Intuition:} The signature is a topological invariant for 4-manifolds that describes the intersection of 2-cycles.

1. $M = S^4$: $H^2(S^4) = 0$. The bilinear form is on a zero-dimensional space. Thus $\tau(S^4) = 0$.
2. $M = \mathbb{CP}^2$: $H^2(\mathbb{CP}^2) \cong \mathbb{Z}$, generated by the hyperplane class $h$. 
The intersection form is given by $h \cdot h = 1$ (the fundamental class of $\mathbb{CP}^2$). 
The matrix is $(1)$, which has one positive eigenvalue. Thus $\tau(\mathbb{CP}^2) = 1$.
3. $M = S^2 \times S^2$: $H^2(S^2 \times S^2)$ is generated by $a = [S^2 \times \{pt\}]$ and $b = [\{pt\} \times S^2]$.
The intersections are $a \cdot a = 0$ (self-intersection of a factor), $b \cdot b = 0$, and $a \cdot b = 1$ (the factors intersect at one point).
The intersection matrix is $\begin{pmatrix} 0 & 1 \\ 1 & 0 \end{pmatrix}$.
The eigenvalues are $+1$ and $-1$. Thus $\tau(S^2 \times S^2) = 1 - 1 = 0$.
\end{solution}

\begin{exercise}
Let $X = \mathbb{R}^{4} / \sim$, with identified relations given. Compute $H_{1}(X, \mathbb{Z})$.
\end{exercise}

\begin{solution}
The space $X$ is a quotient of $\mathbb{R}^4$ by a discrete group of isometries $G$. $H_1(X, \mathbb{Z})$ is the abelianization of $\pi_1(X) \cong G$.
The generators of $G$ are:
$e_1(x) = (x_1+1, x_2, x_3, x_4)$
$e_2(x) = (x_1, x_2+1, x_3, x_4)$
$e_3(x) = (x_1, x_2+x_4, x_3+1, x_4)$
$e_4(x) = (x_1, x_2, x_3, x_4+1)$
We check the commutators $[e_i, e_j] = e_i e_j e_i^{-1} e_j^{-1}$.
- $e_1, e_2, e_4$ all commute with each other because they are translations.
- For $e_3$ and $e_4$: 
$e_3 e_4 (x) = e_3(x_1, x_2, x_3, x_4+1) = (x_1, x_2+x_4+1, x_3+1, x_4+1)$
$e_4 e_3 (x) = e_4(x_1, x_2+x_4, x_3+1, x_4) = (x_1, x_2+x_4, x_3+1, x_4+1)$
Thus $e_3 e_4 = e_2 e_4 e_3$, or $[e_3, e_4] = e_2$.
- For $e_3$ and $e_1$ or $e_2$: they commute.
So the relations are $[e_3, e_4] = e_2$ and all other pairs commute.
The abelianization $G_{ab}$ is $G$ with the additional relation $e_2 = 1$.
Thus $H_1(X, \mathbb{Z}) \cong \mathbb{Z} \langle e_1, e_3, e_4 \rangle \cong \mathbb{Z}^3$.
\end{solution}

\begin{exercise}
(a) Show that $\operatorname{tr}[f(R^{E})]$ is closed.
(b) Show that $\operatorname{tr}[f(R^{E})] - \operatorname{tr}[f(\widetilde{R}^{E})]$ is exact.
\end{exercise}

\begin{solution}
(a) By the Bianchi identity, $d_\nabla R^E = 0$. For any matrix power, $d \operatorname{tr}((R^E)^k) = k \operatorname{tr}(d_\nabla R^E \wedge (R^E)^{k-1}) = 0$. Since $f$ is a polynomial, the trace is a sum of these closed forms.
(b) Consider a path of connections $\nabla_t = (1-t)\widetilde{\nabla} + t\nabla$. Let $R_t$ be the curvature. 
The \textit{Transgression Formula} states $\operatorname{tr}(f(R)) - \operatorname{tr}(f(\widetilde{R})) = d \int_0^1 k \operatorname{tr}(\dot{\nabla}_t \wedge f'(R_t)) dt$, where $\dot{\nabla}_t = \nabla - \widetilde{\nabla}$ is an End(E)-valued 1-form. This defines the required $\omega$.
\end{solution}

\begin{exercise}
(a) Prove Bochner's formula. (b) Solve $\Delta\ln f + Ef^{2} = 1$ on $S^2$.
\end{exercise}

\begin{solution}
(a) Standard calculation using Ricci identity: $\nabla_i \nabla_j \nabla_k u - \nabla_j \nabla_i \nabla_k u = R_{ijk}^l \nabla_l u$.
(b) This is related to the Uniformization Theorem and the Nirenberg problem. On $S^2$, the equation $\Delta \ln f + K' f^2 = K$ describes a change of metric $g' = f^2 g$ with new curvature $K'$. For $K'=E$, the solution is unique up to Mobius transformations. The specific form follows from the fact that the first eigenfunctions $\phi$ of $S^2$ correspond to coordinates $x, y, z$.
\end{solution}

\subsection{Team Exam}

\begin{exercise}
Is $TS^{2}$ diffeomorphic to $S^{2} \times \mathbb{R}^{2}$?
\end{exercise}

\begin{solution}
No. $TS^2$ is the tangent bundle. If it were diffeomorphic to $S^2 \times \mathbb{R}^2$, it would be a trivial bundle. 
However, by the \textit{Hairy Ball Theorem}, $S^2$ does not admit any nowhere-zero continuous vector field. A trivial bundle $S^2 \times \mathbb{R}^2$ admits 2 linearly independent nowhere-zero sections (e.g., constant ones). Thus the bundles are not isomorphic, and the manifolds are not diffeomorphic as bundles. Topologically, $TS^2$ and $S^2 \times \mathbb{R}^2$ have different Euler classes ($\chi(S^2)=2$ vs $0$).
\end{solution}

\begin{exercise}
Compute $\dim(V/W) = H^1(\mathbb{R}^3 \setminus X, \mathbb{R})$.
\end{exercise}

\begin{solution}
The space $V$ consists of closed forms and $W$ consists of exact forms. Thus $V/W \cong H^1(\mathbb{R}^3 \setminus X, \mathbb{R})$.
(a) $X = \{0\}$. $\mathbb{R}^3 \setminus \{0\} \simeq S^2$. $H^1(S^2) = 0$. Dimension 0.
(b) $X = \{z\text{-axis}\}$. $\mathbb{R}^3 \setminus L \simeq S^1$. $H^1(S^1) = \mathbb{R}$. Dimension 1.
(c) $X = \{z=0\text{ plane}\}$. $\mathbb{R}^3 \setminus P$ is two disjoint half-spaces, each contractible. $H^1 \cong H^1(pt) \oplus H^1(pt) = 0$. Dimension 0.
\end{solution}

\begin{exercise}
$F: T^2 \to T^2$, $\deg F = 1$, $F^2 = Id$, no fixed points. Prove $F^* = Id$ on $H^1$.
\end{exercise}

\begin{solution}
Let $T^2 = \mathbb{R}^2 / \mathbb{Z}^2$. Since $F$ has no fixed points and $F^2=Id$, the quotient $M = T^2 / \langle F \rangle$ is a smooth compact surface. 
The covering map $p: T^2 \to M$ is a 2-sheeted unbranched cover.
By the Riemann-Hurwitz formula (or Euler characteristic properties): $\chi(T^2) = 2 \chi(M)$.
Since $\chi(T^2) = 0$, we have $\chi(M) = 0$. Thus $M$ is also a torus (it must be oriented because $F$ has degree 1).
The map $F$ acts on $H^1(T^2) \cong \mathbb{Z}^2$. Since $F^2=Id$, the eigenvalues of $F^*$ are $\pm 1$.
The Lefschetz fixed point number is $L(F) = \sum (-1)^k \operatorname{tr}(F^* |_{H^k})$.
$L(F) = \operatorname{tr}(F^* |_{H^0}) - \operatorname{tr}(F^* |_{H^1}) + \operatorname{tr}(F^* |_{H^2})$.
Since $F$ has no fixed points, $L(F) = 0$.
$F^*$ on $H^0$ is $1$. $F^*$ on $H^2$ is $\deg(F) = 1$.
So $1 - \operatorname{tr}(F^* |_{H^1}) + 1 = 0 \implies \operatorname{tr}(F^* |_{H^1}) = 2$.
For a $2 \times 2$ matrix with eigenvalues $\pm 1$, the trace is 2 only if both eigenvalues are 1.
Thus $F^* = Id$.
\end{solution}

\begin{exercise}
(a) Compute dims of $SU(n), SO(n)$. (b) Compute $\pi_1$.
\end{exercise}

\begin{solution}
(a) $\dim(U(n)) = n^2$. $SU(n)$ is a hypersurface $\det A = 1$, so $\dim(SU(n)) = n^2 - 1$.
$\dim(O(n)) = \frac{n(n-1)}{2}$. $SO(n)$ is the identity component, so same dimension $\frac{n(n-1)}{2}$.
(b) $SU(n) \to S^{2n-1}$ fibration shows $\pi_1(SU(n)) = 0$ for $n \geq 2$.
$SO(n)$ has $\pi_1(SO(2)) = \mathbb{Z}$ and $\pi_1(SO(n)) = \mathbb{Z}_2$ for $n \geq 3$ (the cover is $Spin(n)$).
\end{solution}

\begin{exercise}
Killing fields on weakly negative manifolds.
\end{exercise}

\begin{solution}
(a) Use Bochner formula for Killing fields. A Killing field $X$ satisfies $\nabla_i X_j + \nabla_j X_i = 0$. Then $\Delta X_i = - R_i^j X_j$.
The Hessian of $f = |X|^2/2$ is $\nabla_j \nabla_k (X_i X^i / 2) = \nabla_j (X^i \nabla_k X_i) = \nabla_j X^i \nabla_k X_i + X^i \nabla_j \nabla_k X_i$.
Using the Killing property and curvature identity, we get $(\text{Hess } f)(V, V) = |\nabla_V X|^2 - R(V, X, X, V)$.
(b) If $M$ is weakly negative, $-R(V, X, X, V) \geq 0$ and can be positive. At a maximum of $f$, Hess $f \leq 0$. But the formula gives Hess $f = |\nabla X|^2 - R(V, X, X, V) \geq 0$. This forces $\nabla X = 0$ and $R(V, X, X, V) = 0$. By weak negativity, $X$ must be zero.
\end{solution}
