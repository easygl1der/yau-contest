\chapter{Algebra \& Number Theory}

This chapter outlines the core prerequisite knowledge for the Algebra \& Number Theory track. Based on the past 16 years of problems, the content heavily emphasizes abstract algebra, representation theory, field extensions, and algebraic number theory.

\section{Group Theory and Representations}
\textbf{Prerequisite Knowledge}: Sets, mappings, equivalence relations, basic linear algebra (matrices, vector spaces).

\begin{definition}[Group Action \& Orbits]
A group action of a group $G$ on a set $X$ is a homomorphism from $G$ to the symmetric group on $X$, denoted by $g \cdot x$. The orbit of $x \in X$ is $O_x = \{g \cdot x \mid g \in G\}$. The stabilizer of $x \in X$ is the subgroup $G_x = \{g \in G \mid g \cdot x = x\}$. The Orbit-Stabilizer Theorem states that $|O_x| = [G : G_x] = |G| / |G_x|$.
\end{definition}

\begin{definition}[Sylow $p$-subgroups]
Let $G$ be a finite group of order $p^n m$, where $p$ is a prime and $p \nmid m$. A subgroup of order $p^n$ is called a Sylow $p$-subgroup of $G$.
\end{definition}

\begin{theorem}[Sylow Theorems]
Let $G$ be a group of order $p^n m$ with $\gcd(p, m) = 1$.
\begin{enumerate}
    \item $G$ possesses at least one Sylow $p$-subgroup.
    \item Any two Sylow $p$-subgroups are conjugate in $G$.
    \item The number of Sylow $p$-subgroups, denoted $n_p$, satisfies $n_p \equiv 1 \pmod p$ and $n_p \mid m$.
\end{enumerate}
\end{theorem}

\begin{example}
Let $G = S_3$, the symmetric group on 3 elements, with order 6. A Sylow 2-subgroup has order 2, and there are exactly 3 such subgroups (generated by transpositions). A Sylow 3-subgroup has order 3, and there is exactly 1 such subgroup (generated by 3-cycles), which must be a normal subgroup.
\end{example}

\begin{definition}[Semi-direct Product]
Given groups $N$ and $H$, and a group homomorphism $\phi: H \to \mathrm{Aut}(N)$, the semi-direct product $N \rtimes_\phi H$ is the set $N \times H$ equipped with the binary operation:
\[ (n_1, h_1)(n_2, h_2) = (n_1 \phi(h_1)(n_2), h_1 h_2) \]
\end{definition}

\begin{remark}
Semi-direct products are frequently used to construct non-abelian groups or classify groups of certain orders. For instance, the dihedral group $D_{2n}$ can be expressed as $\mathbb{Z}_n \rtimes \mathbb{Z}_2$.
\end{remark}

\begin{definition}[Solvable and Free Groups]
A group $G$ is solvable if it has a subnormal series $\{1\} = G_0 \triangleleft G_1 \triangleleft \dots \triangleleft G_k = G$ where each quotient $G_{i+1}/G_i$ is abelian. A free group on a set $S$ is a group where every element is uniquely represented as a reduced word of elements from $S$ and their inverses.
\end{definition}

\begin{definition}[Linear Representation \& Character]
A linear representation of a group $G$ on a vector space $V$ over a field $F$ is a group homomorphism $\rho \colon G \to GL(V)$. The character of $\rho$ is the class function $\chi_\rho(g) = \mathrm{tr}(\rho(g))$.
\end{definition}

\begin{theorem}[Orthogonality of Characters]
Let $\chi_1, \dots, \chi_k$ be the irreducible characters of a finite group $G$ over $\mathbb{C}$. They form an orthonormal basis for the space of class functions on $G$ under the inner product:
\[ \langle \chi_i, \chi_j \rangle = \frac{1}{|G|} \sum_{g \in G} \chi_i(g) \overline{\chi_j(g)} = \delta_{ij}. \]
Furthermore, $\sum_{i=1}^k \chi_i(1)^2 = |G|$, and the number of irreducible representations equals the number of conjugacy classes of $G$.
\end{theorem}

\begin{example}
For $G = S_4$, there are 5 conjugacy classes, so 5 irreducible representations. The dimensions satisfy $1^2 + 1^2 + 2^2 + 3^2 + 3^2 = 24 = |S_4|$. The irreducible representations include the trivial, sign, standard (dim 3), its twist by sign (dim 3), and a 2-dimensional representation coming from the quotient $S_4 / V_4 \cong S_3$.
\end{example}

\section{Field Theory and Galois Theory}
\textbf{Prerequisite Knowledge}: Polynomial rings, irreducibility (Eisenstein's criterion), vector spaces.

\begin{definition}[Splitting Field]
A splitting field of a polynomial $f(x) \in F[x]$ is an extension field $E/F$ such that $f(x)$ factors completely into linear factors in $E[x]$ and $E$ is generated over $F$ by the roots of $f(x)$.
\end{definition}

\begin{definition}[Separability and Galois Extension]
An algebraic extension $E/F$ is separable if the minimal polynomial of every element in $E$ has no multiple roots in its splitting field. The extension $E/F$ is Galois if it is finite, normal (is a splitting field of some family of polynomials), and separable.
\end{definition}

\begin{theorem}[Fundamental Theorem of Galois Theory]
Let $E/F$ be a finite Galois extension with Galois group $G = \mathrm{Gal}(E/F)$. There is an inclusion-reversing bijection between the intermediate fields $K$ ($F \subseteq K \subseteq E$) and the subgroups $H$ of $G$, given by $K \mapsto \mathrm{Gal}(E/K)$ and $H \mapsto E^H$ (the fixed field of $H$). Moreover, $K/F$ is Galois if and only if $\mathrm{Gal}(E/K) \triangleleft \mathrm{Gal}(E/F)$.
\end{theorem}

\begin{example}
Consider the polynomial $f(x) = x^4 - 2$ over $\mathbb{Q}$. Its roots are $\pm \sqrt[4]{2}$ and $\pm i\sqrt[4]{2}$. The splitting field is $E = \mathbb{Q}(\sqrt[4]{2}, i)$. The degree of $E/\mathbb{Q}$ is 8. The Galois group $\mathrm{Gal}(E/\mathbb{Q})$ is isomorphic to the dihedral group $D_4$.
\end{example}

\begin{theorem}[Kummer and Artin-Schreier Theory]
Let $F$ be a field containing a primitive $n$-th root of unity (with characteristic not dividing $n$). Any cyclic extension of degree $n$ over $F$ is of the form $F(\sqrt[n]{a})$ for some $a \in F$. In characteristic $p$, a cyclic extension of degree $p$ is generated by a root of an Artin-Schreier polynomial $x^p - x - a$ for some $a \in F$.
\end{theorem}

\section{Ring and Module Theory}
\textbf{Prerequisite Knowledge}: Basic definition of rings, ideals, homomorphisms.

\begin{definition}[PID and UFD]
A Principal Ideal Domain (PID) is an integral domain where every ideal is generated by a single element. A Unique Factorization Domain (UFD) is an integral domain where every non-zero non-unit factors uniquely (up to associates and ordering) into irreducible elements. Note that every PID is a UFD.
\end{definition}

\begin{definition}[Tensor Product]
Let $M$ and $N$ be modules over a commutative ring $R$. The tensor product $M \otimes_R N$ is an $R$-module generated by symbols $m \otimes n$ subject to relations ensuring the map $(m,n) \mapsto m \otimes n$ is $R$-bilinear.
\end{definition}

\begin{theorem}[Structure Theorem for Finitely Generated Modules over a PID]
Let $R$ be a PID and $M$ a finitely generated $R$-module. Then $M \cong R^r \oplus R/(a_1) \oplus \dots \oplus R/(a_m)$, where $r \geq 0$ is the free rank of $M$, and $a_1, \dots, a_m$ are non-zero non-units in $R$ satisfying $a_1 \mid a_2 \mid \dots \mid a_m$ (invariant factors).
\end{theorem}

\begin{remark}
Applying this theorem to the $F[x]$-module $V$ induced by a linear transformation $T: V \to V$ leads directly to the Rational Canonical Form and Jordan Canonical Form of $T$.
\end{remark}

\begin{theorem}[Snake Lemma]
Given a commutative diagram of $R$-modules with exact rows:
\[
\begin{tikzcd}
& M' \arrow[r] \arrow[d, "f'"] & M \arrow[r] \arrow[d, "f"] & M'' \arrow[r] \arrow[d, "f''"] & 0 \\
0 \arrow[r] & N' \arrow[r] & N \arrow[r] & N'' &
\end{tikzcd}
\]
There exists a natural long exact sequence:
\[ \ker f' \to \ker f \to \ker f'' \xrightarrow{\delta} \mathrm{coker} f' \to \mathrm{coker} f \to \mathrm{coker} f'' \]
where $\delta$ is the connecting homomorphism.
\end{theorem}

\section{Commutative Algebra and Algebraic Geometry}
\textbf{Prerequisite Knowledge}: Ideals (prime, maximal), polynomial rings, topological spaces.

\begin{definition}[Coordinate Ring and Zariski Topology]
For an affine algebraic variety $V \subseteq \mathbb{A}^n(k)$ defined by an ideal $I \subseteq k[x_1, \dots, x_n]$, its coordinate ring is $k[V] = k[x_1, \dots, x_n]/I$. The Zariski topology on $\mathbb{A}^n(k)$ has closed sets defined as the zero loci of collections of polynomials.
\end{definition}

\begin{definition}[Integral Closure and Dedekind Domains]
Let $R \subseteq S$ be rings. An element $s \in S$ is integral over $R$ if it is a root of a monic polynomial with coefficients in $R$. $R$ is integrally closed if it equals its integral closure in its field of fractions. A Dedekind domain is a Noetherian, integrally closed integral domain in which every non-zero prime ideal is maximal.
\end{definition}

\begin{theorem}[Hilbert's Nullstellensatz]
Let $k$ be an algebraically closed field. If $I$ is an ideal in $k[x_1, \dots, x_n]$, then the ideal of polynomials vanishing on the algebraic set $V(I)$ is precisely the radical of $I$, denoted $\sqrt{I}$. Maximal ideals in $k[x_1, \dots, x_n]$ are in bijective correspondence with the points of $\mathbb{A}^n(k)$.
\end{theorem}

\begin{example}
Let $I = (x^2 - y^3) \subseteq \mathbb{C}[x,y]$. The coordinate ring $\mathbb{C}[x,y]/(x^2-y^3)$ represents the plane curve with a cusp at the origin. It is an integral domain but is not integrally closed.
\end{example}

\begin{theorem}[Hilbert's Theorem 90]
Let $E/F$ be a finite cyclic Galois extension with Galois group $G = \langle \sigma \rangle$. 
\begin{enumerate}
    \item (Multiplicative) For $\alpha \in E^\times$, the norm $N_{E/F}(\alpha) = 1$ if and only if $\alpha = \beta / \sigma(\beta)$ for some $\beta \in E^\times$.
    \item (Additive) For $\alpha \in E$, the trace $\mathrm{Tr}_{E/F}(\alpha) = 0$ if and only if $\alpha = \beta - \sigma(\beta)$ for some $\beta \in E$.
\end{enumerate}
\end{theorem}

\section{$p$-adic Analysis and Algebraic Number Theory}
\textbf{Prerequisite Knowledge}: Metric spaces, congruences, absolute values.

\begin{definition}[$p$-adic Valuation and Field]
For a prime $p$, the $p$-adic valuation $v_p(n)$ of a non-zero integer $n$ is the exponent of the highest power of $p$ dividing $n$. The $p$-adic absolute value is $|x|_p = p^{-v_p(x)}$. The field of $p$-adic numbers $\mathbb{Q}_p$ is the completion of $\mathbb{Q}$ with respect to this absolute value. The ring of $p$-adic integers is $\mathbb{Z}_p = \{ x \in \mathbb{Q}_p \mid |x|_p \le 1 \}$.
\end{definition}

\begin{definition}[Ring of Integers]
The ring of integers $\mathcal{O}_K$ of an algebraic number field $K$ (a finite extension of $\mathbb{Q}$) is the integral closure of $\mathbb{Z}$ in $K$. $\mathcal{O}_K$ is always a Dedekind domain.
\end{definition}

\begin{theorem}[Hensel's Lemma]
Let $f(x) \in \mathbb{Z}_p[x]$. If there exists an $a \in \mathbb{Z}_p$ such that $f(a) \equiv 0 \pmod p$ and $f'(a) \not\equiv 0 \pmod p$, then there exists a unique $\alpha \in \mathbb{Z}_p$ such that $f(\alpha) = 0$ and $\alpha \equiv a \pmod p$.
\end{theorem}

\begin{example}
Consider $x^2 + 1 = 0$ in $\mathbb{Z}_5$. Since $x=2$ satisfies $2^2 + 1 = 5 \equiv 0 \pmod 5$ and the derivative $2(2) = 4 \not\equiv 0 \pmod 5$, Hensel's lemma ensures that $-1$ has a square root in $\mathbb{Q}_5$. This illustrates that $\mathbb{Q}_p$ contains the Teichmüller lifts for roots of unity in $\mathbb{F}_p$.
\end{example}

\begin{theorem}[Chevalley-Warning Theorem]
Let $f_1, \dots, f_k \in \mathbb{F}_p[x_1, \dots, x_n]$ be polynomials over a finite field. If $\sum \deg(f_i) < n$, then the number of common zeros of $f_1, \dots, f_k$ in $\mathbb{F}_p^n$ is a multiple of $p$. In particular, if there is a trivial zero $(0,\dots,0)$, there must be at least one non-trivial zero.
\end{theorem}

\section{Linear and Lie Algebras}
\textbf{Prerequisite Knowledge}: Vector spaces, linear transformations, matrices, determinants.

\begin{definition}[Orthogonal and Symplectic Groups]
For a vector space $V$ with a non-degenerate bilinear form $B$, the group of linear automorphisms preserving $B$ is the orthogonal group $O(V)$ if $B$ is symmetric, and the symplectic group $Sp(V)$ if $B$ is skew-symmetric.
\end{definition}

\begin{definition}[Positive Semidefinite Matrix]
A real symmetric matrix $A$ is positive semidefinite if $x^T A x \geq 0$ for all vectors $x \in \mathbb{R}^n$. Equivalently, all its eigenvalues are non-negative.
\end{definition}

\begin{theorem}[Cayley-Hamilton Theorem]
Every square matrix over a commutative ring satisfies its own characteristic equation. If $p(\lambda) = \det(\lambda I - A)$ is the characteristic polynomial of an $n \times n$ matrix $A$, then $p(A) = 0$.
\end{theorem}

\begin{example}
If $A$ and $B$ are commuting matrices ($AB = BA$) over an algebraically closed field, they can be simultaneously upper-triangularized.
\end{example}

\begin{definition}[Lie Algebra and Exponential Map]
A Lie algebra $\mathfrak{g}$ is a vector space equipped with a bilinear, alternating bracket $[\cdot, \cdot]$ satisfying the Jacobi identity. The exponential map $\exp: \mathfrak{g} \to G$ connects a Lie algebra to its corresponding Lie group $G$. For matrix Lie groups like $GL_n(\mathbb{R})$, the map is given by the matrix exponential $\exp(X) = \sum_{k=0}^\infty X^k/k!$.
\end{definition}

\begin{remark}
If $X$ is a skew-symmetric matrix ($X^T = -X$), then $e^X$ is an orthogonal matrix. This provides the correspondence between the Lie algebra $\mathfrak{so}(n)$ and the Lie group $SO(n)$.
\end{remark}

\section{Combinatorics and Elementary Number Theory}
\textbf{Prerequisite Knowledge}: Permutations, combinations, primes, modular arithmetic.

\begin{definition}[Generating Function]
The ordinary generating function for a sequence $(a_0, a_1, a_2, \dots)$ is the formal power series $A(x) = \sum_{n=0}^{\infty} a_n x^n$.
\end{definition}

\begin{definition}[Primitive Root]
An integer $g$ is a primitive root modulo $n$ if the multiplicative order of $g$ modulo $n$ is equal to $\varphi(n)$ (Euler's totient function). This means the powers of $g$ generate the group $(\mathbb{Z}/n\mathbb{Z})^\times$.
\end{definition}

\begin{theorem}[Law of Quadratic Reciprocity]
Let $p$ and $q$ be distinct odd primes. The Legendre symbol $\left(\frac{p}{q}\right)$ satisfies:
\[ \left(\frac{p}{q}\right) \left(\frac{q}{p}\right) = (-1)^{\frac{p-1}{2}\frac{q-1}{2}}. \]
Furthermore, $\left(\frac{-1}{p}\right) = (-1)^{\frac{p-1}{2}}$ and $\left(\frac{2}{p}\right) = (-1)^{\frac{p^2-1}{8}}$.
\end{theorem}

\begin{theorem}[Chinese Remainder Theorem]
Let $m_1, m_2, \dots, m_k$ be pairwise coprime positive integers. For any integers $a_1, a_2, \dots, a_k$, the system of simultaneous congruences $x \equiv a_i \pmod{m_i}$ for $1 \le i \leq k$ has a unique solution modulo $M = m_1 m_2 \dots m_k$.
\end{theorem}

\begin{example}
Find $x$ such that $x \equiv 2 \pmod 3$ and $x \equiv 3 \pmod 5$. Since 3 and 5 are coprime, the CRT guarantees a unique solution modulo 15. The solution is $x = 8$.
\end{example}
