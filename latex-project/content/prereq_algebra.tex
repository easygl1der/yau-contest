\chapter{Algebra \& Number Theory}

This chapter outlines the core prerequisite knowledge for the Algebra \& Number Theory track. Based on the past 16 years of Yau Contest problems, the content heavily emphasizes abstract algebra, representation theory, field extensions, commutative algebra, and algebraic number theory. The granularity below reflects the specific theorems and technical lemmas explicitly tested in past exams.

\section{Group Theory}

\textbf{Prerequisite Knowledge}: Sets, mappings, equivalence relations, basic linear algebra (matrices, vector spaces), definitions of groups, normal subgroups, homomorphisms, and quotient groups.

\begin{theorem}[Sylow Theorems and $GL_n(\mathbb{F}_p)$]
Let $G$ be a finite group of order $p^n m$ with $\gcd(p, m) = 1$. Then $G$ has at least one Sylow $p$-subgroup of order $p^n$, any two are conjugate, and the number of them $n_p$ satisfies $n_p \equiv 1 \pmod p$ and $n_p \mid m$.

\textbf{Application:} The general linear group $GL_n(\mathbb{F}_p)$ over a finite field has order
\[ |GL_n(\mathbb{F}_p)| = (p^n - 1)(p^n - p) \dots (p^n - p^{n-1}) = p^{\frac{n(n-1)}{2}} \prod_{i=1}^n (p^i - 1). \]
The subgroup of upper unitriangular matrices (ones on the diagonal) has order $p^{n(n-1)/2}$, making it a Sylow $p$-subgroup of $GL_n(\mathbb{F}_p)$.
\end{theorem}

\begin{example}
In $GL_3(\mathbb{F}_7)$, the order is $7^3 (7^3-1)(7^2-1)(7-1)$. The Sylow $7$-subgroup has order $7^3$ and consists of the matrices $\begin{pmatrix} 1 & a & b \\ 0 & 1 & c \\ 0 & 0 & 1 \end{pmatrix}$ where $a,b,c \in \mathbb{F}_7$.
\end{example}

\begin{theorem}[Goursat's Lemma]
Let $G_1, G_2$ be groups and $H \le G_1 \times G_2$ a subgroup such that the projections $p_1: H \to G_1$ and $p_2: H \to G_2$ are surjective. Let $N_1 = \ker(p_2) \triangleleft G_1$ and $N_2 = \ker(p_1) \triangleleft G_2$. Then the image of $H$ in $G_1/N_1 \times G_2/N_2$ is the graph of an isomorphism $G_1/N_1 \xrightarrow{\sim} G_2/N_2$.
\end{theorem}

\begin{definition}[Semi-direct Product]
Given groups $N$ and $H$, and a homomorphism $\phi: H \to \mathrm{Aut}(N)$, the semi-direct product $N \rtimes_\phi H$ is the set $N \times H$ with the group operation $(n_1, h_1)(n_2, h_2) = (n_1 \cdot \phi(h_1)(n_2), h_1 h_2)$. 
A group $G$ is a semi-direct product of $N$ and $H$ if $N \triangleleft G$, $H \le G$, $N \cap H = \{1\}$, and $G = NH$. 

\textbf{Application:} Used to classify groups of order $pq, p^2q, pqr$. For example, groups of order 99 ($3^2 \times 11$) and 150 ($2 \times 3 \times 5^2$) are analyzed by finding a normal Sylow subgroup and a complement.
\end{definition}

\begin{remark}[Classification of finite subgroups of $GL_2(\mathbb{R})$]
Any finite subgroup of $GL_2(\mathbb{R})$ is isomorphic to either a cyclic group $C_n$ or a dihedral group $D_{2n}$. This is because any finite subgroup of $GL_n(\mathbb{R})$ can be conjugated into the orthogonal group $O(n)$, and the finite subgroups of $O(2)$ are precisely $C_n$ (rotations) and $D_{2n}$ (rotations and reflections).
\end{remark}

\begin{definition}[Pr\"ufer $p$-group]
The Pr\"ufer $p$-group $\mathbb{Z}(p^\infty)$ is the unique $p$-primary group in which every element has $p$-th roots. It can be realized as the quotient group $\mathbb{Z}[1/p] / \mathbb{Z}$, or as the subgroup of $\mathbb{C}^\times$ consisting of all $p^n$-th roots of unity. It is divisible, and its proper subgroups are totally ordered by inclusion.
\end{definition}

\section{Representation Theory}

\begin{definition}[Character and Orthogonality]
For a complex linear representation $\rho: G \to GL(V)$ of a finite group $G$, its character is $\chi_V(g) = \mathrm{tr}(\rho(g))$. Characters of irreducible representations form an orthonormal basis for the space of class functions under the inner product:
\[ \langle \chi, \psi \rangle_G = \frac{1}{|G|} \sum_{g \in G} \chi(g) \overline{\psi(g)}. \]
Furthermore, $|G| = \sum_{i=1}^k \dim(V_i)^2$, and the number of irreducible representations $k$ equals the number of conjugacy classes. A character $\chi$ is real-valued if and only if $g$ is conjugate to $g^{-1}$ for all $g \in G$.
\end{definition}

\begin{remark}[Faithful Representations]
A finite group $G$ has a faithful 1-dimensional complex representation if and only if $G$ is cyclic. More generally, a finite abelian group has a faithful $n$-dimensional complex representation if and only if it can be generated by $n$ elements.
\end{remark}

\begin{theorem}[Frobenius Reciprocity]
Let $H \le G$ be a subgroup. Let $W$ be a representation of $H$ and $V$ a representation of $G$. The induced representation $\mathrm{Ind}_H^G(W)$ satisfies:
\[ \langle \chi_{\mathrm{Ind}_H^G(W)}, \chi_V \rangle_G = \langle \chi_W, \mathrm{Res}_H^G(\chi_V) \rangle_H. \]
\end{theorem}

\begin{theorem}[Molien's Theorem]
Let $\rho: G \to GL(V)$ be a representation of a finite group $G$ over $\mathbb{C}$. The dimension of the space of $G$-invariant homogeneous polynomials of degree $d$ on $V$, denoted $S^d(V^*)^G$, has the generating function (Poincar\'e series):
\[ \sum_{d=0}^\infty \dim(S^d(V^*)^G) t^d = \frac{1}{|G|} \sum_{g \in G} \frac{1}{\det(I - t \rho(g))}. \]
\end{theorem}

\begin{example}[Representations of $SL_2$]
The irreducible complex representations of $SL_2(\mathbb{C})$ are precisely the symmetric powers $V_k = \mathrm{Sym}^k(\mathbb{C}^2)$ for $k \ge 0$, where $\dim(V_k) = k+1$. The Clebsch-Gordan formula decomposes their tensor product:
\[ V_n \otimes V_m \cong \bigoplus_{j=0}^{\min(n,m)} V_{n+m-2j}. \]
\end{example}

\section{Field Theory and Galois Theory}

\begin{theorem}[Fundamental Theorem of Galois Theory]
For a finite Galois extension $E/F$ with Galois group $G$, there is a bijection between intermediate fields $K$ ($F \subseteq K \subseteq E$) and subgroups $H \le G$, given by $H \mapsto E^H$ and $K \mapsto \mathrm{Gal}(E/K)$. The extension $K/F$ is Galois if and only if $\mathrm{Gal}(E/K) \triangleleft G$.
\end{theorem}

\begin{theorem}[Galois Group of Quartic Polynomials]
Let $f(x) \in F[x]$ be an irreducible separable quartic with roots $\alpha_1, \alpha_2, \alpha_3, \alpha_4$. The resolvent cubic is $R(x) = (x - (\alpha_1\alpha_2 + \alpha_3\alpha_4))(x - (\alpha_1\alpha_3 + \alpha_2\alpha_4))(x - (\alpha_1\alpha_4 + \alpha_2\alpha_3))$. The Galois group $G \le S_4$ is determined by whether $R(x)$ splits in $F$:
\begin{itemize}
    \item If $R(x)$ is irreducible over $F$, $G = S_4$ or $A_4$ (distinguished by the discriminant).
    \item If $R(x)$ splits completely over $F$, $G \cong V_4$ (the Klein four-group).
    \item If $R(x)$ has exactly one root in $F$, $G \cong D_4$ or $\mathbb{Z}/4\mathbb{Z}$.
\end{itemize}
\end{theorem}

\begin{example}
Consider $x^4-2$ over $\mathbb{Q}$. Its roots are $\pm \sqrt[4]{2}, \pm i\sqrt[4]{2}$. The splitting field is $\mathbb{Q}(\sqrt[4]{2}, i)$ which has degree 8. Thus the Galois group is $D_4$. The resolvent cubic $R(x)$ has one root in $\mathbb{Q}$, consistent with the $D_4$ classification.
\end{example}

\begin{theorem}[Kummer and Artin-Schreier Theory]
Let $F$ be a field.
\begin{itemize}
    \item \textbf{Kummer Theory:} If $\mathrm{char}(F) \nmid n$ and $F$ contains a primitive $n$-th root of unity, any cyclic extension of degree $n$ is of the form $F(\sqrt[n]{a})$ for some $a \in F$. For example, a $V_4$ extension over $F$ containing a primitive 2nd root of unity (i.e. $\mathrm{char}(F) \neq 2$) exists if and only if the quotient group $F^\times / F^{\times 2}$ has order at least 4.
    \item \textbf{Artin-Schreier Theory:} If $\mathrm{char}(F) = p$, any cyclic extension of degree $p$ is the splitting field of $x^p - x - a$ for some $a \in F$. The roots are $\alpha, \alpha+1, \dots, \alpha+p-1$.
\end{itemize}
\end{theorem}

\begin{remark}[Imperfect Fields]
Over a field $F$ of characteristic $p > 0$, the polynomial $x^p - a$ is either completely reducible (if $a$ has a $p$-th root in $F$) or irreducible (if it does not). This is a standard test for imperfect fields. For example, $x^p-t$ is irreducible over $\mathbb{F}_p(t)$ but its derivative is identically zero, making the extension $\mathbb{F}_p(t^{1/p}) / \mathbb{F}_p(t)$ purely inseparable.
\end{remark}

\begin{theorem}[Infinite Galois Theory]
Let $E/F$ be an algebraic Galois extension (normal and separable, possibly infinite degree). The Galois group $G = \mathrm{Gal}(E/F)$ is a profinite group (an inverse limit of finite groups $\varprojlim \mathrm{Gal}(E_i/F)$) endowed with the Krull topology. The Galois correspondence holds between intermediate fields and \textbf{closed} subgroups of $G$. Open subgroups of finite index correspond to finite intermediate extensions.
\end{theorem}

\begin{remark}[Transcendental Extensions]
The extension $k((t)) / k(t)$, where $k((t))$ is the field of formal Laurent series and $k(t)$ is the field of rational functions, is transcendental and of infinite degree.
\end{remark}

\section{Ring and Module Theory}

\begin{definition}[Rings of Fractions and Local Rings]
For a commutative ring $R$ and a multiplicatively closed set $S \subset R$, the localization $S^{-1}R$ consists of fractions $r/s$. If $\mathfrak{p}$ is a prime ideal, $R_{\mathfrak{p}} = (R \setminus \mathfrak{p})^{-1}R$ is a local ring (a ring with a unique maximal ideal). 
\end{definition}

\begin{theorem}[Fitting's Lemma]
Let $M$ be an $R$-module of finite length (e.g., a finite-dimensional vector space or a finite abelian group), and $f \in \mathrm{End}_R(M)$. Then there exists an integer $n \ge 1$ such that $M = \ker(f^n) \oplus \mathrm{im}(f^n)$. If $M$ is indecomposable, $f$ is either nilpotent or an automorphism.
\end{theorem}

\begin{theorem}[Nakayama's Lemma]
Let $M$ be a finitely generated $R$-module and $I$ an ideal of $R$ contained in the Jacobson radical of $R$ (the intersection of all maximal ideals). If $IM = M$, then $M = 0$. 

\textbf{Corollary:} If $M$ is a finitely generated module over a local ring $(R, \mathfrak{m})$, and $m_1, \dots, m_k$ map to a basis of the $R/\mathfrak{m}$-vector space $M/\mathfrak{m}M$, then $m_1, \dots, m_k$ generate $M$ as an $R$-module.
\end{theorem}

\begin{definition}[Rees Algebra]
For an ideal $I \subseteq R$, the Rees algebra (or blow-up algebra) is the graded ring $\mathcal{R}(I) = \bigoplus_{n \ge 0} I^n t^n \subseteq R[t]$. It is Noetherian if $R$ is Noetherian.
\end{definition}

\section{Commutative Algebra and Algebraic Geometry}

\begin{definition}[Integral Extensions and Normalization]
An element $s \in S$ is integral over $R \subseteq S$ if it is a root of a monic polynomial in $R[x]$. The ring $R$ is \textit{integrally closed} (or normal) in $S$ if it equals its integral closure in $S$. An integral domain $R$ is normal if it is integrally closed in its fraction field.

\textbf{Example:} The coordinate ring $k[x,y]/(y^2 - x^3)$ is an integral domain but not normal. Its fraction field is $k(t)$ where $t = y/x$. Since $t^2 = y^2/x^2 = x \in R$, $t$ is integral over the ring, but $t \notin R$. The integral closure is $k[t]$.
\end{definition}

\begin{theorem}[Going-Up Theorem]
Let $S$ be an integral extension of $R$. For any chain of prime ideals $\mathfrak{p}_1 \subsetneq \mathfrak{p}_2$ in $R$ and a prime $\mathfrak{q}_1 \subset S$ lying over $\mathfrak{p}_1$ ($\mathfrak{q}_1 \cap R = \mathfrak{p}_1$), there exists a prime $\mathfrak{q}_2 \subset S$ such that $\mathfrak{q}_1 \subsetneq \mathfrak{q}_2$ and $\mathfrak{q}_2$ lies over $\mathfrak{p}_2$.
\end{theorem}

\begin{theorem}[Hilbert's Nullstellensatz]
Let $k$ be an algebraically closed field. 
\begin{enumerate}
    \item \textbf{Weak Nullstellensatz:} The maximal ideals of $k[x_1, \dots, x_n]$ are precisely $(x_1-a_1, \dots, x_n-a_n)$ for $a_i \in k$.
    \item \textbf{Strong Nullstellensatz:} For any ideal $I \subset k[x_1, \dots, x_n]$, the vanishing ideal of the algebraic set $V(I)$ is the radical $I(V(I)) = \sqrt{I}$.
\end{enumerate}
\end{theorem}

\begin{theorem}[Hilbert's Theorem 90]
Let $E/F$ be a finite Galois extension with Galois group $G$. The group cohomology $H^1(G, E^\times) = 1$. In the cyclic case $G = \langle \sigma \rangle$, an element $\alpha \in E$ has norm $N_{E/F}(\alpha) = 1$ if and only if $\alpha = \beta / \sigma(\beta)$ for some $\beta \in E^\times$.
\end{theorem}

\section{$p$-adic Analysis and Algebraic Number Theory}

\begin{definition}[$p$-adic Numbers and Valuations]
For a prime $p$, the $p$-adic valuation $v_p(x)$ of $x \in \mathbb{Q}^\times$ is the power of $p$ in its prime factorization. The absolute value is $|x|_p = p^{-v_p(x)}$. The completion of $\mathbb{Q}$ is $\mathbb{Q}_p$, and the local ring of integers is $\mathbb{Z}_p = \{x \in \mathbb{Q}_p : |x|_p \le 1\}$. 

\textbf{Structure:} $\mathbb{Z}_p^\times \cong \mathbb{Z}/(p-1)\mathbb{Z} \times (1 + p\mathbb{Z}_p)$ for $p>2$. The finite subgroup $\mathbb{Z}/(p-1)\mathbb{Z}$ consists of the $(p-1)$-th roots of unity, lifted uniquely from $\mathbb{F}_p^\times$ via the \textbf{Teichm\"uller lift}.
\end{definition}

\begin{definition}[$p$-adic Exponential and Logarithm]
The $p$-adic exponential series $\exp(x) = \sum_{n=0}^\infty \frac{x^n}{n!}$ converges for $x \in p\mathbb{Z}_p$ (if $p>2$) or $x \in 4\mathbb{Z}_2$ (if $p=2$). The logarithm $\log(1+x) = \sum_{n=1}^\infty (-1)^{n-1} \frac{x^n}{n}$ converges for $x \in p\mathbb{Z}_p$. These maps provide an isomorphism between the additive group $p\mathbb{Z}_p$ and the multiplicative group $1+p\mathbb{Z}_p$ (for $p>2$).
\end{definition}

\begin{theorem}[Hensel's Lemma]
Let $f(x_1, \dots, x_n) \in \mathbb{Z}_p[x_1, \dots, x_n]$. If there exists $\mathbf{a} \in \mathbb{Z}_p^n$ such that $f(\mathbf{a}) \equiv 0 \pmod p$ and the gradient vector $\nabla f(\mathbf{a}) \not\equiv \mathbf{0} \pmod p$, then $\mathbf{a}$ can be lifted to a true root $\mathbf{\alpha} \in \mathbb{Z}_p^n$ such that $f(\mathbf{\alpha}) = 0$ and $\mathbf{\alpha} \equiv \mathbf{a} \pmod p$.
\end{theorem}

\begin{definition}[Dedekind Domain and Fractional Ideals]
A Dedekind domain is a Noetherian, integrally closed integral domain of Krull dimension 1 (every non-zero prime ideal is maximal). The ring of integers $\mathcal{O}_K$ of a number field $K$ is a Dedekind domain. In a Dedekind domain, every non-zero fractional ideal is invertible and factors uniquely into prime ideals. Furthermore, invertible fractional ideals are exactly the finitely generated \textbf{projective modules} of rank 1.
\end{definition}

\begin{theorem}[Chevalley-Warning Theorem]
Let $f_1, \dots, f_k \in \mathbb{F}_p[x_1, \dots, x_n]$ be polynomials over a finite field. If the sum of their degrees is strictly less than the number of variables ($\sum \deg(f_i) < n$), then the number of common zeros of $f_1, \dots, f_k$ in $\mathbb{F}_p^n$ is a multiple of $p$.
\end{theorem}

\begin{remark}[Unramified Extensions of Local Fields]
For a local field $K$ (e.g., $\mathbb{Q}_p$ or a finite extension thereof) with residue field $\mathbb{F}_q$, there is a unique unramified extension of degree $f$, which corresponds to the extension $\mathbb{F}_{q^f} / \mathbb{F}_q$ of the residue fields. The Galois group of the unramified extension is cyclic, generated by the Frobenius automorphism.
\end{remark}

\section{Linear Algebras and Matrix Theory}

\begin{theorem}[Cayley-Hamilton and Minimal Polynomials]
For an $n \times n$ matrix $A$ over a commutative ring, the characteristic polynomial $p(x) = \det(xI - A)$ satisfies $p(A) = 0$. The minimal polynomial $m(x)$ divides $p(x)$ and has exactly the same roots as $p(x)$ (ignoring multiplicities) over the algebraic closure.

\textbf{Application:} Two commuting diagonalizable matrices $A$ and $B$ ($AB=BA$) can be \textbf{simultaneously diagonalized}. Two commuting matrices over $\mathbb{C}$ can be \textbf{simultaneously upper-triangularized}. If $\det(A^2+B^2)=0$ for commuting real matrices, $A+iB$ has a zero eigenvalue.
\end{theorem}

\begin{theorem}[Matrix Decompositions: Cartan and Iwasawa]
Let $G = GL_n(\mathbb{R})$.
\begin{itemize}
    \item \textbf{Cartan Decomposition ($G = KAK$):} Any $g \in G$ can be written as $g = k_1 a k_2$, where $k_1, k_2 \in O(n)$ and $a$ is a diagonal matrix with positive entries. This is equivalent to the Singular Value Decomposition (SVD).
    \item \textbf{Iwasawa Decomposition ($G = KAN$):} Any $g \in G$ can be written as $g = k a n$, where $k \in O(n)$, $a$ is diagonal with positive entries, and $n$ is upper unitriangular. This is related to Gram-Schmidt orthogonalization.
\end{itemize}
\end{theorem}

\begin{definition}[Orthogonal and Symplectic Groups]
Let $V$ be a vector space with a non-degenerate bilinear form $B$. 
\begin{itemize}
    \item If $B$ is symmetric, the isometry group is the orthogonal group $O(V)$. Its Lie algebra $\mathfrak{so}(V)$ consists of skew-symmetric matrices $X^T = -X$. The exponential $e^X$ is orthogonal.
    \item If $B$ is skew-symmetric, the isometry group is the symplectic group $Sp(V)$.
\end{itemize}
\end{definition}

\begin{theorem}[Cartan-Dieudonn\'e Theorem]
Every orthogonal transformation in the orthogonal group $O(V)$ of an $n$-dimensional regular metric space over a field (of characteristic $\neq 2$) can be expressed as the composition of at most $n$ reflections.
\end{theorem}

\begin{definition}[Weyl Algebra]
The Weyl algebra over a field $k$ is the non-commutative ring of differential operators on polynomial rings. For one variable, it is generated by $x$ (multiplication by $x$) and $\partial = \frac{d}{dx}$, subject to the canonical commutation relation $[\partial, x] = \partial x - x \partial = 1$.
\end{definition}

\section{Polynomials and Combinatorics}

\begin{theorem}[Mason-Stothers Theorem]
Let $a(t), b(t), c(t) \in \mathbb{C}[t]$ be relatively prime polynomials satisfying $a + b = c$. If they are not all constant, then
\[ \max(\deg(a), \deg(b), \deg(c)) \le N_0(abc) - 1 \]
where $N_0(P)$ is the number of \textit{distinct} roots of $P$. This is the polynomial analogue of the ABC conjecture.
\end{theorem}

\begin{theorem}[Newton's Identities]
Let $x_1, \dots, x_n$ be variables. The elementary symmetric polynomials $e_k$ and the power sum symmetric polynomials $p_k = \sum_{i=1}^n x_i^k$ are related by:
\[ k e_k = \sum_{i=1}^k (-1)^{i-1} e_{k-i} p_i. \]
For example: $e_1 = p_1$, $2e_2 = e_1 p_1 - p_2$, and $3e_3 = e_2 p_1 - e_1 p_2 + p_3$.
\end{theorem}

\begin{definition}[Cyclotomic Polynomials]
The $n$-th cyclotomic polynomial $\Phi_n(x) = \prod_{(k,n)=1, 1 \le k \le n} (x - e^{2\pi i k / n})$ is the minimal polynomial of a primitive $n$-th root of unity over $\mathbb{Q}$. It has degree $\varphi(n)$, has integer coefficients, and is irreducible over $\mathbb{Z}[x]$. Modulo a prime $p \nmid n$, $\Phi_n(x)$ splits into distinct irreducible factors of degree $d$, where $d$ is the multiplicative order of $p$ modulo $n$.
\end{definition}
