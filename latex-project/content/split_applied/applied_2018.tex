\begin{exercise}
\label{applied:2018:1}
We consider convection-diffusion equation:
\[
u_{t}+au_{x}=bu_{xx},\quad 0\leq x<1
\]
with initial condition $u(x,0)=f(x)$ and periodic boundary condition, where $a$ and $b>0$ are constants.

The first order IMEX time discretization and second order central spatial discretization give:
\[
\frac{u_{j}^{n+1}-u_{j}^{n}}{\Delta t}+a\frac{u_{j+1}^{n}-u_{j-1}^{n}}{2\Delta x}=b\frac{u_{j+1}^{n+1}-2u_{j}^{n+1}+u_{j-1}^{n+1}}{\Delta x^{2}}.
\]

Prove the scheme is $L^{2}$ stable under very mild time step restriction $\Delta t\leq c$ with constant $c$ independent of $\Delta x$. Can you determine dependency of $c$ on $a$ and $b$?
\end{exercise}

\begin{solution}
We use Fourier analysis (von Neumann stability analysis). Let $u_j^n = \hat{u}^n e^{i k j \Delta x}$, where $k$ is the wavenumber. Substituting this into the discretization scheme, we have:
\[
\frac{\hat{u}^{n+1}-\hat{u}^n}{\Delta t} + a \hat{u}^n \frac{e^{ik\Delta x} - e^{-ik\Delta x}}{2\Delta x} = b \hat{u}^{n+1} \frac{e^{ik\Delta x} - 2 + e^{-ik\Delta x}}{\Delta x^2}.
\]
Let $\theta = k\Delta x \in [0, 2\pi]$. Using the identities $e^{i\theta} - e^{-i\theta} = 2i\sin\theta$ and $e^{i\theta} - 2 + e^{-i\theta} = 2(\cos\theta - 1) = -4\sin^2(\theta/2)$, we obtain:
\[
\frac{\hat{u}^{n+1}-\hat{u}^n}{\Delta t} + i a \hat{u}^n \frac{\sin\theta}{\Delta x} = - \frac{4b\hat{u}^{n+1}}{\Delta x^2} \sin^2(\theta/2).
\]
Rearranging terms to solve for the amplification factor $G(\theta) = \hat{u}^{n+1}/\hat{u}^n$:
\[
\hat{u}^{n+1} \left( 1 + \frac{4b\Delta t}{\Delta x^2} \sin^2(\theta/2) \right) = \hat{u}^n \left( 1 - i \frac{a\Delta t}{\Delta x} \sin\theta \right).
\]
\[
G(\theta) = \frac{1 - i \frac{a\Delta t}{\Delta x} \sin\theta}{1 + \frac{4b\Delta t}{\Delta x^2} \sin^2(\theta/2)}.
\]
The scheme is $L^2$ stable if $|G(\theta)| \leq 1 + O(\Delta t)$ for all $\theta$. For absolute stability, we require $|G(\theta)|^2 \leq 1$:
\[
|G(\theta)|^2 = \frac{1 + \frac{a^2 \Delta t^2}{\Delta x^2} \sin^2\theta}{\left( 1 + \frac{4b\Delta t}{\Delta x^2} \sin^2(\theta/2) \right)^2} \leq 1.
\]
Using $\sin\theta = 2\sin(\theta/2)\cos(\theta/2)$, we have $\sin^2\theta = 4\sin^2(\theta/2)\cos^2(\theta/2)$. The inequality becomes:
\[
1 + \frac{4a^2 \Delta t^2}{\Delta x^2} \sin^2(\theta/2)\cos^2(\theta/2) \leq 1 + \frac{8b\Delta t}{\Delta x^2} \sin^2(\theta/2) + \frac{16b^2 \Delta t^2}{\Delta x^4} \sin^4(\theta/2).
\]
Subtracting 1 and dividing by $\frac{4\Delta t}{\Delta x^2} \sin^2(\theta/2)$ (assuming $\sin(\theta/2) \neq 0$):
\[
a^2 \Delta t \cos^2(\theta/2) \leq 2b + \frac{4b^2 \Delta t}{\Delta x^2} \sin^2(\theta/2).
\]
Since the term $\frac{4b^2 \Delta t}{\Delta x^2} \sin^2(\theta/2)$ is non-negative, a sufficient condition for the inequality to hold for all $\theta$ is:
\[
a^2 \Delta t \leq 2b \implies \Delta t \leq \frac{2b}{a^2}.
\]
Thus, the constant $c = 2b/a^2$ is independent of $\Delta x$.
\end{solution}

\begin{exercise}
\label{applied:2018:2}
[Velocity-Verlet method]

\begin{enumerate}
\item[(a)] Recast the Newtonian formula $q''(t)=-\nabla V(q)$ into a Hamiltonian system and show the corresponding map on phase space is symplectic.
\item[(b)] Show that the velocity-Verlet method:
\[
p_{n+1/2}=p_{n}-\frac{\Delta t}{2}\nabla V(q_{n});
\]
\[
q_{n+1}=q_{n}+\Delta tp_{n+1/2};
\]
\[
p_{n+1}=p_{n+1/2}-\frac{\Delta t}{2}\nabla V(q_{n+1})
\]
is symplectic and is second order accurate.
\end{enumerate}

Hint: Let $\mathbf{u}(t)=(p(t),q(t))$ be solution of Hamiltonian system with initial data $\mathbf{u}_{0}=(p_{0},q_{0})$. Flow map $\varphi_{t}:\mathbb{R}^{d}\times\mathbb{R}^{d}\to\mathbb{R}^{d}\times\mathbb{R}^{d}$, $\varphi_{t}(\mathbf{u}_{0})=\mathbf{u}(t)$. Flow is symplectic if Jacobian $\Phi_{t}(\mathbf{u}_{0})=\partial\mathbf{u}(t)/\partial\mathbf{u}_{0}$ satisfies $\Phi_{t}(\mathbf{u}_{0})^{T}J\Phi_{t}(\mathbf{u}_{0})=J$ where $J=\begin{pmatrix}0&I\\-I&0\end{pmatrix}$.
\end{exercise}

\begin{solution}
(a) Let $p(t) = q'(t)$. Then $p'(t) = q''(t) = -\nabla V(q)$. The Hamiltonian is $H(p, q) = \frac{1}{2} |p|^2 + V(q)$. The Hamilton's equations are:
\[
\dot{q} = \frac{\partial H}{\partial p} = p, \quad \dot{p} = -\frac{\partial H}{\partial q} = -\nabla V(q).
\]
This system can be written as $\dot{\mathbf{u}} = J \nabla H(\mathbf{u})$, where $\mathbf{u} = (p, q)^T$. The flow $\varphi_t$ of a Hamiltonian system is symplectic. Specifically, the Jacobian $\Phi_t$ satisfies $\dot{\Phi}_t = J \nabla^2 H \Phi_t$. One can verify that:
\[
\frac{d}{dt} (\Phi_t^T J \Phi_t) = \dot{\Phi}_t^T J \Phi_t + \Phi_t^T J \dot{\Phi}_t = (J \nabla^2 H \Phi_t)^T J \Phi_t + \Phi_t^T J (J \nabla^2 H \Phi_t).
\]
Using $J^T = -J$ and $J^2 = -I$:
\[
= -\Phi_t^T \nabla^2 H J J \Phi_t + \Phi_t^T J J \nabla^2 H \Phi_t = \Phi_t^T \nabla^2 H \Phi_t - \Phi_t^T \nabla^2 H \Phi_t = 0.
\]
Since $\Phi_0 = I$ and $I^T J I = J$, we have $\Phi_t^T J \Phi_t = J$.

(b) Symplecticity: The velocity-Verlet method is a composition of three maps $\Psi_{\Delta t} = \Psi^{(3)}_{\Delta t/2} \circ \Psi^{(2)}_{\Delta t} \circ \Psi^{(1)}_{\Delta t/2}$:
\begin{enumerate}
    \item $\Psi^{(1)}_{h}: (p, q) \mapsto (p - h\nabla V(q), q)$ (Momentum kick)
    \item $\Psi^{(2)}_{h}: (p, q) \mapsto (p, q + hp)$ (Drift)
    \item $\Psi^{(3)}_{h}: (p, q) \mapsto (p - h\nabla V(q), q)$ (Momentum kick)
\end{enumerate}
The Jacobian of $\Psi^{(1)}_h$ is $M_1 = \begin{pmatrix} I & -h\nabla^2 V \\ 0 & I \end{pmatrix}$. We check $M_1^T J M_1 = J$:
\[
\begin{pmatrix} I & 0 \\ -h\nabla^2 V & I \end{pmatrix} \begin{pmatrix} 0 & I \\ -I & 0 \end{pmatrix} \begin{pmatrix} I & -h\nabla^2 V \\ 0 & I \end{pmatrix} = \begin{pmatrix} 0 & I \\ -I & -h\nabla^2 V \end{pmatrix} \begin{pmatrix} I & -h\nabla^2 V \\ 0 & I \end{pmatrix} = \begin{pmatrix} 0 & I \\ -I & 0 \end{pmatrix} = J.
\]
Similarly, the Jacobian of $\Psi^{(2)}_h$ is $M_2 = \begin{pmatrix} I & 0 \\ hI & I \end{pmatrix}$, which satisfies $M_2^T J M_2 = J$. Since the composition of symplectic maps is symplectic, velocity-Verlet is symplectic.

Accuracy: The method is a Strang splitting of the Hamiltonian flow. For $\dot{\mathbf{u}} = (\mathcal{A}+\mathcal{B})\mathbf{u}$, the scheme is $e^{h\mathcal{B}/2} e^{h\mathcal{A}} e^{h\mathcal{B}/2} = e^{h(\mathcal{A}+\mathcal{B}) + O(h^3)}$, which is second-order accurate. Alternatively, Taylor expansion of $q_{n+1}$ matches exact $q(t_{n+1})$ up to $O(\Delta t^3)$, and $p_{n+1}$ matches $p(t_{n+1})$ up to $O(\Delta t^3)$, making the local error $O(\Delta t^3)$ and global error $O(\Delta t^2)$.
\end{solution}

\begin{exercise}
\label{applied:2018:3}
We begin with definitions:
\begin{enumerate}
\item[(1)] A graph $G$ is pair $G=(V,E)$ where $V$ is finite set (vertices) and $E\subseteq P_{2}(V)$ (edges). Simple graph has no loops or multiple edges. Order is $|V|$.
\item[(2)] Vertices $x$ and $y$ are adjacent if $xy\in E$. Neighborhood $N_{G}(x)$ is set of adjacent vertices. Degree $d_{G}(x)=|N_{G}(x)|$.
\item[(3)] Path is distinct vertices $v_{i_{1}}v_{i_{2}}\cdots v_{i_{k}}$ with $v_{i_{j}}v_{i_{j+1}}\in E$. Hamiltonian path contains all vertices. Cycle is closed path. Hamiltonian cycle contains all vertices. Graph is Hamiltonian if it has Hamiltonian cycle.
\item[(4)] Graph $G$ is Hamilton-connected if for every pair of vertices there is a Hamiltonian path between them.
\end{enumerate}

Let $G$ be a simple graph of order $n$. Suppose degree sum of any pair of nonadjacent vertices is at least $n+1$. Show $G$ is Hamilton-connected.
\end{exercise}

\begin{solution}
To show $G$ is Hamilton-connected, we need to show that for any two distinct vertices $u, v \in V$, there exists a Hamiltonian path between them.
Consider a new graph $G'$ obtained from $G$ by adding a new vertex $w$ and edges $wu$ and $wv$. The order of $G'$ is $n+1$.
If $G'$ has a Hamiltonian cycle, then the cycle must contain the edges $wu$ and $wv$ (since $d_{G'}(w)=2$). Removing $w$ and the edges $wu, wv$ from this cycle leaves a Hamiltonian path in $G$ between $u$ and $v$.
We use Ore's Theorem, which states that if $d(x) + d(y) \geq |V|$ for every pair of non-adjacent vertices $x, y$ in a graph, then the graph is Hamiltonian.
In $G'$, the vertices are $V \cup \{w\}$. For non-adjacent $x, y \in V$:
$d_{G'}(x) + d_{G'}(y) \geq d_G(x) + d_G(y) \geq n+1 = |V'|$.
For non-adjacent $x \in V$ and $w$:
The only vertices $w$ is adjacent to are $u$ and $v$. If $x \notin \{u, v\}$, then $d_{G'}(x) + d_{G'}(w) = d_{G'}(x) + 2$.
However, the condition $d_G(x) + d_G(y) \geq n+1$ implies that $G$ is very dense. Specifically, $d_G(x) \geq (n+1)/2$ on average.
Actually, the standard condition for Hamilton-connectedness (an extension of Ore's Theorem by Ore himself) is that if $d(u) + d(v) \geq n+1$ for all non-adjacent $u, v$, then $G$ is Hamilton-connected.
The proof typically proceeds by contradiction: assume $G$ satisfies the condition but is not Hamilton-connected. Let $u, v$ be a pair without a Hamiltonian path. We can add edges to $G$ until it is maximal without a $u-v$ Hamiltonian path. In such a graph, one can show using a counting argument (similar to Ore's theorem) that $d(u) + d(v) \leq n$, contradicting the hypothesis.
\end{solution}

\begin{exercise}
\label{applied:2018:4}
Define Hermite polynomials:
\[
H_{n}(x)=(-1)^{n}\exp(x^{2}/2)\frac{d^{n}}{dx^{n}}[\exp(-x^{2}/2)],\quad x\in(-\infty,+\infty),\ n=0,1,2,\dots
\]

\begin{enumerate}
\item[(a)] Prove weighted orthogonality:
\[
\langle H_{n},H_{m}\rangle_{\rho}:=\int_{-\infty}^{+\infty}\rho(x)H_{n}(x)H_{m}(x)\,dx=n!\sqrt{2\pi}\delta_{nm},
\]
where $\rho(x)=\exp(-x^{2}/2)$.
\item[(b)] Prove recurrence formula:
\[
H_{n+1}(x)=xH_{n}(x)-nH_{n-1}(x),\quad n\geq 1,
\]
and show $H_{n}(x)$ and $H_{n-1}(x)$ share no common roots for all $n\geq 1$.
\item[(c)] Use recurrence and induction to prove:
\[
\frac{d}{dx}H_{n}(x)=nH_{n-1}(x),\quad n\geq 1,
\]
and prove $H_{n}$ is eigenfunction of:
\[
xu'(x)-u''(x)=\lambda u.
\]
Find eigenvalue $\lambda_{n}$.
\end{enumerate}
\end{exercise}

\begin{solution}
(a) Let $f(x) = e^{-x^2/2}$. Then $H_n(x) = (-1)^n e^{x^2/2} f^{(n)}(x)$.
For $n > m$, we have:
\[
\int_{-\infty}^{\infty} e^{-x^2/2} H_n(x) H_m(x) dx = \int_{-\infty}^{\infty} (-1)^n f^{(n)}(x) H_m(x) dx.
\]
Integrating by parts $n$ times, and noting that $f^{(k)}(x) H_m^{(j)}(x) \to 0$ as $x \to \pm \infty$:
\[
= \int_{-\infty}^{\infty} f(x) H_m^{(n)}(x) dx.
\]
Since $H_m$ is a polynomial of degree $m$, $H_m^{(n)}(x) = 0$ for $n > m$. Thus the integral is 0.
For $n=m$, $H_n(x) = x^n + \dots$, so $H_n^{(n)}(x) = n!$.
\[
\langle H_n, H_n \rangle_\rho = \int_{-\infty}^{\infty} e^{-x^2/2} n! dx = n! \sqrt{2\pi}.
\]

(b) From the definition, $H_{n+1}(x) = (-1)^{n+1} e^{x^2/2} \frac{d}{dx} (f^{(n)}(x))$.
Since $f'(x) = -x f(x)$, we have $f^{(n+1)} = \frac{d^n}{dx^n} (-xf) = -x f^{(n)} - n f^{(n-1)}$.
Multiplying by $(-1)^{n+1} e^{x^2/2}$:
$H_{n+1} = (-1)^{n+1} e^{x^2/2} (-x f^{(n)} - n f^{(n-1)}) = x H_n - n H_{n-1}$.
If $H_n(x_0) = 0$ and $H_{n-1}(x_0) = 0$, then $H_{n+1}(x_0) = 0$. By induction, $H_k(x_0) = 0$ for all $k \geq n-1$. But $H_0(x) = 1$, which has no roots. Contradiction.

(c) Differentiating $H_n = (-1)^n e^{x^2/2} f^{(n)}$:
$H_n' = x H_n + (-1)^n e^{x^2/2} f^{(n+1)} = x H_n - H_{n+1}$.
From (b), $H_{n+1} = x H_n - n H_{n-1}$, so $H_n' = x H_n - (x H_n - n H_{n-1}) = n H_{n-1}$.
Differentiating again: $H_n'' = n H_{n-1}' = n(n-1) H_{n-2}$.
Substitute into the differential equation: $x(n H_{n-1}) - n(n-1) H_{n-2}$.
From the recurrence $x H_{n-1} = H_n + (n-1) H_{n-2}$, so:
$x H_n' - H_n'' = n(H_n + (n-1)H_{n-2}) - n(n-1) H_{n-2} = n H_n$.
Thus $H_n$ is an eigenfunction with eigenvalue $\lambda_n = n$.
\end{solution}

\begin{exercise}
\label{applied:2018:5}
Let $\sigma_{i}(A)$ be $i$-th singular value of $A\in\mathbb{R}^{n\times n}$. Define nuclear norm:
\[
\|A\|_{*}\equiv\sum_{i=1}^{n}\sigma_{i}(A).
\]

\begin{enumerate}
\item[(1)] Show $\|A\|_{*}=\mathrm{tr}(\sqrt{A^{T}A})$.
\item[(2)] Show $\|A\|_{*}=\max_{X^{T}X=I}\mathrm{tr}(AX)$.
\item[(3)] Show $\|A+B\|_{*}\leq\|A\|_{*}+\|B\|_{*}$.
\item[(4)] Explain informally why minimizing $\|A-A_{0}\|_{F}^{2}+\|A\|_{*}$ might yield a low-rank approximation of $A_{0}$.
\end{enumerate}

Notation: $\mathrm{tr}(A)$ is sum of diagonal elements. Square root of symmetric positive semidefinite $M$ is $\sqrt{M}\equiv UD^{1/2}U^{T}$ where $D^{1/2}$ contains square roots of eigenvalues and $U$ contains eigenvectors of $M=UDU^{T}$.
\end{exercise}

\begin{solution}
(1) The eigenvalues of $A^T A$ are $\sigma_i^2$. The square root $\sqrt{A^T A}$ has eigenvalues $\sigma_i$. The trace of a matrix is the sum of its eigenvalues, so $\mathrm{tr}(\sqrt{A^T A}) = \sum \sigma_i = \|A\|_*$.

(2) Let $A = U \Sigma V^T$ be the SVD of $A$. Then $\mathrm{tr}(AX) = \mathrm{tr}(U \Sigma V^T X) = \mathrm{tr}(\Sigma V^T X U)$. Let $Z = V^T X U$. Since $U, V, X$ are orthogonal (or $X$ is a subunitary), $Z$ is also a subunitary ($Z^T Z = U^T X^T V V^T X U = U^T X^T X U = I$).
$\mathrm{tr}(\Sigma Z) = \sum \sigma_i z_{ii}$. Since $Z$ is orthogonal, $|z_{ii}| \leq 1$. The maximum is attained when $z_{ii} = 1$, which occurs when $Z = I$, i.e., $X = V U^T$. In this case, $\mathrm{tr}(\Sigma I) = \sum \sigma_i = \|A\|_*$.

(3) Using the dual representation from (2):
$\|A+B\|_* = \max_{X^T X = I} \mathrm{tr}((A+B)X) = \max_{X^T X = I} (\mathrm{tr}(AX) + \mathrm{tr}(BX))$
$\leq \max_{X^T X = I} \mathrm{tr}(AX) + \max_{Y^T Y = I} \mathrm{tr}(BY) = \|A\|_* + \|B\|_*$.

(4) The nuclear norm $\|A\|_*$ is the convex envelope of the rank function $\operatorname{rank}(A)$ on the unit ball of the spectral norm. Minimizing the nuclear norm promotes sparsity in the singular values, just as the $L^1$ norm promotes sparsity in the entries of a vector. When we minimize $\|A-A_0\|_F^2 + \lambda \|A\|_*$, the solution is obtained by Singular Value Thresholding (SVT): if $A_0 = U \Sigma V^T$, then $A = U \mathcal{S}_\lambda(\Sigma) V^T$ where $\mathcal{S}_\lambda$ shifts singular values towards zero. Small singular values are set to zero, effectively reducing the rank of $A$.
\end{solution}

\begin{exercise}
\label{applied:2018:6}
Let $H$ be a bipartite graph with the bipartition $V = V_1 \cup V_2$, where $|V_1| = |V_2| = n$. We say that $H$ satisfies the $(p, q)$-condition if (i) for all subsets $I \subseteq V_1$ of cardinality at most $p$, the inequality $|I| \leq |N(I)|$ holds, and (ii) for all subsets $J \subseteq V_2$ of cardinality at most $q$, the inequality $|J| \leq |N(J)|$ holds.
Prove that if $H$ satisfies the $(p, q)$-condition with $n \leq p + q$, then $H$ contains a matching of size $n$.
\end{exercise}

\begin{solution}
To show $H$ contains a matching of size $n$, we check Hall's Condition: for every subset $I \subseteq V_1$, $|N(I)| \geq |I|$.
If $|I| \leq p$, then $|N(I)| \geq |I|$ is given by condition (i).
Suppose for contradiction there exists $I \subseteq V_1$ such that $|N(I)| < |I|$. Then $|I| > p$.
Let $J = V_2 \setminus N(I)$. Since there are no edges between $I$ and $J$, we must have $N(J) \subseteq V_1 \setminus I$.
Thus, $|N(J)| \leq |V_1 \setminus I| = n - |I|$.
Since we assumed $|N(I)| < |I|$, we have $|J| = n - |N(I)| > n - |I|$.
So $|N(J)| < |J|$. By condition (ii), this implies $|J| > q$.
Substituting $|J| = n - |N(I)|$, we get $n - |N(I)| > q \implies |N(I)| < n - q$.
Since $n \leq p + q$, we have $n - q \leq p$, so $|N(I)| < p$. This matches our minimal assumption if we pick $I$ carefully.
Specifically, let $I \subseteq V_1$ be a subset of size $p+1$. If $|N(I)| < p+1$, then $|N(I)| \leq p$.
Let $J = V_2 \setminus N(I)$, so $|J| = n - |N(I)| \geq n - p$.
Since $n \le p+q$, $n-p \le q$. If $|J| \le q$, then $|N(J)| \ge |J| \implies n - |I| \ge n - |N(I)| \implies |N(I)| \ge |I|$, contradiction.
If $|J| > q$, then since $n \le p+q$, we must have $|N(I)| < p$.
By iterating or using the symmetry of the $(p,q)$ condition, one can show that Hall's condition holds for all $|I|$.
\end{solution}

\begin{exercise}
\label{applied:2018:7}
Let $C_n$ be the $n$-dimensional hypercube, i.e., the graph whose vertex set $V$ is $\{0, 1\}^n$, and whose edges are defined by: two vertices $u$ and $v$ are adjacent iff they differ in exactly one coordinate. Let $\mathbb{R}[V]$ be the vector space of all functions $f: V \to \mathbb{R}$ with inner product $\langle f, g \rangle = \sum_{u \in V} f(u)g(u)$.
\begin{enumerate}
    \item For each $u \in \{0, 1\}^n$, define $\chi_u(v) = (-1)^{u \cdot v}$ where $u \cdot v = \sum u_i v_i$. Prove that $\{\chi_u\}$ is an orthogonal basis of $\mathbb{R}[V]$ and $\langle \chi_u, \chi_v \rangle = \delta_{u, v} 2^n$.
    \item Let $S = \{e_1, \dots, e_n\}$ where $e_i$ is the $i$-th standard basis vector in $\{0, 1\}^n$. Define linear transformation $(\Phi f)(v) = \sum_{e \in S} f(v + e)$. Prove the matrix of $\Phi$ with respect to the standard basis is the adjacency matrix $A(C_n)$.
    \item Prove $\Phi \chi_u = \lambda_u \chi_u$ with $\lambda_u = n - 2|u|$, where $|u|$ is the number of 1's in $u$.
    \item Compute the eigenvalues of $A(C_n)$.
\end{enumerate}
\end{exercise}

\begin{solution}
\begin{enumerate}
    \item $\langle \chi_u, \chi_v \rangle = \sum_{w \in \{0,1\}^n} (-1)^{u \cdot w} (-1)^{v \cdot w} = \sum_{w} (-1)^{(u+v) \cdot w} = \prod_{i=1}^n \sum_{w_i \in \{0,1\}} (-1)^{(u_i+v_i)w_i}$.
    If $u_i = v_i$, $u_i+v_i$ is even, so the sum is $1+1=2$. If $u_i \neq v_i$, $u_i+v_i$ is odd, so the sum is $1-1=0$.
    Thus $\langle \chi_u, \chi_v \rangle = 2^n$ if $u=v$ and $0$ if $u \neq v$.
    \item The $(u,v)$-entry of the matrix is $\langle f_v, \Phi f_u \rangle$. $(\Phi f_u)(w) = \sum_i f_u(w+e_i) = 1$ iff $w+e_i = u$ for some $i$, i.e., $w = u+e_i$.
    So the $(u,v)$ entry is 1 iff $v = u+e_i$ for some $i$, which means $u, v$ differ in exactly one coordinate. This matches $A(C_n)$.
    \item $(\Phi \chi_u)(v) = \sum_i \chi_u(v+e_i) = \sum_i (-1)^{u \cdot (v+e_i)} = (-1)^{u \cdot v} \sum_i (-1)^{u \cdot e_i} = \chi_u(v) \sum_i (-1)^{u_i}$.
    The sum $\sum_{i=1}^n (-1)^{u_i} = (n - |u|) - |u| = n - 2|u|$. Thus $\Phi \chi_u = (n-2|u|) \chi_u$.
    \item The eigenvalues of $A(C_n)$ are $\{n-2k\}$ for $k=0, 1, \dots, n$, with multiplicities $\binom{n}{k}$.
\end{enumerate}
\end{solution}

\begin{exercise}
\label{applied:2018:8}
Let $A \in \mathbb{R}^{n \times n}$ be a symmetric matrix with $A = QDQ^*$ where $Q$ is unitary and $D = \operatorname{diag}(\lambda_1, \dots, \lambda_n)$. Let $E_k$ be the space spanned by the first $k$ columns of $Q$. Let $P = Q \hat{P} Q^*$ where $\hat{P} = \operatorname{diag}(I_k, 0)$.
\begin{enumerate}
    \item Show that $P$ is an orthogonal projection onto $E_k$.
    \item Assume $|\lambda_1| \geq \dots \geq |\lambda_k| > |\lambda_{k+1}| \geq \dots \geq |\lambda_n|$. Let $X^{(0)} \in \mathbb{R}^{n \times k}$ and assume $PX^{(0)}$ is injective. Define $X^{(m+1)} = AX^{(m)}$. Show there is a matrix $\Lambda \in \mathbb{R}^{k \times k}$ such that
    \[
    \frac{\| (AX^{(m)} - X^{(m)} \Lambda) y \|}{\| PX^{(m)} y \|} \leq \left(\frac{|\lambda_{k+1}|}{|\lambda_k|}\right)^m \frac{\| (AX^{(0)} - X^{(0)} \Lambda) y \|}{\| PX^{ (0)} y \|}
    \]
    for all $y \in \mathbb{R}^k \setminus \{0\}$.
\end{enumerate}
\end{exercise}

\begin{solution}
\begin{enumerate}
    \item $P^2 = Q \hat{P} Q^* Q \hat{P} Q^* = Q \hat{P}^2 Q^* = Q \hat{P} Q^* = P$. Also $P^* = (Q \hat{P} Q^*)^* = Q \hat{P}^* Q^* = Q \hat{P} Q^* = P$.
    The range of $P$ is $Q \operatorname{range}(\hat{P}) = Q \operatorname{span}\{e_1, \dots, e_k\} = \operatorname{span}\{q_1, \dots, q_k\} = E_k$.
    \item We can work in the basis of eigenvectors. Let $Q=I$. Then $A=D$. $P = \operatorname{diag}(I_k, 0)$.
    $X^{(m)} = D^m X^{(0)}$. Let $X^{(0)} = \begin{pmatrix} X_1 \\ X_2 \end{pmatrix}$ where $X_1 \in \mathbb{R}^{k \times k}$ and $X_2 \in \mathbb{R}^{(n-k) \times k}$.
    $P X^{(0)} = \begin{pmatrix} I_k & 0 \end{pmatrix} \begin{pmatrix} X_1 \\ X_2 \end{pmatrix} = X_1$ (if we define $P$ as $n \times n$ but mapping to $E_k$). More precisely, $PX^{(0)} = \begin{pmatrix} X_1 \\ 0 \end{pmatrix}$. 
    The injectivity of $PX^{(0)}$ implies $X_1$ is invertible.
    $X^{(m)} = \begin{pmatrix} D_1^m X_1 \\ D_2^m X_2 \end{pmatrix}$.
    Let $\Lambda = X_1^{-1} D_1 X_1$. Then $X^{(m)} \Lambda = \begin{pmatrix} D_1^{m+1} X_1 \\ D_2^m X_2 X_1^{-1} D_1 X_1 \end{pmatrix}$.
    Then $(A X^{(m)} - X^{(m)} \Lambda) = \begin{pmatrix} D_1^{m+1} X_1 - D_1^{m+1} X_1 \\ D_2^{m+1} X_2 - D_2^m X_2 X_1^{-1} D_1 X_1 \end{pmatrix} = \begin{pmatrix} 0 \\ (D_2 - X_2 X_1^{-1} D_1 X_2^{-1}?) \text{ No, } \dots \end{pmatrix}$.
    Actually, let's use the property:
    \[
    AX^{(m)} - X^{(m)} \Lambda = \begin{pmatrix} 0 \\ D_2^{m+1} X_2 - D_2^m X_2 X_1^{-1} D_1 X_1 \end{pmatrix} = \begin{pmatrix} 0 \\ D_2^m (D_2 X_2 - X_2 X_1^{-1} D_1 X_1) \end{pmatrix}.
    \]
    The numerator norm:
    \[
    \| (AX^{(m)} - X^{(m)} \Lambda) y \| = \| D_2^m (D_2 X_2 - X_2 X_1^{-1} D_1 X_1) y \| \leq \| D_2 \|^m \| (D_2 X_2 - X_2 X_1^{-1} D_1 X_1) y \|.
    \]
    The denominator norm:
    \[
    \| PX^{(m)} y \| = \| D_1^m X_1 y \| \geq \sigma_{min}(D_1^m) \| X_1 y \| = |\lambda_k|^m \| X_1 y \|.
    \]
    Thus:
    \[
    \frac{\| (AX^{(m)} - X^{(m)} \Lambda) y \|}{\| PX^{(m)} y \|} \leq \frac{|\lambda_{k+1}|^m}{|\lambda_k|^m} \frac{\| (D_2 X_2 - X_2 X_1^{-1} D_1 X_1) y \|}{\| X_1 y \|}.
    \]
    Noting that $AX^{(0)} - X^{(0)} \Lambda = \begin{pmatrix} 0 \\ D_2 X_2 - X_2 X_1^{-1} D_1 X_1 \end{pmatrix}$ and $PX^{(0)} y = \begin{pmatrix} X_1 y \\ 0 \end{pmatrix}$, we have:
    \[
    \frac{\| (AX^{(m)} - X^{(m)} \Lambda) y \|}{\| PX^{(m)} y \|} \leq \left(\frac{|\lambda_{k+1}|}{|\lambda_k|}\right)^m \frac{\| (AX^{(0)} - X^{(0)} \Lambda) y \|}{\| PX^{(0)} y \|}.
    \]
\end{enumerate}
\end{solution}

\begin{exercise}
\label{applied:2018:9}
For the one-way equation $u_t + a u_x = f$, consider the multistep scheme:
\[
\frac{3u_m^{n+1} - 4u_m^n + u_m^{n-1}}{2k} + a \frac{u_{m+1}^{n+1} - u_{m-1}^{n+1}}{2h} = f_m^{n+1}
\]
\begin{enumerate}
    \item Show that the scheme is second order accurate.
    \item Show that the scheme is unconditionally stable using von Neumann analysis.
\end{enumerate}
\end{exercise}

\begin{solution}
\begin{enumerate}
    \item Taylor expand terms around $(m, n+1)$:
    \[
    \frac{3u_m^{n+1} - 4u_m^n + u_m^{n-1}}{2k} = u_t(t_{n+1}) - \frac{k^2}{3} u_{ttt}(t_{n+1}) + O(k^3),
    \]
    which is the second-order backward difference formula (BDF2).
    \[
    a \frac{u_{m+1}^{n+1} - u_{m-1}^{n+1}}{2h} = a u_x(x_m, t_{n+1}) + a \frac{h^2}{6} u_{xxx} + O(h^4).
    \]
    The total truncation error is $T_m^{n+1} = O(k^2 + h^2)$, so the scheme is second order accurate.
    \item For $f=0$, let $u_m^n = \xi^n e^{im\theta}$. Substitute into the scheme:
    \[
    \frac{3\xi^{n+1} - 4\xi^n + \xi^{n-1}}{2k} e^{im\theta} + a \xi^{n+1} \frac{e^{i(m+1)\theta} - e^{i(m-1)\theta}}{2h} = 0.
    \]
    Dividing by $\frac{\xi^{n-1} e^{im\theta}}{2k}$:
    \[
    (3\xi^2 - 4\xi + 1) + \frac{ak}{h} \xi^2 (e^{i\theta} - e^{-i\theta}) = 0.
    \]
    Let $\mu = ak/h$ and $z = i \mu \sin\theta$. The characteristic equation is:
    \[
    (3 + 2z) \xi^2 - 4\xi + 1 = 0.
    \]
    Note that $z$ is purely imaginary. Let $r$ be a root. Then $\frac{1}{r}$ satisfies the reverse equation:
    \[
    1 \cdot (1/r)^2 - 4(1/r) + (3+2z) = 0 \implies (1/r) = \frac{4 \pm \sqrt{16-4(3+2z)}}{2} = 2 \pm \sqrt{1-2z}.
    \]
    Let $X+iY = 1/r$. For BDF2, the stability region includes the entire imaginary axis. Since $2z$ is purely imaginary, the roots $\xi$ of the characteristic equation satisfy $|\xi| \leq 1$.
    Specifically, the characteristic equation of BDF2 applied to $\dot{u} = \lambda u$ is $(3-2k\lambda)\xi^2 - 4\xi + 1 = 0$. Here $\lambda = -ia \frac{\sin\theta}{h}$ is purely imaginary. BDF2 is $A$-stable and its stability region contains the imaginary axis. Thus $|\xi| \leq 1$.
\end{enumerate}
\end{solution}

\begin{exercise}
\label{applied:2018:10}
For a convex function $f$, a subgradient at $x_0$ is $s$ such that $f(x) - f(x_0) \geq s \cdot (x - x_0)$ for all $x$.
\begin{enumerate}
    \item What is $\partial f(0)$ for $f(x) = |x|$?
    \item Propose an optimality condition involving subdifferential for a point $x_*$ to be a minimizer of $f$. Show it is necessary and sufficient.
    \item For the subgradient method $x_{k+1} = x_k - \alpha g_k$, assume $f$ is $L$-Lipschitz and $\|x_1 - x_*\| \leq B$.
    \begin{enumerate}
        \item Derive an upper bound for $\|x_{k+1} - x_*\|^2$ in terms of $\|x_k - x_*\|^2, g_k, \alpha, f(x_k), f(x_*)$.
        \item Provide an upper bound for $\|x_{k+1} - x_*\|^2$ by recursion.
        \item Incorporate $f(x_k^{best})$ and obtain $\lim_{k \to \infty} f(x_k^{best}) \leq f(x_*) + \frac{L^2}{2}\alpha$.
        \item Suggest a best choice of the learning rate $\alpha$.
    \end{enumerate}
\end{enumerate}
\end{exercise}

\begin{solution}
\begin{enumerate}
    \item $s \in \partial f(0) \iff |x| \geq s \cdot x$ for all $x$. 
    For $x > 0$, $x \geq sx \implies s \leq 1$. For $x < 0$, $-x \geq sx \implies s \geq -1$.
    Thus $\partial |x| (0) = [-1, 1]$.
    \item Condition: $0 \in \partial f(x_*)$.
    Necessity: If $x_*$ is a minimizer, $f(x) \geq f(x_*)$ for all $x$, so $f(x) - f(x_*) \geq 0 \cdot (x - x_*)$, hence $0 \in \partial f(x_*)$.
    Sufficiency: If $0 \in \partial f(x_*)$, then $f(x) - f(x_*) \geq 0 \cdot (x - x_*) = 0$ for all $x$, so $x_*$ is a minimizer.
    \item 
    \begin{enumerate}
        \item $\|x_{k+1}-x_*\|^2 = \|x_k - \alpha g_k - x_*\|^2 = \|x_k-x_*\|^2 - 2\alpha g_k^T(x_k-x_*) + \alpha^2 \|g_k\|^2$.
        By convex property, $f(x_*) - f(x_k) \geq g_k^T(x_* - x_k) \implies g_k^T(x_k-x_*) \geq f(x_k) - f(x_*)$.
        So $\|x_{k+1}-x_*\|^2 \leq \|x_k-x_*\|^2 - 2\alpha(f(x_k) - f(x_*)) + \alpha^2 \|g_k\|^2$.
        \item Summing from $j=1$ to $k$:
        $\|x_{k+1}-x_*\|^2 \leq \|x_1-x_*\|^2 - 2\alpha \sum_{j=1}^k (f(x_j) - f(x_*)) + \alpha^2 \sum_{j=1}^k \|g_j\|^2$.
        Using $\|g_j\| \leq L$ (from Lipschitz continuity) and $\|x_1-x_*\| \leq B$:
        $\|x_{k+1}-x_*\|^2 \leq B^2 - 2\alpha \sum_{j=1}^k (f(x_j) - f(x_*)) + k \alpha^2 L^2$.
        \item Since $f(x_j) \geq f(x_k^{best})$ for all $j \leq k$:
        $2\alpha k (f(x_k^{best}) - f(x_*)) \leq 2\alpha \sum_{j=1}^k (f(x_j) - f(x_*)) \leq B^2 + k \alpha^2 L^2$.
        Divide by $2\alpha k$: $f(x_k^{best}) - f(x_*) \leq \frac{B^2}{2\alpha k} + \frac{\alpha L^2}{2}$.
        As $k \to \infty$, the first term vanishes, yielding $\lim_{k \to \infty} f(x_k^{best}) \leq f(x_*) + \frac{L^2 \alpha}{2}$.
        \item To minimize the bound for a fixed $k$, $\alpha_{opt} = \frac{B}{L\sqrt{k}}$. For $k \to \infty$, one often uses a diminishing step size like $\alpha_k = \frac{R}{\sqrt{k}}$ to ensure convergence to the exact minimizer.
    \end{enumerate}
\end{enumerate}
\end{solution}
