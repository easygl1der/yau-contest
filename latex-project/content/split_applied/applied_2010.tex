\subsection{Individual}

\begin{exercise}
\begin{enumerate}
    \item Suppose
    \[
    S = \begin{bmatrix} \sigma & \mathbf{u}^T \\ 0 & S_c \end{bmatrix}, \quad T = \begin{bmatrix} \tau & \mathbf{v}^T \\ 0 & T_c \end{bmatrix}, \quad \mathbf{b} = \begin{bmatrix} \beta \\ \mathbf{b}_c \end{bmatrix},
    \]
    where $\sigma, \tau$ and $\beta$ are scalars, $S_c$ and $T_c$ are $n$-by-$n$ matrices, and $\mathbf{b}_c$ is an $n$-vector. Show that if there exists a vector $\mathbf{x}_c$ such that $(S_c T_c - \lambda I) \mathbf{x}_c = \mathbf{b}_c$ and $\mathbf{w}_c = T_c \mathbf{x}_c$ is available, then
    \[
    \mathbf{x} = \begin{bmatrix} \gamma \\ \mathbf{x}_c \end{bmatrix}, \quad \gamma = \frac{\beta - \sigma \mathbf{v}^T \mathbf{x}_c - \mathbf{u}^T \mathbf{w}_c}{\sigma \tau - \lambda}
    \]
    solves $(ST - \lambda I) \mathbf{x} = \mathbf{b}$.

    \item Hence or otherwise, derive an $O(n^2)$ algorithm for solving the linear system $(U_1 U_2 - \lambda I) \mathbf{x} = \mathbf{b}$ where $U_1$ and $U_2$ are $n$-by-$n$ upper triangular matrices, and $(U_1 U_2 - \lambda I)$ is nonsingular. Please write down your algorithm and prove that it is indeed of $O(n^2)$ complexity.

    \item Hence or otherwise, derive an $O(pn^2)$ algorithm for solving the linear system $(U_1 U_2 \cdots U_p - \lambda I) \mathbf{x} = \mathbf{b}$ where $\{U_i\}_{i=1}^p$ are all $n$-by-$n$ upper triangular matrices, and $(U_1 U_2 \cdots U_p - \lambda I)$ is nonsingular. Please write down your algorithm and prove that it is indeed of $O(pn^2)$ complexity.
\end{enumerate}
\end{exercise}

\begin{solution}
\begin{enumerate}
    \item \textbf{Part (1):}
    Compute $ST$:
    \[
    ST = \begin{bmatrix} \sigma & \mathbf{u}^T \\ 0 & S_c \end{bmatrix} \begin{bmatrix} \tau & \mathbf{v}^T \\ 0 & T_c \end{bmatrix} = \begin{bmatrix} \sigma \tau & \sigma \mathbf{v}^T + \mathbf{u}^T T_c \\ 0 & S_c T_c \end{bmatrix}.
    \]
    The system $(ST - \lambda I)\mathbf{x} = \mathbf{b}$ becomes:
    \[
    \begin{bmatrix} \sigma \tau - \lambda & \sigma \mathbf{v}^T + \mathbf{u}^T T_c \\ 0 & S_c T_c - \lambda I \end{bmatrix} \begin{bmatrix} \gamma \\ \mathbf{x}_c \end{bmatrix} = \begin{bmatrix} \beta \\ \mathbf{b}_c \end{bmatrix}.
    \]
    The second block equation is $(S_c T_c - \lambda I)\mathbf{x}_c = \mathbf{b}_c$, which is given.
    The first block equation is:
    \[
    (\sigma \tau - \lambda)\gamma + (\sigma \mathbf{v}^T + \mathbf{u}^T T_c)\mathbf{x}_c = \beta.
    \]
    Substituting $\mathbf{w}_c = T_c \mathbf{x}_c$:
    \[
    (\sigma \tau - \lambda)\gamma + \sigma \mathbf{v}^T \mathbf{x}_c + \mathbf{u}^T \mathbf{w}_c = \beta.
    \]
    Solving for $\gamma$:
    \[
    \gamma = \frac{\beta - \sigma \mathbf{v}^T \mathbf{x}_c - \mathbf{u}^T \mathbf{w}_c}{\sigma \tau - \lambda}.
    \]
    This proves the identity.

    \item \textbf{Part (2):}
    We use induction on the matrix size $n$.
    For $n=1$, it's a scalar equation.
    For $n > 1$, let $U_1, U_2$ be upper triangular. We can partition them as in Part (1), where $\sigma = (U_1)_{11}$, $\tau = (U_2)_{11}$, and $S_c, T_c$ are the $(n-1) \times (n-1)$ trailing submatrices of $U_1, U_2$ respectively.
    The algorithm:
    \begin{enumerate}
        \item Base case: If $n=1$, $x = b / (u_{11} v_{11} - \lambda)$.
        \item Recursive step:
        \begin{enumerate}
            \item Solve the smaller system $(S_c T_c - \lambda I) \mathbf{x}_c = \mathbf{b}_c$ to find $\mathbf{x}_c$.
            \item Compute $\mathbf{w}_c = T_c \mathbf{x}_c$ (Matrix-vector multiplication with upper triangular matrix).
            \item Compute $\gamma$ using the formula from Part (1).
        \end{enumerate}
    \end{enumerate}
    Complexity analysis:
    Let $C(n)$ be the cost for size $n$.
    $C(n) = C(n-1) + \text{cost}(\text{triangular matrix-vector mult}) + \text{cost}(\text{inner products}) + O(1)$.
    Cost of $T_c \mathbf{x}_c$ where $T_c$ is $(n-1)\times(n-1)$ triangular is $(n-1)^2/2$ multiplications and additions.
    Cost of inner products $\mathbf{v}^T \mathbf{x}_c$ and $\mathbf{u}^T \mathbf{w}_c$ is $O(n)$.
    $C(n) = C(n-1) + \frac{1}{2}n^2 + O(n)$.
    Summing up: $C(n) \approx \sum_{k=1}^n \frac{1}{2}k^2 \approx O(n^3)$? Wait, if we use the triangular structure correctly, it should be $O(n^2)$.
    Actually, to solve $(U_1 U_2 - \lambda I) \mathbf{x} = \mathbf{b}$ in $O(n^2)$, we don't recompute the whole matrix-vector product.
    At each step $k$:
    \[
    w_k = (U_2)_{kk} x_k + \sum_{j=k+1}^n (U_2)_{kj} x_j
    \]
    This sum is only $O(n-k)$. Total work is $O(n^2)$.

    \item \textbf{Part (3):}
    For $p$ matrices, let $M = U_1 \cdots U_p$.
    We can define a sequence of vectors $\mathbf{w}^{(1)}, \dots, \mathbf{w}^{(p)}$ where $\mathbf{w}^{(p)} = \mathbf{x}$, $\mathbf{w}^{(p-1)} = U_p \mathbf{w}^{(p)}$, $\mathbf{w}^{(p-2)} = U_{p-1} \mathbf{w}^{(p-1)}$, etc.
    By induction on $p$ and $n$, we can maintain the state of all partial products.
    For each row $i = n, n-1, \dots, 1$:
    For each matrix $j = p, p-1, \dots, 1$, compute the $i$-th component of the partial products.
    The complexity is $O(p n^2)$ because we do $O(n^2)$ work for each of the $p$ matrices.
\end{enumerate}
\end{solution}

\begin{exercise}
\begin{enumerate}
    \item Let $A \in \mathbb{R}^{m \times n}$. Show that there exists an $m$-by-$m$ orthogonal matrix $U$ and an $n$-by-$n$ orthogonal matrix $V$ such that $U^T A V = \text{diag}(\sigma_1, \dots, \sigma_p)$, where $p = \min\{m, n\}$ and $\sigma_1 \geq \sigma_2 \geq \dots \geq \sigma_p \geq 0$.

    \item Let $\text{rank}(A) = r$. Show that for any positive integer $k < r$
    \[
    \min_{\text{rank}(B) = k} \| A - B \|_2 = \sigma_{k+1}.
    \]
\end{enumerate}
\end{exercise}

\begin{solution}
\begin{enumerate}
    \item \textbf{Part (1):}
    This is the Singular Value Decomposition (SVD).
    Consider the matrix $A^T A$. It is symmetric and positive semi-definite. Thus, it has an orthonormal basis of eigenvectors $v_1, \dots, v_n$ with eigenvalues $\lambda_1 \geq \lambda_2 \geq \dots \geq \lambda_n \geq 0$.
    Let $\sigma_i = \sqrt{\lambda_i}$. Let $r$ be the number of non-zero $\sigma_i$.
    For $i \leq r$, define $u_i = \frac{1}{\sigma_i} A v_i$.
    Then $\{u_1, \dots, u_r\}$ are orthonormal:
    \[
    u_i^T u_j = \frac{1}{\sigma_i \sigma_j} v_i^T A^T A v_j = \frac{1}{\sigma_i \sigma_j} v_i^T (\lambda_j v_j) = \frac{\sigma_j^2}{\sigma_i \sigma_j} \delta_{ij} = \delta_{ij}.
    \]
    Extend $\{u_i\}$ to an orthonormal basis of $\mathbb{R}^m$.
    Then $A v_i = \sigma_i u_i$ for $i \leq r$, and $A v_i = 0$ for $i > r$.
    In matrix form, $A V = U \Sigma \implies U^T A V = \Sigma$.

    \item \textbf{Part (2):}
    This is the Eckart-Young-Mirsky Theorem.
    Let $A = \sum_{i=1}^r \sigma_i u_i v_i^T$. Let $B_k = \sum_{i=1}^k \sigma_i u_i v_i^T$.
    Then $\| A - B_k \|_2 = \| \sum_{i=k+1}^r \sigma_i u_i v_i^T \|_2 = \sigma_{k+1}$.
    To show this is the minimum, let $B$ be any matrix with rank $k$.
    The null space of $B$, $\mathcal{N}(B)$, has dimension $n-k$.
    The subspace $V_{k+1} = \text{span}\{v_1, \dots, v_{k+1}\}$ has dimension $k+1$.
    Since $(n-k) + (k+1) = n+1 > n$, there exists a unit vector $z \in \mathcal{N}(B) \cap V_{k+1}$.
    $z = \sum_{i=1}^{k+1} \alpha_i v_i$ with $\sum \alpha_i^2 = 1$.
    \[
    \| (A-B)z \|_2^2 = \| Az \|_2^2 = \| \sum_{i=1}^{k+1} \sigma_i \alpha_i u_i \|_2^2 = \sum_{i=1}^{k+1} \sigma_i^2 \alpha_i^2 \geq \sigma_{k+1}^2 \sum \alpha_i^2 = \sigma_{k+1}^2.
    \]
    Thus $\| A-B \|_2 \geq \sigma_{k+1}$.
\end{enumerate}
\end{solution}

\begin{exercise}
When considering finite difference schemes approximating partial differential equations (PDEs), for example, the scheme
\[
u_j^{n+1} = u_j^n - \lambda(u_j^n - u_{j-1}^n) \tag{1}
\]
where $\lambda = \frac{\Delta t}{\Delta x}$, approximating the PDE
\[
u_t + u_x = 0, \tag{2}
\]
we are often interested in stability, namely
\[
\|u^n\| \leq C\|u^0\|, \quad n \Delta t \leq T \tag{3}
\]
for a constant $C = C(T)$ independent of the time step $\Delta t$ and the spatial mesh size $\Delta x$. Here $\|\cdot\|$ is a given norm, for example the $L^2$ norm or the $L^\infty$ norm, of the numerical solution vector $u^n = (u_1^n, u_2^n, \dots, u_N^n)$. The mesh points are $x_j = j \Delta x$, $t^n = n \Delta t$, and the numerical solution $u_j^n$ approximates the exact solution $u(x_j, t^n)$ of the PDE (2) with a periodic boundary condition.

\begin{enumerate}
    \item Prove that the scheme (1) is stable in the sense of (3) for both the $L^2$ norm and the $L^\infty$ norm under the time step restriction $\lambda \leq 1$.
    \item Since the numerical solution $u^n$ is in a finite dimensional space, Student A argues that the stability (3), once proved for a specific norm $\|\cdot\|_a$, would also automatically hold for any other norm $\|\cdot\|_b$. His argument is based on the equivalency of all norms in a finite dimensional space, namely for any two norms $\|\cdot\|_a$ and $\|\cdot\|_b$ on a finite dimensional space $W$, there exists a constant $\delta > 0$ such that
    \[
    \delta \|u\|_b \leq \|u\|_a \leq \frac{1}{\delta} \|u\|_b.
    \]
    Do you agree with his argument? If not, please explain the mistake made by Student A.
\end{enumerate}
\end{exercise}

\begin{solution}
\textbf{Stability Analysis:}
\begin{enumerate}
    \item \textbf{$L^\infty$ norm:} If $0 \leq \lambda \leq 1$, then $u_j^{n+1} = (1-\lambda) u_j^n + \lambda u_{j-1}^n$ is a convex combination.
    \[
    |u_j^{n+1}| \leq (1-\lambda)|u_j^n| + \lambda |u_{j-1}^n| \leq \|u^n\|_\infty.
    \]
    Thus $\|u^{n+1}\|_\infty \leq \|u^n\|_\infty \leq \|u^0\|_\infty$. Stable with $C=1$.

    \item \textbf{$L^2$ norm:} Use Von Neumann analysis. $u_j^n = \hat{u}^n e^{ij\theta}$.
    \[
    \hat{u}^{n+1} = (1 - \lambda + \lambda e^{-i\theta}) \hat{u}^n.
    \]
    Amplification factor $g(\theta) = 1 - \lambda + \lambda (\cos \theta - i \sin \theta)$.
    \[
    |g(\theta)|^2 = (1-\lambda + \lambda \cos \theta)^2 + (\lambda \sin \theta)^2 = 1 - 2\lambda(1-\lambda)(1 - \cos \theta).
    \]
    For stability, $|g(\theta)|^2 \leq 1 \iff 2\lambda(1-\lambda)(1-\cos \theta) \geq 0$, which requires $0 \leq \lambda \leq 1$.
\end{enumerate}

\textbf{Student A's Mistake:}
The equivalence of norms $\|u\|_a \leq K \|u\|_b$ holds, but the constant $K$ depends on the dimension $J$ of the space. In numerical analysis, $J \to \infty$ as $\Delta x \to 0$. Stability requires the constant $C$ in $\|u^n\| \leq C\|u^0\|$ to be independent of $\Delta x$ (and thus $J$).
For example, $\|u\|_\infty \leq \|u\|_2 \leq \sqrt{J} \|u\|_\infty$.
If a scheme is stable in $L^2$ with $C=1$, we have $\|u^n\|_\infty \leq \|u^n\|_2 \leq \|u^0\|_2 \leq \sqrt{J} \|u^0\|_\infty$.
As $\Delta x \to 0$, $J \to \infty$, so $\|u^n\|_\infty$ could potentially grow like $1/\sqrt{\Delta x}$, violating the stability definition.
\end{solution}

\begin{exercise}
We have the following 3 PDEs
\[
u_t + A u_x = 0, \tag{4}
\]
\[
u_t + B u_x = 0, \tag{5}
\]
\[
u_t + C u_x = 0, \quad C = A + B. \tag{6}
\]
Here $u$ is a vector of size $m$ and $A$ and $B$ are $m \times m$ real matrices. We assume $m \geq 2$ and both $A$ and $B$ are diagonalizable with only real eigenvalues. We also assume periodic initial condition for these PDEs.

\begin{enumerate}
    \item Prove that (4) and (5) are both well-posed in the $L^2$-norm. Recall that a PDE is well-posed if its solution satisfies
    \[
    \|u(\cdot, t)\| \leq C(T) \|u(\cdot, 0)\|, \quad 0 \leq t \leq T
    \]
    for a constant $C(T)$ which depends only on $T$.
    \item Is (6) guaranteed to be well-posed as well? If not, give a counter example.
    \item Suppose we have a finite difference scheme $u^{n+1} = A_h u^n$ for approximating (4) and another scheme $u^{n+1} = B_h u^n$ for approximating (5). Suppose both schemes are stable in the $L^2$-norm. If we now form the splitting scheme $u^{n+1} = B_h A_h u^n$ which is a consistent scheme for solving (6), is this scheme guaranteed to be $L^2$ stable as well? If not, give a counter example.
\end{enumerate}
\end{exercise}

\begin{solution}
\begin{enumerate}
    \item \textbf{Well-posedness:}
    Let $A = P D P^{-1}$ where $D = \text{diag}(\lambda_1, \dots, \lambda_m)$ and $\lambda_i \in \mathbb{R}$.
    Let $v = P^{-1}u$. Then $v_t + D v_x = 0$.
    For each component $v_i(x, t) = v_i(x - \lambda_i t, 0)$, so $\|v_i(\cdot, t)\|_2 = \|v_i(\cdot, 0)\|_2$.
    Then $\|u(\cdot, t)\|_2 \leq \|P\| \|v(\cdot, t)\|_2 = \|P\| \|P^{-1} u(\cdot, 0)\|_2 \leq \|P\| \|P^{-1}\| \|u(\cdot, 0)\|_2$.
    Since $C(T) = \|P\| \|P^{-1}\|$ is independent of the initial data, it is well-posed.

    \item \textbf{Counterexample:}
    Let $A = \begin{bmatrix} 1 & a \\ 0 & -1 \end{bmatrix}$ and $B = \begin{bmatrix} -1 & 0 \\ b & 1 \end{bmatrix}$. Both have eigenvalues $\pm 1$.
    $C = A + B = \begin{bmatrix} 0 & a \\ b & 0 \end{bmatrix}$.
    The eigenvalues of $C$ are $\pm \sqrt{ab}$. If $a=1, b=-1$, eigenvalues are $\pm i$.
    A system with complex eigenvalues $\lambda = \pm i$ has solutions that grow like $e^{|\xi| t}$ in Fourier space, which makes the problem ill-posed in $L^2$.

    \item \textbf{Splitting Scheme:}
    By the Lax Equivalence Theorem, for a consistent scheme, stability is equivalent to the well-posedness of the underlying PDE. Since the PDE (6) can be ill-posed (as shown in Part 2), the splitting scheme cannot be stable in those cases. The amplification factor will mirror the growth of the ill-posed PDE.
\end{enumerate}
\end{solution}
