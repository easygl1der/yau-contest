\section{2025 Applied \& Computational Mathematics}

\begin{exercise}
\label{applied:2025:1}
We consider the multipoint iteration method
\[
x_{k+1} = x_{k} - \alpha \frac{f(x_{k})}{f' \left(x_{k} - \beta f(x_{k}) / f'(x_{k})\right)},
\]
where $\alpha$ and $\beta$ are arbitrary parameters, for solving the equation $f(x)=0$. Determine the values $\alpha$ and $\beta$ such that the multipoint method achieves the highest possible order of convergence for finding $\xi$, a simple root of $f(x)=0$.
\end{exercise}

\begin{solution}
\textbf{Intuition:} This iterative scheme is a two-step method where the derivative is evaluated at an intermediate point $y_k = x_k - \beta \frac{f(x_k)}{f'(x_k)}$. This is similar to the Midpoint method or Jarratt-type methods. We aim to find $\alpha, \beta$ that cancel the lower-order error terms in the Taylor expansion.

\textbf{Analysis:} Let $e_k = x_k - \xi$ be the error at step $k$. Since $\xi$ is a simple root, $f(\xi)=0$ and $f'(\xi) \neq 0$. Without loss of generality, assume $f'(\xi)=1$. Let $c_n = \frac{f^{(n)}(\xi)}{n! f'(\xi)}$. The Taylor expansions are:
\begin{align*}
f(x_k) &= e_k + c_2 e_k^2 + c_3 e_k^3 + O(e_k^4), \\
f'(x_k) &= 1 + 2c_2 e_k + 3c_3 e_k^2 + O(e_k^3).
\end{align*}
The intermediate point $y_k$ is:
\[
y_k = x_k - \beta \frac{f(x_k)}{f'(x_k)} = \xi + e_k - \beta \frac{e_k + c_2 e_k^2 + O(e_k^3)}{1 + 2c_2 e_k + O(e_k^2)} = \xi + (1-\beta)e_k + \beta c_2 e_k^2 + O(e_k^3).
\]
Now expand the derivative at $y_k$ around $\xi$ (let $\hat{e}_k = y_k - \xi = (1-\beta)e_k + \beta c_2 e_k^2$):
\[
f'(y_k) = 1 + 2c_2 \hat{e}_k + 3c_3 \hat{e}_k^2 + O(\hat{e}_k^3) = 1 + 2c_2(1-\beta)e_k + [2\beta c_2^2 + 3c_3(1-\beta)^2] e_k^2 + O(e_k^3).
\]
The next error $e_{k+1}$ is:
\begin{align*}
e_{k+1} &= e_k - \alpha \frac{f(x_k)}{f'(y_k)} = e_k - \alpha \frac{e_k + c_2 e_k^2 + c_3 e_k^3}{1 + 2c_2(1-\beta)e_k + [2\beta c_2^2 + 3c_3(1-\beta)^2] e_k^2} \\
&= e_k - \alpha e_k (1 + c_2 e_k + c_3 e_k^2) \left[ 1 - 2c_2(1-\beta)e_k + (4c_2^2(1-\beta)^2 - 2\beta c_2^2 - 3c_3(1-\beta)^2)e_k^2 \right] + O(e_k^4) \\
&= (1-\alpha)e_k + \alpha [2c_2(1-\beta) - c_2] e_k^2 + \alpha [c_2^2(2\beta - 4(1-\beta)^2) + c_3(3(1-\beta)^2 - 1)] e_k^3 + O(e_k^4).
\end{align*}

\textbf{Optimization:}
\begin{itemize}
    \item To achieve order $p \geq 2$, we need $1-\alpha = 0 \implies \mathbf{\alpha = 1}$.
    \item To achieve order $p \geq 3$, we need $2c_2(1-\beta) - c_2 = 0 \implies 2-2\beta-1=0 \implies \mathbf{\beta = 1/2}$.
\end{itemize}
With $\alpha=1, \beta=1/2$, the method becomes the Midpoint method. Let's check if the $e_k^3$ term can be zero for all $f$. The coefficient of $e_k^3$ becomes:
\[
c_2^2(1 - 4(1/4)) + c_3(3(1/4) - 1) = 0 \cdot c_2^2 - \frac{1}{4} c_3.
\]
This term depends on $c_3 = f'''(\xi)/6$, so it is not zero in general. Thus, the highest possible order is \textbf{3}, achieved for $\alpha = 1, \beta = 1/2$.
\end{solution}

\begin{exercise}
\label{applied:2025:2}
Compute the spectral radius of the matrix $\mathbf { A } ^ { - 1 }$ , where $\mathbf{A} = \operatorname{tridiag}(1, 0, 1) \in \mathbb{R}^{8 \times 8}$.
\end{exercise}

\begin{solution}
\textbf{Intuition:} The matrix $A$ is the adjacency matrix of a path graph on 8 vertices. Its eigenvalues are well-known. The spectral radius of $A^{-1}$ is the reciprocal of the smallest absolute eigenvalue of $A$.

\textbf{Eigenvalues of A:} For a tridiagonal matrix with diagonal $d$ and off-diagonals $b$, the eigenvalues are given by $\lambda_k = d + 2b \cos\left(\frac{k\pi}{n+1}\right)$. Here $d=0, b=1, n=8$:
\[
\lambda_k = 2 \cos\left(\frac{k\pi}{9}\right), \quad k=1, 2, \dots, 8.
\]
The eigenvalues of $A^{-1}$ are $1/\lambda_k$. The spectral radius is $\rho(A^{-1}) = \max_k |1/\lambda_k| = \frac{1}{\min_k |\lambda_k|}$.

\textbf{Minimizing $|\lambda_k|$:} We need to find $k \in \{1, \dots, 8\}$ such that $\frac{k\pi}{9}$ is closest to $\pi/2$. 
\[
\frac{4\pi}{9} = 80^\circ, \quad \frac{5\pi}{9} = 100^\circ.
\]
Both are equidistant from $90^\circ$ (by $10^\circ$). Thus:
\[
\min |\lambda_k| = |2 \cos(4\pi/9)| = 2 \sin(\pi/18).
\]
Therefore, the spectral radius is:
\[
\rho(A^{-1}) = \frac{1}{2 \sin(\pi/18)}.
\]
\end{solution}

\begin{exercise}
\label{applied:2025:3}
Let $P_k(x)$ be Legendre polynomials and associated Legendre functions $P_k^m = (-1)^m (1-x^2)^{m/2} \frac{d^m}{dx^m} P_k(x)$.
\begin{enumerate}
    \item[(a)] Prove that $\sum_{k=1}^N a_k P_k^1(x_i) = y_i$ is nonsingular for distinct $x_i \in (-1, 1)$.
    \item[(b)] Derive the linear system for coefficients $a_k$ to minimize $\| f(x) - \sum a_k P_k^1(x) \|_{L^2}^2$ and explain why it is nonsingular.
    \item[(c)] Prove that $M_{kj} = 0$ when $k+j$ is odd, where $M$ is the matrix from part (b).
\end{enumerate}
\end{exercise}

\begin{solution}
\textbf{(a) Nonsingularity of Interpolation:}
The function is $P_k^1(x) = -(1-x^2)^{1/2} P_k'(x)$. The system is:
\[
\sum_{k=1}^N a_k [-(1-x_i^2)^{1/2} P_k'(x_i)] = y_i \implies \sum_{k=1}^N a_k P_k'(x_i) = \frac{-y_i}{\sqrt{1-x_i^2}}.
\]
Let $q_k(x) = P_k'(x)$. Since $P_k$ is a polynomial of degree $k$, $q_k$ is a polynomial of degree $k-1$. For $k=1, \dots, N$, the degrees are $0, 1, \dots, N-1$. Thus $\{q_1, \dots, q_N\}$ forms a basis for $\mathcal{P}_{N-1}$. 
The matrix $V_{ik} = q_k(x_i)$ is essentially a Vandermonde-like matrix for the polynomial space $\mathcal{P}_{N-1}$. Since the $N$ points $x_i$ are distinct and the basis spans $\mathcal{P}_{N-1}$, the interpolation problem is unique and the matrix is nonsingular.

\textbf{(b) Least Squares System:}
To minimize $J(\mathbf{a}) = \int_{-1}^1 (f(x) - \sum a_k P_k^1(x))^2 dx$, we take derivatives $\frac{\partial J}{\partial a_j} = 0$:
\[
\int_{-1}^1 2(f(x) - \sum a_k P_k^1(x))(-P_j^1(x)) dx = 0 \implies \sum_{k=1}^N a_k \int_{-1}^1 P_k^1(x) P_j^1(x) dx = \int_{-1}^1 f(x) P_j^1(x) dx.
\]
The system is $M\mathbf{a} = \mathbf{b}$ where $M_{kj} = \langle P_k^1, P_j^1 \rangle_{L^2}$. This is the Gram matrix of the set $\{P_1^1, \dots, P_N^1\}$. Since these functions are linearly independent (they are polynomials multiplied by $\sqrt{1-x^2}$ and have distinct degrees), the Gram matrix is positive definite and thus nonsingular. In fact, $P_k^m$ are orthogonal on $[-1, 1]$ for fixed $m$:
\[
\int_{-1}^1 P_k^1(x) P_j^1(x) dx = \delta_{kj} \frac{2k(k+1)}{2k+1}.
\]
Thus $M$ is a diagonal matrix with positive diagonal entries.

\textbf{(c) Parity Argument:}
The parity of Legendre polynomials is $P_k(-x) = (-1)^k P_k(x)$.
The derivative $P_k'(x)$ has parity $P_k'(-x) = (-1)^{k-1} P_k'(x)$.
The factor $(1-x^2)^{1/2}$ is even. Thus $P_k^1(x)$ has parity $(-1)^{k-1}$.
The product $P_k^1(x) P_j^1(x)$ has parity $(-1)^{k-1 + j-1} = (-1)^{k+j}$.
If $k+j$ is odd, the product is an odd function. Integrating an odd function over the symmetric interval $[-1, 1]$ yields:
\[
M_{kj} = \int_{-1}^1 P_k^1(x) P_j^1(x) dx = 0.
\]
\end{solution}

\begin{exercise}
\label{applied:2025:4}
Given vectors $y_1, \dots, y_n \in \mathbb{R}^m, find $\ell \leq \dim \operatorname{span}\{y_j\}$ orthonormal vectors $\{\psi_i\}_{i=1}^\ell$ minimizing $J = \sum_{j=1}^n \| y_j - \sum_{i=1}^\ell (y_j^\top \psi_i) \psi_i \|^2$.
\end{exercise}

\begin{solution}
\textbf{Intuition:} This is the Principal Component Analysis (PCA) problem. We seek an $\ell$-dimensional subspace that minimizes the sum of squared distances to the data points.

\textbf{Derivation:} Let $\Psi = [\psi_1, \dots, \psi_\ell] \in \mathbb{R}^{m \times \ell}$ with $\Psi^\top \Psi = I$. The projection of $y_j$ onto the subspace is $P y_j = \Psi \Psi^\top y_j$.
The error for $y_j$ is $\|y_j - \Psi \Psi^\top y_j\|^2 = \|y_j\|^2 - \| \Psi \Psi^\top y_j\|^2 = \|y_j\|^2 - y_j^\top \Psi \Psi^\top y_j$.
Minimizing $J$ is equivalent to maximizing:
\[
\sum_{j=1}^n y_j^\top \Psi \Psi^\top y_j = \operatorname{tr}(\Psi^\top (\sum_{j=1}^n y_j y_j^\top) \Psi).
\]
Let $C = YY^\top = \sum y_j y_j^\top$ be the (unnormalized) covariance matrix. We maximize $\operatorname{tr}(\Psi^\top C \Psi)$ subject to $\Psi^\top \Psi = I$.
By the Ky Fan Theorem (or Rayleigh-Ritz theorem), the maximum is achieved when the columns of $\Psi$ are the $\ell$ eigenvectors of $C$ corresponding to its $\ell$ largest eigenvalues.
These $\psi_i$ are the same as the left singular vectors of the data matrix $Y = [y_1, \dots, y_n]$.
\end{solution}

\begin{exercise}
\label{applied:2025:5}
Consider $u_t + u = u_{xx}$ with $\theta$-scheme:
\[
\frac{U_j^{m+1} - U_j^m}{\Delta t} + [ \theta U_j^{m+1} + (1 - \theta) U_j^m ] = \theta \frac{\delta^2 U_j^{m+1}}{\Delta x^2} + (1 - \theta) \frac{\delta^2 U_j^m}{\Delta x^2}
\]
where $\delta^2 U_j = U_{j+1} - 2U_j + U_{j-1}$.
\begin{enumerate}
    \item[(a)] Prove $\| U^{m} \|_{\ell_\infty} \leq \left(\frac{1 - (1 - \theta) \Delta t}{1 + \theta \Delta t}\right)^m \| U^0 \|_{\ell_\infty}$ under $A(\theta) \Delta t \leq \frac{\Delta x^2}{2 + \Delta x^2}$.
    \item[(b)] Show $\| U^m \|_{\ell_2} \leq \| U^0 \|_{\ell_2}$ for $\theta \geq 1/2$ unconditionally, and for $\theta < 1/2$ under $B(\theta) \Delta t \leq \frac{2\Delta x^2}{4 + \Delta x^2}$.
\end{enumerate}
\end{exercise}

\begin{solution}
\textbf{(a) $\ell_\infty$ Stability:}
Let $\lambda = \Delta t / \Delta x^2$. Rearrange the scheme:
\[
U_j^{m+1} (1 + \theta \Delta t + 2\theta \lambda) = U_j^m (1 - (1-\theta)\Delta t - 2(1-\theta)\lambda) + (1-\theta)\lambda (U_{j+1}^m + U_{j-1}^m) + \theta \lambda (U_{j+1}^{m+1} + U_{j-1}^{m+1}).
\]
For the coefficients on the RHS to be non-negative, we need $1 - (1-\theta)(\Delta t + 2\lambda) \geq 0$:
\[
(1-\theta) \Delta t \left( 1 + \frac{2}{\Delta x^2} \right) \leq 1 \implies (1-\theta) \Delta t \leq \frac{\Delta x^2}{\Delta x^2 + 2}.
\]
Thus $A(\theta) = 1 - \theta$. Under this condition:
\[
(1 + \theta \Delta t + 2\theta \lambda) |U_j^{m+1}| \leq [1 - (1-\theta)\Delta t - 2(1-\theta)\lambda + 2(1-\theta)\lambda] \|U^m\|_\infty + 2\theta \lambda \|U^{m+1}\|_\infty.
\]
Taking the maximum over $j$:
\[
(1 + \theta \Delta t) \|U^{m+1}\|_\infty \leq (1 - (1-\theta)\Delta t) \|U^m\|_\infty \implies \|U^{m+1}\|_\infty \leq \frac{1 - (1-\theta)\Delta t}{1 + \theta \Delta t} \|U^m\|_\infty.
\]

\textbf{(b) $\ell_2$ Stability (Von Neumann):}
The amplification factor $G(\xi)$ for $U_j^m = G^m e^{ij\xi}$ satisfies:
\[
G = \frac{1 - (1-\theta)(\Delta t + 4\lambda \sin^2(\xi/2))}{1 + \theta(\Delta t + 4\lambda \sin^2(\xi/2))}.
\]
Let $Z = \Delta t + 4\lambda \sin^2(\xi/2) \in [\Delta t, \Delta t(1 + 4/\Delta x^2)]$. The factor is $G(Z) = \frac{1-(1-\theta)Z}{1+\theta Z}$.
We need $|G(Z)| \leq 1$ for all $Z \in [Z_{min}, Z_{max}]$.
\begin{itemize}
    \item $G \leq 1 \iff 1 - (1-\theta)Z \leq 1 + \theta Z \iff Z \geq 0$, which is always true.
    \item $G \geq -1 \iff 1 - (1-\theta)Z \geq -1 - \theta Z \iff (1-2\theta)Z \leq 2$.
\end{itemize}
Case 1: $\theta \in [1/2, 1]$. Then $1-2\theta \leq 0$, so $(1-2\theta)Z \leq 0 \leq 2$ is always true. Unconditionally stable.
Case 2: $\theta \in [0, 1/2)$. We need $Z_{max} \leq \frac{2}{1-2\theta}$:
\[
\Delta t \left( 1 + \frac{4}{\Delta x^2} \right) \leq \frac{2}{1-2\theta} \implies (1-2\theta) \Delta t \leq \frac{2 \Delta x^2}{\Delta x^2 + 4}.
\]
Thus $B(\theta) = 1 - 2\theta$.
\end{solution}

\begin{exercise}
\label{applied:2025:6}
Stiff system: $\dot{y}_1 = -1000y_1 + 999y_2$, $\dot{y}_2 = -y_2$ with $\mathbf{y}_0 = \binom{2}{1}$.
\begin{enumerate}
    \item[(a)] Find exact solution.
    \item[(b)] Stability of Explicit Euler ($h > 0.002$).
    \item[(c)] Unconditional stability of Implicit Euler.
    \item[(d)] Stability of Trapezoidal Rule.
\end{enumerate}
\end{exercise}

\begin{solution}
\textbf{(a) Exact Solution:}
The second equation $\dot{y}_2 = -y_2$ gives $y_2(t) = e^{-t}$ (as $y_2(0)=1$).
The first equation is $\dot{y}_1 + 1000y_1 = 999 e^{-t}$. Multiply by integrating factor $e^{1000t}$:
\[
\frac{d}{dt}(y_1 e^{1000t}) = 999 e^{999t} \implies y_1 e^{1000t} = e^{999t} + C.
\]
At $t=0$, $y_1=2$, so $2 = 1 + C \implies C=1$.
$y_1(t) = e^{-t} + e^{-1000t}$.
Solution: $\mathbf{y}(t) = \binom{e^{-t} + e^{-1000t}}{e^{-t}}$.

\textbf{(b) Explicit Euler:}
The system is $\dot{\mathbf{y}} = A\mathbf{y}$ with $A = \begin{pmatrix} -1000 & 999 \\ 0 & -1 \end{pmatrix}$. Eigenvalues are $\lambda_1 = -1000, \lambda_2 = -1$.
For stability, we need $|1 + h\lambda_i| \leq 1$ for all $i$.
$|1 - 1000h| \leq 1 \implies -1 \leq 1 - 1000h \leq 1 \implies 1000h \leq 2 \implies h \leq 0.002$.
If $h > 0.002$, then $(1-1000h)^n$ diverges/oscillates as $n \to \infty$.

\textbf{(c) Implicit Euler:}
Stability function $R(z) = \frac{1}{1-z}$.
For $\text{Re}(\lambda) < 0$, $|1 - h\lambda| > 1$, so $|R(z)| < 1$ for all $h > 0$. Unconditionally stable.

\textbf{(d) Trapezoidal Rule:}
Stability function $R(z) = \frac{1 + z/2}{1 - z/2}$.
Since $|R(z)| \leq 1 \iff \text{Re}(z) \leq 0$, the method is A-stable.
For $\lambda_1, \lambda_2 < 0$, $|R(h\lambda_i)| < 1$ for all $h > 0$. It is unconditionally stable, making it suitable for this stiff problem. Note that as $h \to \infty$, $R(z) \to -1$, which might cause ringing (non-decaying oscillations) if the step size is extremely large, unlike Implicit Euler where $R(z) \to 0$.
\end{solution}
