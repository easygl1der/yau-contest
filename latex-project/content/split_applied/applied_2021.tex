\begin{exercise}
\begin{enumerate}
\item[(a)] Show that
\[
T_{n}(x)=\cos(n\arccos x), \quad x \in [-1, 1],
\]
is a polynomial of degree $n$ with extrema at $x_{k}=\cos(k\pi/n)$, $k=0,1,\dots,n$ and leading coefficient $2^{n-1}$.
\item[(b)] Show that if $f\in C^{n+1}[-1,1]$ and if $P(x)$ is the polynomial with degree at most $n$ that interpolates $f$ at $x_{k}$, $k=0,1,\dots,n$, then
\[
\|f(x)-P(x)\|_{\infty}\leq\frac{1}{2^{n-1}(n+1)!}\|f^{(n+1)}\|_{\infty}.
\]
\end{enumerate}
\end{exercise}

\begin{solution}
\begin{enumerate}
\item[(a)] Let $x = \cos \theta$, where $\theta \in [0, \pi]$. Then $T_n(x) = \cos(n\theta)$. Using the identity $\cos((n+1)\theta) + \cos((n-1)\theta) = 2\cos \theta \cos(n\theta)$, we obtain the recurrence relation:
\[ T_{n+1}(x) = 2x T_n(x) - T_{n-1}(x) \]
With $T_0(x) = 1$ and $T_1(x) = x$, it follows by induction that $T_n(x)$ is a polynomial of degree $n$. For $n \geq 1$, the leading coefficient $a_n$ satisfies $a_{n+1} = 2a_n$ with $a_1 = 1$, thus $a_n = 2^{n-1}$.
The extrema of $T_n(x)$ in $[-1, 1]$ occur when $|\cos(n\theta)| = 1$, i.e., $n\theta = k\pi$ for $k=0, 1, \dots, n$. This yields $x_k = \cos(k\pi/n)$. At these points, $T_n(x_k) = (-1)^k$, achieving the maximum and minimum values of $\pm 1$.

\item[(b)] The error in polynomial interpolation for $n+1$ nodes $x_0, \dots, x_n$ is given by:
\[ f(x) - P(x) = \frac{f^{(n+1)}(\xi)}{(n+1)!} \prod_{k=0}^n (x - x_k) \]
for some $\xi \in (-1, 1)$. Let $\omega(x) = \prod_{k=0}^n (x - x_k)$. Since $x_k$ are the extrema of $T_n(x)$, they are the roots of $(1-x^2)T_n'(x)$. Note that $T_n'(x) = n U_{n-1}(x)$, where $U_{n-1}$ is the Chebyshev polynomial of the second kind with leading coefficient $2^{n-1}$.
Thus, $(1-x^2)U_{n-1}(x)$ has leading coefficient $-2^{n-1}$. The monic polynomial $\omega(x)$ of degree $n+1$ is then
\[ \omega(x) = -\frac{1}{2^{n-1}}(x^2-1)U_{n-1}(x). \]
On $[-1, 1]$, let $x = \cos \theta$. Then $|(x^2-1)U_{n-1}(x)| = |-\sin^2 \theta \frac{\sin n\theta}{\sin \theta}| = |\sin \theta \sin n\theta| \leq 1$.
Therefore, $\|\omega(x)\|_{\infty} \leq \frac{1}{2^{n-1}}$, and the bound follows.
\end{enumerate}
\end{solution}

\begin{exercise}
Let $S(x)$ be a cubic spline with knots $\{t_{i}\}_{i=0}^{n}$. If it is determined that $S(x)$ is linear over $[t_{1},t_{2}]$ and $[t_{3},t_{4}]$, prove that $S(x)$ is also linear over $[t_{2},t_{3}]$.
\end{exercise}

\begin{solution}
By definition, a cubic spline $S(x)$ belongs to $C^2[t_0, t_n]$. On each subinterval $[t_i, t_{i+1}]$, $S''(x)$ is a linear polynomial.
If $S(x)$ is linear over $[t_1, t_2]$, then $S''(x) = 0$ for all $x \in (t_1, t_2)$. By the continuity of $S''$, we have $S''(t_1) = 0$ and $S''(t_2) = 0$.
Similarly, if $S(x)$ is linear over $[t_3, t_4]$, then $S''(x) = 0$ for all $x \in (t_3, t_4)$, implying $S''(t_4) = 0$ and $S''(t_3) = 0$.
Now consider the interval $[t_2, t_3]$. On this interval, $S''(x)$ is a linear function. Since we have $S''(t_2) = 0$ and $S''(t_3) = 0$, and a linear function is uniquely determined by its values at two points, it follows that $S''(x) \equiv 0$ for all $x \in [t_2, t_3]$.
Integrating $S''(x) = 0$ twice shows that $S(x)$ is a linear function on $[t_2, t_3]$.
\end{solution}

\begin{exercise}
Let $f:\mathbb{R} \to \mathbb{R}$ be defined by $f(x)=2x-\cos x$.
\begin{enumerate}
\item[(a)] Prove that $f(x)=0$ has a unique solution $x^{*}\in\mathbb{R}$ in the interval $(1/4,1/2)$.
\item[(b)] Prove that the sequence defined by the fixed point iteration $x_{n}=\frac{1}{2}\cos x_{n-1}$ converges to $x^{*}$ for any initial guess $x_{0}$.
\item[(c)] For $x_{0}=\pi/6$, determine $n$ to guarantee $|x_{n}-x^{*}| < \frac{1}{2}\times 10^{-8}$. For $x_{0}=20$, determine $n$ to guarantee $|x_{n}-x^{*}| < 1/4$.
\end{enumerate}
\end{exercise}

\begin{solution}
\begin{enumerate}
\item[(a)] $f(1/4) = 1/2 - \cos(1/4) \approx 0.5 - 0.9689 < 0$ and $f(1/2) = 1 - \cos(1/2) \approx 1 - 0.8776 > 0$. By the Intermediate Value Theorem, there exists $x^* \in (1/4, 1/2)$ such that $f(x^*) = 0$. Since $f'(x) = 2 + \sin x \geq 1 > 0$, $f$ is strictly increasing, so $x^*$ is unique.
\item[(b)] The equation $f(x)=0$ is equivalent to $x = g(x)$ where $g(x) = \frac{1}{2}\cos x$. For any $x, y \in \mathbb{R}$, $|g'(x)| = |-\frac{1}{2}\sin x| \leq \frac{1}{2} = L < 1$. By the Banach Fixed Point Theorem, $g$ is a contraction mapping on $\mathbb{R}$, and the iteration $x_n = g(x_{n-1})$ converges to the unique fixed point $x^*$ for any $x_0$.
\item[(c)] The error estimate is $|x_n - x^*| \leq \frac{L^n}{1-L}|x_1 - x_0| = 2^{1-n}|x_1 - x_0|$.
For $x_0 = \pi/6$, $x_1 = \frac{1}{2}\cos(\pi/6) = \frac{\sqrt{3}}{4}$, so $|x_1 - x_0| \approx |0.4330 - 0.5236| = 0.0906$.
We require $2^{1-n} (0.0906) < 0.5 \times 10^{-8} \implies 2^n > 3.624 \times 10^7 \implies n \geq 26$.
For $x_0 = 20$, $x_1 = \frac{1}{2}\cos(20) \approx 0.204$. $|x_1 - x_0| \approx 19.796$.
We require $2^{1-n} (19.796) < 0.25 \implies 2^n > 158.37 \implies n \geq 8$.
\end{enumerate}
\end{solution}

\begin{exercise}
Let $\mathbf{A}\in\mathbb{R}^{m\times n}$ with $m\geq n$ and $r=\mathrm{rank}(\mathbf{A})<n$. Assume $\mathbf{A}$ has the SVD decomposition:
\[
\mathbf{A}=[\mathbf{U}_{1},\mathbf{U}_{2}]\begin{bmatrix}\Sigma_{1}&0\\0&0\end{bmatrix}[\mathbf{V}_{1},\mathbf{V}_{2}]^{T}=\mathbf{U}_{1}\Sigma_{1}\mathbf{V}_{1}^{T},
\]
where $\Sigma_{1}$ is $r\times r$ nonsingular and $\sigma=\sigma_{\min}(\Sigma_{1})$.
Consider the least square problem: $\min_{\mathbf{x}\in\mathbb{R}^{n}}\|\mathbf{Ax}-\mathbf{b}\|_{2}$.
\begin{enumerate}
\item[(a)] Show that all solutions can be written as $\mathbf{x}=\mathbf{V}_{1}\Sigma_{1}^{-1}\mathbf{U}_{1}^{T}\mathbf{b}+\mathbf{V}_{2}\mathbf{z}_{2}$ with arbitrary $\mathbf{z}_{2}$.
\item[(b)] Show that the solution has minimal norm precisely when $\mathbf{z}_{2}=0$, and in which case $\|\mathbf{x}\|_{2}\leq\|\mathbf{b}\|_{2}/\sigma$.
\end{enumerate}
\end{exercise}

\begin{solution}
\begin{enumerate}
\item[(a)] Let $\mathbf{x} = \mathbf{V}\mathbf{y} = \mathbf{V}_1 \mathbf{y}_1 + \mathbf{V}_2 \mathbf{y}_2$. Then $\|\mathbf{Ax} - \mathbf{b}\|_2 = \|\mathbf{U}\Sigma\mathbf{y} - \mathbf{b}\|_2 = \|\Sigma \mathbf{y} - \mathbf{U}^T \mathbf{b}\|_2$.
Expanding the blocks:
\[ \|\mathbf{Ax} - \mathbf{b}\|_2^2 = \|\Sigma_1 \mathbf{y}_1 - \mathbf{U}_1^T \mathbf{b}\|_2^2 + \|\mathbf{U}_2^T \mathbf{b}\|_2^2. \]
The minimum is achieved when $\Sigma_1 \mathbf{y}_1 = \mathbf{U}_1^T \mathbf{b} \implies \mathbf{y}_1 = \Sigma_1^{-1} \mathbf{U}_1^T \mathbf{b}$. The vector $\mathbf{y}_2$ (denoted as $\mathbf{z}_2$) is unconstrained and thus arbitrary.
Therefore, $\mathbf{x} = \mathbf{V}_1 \Sigma_1^{-1} \mathbf{U}_1^T \mathbf{b} + \mathbf{V}_2 \mathbf{z}_2$.
\item[(b)] Using the orthogonality of $\mathbf{V}_1$ and $\mathbf{V}_2$, $\|\mathbf{x}\|_2^2 = \|\mathbf{y}_1\|_2^2 + \|\mathbf{y}_2\|_2^2$. This is minimized for $\mathbf{y}_2 = \mathbf{z}_2 = 0$.
The minimal norm solution is $\mathbf{x}^\dagger = \mathbf{V}_1 \Sigma_1^{-1} \mathbf{U}_1^T \mathbf{b}$. Its norm is
\[ \|\mathbf{x}^\dagger\|_2^2 = \|\Sigma_1^{-1} \mathbf{U}_1^T \mathbf{b}\|_2^2 \leq \|\Sigma_1^{-1}\|_2^2 \|\mathbf{U}_1^T \mathbf{b}\|_2^2 \leq \frac{1}{\sigma^2} \|\mathbf{b}\|_2^2. \]
Thus $\|\mathbf{x}\|_2 \leq \|\mathbf{b}\|_2/\sigma$.
\end{enumerate}
\end{solution}

\begin{exercise}
Consider the family of semi-implicit Runge-Kutta methods:
\[
\begin{array}{l}
k_{1}=f(y_{n}+\beta hk_{1}),\quad k_{2}=f(y_{n}+hk_{1}+\beta hk_{2}),\\
y_{n+1}=y_{n}+h((\frac{1}{2}+\beta)k_{1}+(\frac{1}{2}-\beta)k_{2}).
\end{array}
\]
\begin{enumerate}
\item[(a)] Determine the order and the principal part of the local truncation error.
\item[(b)] Show that if $\beta > 1/2$, then the negative real axis is contained in the region of absolute stability.
\end{enumerate}
\end{exercise}

\begin{solution}
\begin{enumerate}
\item[(a)] Apply the method to $y' = \lambda y$. Let $z = \lambda h$.
$k_1 = \lambda (y_n + \beta z k_1) \implies k_1 = \frac{\lambda y_n}{1 - \beta z}$.
$k_2 = \lambda (y_n + z k_1 + \beta z k_2) \implies k_2 = \frac{\lambda y_n (1 - \beta z + z)}{(1 - \beta z)^2}$.
Substituting into $y_{n+1} = y_n + h(b_1 k_1 + b_2 k_2)$ gives the stability function:
\[ R(z) = \frac{1 + (1-2\beta) z + (\beta^2 - 2\beta + 1/2) z^2}{(1 - \beta z)^2} \]
Expanding $R(z)$ in powers of $z$:
\[ R(z) = 1 + z + \frac{1}{2} z^2 + (\beta - \beta^2) z^3 + O(z^4) \]
Comparing with $e^z = 1 + z + \frac{1}{2} z^2 + \frac{1}{6} z^3 + O(z^4)$, we see the method is of order $2$ for any $\beta$.
The principal part of the local truncation error is $(\beta - \beta^2 - 1/6) h^3 y^{(3)}(t_n)$.
\item[(b)] For $z = x < 0$, let $x = -X$ ($X > 0$). Then $R(-X) = \frac{1 - (1-2\beta)X + (\beta^2 - 2\beta + 1/2) X^2}{(1 + \beta X)^2}$.
We have $|R(-X)| \leq 1 \iff -1 \leq R(-X) \leq 1$.
$R(-X) \leq 1 \iff 1 + (2\beta-1)X + (\beta^2-2\beta+1/2)X^2 \leq 1 + 2\beta X + \beta^2 X^2 \iff -X + (-2\beta+1/2)X^2 \leq 0$.
For $\beta > 1/2$, both coefficients are negative, so $R(-X) < 1$.
$R(-X) \geq -1 \iff 2 + (4\beta-1)X + (2\beta^2-2\beta+1/2)X^2 \geq 0 \iff 2 + (4\beta-1)X + 2(\beta-1/2)^2 X^2 \geq 0$.
Since $\beta > 1/2$, all coefficients are positive for $X > 0$.
Thus the method is absolutely stable on the negative real axis.
\end{enumerate}
\end{solution}

\begin{exercise}
Consider the Beam equation: $u^{(4)}=f(x)$ on $(0,1)$ with $u(0)=u'(0)=u''(1)=u'''(1)=0$.
\begin{enumerate}
\item[(a)] Recast this equation into a variational problem.
\item[(b)] Interpret as an energy minimization problem and prove equivalence.
\item[(c)] Develop a CG(3) (cubic continuous Galerkin) finite element method.
\item[(d)] Prove an a priori error estimate in the energy norm: $\|e\|_{E}=(\int_{0}^{1}(e'')^{2}dx)^{1/2}$.
\item[(e)] Prove an a priori error estimate in the $L_{2}$ norm.
\end{enumerate}
\end{exercise}

\begin{solution}
\begin{enumerate}
\item[(a)] Let $V = \{ v \in H^2(0, 1) : v(0) = v'(0) = 0 \}$. Multiplying by $v$ and integrating by parts twice:
\[ \int_0^1 u^{(4)}v \, dx = [u'''v - u''v']_0^1 + \int_0^1 u'' v'' \, dx = \int_0^1 f v \, dx. \]
Given $u''(1)=u'''(1)=0$ and $v(0)=v'(0)=0$, the boundary terms vanish.
Variational Problem: Find $u \in V$ such that $a(u, v) = L(v)$ for all $v \in V$, with $a(u, v) = \int_0^1 u'' v'' \, dx$ and $L(v) = \int_0^1 f v \, dx$.
\item[(b)] The energy functional is $J(v) = \frac{1}{2} a(v, v) - L(v)$. Since $a(\cdot, \cdot)$ is symmetric and coercive on $V$ (by Poincar\'{e}), $u = \arg\min J$ iff $dJ(u; v) = a(u, v) - L(v) = 0$.
\item[(c)] Let $V_h \subset V$ consist of piecewise cubic polynomials with $C^1$ continuity (Hermite elements) on a mesh of size $h$. The FEM solution $U \in V_h$ satisfies $a(U, v_h) = L(v_h)$ for all $v_h \in V_h$.
\item[(d)] By Cea's Lemma, $\|u - U\|_E \le \min_{v_h \in V_h} \|u - v_h\|_E$. For cubic elements, interpolation theory gives $\|u - I_h u\|_{H^2} \leq C h^{4-2} |u|_{H^4} = C h^2 \|f\|_{L^2}$. Thus $\|e\|_E \leq C h^2 \|f\|_{L^2}$.
\item[(e)] Using the Aubin-Nitsche trick, solve $\psi^{(4)} = e$ with BCs. Then $\|e\|_{L^2}^2 = a(e, \psi) = a(e, \psi - I_h \psi) \leq \|e\|_E \|\psi - I_h \psi\|_E \leq (C h^2 \|f\|) (C h^2 |u|_{H^4}) \leq C h^4 \|f\|_{L^2}$.
\end{enumerate}
\end{solution}
