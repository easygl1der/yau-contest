\begin{exercise}
\label{applied:2011:1}
Ming Antu derived the following half-angle formula in 1730:
\[
\sin^{2}\frac{\theta}{2} = \sum_{n=1}^{\infty} c_{n-1}\left(\frac{\sin\theta}{2}\right)^{2n}
\]
where the Catalan numbers are
\[
c_{n} = \frac{1}{n+1}\binom{2n}{n}, \quad n=0,1,2,\dots
\]

Prove this formula.

\textit{Hint: Use the generating function $F(z) = \sum_{n=0}^{\infty}c_{n}z^{n}$ and show that $\sum_{m+k=n}c_{m}c_{k}=c_{n+1}$, then show $zF(z)^{2}=F(z)-1$.}
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item Definition and recurrence relations of Catalan numbers.
    \item Methods of generating functions.
    \item Trigonometric half-angle identities ($\sin^2(\theta/2) = (1-\cos\theta)/2$).
\end{itemize}

\textbf{Steps:}
\begin{enumerate}
    \item \textbf{Catalan Generating Function:} Let $F(z) = \sum_{n=0}^\infty c_n z^n$. The recurrence $\sum_{m+k=n} c_m c_k = c_{n+1}$ implies $F(z)^2 = \sum_{n=0}^\infty c_{n+1} z^n = (F(z) - 1)/z$. Thus, $z F(z)^2 - F(z) + 1 = 0$. Solving the quadratic equation gives $F(z) = \frac{1 - \sqrt{1-4z}}{2z}$ (the minus sign is chosen as $F(0)=c_0=1$).

    \item \textbf{Series Summation:} Let $z = (\frac{\sin\theta}{2})^2 = \frac{\sin^2\theta}{4}$. The given sum is $S = \sum_{n=1}^\infty c_{n-1} z^n = z \sum_{m=0}^\infty c_m z^m = z F(z)$.
    
    \item \textbf{Trigonometric Substitution:} Substituting the expression for $F(z)$:
    \[ z F(z) = z \left( \frac{1 - \sqrt{1-4z}}{2z} \right) = \frac{1 - \sqrt{1 - \sin^2\theta}}{2} = \frac{1 - |\cos\theta|}{2}. \]
    For small $\theta$, $\cos\theta > 0$, so $\frac{1-\cos\theta}{2} = \sin^2\frac{\theta}{2}$.
\end{enumerate}

\textbf{Intuition:}
The problem reveals a deep connection between combinatorial structures (Catalan numbers) and geometry (trigonometry). The square root in the Catalan generating function is essentially the same square root appearing in the Pythagorean identity $\cos\theta = \sqrt{1-\sin^2\theta}$, allowing the power series to sum exactly to a trigonometric function.
\end{solution}

\begin{exercise}
\label{applied:2011:2}
Many algorithms, including polynomial factorisation in finite fields, require computing $\gcd(f(X), X^{N}-1)$ for a polynomial $f$ of reasonably small degree $n$ and a binomial $X^{N}-1$ of very large degree $N$.

\begin{enumerate}
\item[(i)] Suggest a more efficient approach for computing $\gcd(f(X), X^{N}-1)$ via the Euclid algorithm.
\item[(ii)] Generalise it to $\gcd(f(X), A_{1}X^{N_{1}}+\dots+A_{m}X^{N_{m}}+A_{m+1})$.
\end{enumerate}

\textit{Hint: If for three polynomials $f$, $g$ and $h$ we have $g\equiv h\pmod{f}$, then $\gcd(f,g)=\gcd(f,h)$.}
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item Euclidean algorithm for GCD of polynomials.
    \item Properties of modular arithmetic in the ring $\mathbb{K}[X]/(f(X))$.
    \item Complexity of binary exponentiation.
\end{itemize}

\textbf{Steps:}
\begin{enumerate}
    \item \textbf{Efficient Powering:} To compute $\gcd(f(X), X^N-1)$, we first compute $R(X) = X^N \pmod{f(X)}$ using binary exponentiation (square-and-multiply). This takes $O(\log N)$ polynomial multiplications. Since $\text{deg}(f) = n$, each multiplication takes $O(n^2)$ (or $O(n \log n)$ with FFT).

    \item \textbf{GCD Reduction:} Use the identity $\gcd(f, X^N-1) = \gcd(f, R(X)-1)$. Once $R(X)$ is found, the second GCD involves polynomials of degree at most $n$, which is solved in $O(n^2)$ time. Total complexity: $O(n^2 \log N)$.

    \item \textbf{Generalization:} For $g(X) = \sum A_i X^{N_i} + A_{m+1}$, compute each $R_i(X) = X^{N_i} \pmod{f(X)}$ individually in $O(n^2 \log N_i)$. Then compute $R(X) = (\sum A_i R_i(X) + A_{m+1}) \pmod{f(X)}$ and perform $\gcd(f, R)$.
\end{enumerate}

\textbf{Intuition:}
In GCD computations, the primary cost is the initial reduction. When one polynomial has a very high degree but sparse or structured exponents, we can bypass the $O(N)$ division process by using modular exponentiation, which is exponentially faster.
\end{solution}

\begin{exercise}
\label{applied:2011:3}
For solving the partial differential equation
\[
u_{t}+f(u)_{x}=0, \quad 0\leq x\leq 1
\]
where $f'(u)\geq 0$, with periodic boundary condition, we use the semi-discrete upwind scheme:
\[
\frac{d}{dt}u_{j}+\frac{f(u_{j})-f(u_{j-1})}{\Delta x}=0, \quad j=1,2,\dots,N
\]
with periodic boundary condition $u_{0}=u_{N}$.

\begin{enumerate}
\item[(i)] Prove the $L^{2}$ stability of the scheme: $\frac{d}{dt}E(t)\leq 0$ where $E(t)=\sum_{j=1}^{N}|u_{j}|^{2}\Delta x$.
\item[(ii)] Is the result true for $E(t)=\sum_{j=1}^{N}|u_{j}|^{2p}\Delta x$ for arbitrary integer $p\geq 1$?
\end{enumerate}
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item $L^p$ energy estimates for PDEs.
    \item Periodic summation by parts.
    \item Properties of monotonically increasing functions ($f' \ge 0$).
\end{itemize}

\textbf{Steps:}
\begin{enumerate}
    \item \textbf{$L^2$ Stability:} Differentiate $E(t)$ to get $\frac{d}{dt}E(t) = -2 \sum_{j=1}^N u_j(f(u_j) - f(u_{j-1}))$. Define $G(u) = \int^u f(s) ds$. Note that $u f(u) - G(u)$ is increasing since its derivative is $u f'(u)$. By the Mean Value Theorem for integrals, $u_j (f(u_j) - f(u_{j-1})) \ge G(u_j) - G(u_{j-1})$ given $f' \ge 0$. Summing over the periodic domain makes the $G$ terms cancel, leaving $\frac{d}{dt}E \le 0$.

    \item \textbf{$L^{2p}$ Generalization:} Use $h(u) = u^{2p-1}$, which is non-decreasing. The energy derivative is proportional to $-\sum h(u_j)(f(u_j) - f(u_{j-1}))$. Applying the same integral argument to the pair $(h, f)$, the positive semi-definiteness of the dissipation term holds for any $p \ge 1$.
\end{enumerate}

\textbf{Intuition:}
The upwind scheme is designed to respect the physical direction of information. When $f' \ge 0$, information travels from left to right. By using the left neighbor, the scheme introduces a numerical dissipation (entropy production) that stabilizes the solution by damping oscillations in any $L^p$ norm.
\end{solution}

\begin{exercise}
\label{applied:2011:4}
Let $A$ be an $n\times n$ matrix with real and positive eigenvalues and $b$ be a given vector. Consider Richardson's iteration:
\[
x^{(k+1)}=(I-\omega A)x^{(k)}+\omega b
\]
Let $\lambda_{1}, \dots, \lambda_{n}$ be the eigenvalues of $A$ in $[ \lambda_1, \lambda_n ]$.

\begin{enumerate}
\item[(i)] Prove Richardson's iteration converges if and only if $0<\omega<\frac{2}{\lambda_{n}}$.
\item[(ii)] Prove the optimal choice is $\omega_{\mathrm{opt}}=\frac{2}{\lambda_{1}+\lambda_{n}}$ and find the spectral radius $\rho(G_{\omega})$.
\item[(iii)] If $A$ is SPD, prove $\rho(G_{\omega_{\mathrm{opt}}})=\frac{\kappa_{2}(A)-1}{\kappa_{2}(A)+1}$.
\end{enumerate}
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item Linear iterative methods for systems $Ax=b$.
    \item Spectral radius and convergence criteria ($\rho(G) < 1$).
    \item Condition number $\kappa_2(A)$ for symmetric positive definite (SPD) matrices.
\end{itemize}

\textbf{Steps:}
\begin{enumerate}
    \item \textbf{Convergence Boundary:} The error satisfies $e^{(k+1)} = (I - \omega A) e^{(k)}$. Convergence requires eigenvalues $1 - \omega \lambda_i \in (-1, 1)$. This implies $0 < \omega \lambda_i < 2$. The most restrictive eigenvalue is $\lambda_n$, hence $0 < \omega < 2/\lambda_n$.

    \item \textbf{Optimal Parameter:} We minimize $\rho(I - \omega A) = \max_i |1 - \omega \lambda_i|$. The minimum occurs when the errors for the smallest and largest eigenvalues are equal in magnitude but opposite in sign: $1 - \omega \lambda_1 = -(1 - \omega \lambda_n)$. Solving gives $\omega_{\mathrm{opt}} = 2/(\lambda_1 + \lambda_n)$. Then $\rho_{\mathrm{opt}} = (\lambda_n - \lambda_1)/(\lambda_n + \lambda_1)$.

    \item \textbf{Condition Number:} For SPD matrices, $\lambda_n = \lambda_{\max}$ and $\lambda_1 = \lambda_{\min}$. Dividing numerator and denominator of $\rho_{\mathrm{opt}}$ by $\lambda_1$ gives $(\kappa_2(A)-1)/(\kappa_2(A)+1)$.
\end{enumerate}

\textbf{Intuition:}
Richardson's iteration is a gradient descent process with a fixed step size. To converge as fast as possible, we must choose a step size that prevents the "fastest" modes (large eigenvalues) from overshooting while still allowing the "slowest" modes (small eigenvalues) to progress.
\end{solution}

\begin{exercise}
\label{applied:2011:5}
Show that for the backward Euler discretization of the heat equation $u_t = u_{xx}$:
\[
\frac{1}{2}\sum_{j=1}^{N-1}\left((U_{j}^{n+1})^{2}-(U_{j}^{n})^{2}\right)\leq-\lambda\sum_{j=1}^{N-2}(U_{j+1}^{n+1}-U_{j}^{n+1})^{2}-\frac{\lambda}{2}\left((U_{1}^{n+1})^{2}+(U_{N-1}^{n+1})^{2}\right)+\frac{\lambda}{2}(u_{0}^{2}+u_{1}^{2}).
\]
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item Backward Euler scheme for parabolic equations.
    \item Discrete energy method.
    \item Summation by parts and Cauchy-Schwarz inequality.
\end{itemize}

\textbf{Steps:}
\begin{enumerate}
    \item \textbf{Energy Expansion:} Start with the scheme $U_j^{n+1} - U_j^n = \lambda \delta^2 U_j^{n+1}$. Multiply by $U_j^{n+1}$ and sum over $j$. Use the identity $(a-b)a \ge \frac{1}{2}(a^2 - b^2)$ for the LHS.

    \item \textbf{Summation by Parts:} The RHS sum $\sum U_j^{n+1} (U_{j+1}^{n+1} - 2U_j^{n+1} + U_{j-1}^{n+1})$ is rearranged using discrete summation by parts. This generates negative quadratic terms like $-(U_{j+1} - U_j)^2$ (representing dissipation) and boundary products.

    \item \textbf{Boundary Term Bounding:} Use $ab \le \frac{1}{2}(a^2 + b^2)$ on the boundary terms $u_0 U_1$ and $u_1 U_{N-1}$ to transform them into squares of the grid values and boundary data, matching the structure of the target inequality.
\end{enumerate}

\textbf{Intuition:}
This estimate proves the unconditional stability of the backward Euler method. Physically, it shows that the thermal energy of the system decreases at every time step due to internal diffusion, with the only possible increase coming from the boundary values.
\end{solution}

\begin{exercise}
\label{applied:2011:6}
Show for the Conjugate Gradient (CG) method that:
\begin{enumerate}
\item[(a)] $\alpha_{n}$ minimizes $f(\mathbf{x}_{n}+\alpha\mathbf{p}_{n})$ for $f(\mathbf{x})\equiv\frac{1}{2}\mathbf{x}^{T}A\mathbf{x}-\mathbf{b}^{T}\mathbf{x}$.
\item[(b)] $\mathbf{p}_{i}^{T}A\mathbf{p}_{j}=0$ for $i \neq j$.
\item[(c)] $K_n = \mathrm{Span}\{\mathbf{r}_{0}, A\mathbf{r}_{0}, \dots, A^{n-1}\mathbf{r}_{0}\}$.
\end{enumerate}
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item Quadratic minimization problems.
    \item Gram-Schmidt $A$-orthogonalization.
    \item Definition of Krylov subspaces $K_n(A, r_0)$.
\end{itemize}

\textbf{Steps:}
\begin{enumerate}
    \item \textbf{Optimal Step Size:} Set $\frac{d}{d\alpha} f(x_n + \alpha p_n) = 0$. The gradient is $Ax - b = -r$. Thus $(A(x_n + \alpha p_n) - b)^T p_n = 0$, which gives $\alpha_n = (r_n^T p_n) / (p_n^T A p_n)$.

    \item \textbf{Conjugacy:} The search directions $p_n$ are constructed via $p_n = r_n + \beta_{n-1} p_{n-1}$. By choosing $\beta$ to satisfy $p_n^T A p_{n-1} = 0$, and using the symmetry of $A$, induction shows that higher search directions remain $A$-orthogonal to all previous ones.

    \item \textbf{Krylov Subspace:} Since $x_{n+1} = x_n + \alpha_n p_n$ and $r_{n+1} = r_n - \alpha_n A p_n$, the residuals and directions are linear combinations of $A^k r_0$ for $k < n$. Thus, the span of the first $n$ residuals is exactly the $n$-th Krylov subspace.
\end{enumerate}

\textbf{Intuition:}
CG is an optimal solver for SPD systems because it builds an $A$-orthogonal basis for the Krylov subspace. Unlike steepest descent, which might search in the same direction twice, CG ensures that each search direction is "new" with respect to the $A$-norm, covering the subspace as efficiently as possible.
\end{solution}

\begin{exercise}
\label{applied:2011:7}
Find $a, b, c$ for the second-order scheme $u_{j}^{n+1}=au_{j-2}^{n}+bu_{j-1}^{n}+cu_{j}^{n}$ for $u_t + u_x = 0$.
\end{exercise}

\begin{solution}
\textbf{Prerequisites:}
\begin{itemize}
    \item Method of undetermined coefficients for finite differences.
    \item Taylor series expansion of functions of two variables.
    \item Consistency order of numerical schemes.
\end{itemize}

\textbf{Steps:}
\begin{enumerate}
    \item \textbf{Taylor Expansion:} Expand $u(x, t+\Delta t)$ to $O(\Delta t^2)$ and $u(x-k\Delta x, t)$ to $O(\Delta x^2)$. Using $u_t = -u_x$ and $u_{tt} = u_{xx}$, the LHS is $u - \lambda \Delta x u_x + \frac{1}{2} \lambda^2 \Delta x^2 u_{xx}$.

    \item \textbf{Matching Moments:} 
    \begin{itemize}
        \item Sum of coefficients: $a+b+c = 1$.
        \item First moment: $2a + b = \lambda$ (matches $u_x$ term).
        \item Second moment: $4a + b = \lambda^2$ (matches $u_{xx}$ term).
    \end{itemize}

    \item \textbf{Solving for $a, b, c$:} Subtracting the equations gives $2a = \lambda^2 - \lambda \implies a = \frac{\lambda(\lambda-1)}{2}$. Substituting back gives $b = \lambda(2-\lambda)$ and $c = \frac{(1-\lambda)(2-\lambda)}{2}$.
\end{enumerate}

\textbf{Intuition:}
This configuration is known as the Beam-Warming scheme. It is an upwind-biased second-order scheme. By including an additional point further upstream ($u_{j-2}$), the scheme can achieve higher accuracy for advection problems while specifically targeting information coming from the upstream direction, which improves stability compared to centered schemes.
\end{solution}
