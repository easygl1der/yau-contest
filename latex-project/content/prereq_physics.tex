\chapter{Mathematical Physics}
\label{sec:prereq-physics}

This section provides prerequisite knowledge, definitions, theorems, and examples for the Mathematical Physics track.

\section{Classical Mechanics}

\textbf{Key Topics:} Lagrangian and Hamiltonian formalisms, central potentials, orbital stability, constrained motion, and Noether's theorem.

\begin{definition}[Lagrangian and Hamiltonian]
The \emph{Lagrangian} $L$ of a classical system is defined as $L = T - V$, where $T$ is the kinetic energy and $V$ is the potential energy. The \emph{Hamiltonian} $H$ is obtained via a Legendre transformation: $H(q, p, t) = \sum_i p_i \dot{q}_i - L(q, \dot{q}, t)$, where $p_i = \frac{\partial L}{\partial \dot{q}_i}$ are the canonical momenta.
\end{definition}

\begin{theorem}[Noether's Theorem]
Every differentiable symmetry of the action of a physical system has a corresponding conservation law. For example, spatial translation invariance implies conservation of momentum, and time translation invariance implies conservation of energy.
\end{theorem}

\begin{example}[Central Potential]
Consider a particle of mass $m$ in a central potential $V(r)$. The Lagrangian in polar coordinates $(r, \theta)$ is $L = \frac{1}{2}m(\dot{r}^2 + r^2\dot{\theta}^2) - V(r)$. Since $\theta$ is cyclic, the angular momentum $l = mr^2\dot{\theta}$ is conserved. The radial equation of motion reduces to a 1D problem with an effective potential $V_{\text{eff}}(r) = V(r) + \frac{l^2}{2mr^2}$.
\end{example}

\section{Quantum Mechanics}

\textbf{Key Topics:} Harmonic oscillator, ladder operators, perturbation theory, quantum scattering, and coherent states.

\begin{definition}[Ladder Operators]
For a 1D quantum harmonic oscillator with Hamiltonian $\hat{H} = \frac{\hat{p}^2}{2m} + \frac{1}{2}m\omega^2 \hat{x}^2$, the annihilation operator is $\hat{a} = \sqrt{\frac{m\omega}{2\hbar}} \left(\hat{x} + \frac{i}{m\omega}\hat{p}\right)$ and the creation operator is $\hat{a}^\dagger$. They satisfy the commutation relation $[\hat{a}, \hat{a}^\dagger] = 1$. The Hamiltonian is $\hat{H} = \hbar\omega \left(\hat{a}^\dagger \hat{a} + \frac{1}{2}\right)$.
\end{definition}

\begin{theorem}[First-Order Perturbation Theory]
Let $\hat{H} = \hat{H}_0 + \lambda \hat{V}$, where $\hat{H}_0$ has known non-degenerate eigenstates $|n^{(0)}\rangle$ and energies $E_n^{(0)}$. The first-order correction to the energy is given by $E_n^{(1)} = \langle n^{(0)} | \hat{V} | n^{(0)} \rangle$.
\end{theorem}

\begin{example}[Quantum Scattering at a Step Potential]
Consider a particle of mass $m$ and energy $E > V_0$ encountering a potential step $V(x) = V_0 \Theta(x)$. By matching wavefunctions and their derivatives at $x=0$, the reflection coefficient is found to be $\mathcal{R} = \left(\frac{k-k'}{k+k'}\right)^2$, where $k = \sqrt{2mE}/\hbar$ and $k' = \sqrt{2m(E-V_0)}/\hbar$.
\end{example}

\section{Electromagnetism}

\textbf{Key Topics:} Maxwell equations, wave propagation in media, retarded potentials, and the Lorentz gauge.

\begin{definition}[Maxwell Equations]
Maxwell's equations are a set of coupled partial differential equations that describe electromagnetism:
\begin{enumerate}
    \item $\nabla \cdot \mathbf{E} = \frac{\rho}{\varepsilon_0}$ (Gauss's law)
    \item $\nabla \cdot \mathbf{B} = 0$ (Gauss's law for magnetism)
    \item $\nabla \times \mathbf{E} = -\frac{\partial \mathbf{B}}{\partial t}$ (Faraday's law)
    \item $\nabla \times \mathbf{B} = \mu_0 \mathbf{J} + \mu_0 \varepsilon_0 \frac{\partial \mathbf{E}}{\partial t}$ (Amp\`ere-Maxwell law)
\end{enumerate}
\end{definition}

\begin{theorem}[Skin Effect]
An electromagnetic wave penetrating a good conductor with conductivity $\sigma$ is exponentially attenuated. The amplitude decays as $e^{-z/\delta}$, where $\delta = \sqrt{\frac{2}{\mu_0\omega\sigma}}$ is the skin depth and $\omega$ is the angular frequency.
\end{theorem}

\begin{example}[Retarded Potentials]
In the Lorentz gauge $\nabla \cdot \mathbf{A} + \frac{1}{c^2}\frac{\partial V}{\partial t} = 0$, the scalar potential $V$ satisfies a wave equation with the solution $V(\mathbf{r}, t) = \frac{1}{4\pi\varepsilon_0} \int \frac{\rho(\mathbf{r}', t_r)}{|\mathbf{r} - \mathbf{r}'|} d^3r'$, where $t_r = t - \frac{|\mathbf{r} - \mathbf{r}'|}{c}$ is the retarded time.
\end{example}

\section{Statistical Mechanics and General Relativity}

\textbf{Key Topics:} Ising model, Fermi-Dirac statistics, Schwarzschild metric, Killing vectors, and gravitational waves.

\begin{definition}[Partition Function]
In statistical mechanics, the canonical partition function for a system in thermal equilibrium is $Z = \sum_s e^{-\beta E_s}$, where the sum is over all microstates $s$, $E_s$ is the state's energy, and $\beta = \frac{1}{k_B T}$. The free energy is $F = -k_B T \ln Z$.
\end{definition}

\begin{definition}[Schwarzschild Metric]
The Schwarzschild metric is the spherically symmetric vacuum solution to the Einstein field equations. In geometric units ($G=c=1$), it is:
\[ ds^2 = -\left(1 - \frac{2M}{r}\right) dt^2 + \left(1 - \frac{2M}{r}\right)^{-1} dr^2 + r^2 (d\theta^2 + \sin^2\theta d\phi^2) \]
\end{definition}

\begin{theorem}[Killing Vectors and Conserved Quantities]
If a spacetime has a symmetry generated by a Killing vector field $K^\mu$, then for a freely falling particle with four-momentum $p^\mu$, the quantity $Q = K_\mu p^\mu$ is conserved along its geodesic.
\end{theorem}

\begin{example}[Fermi Gas at Zero Temperature]
For a gas of non-interacting fermions at absolute zero ($T=0$), particles fill the lowest available energy states up to the Fermi energy $E_F = \frac{\hbar^2 k_F^2}{2m}$. This yields a degeneracy pressure that stabilizes white dwarf stars.
\end{example}

\section{Quantum Field Theory}

\textbf{Key Topics:} Feynman diagrams, renormalization, scalar field theories, and conformal invariance.

\begin{definition}[Feynman Diagrams]
Feynman diagrams graphically represent the perturbative expansion of scattering amplitudes in QFT. Propagators represent free particles, while vertices correspond to interactions between them.
\end{definition}

\begin{theorem}[Dimensional Regularization]
To handle divergent loop integrals in QFT, dimensional regularization evaluates integrals in $d=4-\epsilon$ dimensions. Divergences appear as $1/\epsilon$ poles, which can be systematically subtracted (e.g., in the $\overline{\text{MS}}$ scheme) via renormalization to yield finite physical predictions.
\end{theorem}

\begin{example}[One-Loop Self-Energy in $\phi^4$ Theory]
In massless scalar $\phi^4$ theory, the interaction term is $-\frac{\lambda}{4!} \phi^4$. The one-loop correction to the propagator is quadratically divergent in 4 dimensions. Renormalization introduces a mass counterterm to absorb this divergence, allowing for a well-defined renormalized mass.
\end{example}
